142 static void put_chunk_of_file_finished(KineticCompletionData* kinetic_data, void* clientData)
143 {
144  AsyncWriteClosureData* closureData = clientData;
145  FileTransferProgress* currentTransfer = closureData->currentTransfer;
146  free(closureData);
147  currentTransfer->opsInProgress--;
148 
149  if (kinetic_data->status == KINETIC_STATUS_SUCCESS) {
150  int bytesPut = put_chunk_of_file(currentTransfer);
151  if (bytesPut <= 0 && currentTransfer->opsInProgress == 0) {
152  if (currentTransfer->status == KINETIC_STATUS_NOT_ATTEMPTED) {
153  currentTransfer->status = KINETIC_STATUS_SUCCESS;
154  }
155  pthread_cond_signal(&currentTransfer->completeCond);
156  }
157  }
158  else {
159  currentTransfer->status = kinetic_data->status;
160  // only signal when finished
161  // keep track of outstanding operations
162  // if there is no more data to read (or error), and no outstanding operations,
163  // then signal
164  pthread_cond_signal(&currentTransfer->completeCond);
165  fprintf(stderr, "Failed writing chunk! PUT response reported status: %s\n",
