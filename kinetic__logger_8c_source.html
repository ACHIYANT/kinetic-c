201 static int bytetoa(char* p_buf, uint8_t val)
202 {
203  // KineticLogger_LogPrintf(log_level, "Converting byte=%02u", val);
204  const uint8_t base = 16;
205  const int width = 2;
206  int i = width;
207  char c = 0;
208 
209  p_buf += width - 1;
210  do {
211  c = val % base;
212  val /= base;
213  if (c >= 10) c += 'A' - '0' - 10;
214  c += '0';
215  *p_buf-- = c;
216  }
217  while (--i);
218  return width;
219 }

233 static void LogUnboxed(int log_level,
234  void const * const fieldData,
235  ProtobufCFieldDescriptor const * const fieldDesc,
236  size_t const i,
237  char* indent)
238 {
239  switch (fieldDesc->type) {
240  case PROTOBUF_C_TYPE_INT32:
241  case PROTOBUF_C_TYPE_SINT32:
242  case PROTOBUF_C_TYPE_SFIXED32:
243  {
244  int32_t const * value = (int32_t const *)fieldData;
245  KineticLogger_LogPrintf(log_level, "%s%s: %ld", indent, fieldDesc->name, value[i]);
246  }
247  break;
248 
249  case PROTOBUF_C_TYPE_INT64:
250  case PROTOBUF_C_TYPE_SINT64:
251  case PROTOBUF_C_TYPE_SFIXED64:
252  {
253  int64_t* value = (int64_t*)fieldData;
254  KineticLogger_LogPrintf(log_level, "%s%s: %lld", indent, fieldDesc->name, value[i]);
255  }
256  break;
257 
258  case PROTOBUF_C_TYPE_UINT32:
259  case PROTOBUF_C_TYPE_FIXED32:
260  {
261  uint32_t* value = (uint32_t*)fieldData;
262  KineticLogger_LogPrintf(log_level, "%s%s: %lu", indent, fieldDesc->name, value[i]);
263  }
264  break;
265 
266  case PROTOBUF_C_TYPE_UINT64:
267  case PROTOBUF_C_TYPE_FIXED64:
268  {
269  uint64_t* value = (uint64_t*)fieldData;
270  KineticLogger_LogPrintf(log_level, "%s%s: %llu", indent, fieldDesc->name, value[i]);
271  }
272  break;
273 
274  case PROTOBUF_C_TYPE_FLOAT:
275  {
276  float* value = (float*)fieldData;
277  KineticLogger_LogPrintf(log_level, "%s%s: %f", indent, fieldDesc->name, value[i]);
278  }
279  break;
280 
281  case PROTOBUF_C_TYPE_DOUBLE:
282  {
283  double* value = (double*)fieldData;
284  KineticLogger_LogPrintf(log_level, "%s%s: %f", indent, fieldDesc->name, value[i]);
285  }
286  break;
287 
288  case PROTOBUF_C_TYPE_BOOL:
289  {
290  protobuf_c_boolean* value = (protobuf_c_boolean*)fieldData;
291  KineticLogger_LogPrintf(log_level, "%s%s: %s", indent, fieldDesc->name, BOOL_TO_STRING(value[i]));
292  }
293  break;
294 
295  case PROTOBUF_C_TYPE_STRING:
296  {
297  char** strings = (char**)fieldData;
298  KineticLogger_LogPrintf(log_level, "%s%s: %s", indent, fieldDesc->name, strings[i]);
299  }
300  break;
301 
302  case PROTOBUF_C_TYPE_BYTES:
303  {
304  ProtobufCBinaryData* value = (ProtobufCBinaryData*)fieldData;
305  log_proto_level_start(fieldDesc->name);
306  KineticLogger_LogByteArray(log_level, indent,
307  (ByteArray){.data = value[i].data,
308  .len = value[i].len});
309  log_proto_level_end();
310  }
311  break;
312 
313  case PROTOBUF_C_TYPE_ENUM:
314  {
315  int * value = (int*)fieldData;
316  ProtobufCEnumDescriptor const * enumDesc = fieldDesc->descriptor;
317  ProtobufCEnumValue const * enumVal = protobuf_c_enum_descriptor_get_value(enumDesc, value[i]);
318  KineticLogger_LogPrintf(log_level, "%s%s: %s", indent, fieldDesc->name, enumVal->name);
319  }
320  break;
321 
322  case PROTOBUF_C_TYPE_MESSAGE: // nested message
323  {
324  ProtobufCMessage** msg = (ProtobufCMessage**)fieldData;
325  if (msg[i] != NULL)
326  {
327  log_proto_level_start(fieldDesc->name);
328  log_protobuf_message(log_level, msg[i], indent);
329  log_proto_level_end();
330  }
331  } break;
332 
333  default:
334  KineticLogger_LogPrintf(log_level, "Invalid message field type!: %d", fieldDesc->type);
335  KINETIC_ASSERT(false); // should never get here!
336  break;
337  };
338 }

340 static void log_protobuf_message(int log_level, ProtobufCMessage const * msg, char* indent)
341 {
342  if (msg == NULL || msg->descriptor == NULL || !is_level_enabled(log_level)) {
343  return;
344  }
345 
346  ProtobufCMessageDescriptor const * desc = msg->descriptor;
347  uint8_t const * pMsg = (uint8_t const *)msg;
348 
349  for (unsigned int i = 0; i < desc->n_fields; i++) {
350  ProtobufCFieldDescriptor const * fieldDesc = &desc->fields[i];
351 
352  if (fieldDesc == NULL) {
353  continue;
354  }
355 
356  switch(fieldDesc->label)
357  {
358  case PROTOBUF_C_LABEL_REQUIRED:
359  {
360  LogUnboxed(log_level, &pMsg[fieldDesc->offset], fieldDesc, 0, indent);
361  } break;
362  case PROTOBUF_C_LABEL_OPTIONAL:
363  {
364  protobuf_c_boolean const * quantifier = (protobuf_c_boolean const *)(void*)&pMsg[fieldDesc->quantifier_offset];
365  if ((*quantifier) && // only print out if it's there
366  // and a special case: if this is a message, don't show it if the message is NULL
367  (PROTOBUF_C_TYPE_MESSAGE != fieldDesc->type || ((ProtobufCMessage**)(void*)&pMsg[fieldDesc->offset])[0] != NULL))
368  {
369  // special case for nested command packed into commandBytes field
370  if ((protobuf_c_message_descriptor_get_field_by_name(desc, "commandBytes") == fieldDesc ) &&
371  (PROTOBUF_C_TYPE_BYTES == fieldDesc->type))
372  {
373  ProtobufCBinaryData* value = (ProtobufCBinaryData*)(void*)&pMsg[fieldDesc->offset];
374  if ((value->data != NULL) && (value->len > 0)) {
375  log_proto_level_start(fieldDesc->name);
376  KineticProto_Command * cmd = KineticProto_command__unpack(NULL, value->len, value->data);
377  log_protobuf_message(log_level, &cmd->base, indent);
378  log_proto_level_end();
379  KineticProto_command__free_unpacked(cmd, NULL);
380  }
381  }
382  else {
383  LogUnboxed(log_level, &pMsg[fieldDesc->offset], fieldDesc, 0, indent);
384  }
385  }
386  } break;
387  case PROTOBUF_C_LABEL_REPEATED:
388  {
389  unsigned const * quantifier = (unsigned const *)(void*)&pMsg[fieldDesc->quantifier_offset];
390  if (*quantifier > 0) {
391  log_proto_level_start_array(fieldDesc->name, *quantifier);
392  for (uint32_t i = 0; i < *quantifier; i++) {
393  void const ** box = (void const **)(void*)&pMsg[fieldDesc->offset];
394  LogUnboxed(log_level, *box, fieldDesc, i, indent);
395  }
396  log_proto_level_end_array();
397  }
398  } break;
399  }
400  }

415 void KineticLogger_LogStatus(int log_level, KineticProto_Command_Status* status)
416 {
417  if (status == NULL || !is_level_enabled(log_level)) {
418  return;
419  }
420 
421  ProtobufCMessage* protoMessage = &status->base;
422  KineticProto_Command_Status_StatusCode code = status->code;
423 
424  if (code == KINETIC_PROTO_COMMAND_STATUS_STATUS_CODE_SUCCESS) {
425  KineticLogger_LogPrintf(log_level, "Operation completed successfully\n");
426  }
427  else if (code == KINETIC_PROTO_COMMAND_STATUS_STATUS_CODE_INVALID_STATUS_CODE) {
428  KineticLogger_LogPrintf(log_level, "Operation was aborted!\n");
429  }
430  else {
431  // Output status code short name
432  const ProtobufCMessageDescriptor* protoMessageDescriptor = protoMessage->descriptor;
433  const ProtobufCFieldDescriptor* statusCodeDescriptor =
434  protobuf_c_message_descriptor_get_field_by_name(protoMessageDescriptor, "code");
435  const ProtobufCEnumDescriptor* statusCodeEnumDescriptor =
436  (ProtobufCEnumDescriptor*)statusCodeDescriptor->descriptor;
437  const ProtobufCEnumValue* eStatusCodeVal =
438  protobuf_c_enum_descriptor_get_value(statusCodeEnumDescriptor, code);
439  KineticLogger_LogPrintf(log_level, "Operation completed but failed w/error: %s=%d(%s)\n",
440  statusCodeDescriptor->name, code, eStatusCodeVal->name);
441 
442  // Output status message, if supplied
443  if (status->statusMessage) {
444  const ProtobufCFieldDescriptor* statusMsgFieldDescriptor =
445  protobuf_c_message_descriptor_get_field_by_name(protoMessageDescriptor, "statusMessage");
446  const ProtobufCMessageDescriptor* statusMsgDescriptor =
447  (ProtobufCMessageDescriptor*)statusMsgFieldDescriptor->descriptor;
448 
449  KineticLogger_LogPrintf(log_level, " %s: '%s'", statusMsgDescriptor->name, status->statusMessage);
450  }
451 
452  // Output detailed message, if supplied
453  if (status->has_detailedMessage) {
454  char tmp[8], msg[256];
455  const ProtobufCFieldDescriptor* statusDetailedMsgFieldDescriptor =
456  protobuf_c_message_descriptor_get_field_by_name(
457  protoMessageDescriptor, "detailedMessage");
458  const ProtobufCMessageDescriptor* statusDetailedMsgDescriptor =
459  (ProtobufCMessageDescriptor*)
460  statusDetailedMsgFieldDescriptor->descriptor;
461 
462  sprintf(msg, " %s: ", statusDetailedMsgDescriptor->name);
463  for (size_t i = 0; i < status->detailedMessage.len; i++) {
464  sprintf(tmp, "%02hhX", status->detailedMessage.data[i]);
465  strcat(msg, tmp);
466  }
467  KineticLogger_LogPrintf(log_level, " %s", msg);
468  }
469  }
470 }

472 void KineticLogger_LogByteArray(int log_level, const char* title, ByteArray bytes)
473 {
474  if (title == NULL || !is_level_enabled(log_level)) {
475  return;
476  }
477 
478  if (bytes.data == NULL) {
479  KineticLogger_LogPrintf(log_level, "%s: (??? bytes : buffer is NULL)", title);
480  return;
481  }
482  if (bytes.data == NULL) {
483  KineticLogger_LogPrintf(log_level, "%s: (0 bytes)", title);
484  return;
485  }
486  KineticLogger_LogPrintf(log_level, "%s: (%zd bytes)", title, bytes.len);
487  const int byteChars = 4;
488  const int bytesPerLine = 32;
489  const int lineLen = 4 + (bytesPerLine * byteChars);
490  char hex[lineLen + 1];
491  char ascii[lineLen + 1];
492  for (size_t i = 0; i < bytes.len;) {
493  hex[0] = '\0';
494  ascii[0] = '\0';
495  for (int j = 0;
496  j < bytesPerLine && i < bytes.len;
497  j++, i++) {
498  char byHex[8];
499  sprintf(byHex, "%02hhX", bytes.data[i]);
500  strcat(hex, byHex);
501  char byAscii[8];
502  int ch = (int)bytes.data[i];
503  if (ch >= 32 && ch <= 126) {
504  sprintf(byAscii, "%c", bytes.data[i]);
505  }
506  else {
507  byAscii[0] = '.';
508  byAscii[1] = '\0';
509  }
510  strcat(ascii, byAscii);
511  }
512  KineticLogger_LogPrintf(log_level, "%s: %s : %s", title, hex, ascii);
513  }
514 }
