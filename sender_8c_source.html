41 struct sender *sender_init(struct bus *b, struct bus_config *cfg) {
42  struct sender *s = calloc(1, sizeof(*s));
43  if (s == NULL) {
44  return NULL;
45  }
46 
47  /* TODO: use `goto cleanup` idiom to ensure that all resources are
48  * freed if any intermediate resource acquisition fails. */
49 
50  s->bus = b;
51 
52  if ((8 * sizeof(tx_flag_t)) < MAX_CONCURRENT_SENDS) {
53  /* tx_flag_t MUST have >= MAX_CONCURRENT_SENDS bits, since it
54  * is used as bitflags to indicate message availability. */
55  assert(false);
56  }
57 
58  int pipes[2];
59  int res = pipe(pipes);
60  if (res == -1) {
61  free(s);
62  return NULL;
63  }
64 
65  s->incoming_command_pipe = pipes[0];
66  s->commit_pipe = pipes[1];
67 
68  s->fd_hash_table = yacht_init(HASH_TABLE_SIZE2);
69  if (s->fd_hash_table == NULL) {
70  close(pipes[0]);
71  close(pipes[1]);
72  free(s);
73  return NULL;
74  }
75 
76  s->fds[0].fd = s->incoming_command_pipe;
77  s->fds[0].events = POLLIN;
78 
79  for (int i = 0; i < MAX_CONCURRENT_SENDS; i++) {
80  if (0 != pipe(s->pipes[i])) {
81  fprintf(stderr, "Error: pipe(2): %s\n", strerror(errno));
82  free(s);
83  return NULL;
84  }
85 
86  BUS_LOG_SNPRINTF(b, 6, LOG_SENDER, b->udata, 64,
87  "PIPE fds %d and %d\n", s->pipes[i][0], s->pipes[i][1]);
88 
89  s->fds[i + 1].fd = SENDER_FD_NOT_IN_USE;
90  s->fds[i + 1].events = POLLOUT;
91 
92  int *p_id = (int *)(&s->tx_info[i].id);
93  *p_id = i;
94  s->tx_info[i].state = TIS_UNDEF;
95  }
96 
97  BUS_LOG(b, 2, LOG_SENDER, "init success", b->udata);
98  BUS_LOG_SNPRINTF(b, 4, LOG_SENDER, b->udata, 64,
99  "sender tx_info table at %p", (void *)s->tx_info);
100 
101  (void)cfg;
102  return s;
103 }

181 static tx_info_t *get_free_tx_info(struct sender *s) {
182  struct bus *b = s->bus;
183  /* Use atomic compare-and-swap to attempt to reserve a tx_info_t. */
184  for (int i = 0; i < MAX_CONCURRENT_SENDS; i++) {
185  tx_flag_t cur = s->tx_flags;
186  tx_flag_t bit = (1 << i);
187  if ((cur & bit) == 0) {
188  tx_flag_t marked = cur | bit;
189  if (ATOMIC_BOOL_COMPARE_AND_SWAP(&s->tx_flags, cur, marked)) {
190  /* *info is now reserved. */
191  tx_info_t *info = &s->tx_info[i];
192  BUS_LOG_SNPRINTF(b, 6, LOG_SENDER, b->udata, 64,
193  "reserving free TX info %d", info->id);
194 
195  assert(info->id == i);
196  assert(info->state == TIS_UNDEF);
197  info->done_pipe = get_notify_pipe(s, info->id);
198  return info;
199  }
200  }
201  }
202 
203  BUS_LOG(b, 1, LOG_SENDER, "No tx_info cells left", b->udata);
204  return NULL;
205 }

207 static void release_tx_info(struct sender *s, tx_info_t *info) {
208  assert(info->state != TIS_UNDEF);
209  info->state = TIS_UNDEF;
210  struct bus *b = s->bus;
211  assert(info->id < MAX_CONCURRENT_SENDS);
212  assert(s->tx_flags & (1 << info->id));
213 
214  for (;;) {
215  tx_flag_t cur = s->tx_flags;
216  tx_flag_t cleared = cur & ~(1 << info->id);
217  BUS_LOG_SNPRINTF(b, 6, LOG_SENDER, b->udata, 64,
218  "CAS 0x%04x => 0x%04x", cur, cleared);
219  if (ATOMIC_BOOL_COMPARE_AND_SWAP(&s->tx_flags, cur, cleared)) {
220  BUS_LOG_SNPRINTF(b, 6, LOG_SENDER, b->udata, 64,
221  "releasing TX info %d", info->id);
222  break;
223  }
224  }

231 static bool write_commit(struct sender *s, tx_info_t *info) {
232  uint8_t buf[1];
233  buf[0] = info->id;
234  struct bus *b = s->bus;
235 
236  for (;;) {
237  /* Notify the sender thread that a command is available. */
238  ssize_t wr = write(s->commit_pipe, buf, sizeof(buf));
239  if (wr == 1) {
240  return true;
241  } else {
242  assert(wr == -1);
243  if (errno == EINTR) {
244  errno = 0;
245  } else {
246  BUS_LOG_SNPRINTF(b, 10, LOG_SENDER, b->udata, 64,
247  "write_commit errno %d", errno);
248  errno = 0;
249  return false;
250  }
251  }
252  }

255 static tx_error_t commit_event_and_block(struct sender *s, tx_info_t *info) {
256  if (!write_commit(s, info)) {
257  return TX_ERROR_WRITE_FAILURE;
258  }
259 
260  struct bus *b = s->bus;
261  struct pollfd fds[1];
262  assert(info->done_pipe != SENDER_FD_NOT_IN_USE);
263  fds[0].fd = info->done_pipe;
264  fds[0].events = POLLIN;
265 
266  for (;;) {
267  int res = poll(fds, 1, -1);
268  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 64,
269  "polling done_pipe: %d", res);
270  if (res == 1) {
271  short events = fds[0].revents;
272  BUS_LOG_SNPRINTF(b, 8, LOG_SENDER, b->udata, 64,
273  "poll: events %d, errno %d", events, errno);
274  if (events & (POLLHUP | POLLERR | POLLNVAL)) {
275  /* We've been hung up on due to a shutdown event. */
276  close(info->done_pipe);
277  return TX_ERROR_CLOSED;
278  } else if (events & POLLIN) {
279  uint16_t backpressure = 0;
280  uint8_t buf[sizeof(bool) + sizeof(backpressure)];
281  ssize_t rd = read(info->done_pipe, buf, sizeof(buf));
282  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 64,
283  "reading done_pipe: %zd", rd);
284  if (rd == sizeof(buf)) {
285  bool success = buf[0] == 0 ? false : true;
286  backpressure = (buf[1] << 0);
287  backpressure += (buf[2] << 8);
288 
289  /* Push back if message bus is too busy. */
290  backpressure >>= 7; // TODO: further tuning
291 
292  if (backpressure > 0) {
293  BUS_LOG_SNPRINTF(b, 5, LOG_SENDER, b->udata, 64,
294  "reading done_pipe: backpressure %d", backpressure);
295  poll(NULL, 0, backpressure);
296  }
297 
298  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 64,
299  "reading done_pipe: success %d", 1);
300  return success ? TX_ERROR_NONE : TX_ERROR_WRITE_FAILURE;
301  } else if (rd == -1) {
302  if (errno == EINTR) {
303  errno = 0;
304  continue;
305  } else {
306  BUS_LOG_SNPRINTF(b, 10, LOG_SENDER, b->udata, 64,
307  "blocking read on done_pipe: errno %d", errno);
308  errno = 0;
309  return TX_ERROR_CLOSED;
310  }
311  }
312  } else {
313  /* Shouldn't happen -- blocking. */
314  BUS_LOG_SNPRINTF(b, 1, LOG_SENDER, b->udata, 64,
315  "shouldn't happen: events %d, errno %d", events, errno);
316  assert(false);
317  }
318  } else if (res == -1) {
319  BUS_LOG_SNPRINTF(b, 10, LOG_SENDER, b->udata, 64,
320  "blocking poll for done_pipe: errno %d", errno);
321  errno = 0;
322  return TX_ERROR_CLOSED;
323  } else {
324  /* Shouldn't happen -- blocking. */
325  assert(false);
326  }
327  }
328 }

332 void *sender_mainloop(void *arg) {
333  sender *self = (sender *)arg;
334  assert(self);
335  struct bus *b = self->bus;
336 
337  struct timeval tv;
338  gettimeofday(&tv, NULL);
339  time_t last_sec = tv.tv_sec;
340 
341  BUS_LOG(b, 5, LOG_SENDER, "entering main loop", b->udata);
342  while (!self->shutdown) {
343  gettimeofday(&tv, NULL); // TODO: clock_gettime
344  time_t cur_sec = tv.tv_sec;
345  if (cur_sec != last_sec) {
346  BUS_LOG(b, 5, LOG_SENDER, "entering tick handler", b->udata);
347  tick_handler(self); /* handle timeouts and retries */
348  last_sec = cur_sec;
349  }
350 
351  /* Poll on incoming command pipe and any sockets for which we have
352  * an outgoing message queued up.
353  *
354  * Note: Even if epoll(2) or kqueue(2) are available, it still makes
355  * sense to use poll here -- self->active_fds will be small. */
356  BUS_LOG(b, 7, LOG_SENDER, "polling", b->udata);
357 
358  int delay = self->is_idle ? -1 : TIMEOUT_DELAY;
359  int res = poll(self->fds, self->active_fds + CMD_FD, delay);
360 
361  BUS_LOG_SNPRINTF(b, (res == 0 ? 6 : 4), LOG_SENDER, b->udata, 64,
362  "poll res %d, active fds %d", res, self->active_fds);
363 
364  if (res == -1) {
365  if (util_is_resumable_io_error(errno)) {
366  errno = 0;
367  } else {
368  /* unrecoverable poll error -- FD count is bad
369  * or FDS is a bad pointer. */
370  assert(false);
371  }
372  } else if (res > 0) {
373  if (self->fds[0].revents & POLLIN) {
374  (void)check_incoming_commands(self);
375  res--;
376  }
377 
378  if (self->shutdown) { break; }
379 
380  /* If the incoming command pipe isn't the only active FD: */
381  if (res > 0) {
382  attempt_write(self, res);
383  }
384  // work = true;
385  }
386  }
387 
388  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 64,
389  "shutting down, shutdown == %d", self->shutdown);
390  cleanup(self);
391 
392  return NULL;
393 }

405 static void cleanup(sender *s) {
406  struct bus *b = s->bus;
407  BUS_LOG(b, 2, LOG_SHUTDOWN, "sender_cleanup", b->udata);
408  if (s->fd_hash_table) { /* make idempotent */
409  struct yacht *y = s->fd_hash_table;
410  s->fd_hash_table = NULL;
411  int shutdown_id = -1;
412 
413  for (int i = 0; i < MAX_CONCURRENT_SENDS; i++) {
414  if (s->tx_info[i].state == TIS_SHUTDOWN) {
415  shutdown_id = i;
416  } else {
417  close(s->pipes[i][0]);
418  close(s->pipes[i][1]);
419  }
420  }
421 
422  BUS_LOG_SNPRINTF(b, 6 , LOG_SENDER, b->udata, 64,
423  "shutdown_id: %d", shutdown_id);
424  if (shutdown_id != -1) {
425  /* Notify the sender about the shutdown via POLLHUP / POLLNVAL. */
426  BUS_LOG(b, 3, LOG_SENDER, "closing to notify shutdown", b->udata);
427  /* Client thread will close the other end. */
428  close(s->pipes[shutdown_id][1]);
429  }
430 
431  yacht_free(y, free_fd_info_cb, NULL);
432  close(s->incoming_command_pipe);
433  close(s->commit_pipe);
434  }

472 static void decrement_fd_refcount(sender *s, fd_info *fdi) {
473  assert(fdi);
474  assert(fdi->refcount > 0);
475 
476  if (fdi->refcount == 1) {
477  bool found = false;
478  /* Remove from poll watch set */
479  for (int i = 0; i < MAX_CONCURRENT_SENDS; i++) {
480  struct pollfd *pfd = &s->fds[CMD_FD + i];
481  if (pfd->fd == fdi->fd) {
482  if (s->active_fds > 1) {
483  /* Replace this one with last */
484  s->fds[CMD_FD + i].fd = s->fds[CMD_FD + s->active_fds - 1].fd;
485  }
486  s->fds[CMD_FD + s->active_fds - 1].fd = SENDER_FD_NOT_IN_USE;
487  s->active_fds--;
488  found = true;
489  break;
490  }
491  }
492  assert(found);
493  }
494 
495  fdi->refcount--;
496 }

498 static bool release_socket_info(sender *s, int fd) {
499  if (s->shutdown) { return false; }
500 
501  void *old = NULL;
502  if (!yacht_remove(s->fd_hash_table, fd, &old)) {
503  return false;
504  }
505  fd_info *info = (fd_info *)old;
506  if (info) {
507  assert(fd == info->fd);
508  info->errored = true;
509  /* Expire any pending events on this socket. */
510  if (info->refcount > 0) {
511  set_error_for_socket(s, fd, TX_ERROR_CLOSED);
512  }
513  free(info);
514  return true;
515  } else {
516  return false;
517  }

520 static void handle_command(sender *s, int id) {
521  assert(id < MAX_CONCURRENT_SENDS);
522  tx_info_t *info = &s->tx_info[id];
523 
524  s->is_idle = false;
525 
526  switch (info->state) {
527  case TIS_ADD_SOCKET:
528  {
529  int fd = info->u.add_socket.fd;
530  SSL *ssl = info->u.add_socket.ssl;
531  if (register_socket_info(s, fd, ssl)) {
532  notify_caller(s, info, true);
533  } else {
534  notify_caller(s, info, false);
535  }
536  break;
537  }
538  case TIS_RM_SOCKET:
539  {
540  int fd = info->u.rm_socket.fd;
541  if (release_socket_info(s, fd)) {
542  notify_caller(s, info, true);
543  } else {
544  notify_caller(s, info, false);
545  }
546  break;
547  }
548  case TIS_SHUTDOWN:
549  {
550  s->shutdown = true;
551  /* Caller should be notified from cleanup(), by closing the
552  * notification pipe. Block until shutdown is complete. */
553  break;
554  }
555  case TIS_REQUEST_ENQUEUE:
556  {
557  enqueue_write(s, info);
558  break;
559  }
560 
561  /* Should not get anything else as an incoming command. */
562  default:
563  {
564  struct bus *b = s->bus;
565  BUS_LOG_SNPRINTF(b, 0, LOG_SENDER, b->udata, 64,
566  "match_fail: %d", info->state);
567  assert(false);
568  }
569  }
570 }

572 static void enqueue_write(struct sender *s, tx_info_t *info) {
573  assert(info->state == TIS_REQUEST_ENQUEUE);
574 
575  struct bus *b = s->bus;
576  int fd = info->u.enqueue.fd;
577  fd_info *fdi = NULL;
578  int64_t out_seq_id = info->u.enqueue.box->out_seq_id;
579  if (yacht_get(s->fd_hash_table, fd, (void **)&fdi)) {
580  assert(fdi);
581 
582  if (fdi->errored) {
583  set_error_for_socket(s, fd, TX_ERROR_CLOSED);
584  return;
585  }
586 
587  if (fdi->largest_seq_id_seen > out_seq_id && fdi->largest_seq_id_seen != 0
588  && out_seq_id != BUS_NO_SEQ_ID) {
589  BUS_LOG_SNPRINTF(b, 0 , LOG_SENDER, b->udata, 64,
590  "suspicious outgoing sequence ID on %d: got %lld, already sent up to %lld",
591  fd, (long long)out_seq_id, (long long)fdi->largest_seq_id_seen);
592  set_error_for_socket(s, fd, TX_ERROR_BAD_SEQUENCE_ID);
593  return;
594  }
595  fdi->largest_seq_id_seen = out_seq_id;
596 
597  /* Notify the listener that we're about to start writing to a drive,
598  * because (in rare cases) the response may arrive between finishing
599  * the write and the listener processing the notification. In that
600  * case, it should hold onto the unrecognized response until the
601  * sender notifies it (and passes it the callback).
602  *
603  * This timeout is several extra seconds so that we don't have
604  * a window where the HOLD message has timed out, but the
605  * EXPECT hasn't, leading to ambiguity about what to do with
606  * the response (which may or may not have arrived).
607  * */
608  attempt_to_enqueue_sending_request_message_to_listener(s,
609  fd, out_seq_id, info->u.enqueue.timeout_sec + 5);
610 
611  /* Increment the refcount. This will cause poll to watch for the
612  * socket being writable, if it isn't already being watched. */
613  increment_fd_refcount(s, fdi);
614 
615  struct u_write uw = {
616  .fd = info->u.enqueue.fd,
617  .timeout_sec = info->u.enqueue.timeout_sec,
618  .orig_timeout_sec = info->u.enqueue.timeout_sec,

656 /* Get a tx_info write event associated with a socket. If there is one
657  * that is already partially written, prefer that, to avoid interleaving
658  * outgoing messages. */
659 static tx_info_t *get_info_to_write_for_socket(sender *s, int fd) {
660  tx_info_t *res = NULL;
661  for (int i = 0; i < MAX_CONCURRENT_SENDS; i++) {
662  tx_flag_t cur = 1 << i;
663  if (s->tx_flags & cur) {
664  tx_info_t *info = &s->tx_info[i];
665  if (info->state != TIS_REQUEST_WRITE) { continue; }
666  if (info->u.write.fd == fd) {
667  if (res == NULL) {
668  res = info;
669  } else {
670  int64_t cur_seq_id = res->u.write.box->out_seq_id;
671  int64_t new_seq_id = info->u.write.box->out_seq_id;
672  if (new_seq_id < cur_seq_id) {
673  res = info;
674  }
675  }
676  /* If we've already sent part of a message, send the rest
677  * before starting another, otherwise arbitrarily choose
678  * the one with the highest tx_info->id. */
679  if (info->u.write.sent_size > 0) { return res; }
680  }
681  }
682  }
683  return res;

688 static void set_error_for_socket(sender *s, int fd, tx_error_t error) {

716  struct u_error ue = {
717  .error = error,
718  };

721

760  int written = 0;
761  struct bus *b = s->bus;
762  for (int i = 0; i < s->active_fds; i++) {

803 static ssize_t socket_write_plain(sender *s, tx_info_t *info) {
804  struct bus *b = s->bus;

806  boxed_msg *box = info->u.write.box;

813 
814  BUS_LOG_SNPRINTF(b, 10, LOG_SENDER, b->udata, 64,

831  } else if (wrsz > 0) {

842 static ssize_t socket_write_ssl(sender *s, tx_info_t *info, SSL *ssl) {
843  struct bus *b = s->bus;
844  assert(info->state == TIS_REQUEST_WRITE);
845  boxed_msg *box = info->u.write.box;
846  uint8_t *msg = box->out_msg;
847  size_t msg_size = box->out_msg_size;
848  size_t rem = msg_size - info->u.write.sent_size;
849  int fd = info->u.write.fd;
850  ssize_t written = 0;
851 
852  while (rem > 0) {
853  ssize_t wrsz = SSL_write(ssl, &msg[info->u.write.sent_size], rem);
854  if (wrsz > 0) {
855  update_sent(b, s, info, wrsz);
856  written += wrsz;
857  rem -= wrsz;
858  } else if (wrsz < 0) {
859  int reason = SSL_get_error(ssl, wrsz);
860  switch (reason) {
861  case SSL_ERROR_WANT_WRITE:
862  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 64,
863  "SSL_write: socket %d: WANT_WRITE", fd);
864  break;
865 
866  case SSL_ERROR_WANT_READ:
867  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 64,
868  "SSL_write: socket %d: WANT_READ", fd);
869  assert(false); // shouldn't get this; we're writing.
870  break;
871 
872  case SSL_ERROR_SYSCALL:
873  {
874  if (util_is_resumable_io_error(errno)) {
875  errno = 0;
876  /* don't break; we want to retry on EINTR etc. until
877  * we get WANT_WRITE, otherwise poll(2) may not retry
878  * the socket for too long */
879  } else {
880  set_error_for_socket(s, fd, TX_ERROR_WRITE_FAILURE);
881  }
882  }
883  default:
884  {
885  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 64,
886  "SSL_write: socket %d: error %d", info->u.write.fd, reason);
887  break;
888  }
889  }
890  } else {
891  /* SSL_write should give SSL_ERROR_WANT_WRITE when unable
892  * to write further. */
893  }
894  }
895  return written;
896 }

975  "error on notify_caller write: %s", strerror(errno));

988  struct bus *b = s->bus;
989  int tx_info_in_use = 0;
990  bool any_work = false;
991 
992  for (int i = 0; i < MAX_CONCURRENT_SENDS; i++) {
993  tx_flag_t bit = (1 << i);
994  if (s->tx_flags & bit) {
995  tx_info_t *info = &s->tx_info[i];
996  if (info->state != TIS_UNDEF) { tx_info_in_use++; }
997  }
998  }
999 
1000  BUS_LOG_SNPRINTF(b, 2, LOG_SENDER, b->udata, 64,
1001  "tick... %p -- %d of %d tx_info in use, %d active FDs",
1002  (void*)s, tx_info_in_use, MAX_CONCURRENT_SENDS, s->active_fds);
1003 
1004  /* Walk table and expire timeouts & items which have been sent.
1005  *
1006  * All expiration of TX_INFO fields happens in here. */
1007  for (int i = 0; i < MAX_CONCURRENT_SENDS; i++) {
1008  tx_flag_t bit = (1 << i);
1009  if (s->tx_flags & bit) { /* if info is in use */
1010  any_work = true;
1011  tx_info_t *info = &s->tx_info[i];
1012  switch (info->state) {
1013  case TIS_REQUEST_WRITE:
1014  tick_timeout(s, info);
1015  break;
1016  case TIS_RESPONSE_NOTIFY:
1017  tick_timeout(s, info);
1018 
1019  /* if not timed out, attempt to re-send */
1020  if (info->state == TIS_RESPONSE_NOTIFY) {
1021  attempt_to_enqueue_request_sent_message_to_listener(s, info);
1022  }
1023  break;
1024 
1025  default:
1026  break;
1027  }
1028  }
1029  }
1030  if (!any_work) { s->is_idle = true; }
1031 }
