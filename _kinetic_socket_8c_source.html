34 int KineticSocket_Connect(const char* host, int port, bool blocking)
35 {
36  char message[256];
37  char port_str[32];
38  struct addrinfo* result = NULL;
39  struct addrinfo hints;
40  struct addrinfo* ai = NULL;
41  int socket_fd;
42  int res;
43 
44  sprintf(message, "Connecting to %s:%d", host, port);
45  LOG(message);
46 
47  memset(&hints, 0, sizeof(struct addrinfo));
48 
49  // could be inet or inet6
50  hints.ai_family = PF_UNSPEC;
51  hints.ai_socktype = SOCK_STREAM;
52  hints.ai_protocol = IPPROTO_TCP;
53  hints.ai_flags = AI_NUMERICSERV;
54 
55  sprintf(port_str, "%d", port);
56 
57  res = getaddrinfo(host, port_str, &hints, &result);
58  if (res != 0)
59  {
60  LOG("Could not resolve host");
61  return -1;
62  }
63 
64  for (ai = result; ai != NULL; ai = ai->ai_next)
65  {
66  char host[NI_MAXHOST];
67  char service[NI_MAXSERV];
68  int res;
69 
70  res = getnameinfo(ai->ai_addr, ai->ai_addrlen, host, sizeof(host), service,
71  sizeof(service), NI_NUMERICHOST | NI_NUMERICSERV);
72  if (res != 0)
73  {
74  LOG("Could not get name info");
75  continue;
76  }
77  else
78  {
79  LOG("Trying to connect to host");
80  }
81 
82  socket_fd = socket(ai->ai_family, ai->ai_socktype, ai->ai_protocol);
83  if (socket_fd == -1)
84  {
85  LOG("Could not create socket");
86  continue;
87  }
88 
89  // os x won't let us set close-on-exec when creating the socket, so set it separately
90  int current_fd_flags = fcntl(socket_fd, F_GETFD);
91  if (current_fd_flags == -1)
92  {
93  LOG("Failed to get socket fd flags");
94  close(socket_fd);
95  continue;
96  }
97  if (fcntl(socket_fd, F_SETFD, current_fd_flags | FD_CLOEXEC) == -1)
98  {
99  LOG("Failed to set socket close-on-exit");
100  close(socket_fd);
101  continue;
102  }
103 
104  // On BSD-like systems we can set SO_NOSIGPIPE on the socket to prevent it from sending a
105  // PIPE signal and bringing down the whole application if the server closes the socket
106  // forcibly
107 // #ifdef SO_NOSIGPIPE
108 // int set = 1;
109 // int setsockopt_result = setsockopt(socket_fd, SOL_SOCKET, SO_NOSIGPIPE, &set, sizeof(set));
110 // // Allow ENOTSOCK because it allows tests to use pipes instead of real sockets
111 // if (setsockopt_result != 0 && setsockopt_result != ENOTSOCK)
112 // {
113 // LOG("Failed to set SO_NOSIGPIPE on socket");
114 // close(socket_fd);
115 // continue;
116 // }
117 // #endif
118 
119  if (connect(socket_fd, ai->ai_addr, ai->ai_addrlen) == -1)
120  {
121  LOG("Unable to connect");
122  close(socket_fd);
123  continue;
124  }
125  else
126  {
127  LOG("Connected to host!");
128  }
129 
130  if (!blocking && fcntl(socket_fd, F_SETFL, O_NONBLOCK) != 0)
131  {
132  LOG("Failed to set socket nonblocking");
133  close(socket_fd);
134  continue;
135  }
136 
137  break;
138  }
139 
140  freeaddrinfo(result);
141 
142  if (ai == NULL)
143  {
144  // we went through all addresses without finding one we could bind to
145  sprintf(message, "Could not connect to %s:%d", host, port);
146  LOG(message);
147  return -1;
148  }
149 
150  return socket_fd;
151 }

153 void KineticSocket_Close(int socket_descriptor)
154 {
155  char message[64];
156  if (socket_descriptor == -1)
157  {
158  LOG("Not connected so no cleanup needed");
159  }
160  else
161  {
162  sprintf(message, "Closing socket with fd=%d", socket_descriptor);
163  LOG(message);
164  if (close(socket_descriptor))
165  {
166  LOG("Error closing socket file descriptor");
167  }
168  }
169 }
