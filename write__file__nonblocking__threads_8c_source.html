143  int pthreadStatus = pthread_join(ops[i].thread, NULL);
144  if (pthreadStatus == 0) {

146  }

149  fprintf(stderr, "Failed storing '%s' to disk! status: %s\n",

151  success = false;
152  }
153  }
154  printf("Complete!\n");

164 {

176 FileTransferProgress * start_file_transfer(KineticSession* session,
177  char const * const filename, uint64_t prefix, uint32_t maxOverlappedChunks)
178 {
179  FileTransferProgress * transferState = malloc(sizeof(FileTransferProgress));
180  *transferState = (FileTransferProgress) {
181  .session = session,
182  .maxOverlappedChunks = maxOverlappedChunks,
183  .keyPrefix = ByteBuffer_CreateAndAppend(transferState->keyPrefixBuffer,
184  sizeof(transferState->keyPrefixBuffer), &prefix, sizeof(prefix)),
185  .fd = open(filename, O_RDONLY),
186  };
187  pthread_mutex_init(&transferState->transferMutex, NULL);
188  pthread_mutex_init(&transferState->completeMutex, NULL);
189  pthread_cond_init(&transferState->completeCond, NULL);
190 
191  // Start max overlapped PUT operations
192  for (size_t i = 0; i < transferState->maxOverlappedChunks; i++) {
193  put_chunk_of_file(transferState);
194  }
195  return transferState;
196 }

198 KineticStatus wait_for_transfer_complete(FileTransferProgress* const transfer)
199 {
200  pthread_mutex_lock(&transfer->completeMutex);
201  pthread_cond_wait(&transfer->completeCond, &transfer->completeMutex);
202  pthread_mutex_unlock(&transfer->completeMutex);
203 
204  KineticStatus status = transfer->status;
205 
206  pthread_mutex_destroy(&transfer->completeMutex);
207  pthread_cond_destroy(&transfer->completeCond);
208 
209  close(transfer->fd);
210 
211  pthread_mutex_destroy(&transfer->transferMutex);
212  free(transfer);
213 
214  return status;
215 }

217 int put_chunk_of_file(FileTransferProgress* transfer)
218 {
219  AsyncWriteClosureData* closureData = calloc(1, sizeof(AsyncWriteClosureData));
220 
221  pthread_mutex_lock(&transfer->transferMutex);
222 
223  transfer->opsInProgress++;
224  closureData->currentTransfer = transfer;
225 
226  int bytesRead = read(transfer->fd, closureData->value, sizeof(closureData->value));
227  if (bytesRead > 0) {
228  transfer->currentChunk++;
229  closureData->entry = (KineticEntry){
230  .key = ByteBuffer_CreateAndAppend(closureData->key, sizeof(closureData->key),
231  transfer->keyPrefix.array.data, transfer->keyPrefix.bytesUsed),
232  .tag = ByteBuffer_CreateAndAppendFormattedCString(closureData->tag, sizeof(closureData->tag),
233  "some_value_tag..._%04d", transfer->currentChunk),
234  .algorithm = KINETIC_ALGORITHM_SHA1,
235  .value = ByteBuffer_Create(closureData->value, sizeof(closureData->value), (size_t)bytesRead),
236  .synchronization = KINETIC_SYNCHRONIZATION_WRITETHROUGH,
237  };
238  KineticStatus status = KineticClient_Put(transfer->session, &closureData->entry,
239  &(KineticCompletionClosure) {
240  .callback = put_chunk_of_file_finished,
241  .clientData = closureData,
242  });
243  if (status != KINETIC_STATUS_SUCCESS) {
244  transfer->opsInProgress--;
245  free(closureData);
246  fprintf(stderr, "Failed writing chunk! PUT request reported status: %s\n",
247  Kinetic_GetStatusDescription(status));
248  }
249  }
250  else if (bytesRead == 0) { // EOF reached
251  transfer->opsInProgress--;
252  free(closureData);
253  }
254  else {
255  transfer->opsInProgress--;
256  free(closureData);
257  fprintf(stderr, "Failed reading data from file!\n");
258  REPORT_ERRNO(bytesRead, "read");
259  }
260 
261  pthread_mutex_unlock(&transfer->transferMutex);
262 
263  return bytesRead;
264 }

266 void put_chunk_of_file_finished(KineticCompletionData* kinetic_data, void* clientData)
267 {
268  AsyncWriteClosureData* closureData = clientData;
269  FileTransferProgress* transfer = closureData->currentTransfer;
270  free(closureData);
271 
272  pthread_mutex_lock(&transfer->transferMutex);
273  transfer->opsInProgress--;
274  pthread_mutex_unlock(&transfer->transferMutex);
275 
276  if (kinetic_data->status == KINETIC_STATUS_SUCCESS) {
277  int bytesPut = put_chunk_of_file(transfer);
278  if (bytesPut <= 0 && transfer->opsInProgress == 0) {
279  if (transfer->status == KINETIC_STATUS_NOT_ATTEMPTED) {
280  transfer->status = KINETIC_STATUS_SUCCESS;
281  }
282  pthread_cond_signal(&transfer->completeCond);
283  }
284  }
285  else {
286  transfer->status = kinetic_data->status;
287  // only signal when finished
288  // keep track of outstanding operations
289  // if there is no more data to read (or error), and no outstanding operations,
290  // then signal
291  pthread_cond_signal(&transfer->completeCond);
292  pthread_mutex_destroy(&transfer->transferMutex);
293  pthread_mutex_destroy(&transfer->completeMutex);
294  fprintf(stderr, "Failed writing chunk! PUT response reported status: %s\n",
295  Kinetic_GetStatusDescription(kinetic_data->status));
296  }
297 }
