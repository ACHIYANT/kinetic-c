26 #define COPY_VALUE_OPTIONAL(_attr, _dest, _src) { \
27  if (_src->has_##_attr) { _dest->_attr = _src->_attr; } }
28 #define COPY_CSTRING(_attr, _dest, _src, _allocator) \
29  if (_src->_attr != NULL) { \
30  size_t len = strlen(_src->_attr)+1; \
31  _dest->_attr = KineticSerialAllocator_AllocateChunk(_allocator, len); \
32  memcpy(_dest->_attr, _src->_attr, len); \
33  }
34 #define COPY_BYTES_OPTIONAL(_attr, _dest, _src, _allocator) \
35  if (_src->has_##_attr) { \
36  _dest->_attr.data = KineticSerialAllocator_AllocateChunk(_allocator, _src->_attr.len); \
37  memcpy(_dest->_attr.data, _src->_attr.data, _src->_attr.len); \
38  _dest->_attr.len = _src->_attr.len; \
39  }

41 static void KineticDeviceInfo_ExtractUtilizations(
42  KineticDeviceInfo* info,
43  const KineticProto_Command_GetLog* getLog,
44  KineticSerialAllocator* allocator)
45 {
46  info->numUtilizations = getLog->n_utilizations;
47  if (info->numUtilizations > 0) {
48  info->utilizations = KineticSerialAllocator_AllocateChunk(allocator,
49  info->numUtilizations * sizeof(KineticDeviceInfo_Utilization));
50  for (size_t i = 0; i < info->numUtilizations; i++) {
51  KineticDeviceInfo_Utilization* destUtilization = &(info->utilizations[i]);
52  COPY_CSTRING(name, destUtilization, getLog->utilizations[i], allocator);

54  }
55  }

99  COPY_CSTRING(model, info->configuration, getLog->configuration, allocator);

101  COPY_BYTES_OPTIONAL(worldWideName, info->configuration, getLog->configuration, allocator);
102  COPY_CSTRING(version, info->configuration, getLog->configuration, allocator);
103  COPY_CSTRING(compilationDate, info->configuration, getLog->configuration, allocator);
104  COPY_CSTRING(sourceHash, info->configuration, getLog->configuration, allocator);
105  COPY_CSTRING(protocolVersion, info->configuration, getLog->configuration, allocator);
106  COPY_CSTRING(protocolCompilationDate, info->configuration, getLog->configuration, allocator);
107  COPY_CSTRING(protocolSourceHash, info->configuration, getLog->configuration, allocator);
108  if (getLog->configuration->n_interface > 0) {
109  info->configuration->interfaces = KineticSerialAllocator_AllocateChunk(allocator,
110  sizeof(KineticDeviceInfo_Interface) * getLog->configuration->n_interface);
111  info->configuration->numInterfaces = getLog->configuration->n_interface;
112  for (size_t i = 0; i < getLog->configuration->n_interface; i++) {
113  KineticDeviceInfo_Interface* destIface = &info->configuration->interfaces[i];
114  KineticProto_Command_GetLog_Configuration_Interface* srcIface = getLog->configuration->interface[i];
115  COPY_CSTRING(name, destIface, srcIface, allocator);
116  COPY_BYTES_OPTIONAL(MAC, destIface, srcIface, allocator);
117  COPY_BYTES_OPTIONAL(ipv4Address, destIface, srcIface, allocator);
118  COPY_BYTES_OPTIONAL(ipv6Address, destIface, srcIface, allocator);
119  }
120  }
121  COPY_VALUE_OPTIONAL(port, info->configuration, getLog->configuration);
122  COPY_VALUE_OPTIONAL(tlsPort, info->configuration, getLog->configuration);
123  }
124 }

185 KineticDeviceInfo* KineticDeviceInfo_Create(const KineticProto_Command_GetLog* getLog)
186 {
187  assert(getLog != NULL);
188 
189  // Allocate the data
190  KineticSerialAllocator allocator = KineticSerialAllocator_Create(KINETIC_DEVICE_INFO_SCRATCH_BUF_LEN);
191 
192  // Copy data into the nested allocated structure tree
193  KineticDeviceInfo* info = KineticSerialAllocator_AllocateChunk(&allocator, sizeof(KineticDeviceInfo));
194  KineticDeviceInfo_ExtractUtilizations(info, getLog, &allocator);
195  KineticDeviceInfo_ExtractCapacity(info, getLog, &allocator);
196  KineticDeviceInfo_ExtractTemperatures(info, getLog, &allocator);
197  KineticDeviceInfo_ExtractConfiguration(info, getLog, &allocator);
198  KineticDeviceInfo_ExtractStatistics(info, getLog, &allocator);
199  KineticDeviceInfo_ExtractMessages(info, getLog, &allocator);
200  KineticDeviceInfo_ExtractLimits(info, getLog, &allocator);
201  KineticDeviceInfo_ExtractDevice(info, getLog, &allocator);
202 
203  // Trim the buffer prior to returning in order to free up unused memory
204  info->totalLength = allocator.used;
205  info->totalLength = KineticSerialAllocator_TrimBuffer(&allocator);
206  info = KineticSerialAllocator_GetBuffer(&allocator);
207  LOGF2("Created KineticDeviceInfo @ 0x%0llX w/ length=%zu", info, info->totalLength);
208  return info;
209 }
