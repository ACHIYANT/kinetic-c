139 #define KINETIC_MESSAGE_AUTH_HMAC_INIT(_msg, _identity, _hmac) { \
140  assert((_msg) != NULL); \
141  (_msg)->message.has_authType = true; \
142  (_msg)->message.authType = KINETIC_PROTO_MESSAGE_AUTH_TYPE_HMACAUTH; \
143  KineticProto_Message_hmacauth__init(&(_msg)->hmacAuth); \
144  (_msg)->message.hmacAuth = &(_msg)->hmacAuth; \
145  KineticProto_Message_pinauth__init(&(_msg)->pinAuth); \
146  (_msg)->message.pinAuth = NULL; \
147  (_msg)->command.header = &(_msg)->header; \
148  memset((_msg)->hmacData, 0, KINETIC_HMAC_MAX_LEN); \
149  if ((_hmac).len <= KINETIC_HMAC_MAX_LEN \
150  && (_hmac).data != NULL && (_hmac).len > 0 \
151  && (_msg)->hmacData != NULL) { \
152  memcpy((_msg)->hmacData, (_hmac).data, (_hmac).len);} \
153  (_msg)->message.hmacAuth->has_identity = true; \
154  (_msg)->message.hmacAuth->identity = (_identity); \
155  (_msg)->message.hmacAuth->has_hmac = true; \
156  (_msg)->message.hmacAuth->hmac = (ProtobufCBinaryData) { \
157  .data = (_msg)->hmacData, .len = SHA_DIGEST_LENGTH}; \
158 }
