132 #define KINETIC_MESSAGE_AUTH_HMAC_INIT(_msg, _identity, _hmac) { \
133  assert((_msg) != NULL); \
134  (_msg)->message.has_authType = true; \
135  (_msg)->message.authType = KINETIC_PROTO_MESSAGE_AUTH_TYPE_HMACAUTH; \
136  KineticProto_Message_hmacauth__init(&(_msg)->hmacAuth); \
137  (_msg)->message.hmacAuth = &(_msg)->hmacAuth; \
138  KineticProto_Message_pinauth__init(&(_msg)->pinAuth); \
139  (_msg)->message.pinAuth = NULL; \
140  (_msg)->command.header = &(_msg)->header; \
141  memset((_msg)->hmacData, 0, KINETIC_HMAC_MAX_LEN); \
142  if ((_hmac).len <= KINETIC_HMAC_MAX_LEN \
143  && (_hmac).data != NULL && (_hmac).len > 0 \
144  && (_msg)->hmacData != NULL) { \
145  memcpy((_msg)->hmacData, (_hmac).data, (_hmac).len);} \
146  (_msg)->message.hmacAuth->has_identity = true; \
147  (_msg)->message.hmacAuth->identity = (_identity); \
148  (_msg)->message.hmacAuth->has_hmac = true; \
149  (_msg)->message.hmacAuth->hmac = (ProtobufCBinaryData) { \
150  .data = (_msg)->hmacData, .len = SHA_DIGEST_LENGTH}; \
151 }
