73 static void* KineticAllocator_GetNextListItem(KineticList* list, void* item_data)
74 {
75  assert(list != NULL);
76  void* nextData = NULL;
77  KINETIC_LIST_LOCK(list);
78  KineticListItem* current = list->start;
79  while (current != NULL) {
80  void* currData = current->data;
81  if (currData == item_data) {
82  if (current->next != NULL) {
83  nextData = current->next->data;
84  }
85  break;
86  }
87  current = current->next;
88  }
89  KINETIC_LIST_UNLOCK(list);
90  return nextData;
91 }

93 static void* KineticAllocator_NewItem(KineticList* const list, size_t size)
94 {
95  KineticListItem* newItem = (KineticListItem*)malloc(sizeof(KineticListItem));
96  if (newItem == NULL) {
97  LOG0(" Failed allocating new list item!");
98  return NULL;
99  }
100  memset(newItem, 0, sizeof(KineticListItem));
101  newItem->data = malloc(size);
102  if (newItem->data == NULL) {
103  LOG0(" Failed allocating data for list item!");
104  return NULL;
105  }
106  memset(newItem->data, 0, size);
107 
108  // Add the new item to the list
109  KINETIC_LIST_LOCK(list);
110  newItem->next = list->start;
111  list->start = newItem;
112  KINETIC_LIST_UNLOCK(list);
113 
114  LOGF3(" Allocated new list item (0x%0llX) w/data (0x%0llX)",
115  (long long)newItem, (long long)newItem->data);
116 
117  return newItem->data;
118 }

120 static void KineticAllocator_FreeItem(KineticList* const list, void* item, bool lock)
121 {
122  /* Make locking optional, since the lock may already be owned by the caller. */
123  if (lock) {
124  KINETIC_LIST_LOCK(list);
125  }
126  KineticListItem* cur = list->start;
127  KineticListItem* prev = NULL;
128 
129  while (cur->data != item) {
130  if (cur->next == NULL) {
131  LOG1(" Reached end of list before finding item to free!");
132  if (lock) {
133  KINETIC_LIST_UNLOCK(list);
134  }
135  return;
136  }
137  else {
138  prev = cur;
139  cur = cur->next;
140  }
141  }
142  LOG3(" Done searching for item list item");
143 
144  if ((cur != NULL) && (cur->data == item)) {
145  LOG3(" item found! freeing it.");
146 
147  if (prev == NULL) {
148  LOG3(" At start of list.");
149  list->start = cur->next;
150  } else {
151  LOG3(" Not at list start, so relinking list to free item.");
152  prev->next = cur->next;
153  }
154 
155  LOGF3(" Freeing item (0x%0llX) w/data (0x%0llX)", cur, &cur->data);
156  free(cur->data);
157  cur->data = NULL;
158  free(cur);
159  cur = NULL;
160  }
161  if (lock) {
162  KINETIC_LIST_UNLOCK(list);
163  }
164 }

166 static void KineticAllocator_FreeList(KineticList* const list)
167 {
168  if (list != NULL) {
169  LOGF3(" Freeing list (0x%0llX) of all items...", list);
170  KINETIC_LIST_LOCK(list);
171 
172  KineticListItem* next = NULL;
173  for (KineticListItem* item = list->start; item; item = next) {
174  next = item->next;
175 
176  LOGF3(" Freeing list item (0x%0llX) w/ data (0x%llX)",
177  (long long)item, (long long)&item->data);
178  if (item->data != NULL) {
179  free(item->data);
180  item->data = NULL;
181  }
182  free(item);
183  }
184 
185  // Make list empty, but leave mutex alone so the state is retained!
186  list->start = NULL;
187  /* list->last = NULL; */
188  KINETIC_LIST_UNLOCK(list);
189  }
190  else {
191  LOGF3(" Nothing to free from list (0x%0llX)", list);
192  }
193 }

216 void KineticAllocator_FreePDU(KineticConnection* connection, KineticPDU* pdu)
217 {
218  LOGF3("Freeing PDU (0x%0llX) on connection (0x%0llX)", pdu, connection);
219  KINETIC_LIST_LOCK(&connection->pdus);
220  if (pdu && (pdu->proto != NULL) && pdu->protobufDynamicallyExtracted) {
221  LOG3("Freeing dynamically allocated protobuf");
222  KineticProto_Message__free_unpacked(pdu->proto, NULL);
223  pdu->proto = NULL;
224  };
225 
226  /* TODO: We can't unlock until the function below completes, but it
227  * normally also tries to lock, so pass in a flag indicating
228  * we already have it locked. The way that the mutexes are
229  * currently initialized makes adding an attribute of
230  * PTHREAD_MUTEX_RECURSIVE significantly more trouble. */
231  KineticAllocator_FreeItem(&connection->pdus, (void*)pdu, false);
232  KINETIC_LIST_UNLOCK(&connection->pdus);
233  LOGF3("Freed PDU (0x%0llX) on connection (0x%0llX)", pdu, connection);
234 }

248 void KineticAllocator_FreeAllPDUs(KineticConnection* connection)
249 {
250  assert(connection != NULL);
251  if (connection->pdus.start != NULL) {
252  LOG3("Freeing all PDUs...");
253  KINETIC_LIST_LOCK(&connection->pdus);
254  KineticListItem* current = connection->pdus.start;
255  while (current != NULL) {
256  KineticPDU* pdu = (KineticPDU*)current->data;
257  if (pdu != NULL && pdu->proto != NULL
258  && pdu->protobufDynamicallyExtracted) {
259  KineticProto_Message__free_unpacked(pdu->proto, NULL);
260  }
261  current = current->next;
262  }
263  KINETIC_LIST_UNLOCK(&connection->pdus);
264  KineticAllocator_FreeList(&connection->pdus);
265  }
266  else {
267  LOG1(" Nothing to free!");
268  }
269 }

293 void KineticAllocator_FreeOperation(KineticConnection* const connection, KineticOperation* operation)
294 {
295  assert(connection != NULL);
296  assert(operation != NULL);
297  LOGF3("Freeing operation (0x%0llX) on connection (0x%0llX)", operation, connection);
298  if (operation->request != NULL) {
299  LOGF3("Freeing request PDU (0x%0llX) from operation (0x%0llX) on connection (0x%0llX)",
300  operation->request, operation, connection);
301  KineticAllocator_FreePDU(connection, operation->request);
302  operation->request = NULL;
303  }
304  if (operation->response != NULL) {
305  LOGF3("Freeing response PDU (0x%0llX) from operation (0x%0llX) on connection (0x%0llX)",
306  operation->response, operation, connection);
307  KineticAllocator_FreePDU(connection, operation->response);
308  operation->response = NULL;
309  }
310  pthread_mutex_destroy(&operation->timeoutTimeMutex);
311  KineticAllocator_FreeItem(&connection->operations, (void*)operation, true);
312  LOGF3("Freed operation (0x%0llX) on connection (0x%0llX)", operation, connection);
313 }
