30 KineticSession* KineticAllocator_NewSession(struct bus * b, KineticSessionConfig* config)
31 {
32  (void)b; // TODO: combine session w/connection, which will use this variable
33 
34  // Allocate a new session
35  KineticSession* session = KineticCalloc(1, sizeof(KineticSession));
36  if (session == NULL) {
37  LOG0("Failed allocating a new session!");
38  return NULL;
39  }
40 
41  // Copy the supplied config into the session config
42  session->config = *config;
43  memcpy(session->config.keyData, config->hmacKey.data, config->hmacKey.len);
44  // Update pointer to copy of key data
45  session->config.hmacKey.data = session->config.keyData;
46  strncpy(session->config.host, config->host, sizeof(session->config.host));
47  session->timeoutSeconds = config->timeoutSeconds;
48 
49  return session;
50 }

59 KineticConnection* KineticAllocator_NewConnection(struct bus * b, KineticSession* const session)
60 {
61  KineticConnection* connection = KineticCalloc(1, sizeof(KineticConnection));
62  if (connection == NULL) {
63  LOG0("Failed allocating new Connection!");
64  return NULL;
65  }
66  KineticResourceWaiter_Init(&connection->connectionReady);
67  connection->pSession = session;

103 KineticOperation* KineticAllocator_NewOperation(KineticConnection* const connection)
104 {
105  KINETIC_ASSERT(connection != NULL);
106  KINETIC_ASSERT(connection->pSession != NULL);
107  LOGF3("Allocating new operation on connection (0x%0llX)", connection);
108  KineticOperation* newOperation =
109  (KineticOperation*)KineticCalloc(1, sizeof(KineticOperation));
110  if (newOperation == NULL) {
111  LOGF0("Failed allocating new operation on connection (0x%0llX)!", connection);
112  return NULL;
113  }
114  KineticOperation_Init(newOperation, connection->pSession);
115  LOGF3("Allocating new PDU on connection (0x%0llX)", connection);
116  newOperation->request = (KineticRequest*)KineticCalloc(1, sizeof(KineticRequest));
117  if (newOperation->request == NULL) {
118  LOG0("Failed allocating new PDU!");
119  KineticFree(newOperation);
120  return NULL;
121  }
122  KineticRequest_Init(newOperation->request, connection->pSession);
123  LOGF3("Allocated new operation (0x%0llX) on connection (0x%0llX)", newOperation, connection);
124  return newOperation;
125 }

127 void KineticAllocator_FreeOperation(KineticOperation* operation)
128 {
129  KINETIC_ASSERT(operation != NULL);
130  KineticConnection* const connection = operation->connection;
131  LOGF3("Freeing operation (0x%0llX) on connection (0x%0llX)", operation, connection);
132  if (operation->request != NULL) {
133  LOGF3("Freeing request PDU (0x%0llX) from operation (0x%0llX) on connection (0x%0llX)",
134  operation->request, operation, connection);
135  KineticFree(operation->request);
136  LOGF3("Freed PDU (0x%0llX)", operation->request);
137  operation->request = NULL;
138  }
139  if (operation->response != NULL) {
140  LOGF3("Freeing response (0x%0llX) from operation (0x%0llX) on connection (0x%0llX)",
141  operation->response, operation, connection);
142  KineticAllocator_FreeKineticResponse(operation->response);
143  operation->response = NULL;
144  }
145  KineticFree(operation);
146  LOGF3("Freed operation (0x%0llX) on connection (0x%0llX)", operation, connection);
147 }
