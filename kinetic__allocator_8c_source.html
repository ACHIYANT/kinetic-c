42 void* KineticAllocator_NewItem(KineticList* const list, size_t size)
43 {
44  KineticListItem* newItem = (KineticListItem*)malloc(sizeof(KineticListItem));
45  if (newItem == NULL) {
46  LOG("Failed allocating new list item!");
47  return NULL;
48  }
49  memset(newItem, 0, sizeof(KineticListItem));
50  newItem->data = malloc(size);
51  memset(newItem->data, 0, size);
52 
53  // Add the new item to the list
54  KineticAllocator_Lock();
55  if (list->start == NULL) {
56  list->start = newItem;
57  }
58  else {
59  newItem->previous = list->last;
60  list->last->next = newItem;
61  }
62  list->last = newItem;
63  KineticAllocator_Unlock();
64 
65  // LOGF("Allocated new list item @ 0x%0llX w/data @ 0x%0llX",
66  // (long long)newItem, (long long)&newItem->data);
67 
68  return newItem->data;
69 }

71 void KineticAllocator_FreeItem(KineticList* const list, void* item)
72 {
73  KineticAllocator_Lock();
74  KineticListItem* cur = list->start;
75  while (cur->data != item) {
76  if (cur->next == NULL) {
77  LOG(" Reached end of list before finding item to free!");
78  KineticAllocator_Unlock();
79  return;
80  }
81  else {
82  cur = cur->next;
83  }
84  }
85  // LOG(" Done searching for item list item");
86 
87  if ((cur != NULL) && (cur->data == item)) {
88  // LOG(" item found! freeing it.");
89 
90  // Handle PDU list emptied
91  if (cur->previous == NULL) {
92  // LOG(" At start of list.");
93  if (cur->next == NULL) {
94  // LOG(" Making it empty, since all deallocated!");
95  list->start = NULL;
96  list->last = NULL;
97  }
98  else {
99  // LOG(" Moving current item to head, since head deallocated!");
100  list->start = cur->next;
101  list->start->previous = NULL;
102  }
103  }
104  else {
105  // Relink from previous to next, if avaliable
106  // LOG(" Not at list start, so relinking list to free item.");
107  if (cur->previous->next != NULL) {
108  // LOG(" Relinking previous to next");
109  if (cur->next != NULL) {
110  // LOG(" next being reset!");
111  cur->previous->next = cur->next;
112  }
113  else {
114  list->last = cur->previous;
115  list->last->next = NULL;
116  // LOGF(" next is NULL. End of list now @ 0x%0llX",
117  // (long long)list->last);
118  }
119  }
120  else {
121  LOG(" This shouldn't happen!");
122  list->last = cur->previous;
123  }
124  }
125 
126  // LOGF(" Freeing item @ 0x%0llX, item @ 0x%0llX",
127  // (long long)cur, (long long)&cur->data);
128  free(cur->data);
129  cur->data = NULL;
130  free(cur);
131  cur = NULL;
132  }
133  KineticAllocator_Unlock();
134 }

136 void KineticAllocator_FreeList(KineticList* const list)
137 {
138  if (list != NULL) {
139  LOG("Freeing list of all items");
140 
141  KineticAllocator_Lock();
142 
143  KineticListItem* current = list->start;
144 
145  while (current->next != NULL) {
146  // LOG("Advancing to next list item...");
147  current = current->next;
148  }
149 
150  while (current != NULL) {
151  // LOG(" Current item not freed!");
152  // LOGF(" DEALLOCATING item: 0x%0llX, data: 0x%llX, prev: 0x%0llX",
153  // (long long)current, (long long)&current->data, (long long)current->previous);
154  KineticListItem* curItem = current;
155  KineticListItem* prevItem = current->previous;
156  if (curItem != NULL) {
157  // LOG(" Freeing list item");
158  if (curItem->data != NULL) {
159  free(curItem->data);
160  }
161  free(curItem);
162  }
163  current = prevItem;
164  // LOGF(" on to prev=0x%llX", (long long)current);
165  }
166 
167  *list = (KineticList) {

169  };

171  KineticAllocator_Unlock();
172  }

206 {
