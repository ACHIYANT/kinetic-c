38 /* Do a blocking send.
39  *
40  * Returning true indicates that the message has been queued up for
41  * delivery, but the request or response may still fail. Those errors
42  * are handled by giving an error status code to the callback.
43  * Returning false means that the send was rejected outright, and
44  * the callback-based error handling will not be used. */
45 bool send_do_blocking_send(bus *b, boxed_msg *box) {
46  /* Note: assumes that all locking and thread-safe seq_id allocation
47  * has been handled upstream. If multiple client requests are queued
48  * up to go out at the same time, they must go out in monotonic order,
49  * with only a single thread writing to the socket at once. */
50  assert(b);
51  assert(box);
52 
53  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 256,
54  "doing blocking send of box %p, with <fd:%d, seq_id %lld>, msg[%zd]: %p",
55  (void *)box, box->fd, (long long)box->out_seq_id,
56  box->out_msg_size, (void *)box->out_msg);
57 
58  int timeout_msec = box->timeout_sec * 1000;
59 
60  struct timeval start;
61  if (util_timestamp(&start, true)) {
62  box->tv_send_start = start;
63  } else {
64  BUS_LOG_SNPRINTF(b, 0, LOG_SENDER, b->udata, 128,
65  "gettimeofday failure: %d", errno);
66  return false;
67  }
68 
69  struct pollfd fds[1];
70  fds[0].fd = box->fd;
71  fds[0].events = POLLOUT;
72 
73  /* Notify the listener that we're about to start writing to a drive,
74  * because (in rare cases) the response may arrive between finishing
75  * the write and the listener processing the notification. In that
76  * case, it should hold onto the unrecognized response until the
77  * client notifies it (and passes it the callback).
78  *
79  * This timeout is several extra seconds so that we don't have
80  * a window where the HOLD message has timed out, but the
81  * EXPECT hasn't, leading to ambiguity about what to do with
82  * the response (which may or may not have arrived).
83  * */
84  if (!attempt_to_enqueue_sending_request_message_to_listener(b,
85  box->fd, box->out_seq_id, box->timeout_sec + 5)) {
86  return false;
87  }
88  assert(box->out_sent_size == 0);
89 
90  int rem_msec = timeout_msec;
91 
92  while (rem_msec > 0) {
93  struct timeval now;
94  if (util_timestamp(&now, true)) {
95  size_t usec_elapsed = (((now.tv_sec - start.tv_sec) * 1000000)
96  + (now.tv_usec - start.tv_usec));
97  size_t msec_elapsed = usec_elapsed / 1000;
98 
99  rem_msec = timeout_msec - msec_elapsed;
100  } else {
101  /* If gettimeofday fails here, the listener's hold message has
102  * already been sent; it will time out later. We need to treat
103  * this like a TX failure (including closing the socket) because
104  * we don't know what state the connection was left in. */
105  BUS_LOG_SNPRINTF(b, 0, LOG_SENDER, b->udata, 128,
106  "gettimeofday failure in poll loop: %d", errno);
107  handle_failure(b, box, BUS_SEND_TX_FAILURE);
108  return true;
109  }
110 
111  int res = poll(fds, 1, rem_msec);
112  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 256,
113  "handle_write: poll res %d", res);
114  if (res == -1) {
115  if (errno == EINTR || errno == EAGAIN) { /* interrupted/try again */
116  errno = 0;
117  continue;
118  } else {
119  handle_failure(b, box, BUS_SEND_TX_FAILURE);
120  return true;
121  }
122  } else if (res == 1) {
123  short revents = fds[0].revents;
124  if (revents & POLLNVAL) {
125  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 256,
126  "do_blocking_send on %d: POLLNVAL => UNREGISTERED_SOCKET", box->fd);
127  handle_failure(b, box, BUS_SEND_UNREGISTERED_SOCKET);
128  return true;
129  } else if (revents & (POLLERR | POLLHUP)) {
130  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 256,
131  "do_blocking_send on %d: POLLERR/POLLHUP => TX_FAILURE (%d)",
132  box->fd, revents);
133  handle_failure(b, box, BUS_SEND_TX_FAILURE);
134  return true;
135  } else if (revents & POLLOUT) {
136  handle_write_res hw_res = handle_write(b, box);
137 
138  BUS_LOG_SNPRINTF(b, 4, LOG_SENDER, b->udata, 256,
139  "handle_write res %d", hw_res);
140 
141  switch (hw_res) {
142  case HW_OK:
143  continue;
144  case HW_DONE:
145  return true;
146  case HW_ERROR:
147  return true;
148  }
149  } else {
150  BUS_LOG_SNPRINTF(b, 0, LOG_SENDER, b->udata, 256,
151  "match fail %d", revents);
152  assert(false); /* match fail */
153  }
154  } else if (res == 0) { /* timeout */
155  break;
156  }
157  }
158  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 256,
159  "do_blocking_send on <fd:%d, seq_id:%lld>: TX_TIMEOUT",
160  box->fd, (long long)box->out_seq_id);
161  handle_failure(b, box, BUS_SEND_TX_TIMEOUT);
162  return true;
163 }

165 static bool attempt_to_enqueue_sending_request_message_to_listener(struct bus *b,
166  int fd, int64_t seq_id, int16_t timeout_sec) {
167  BUS_LOG_SNPRINTF(b, 5, LOG_SENDER, b->udata, 128,
168  "telling listener to expect response, with <fd%d, seq_id:%lld>",
169  fd, (long long)seq_id);
170 
171  struct listener *l = bus_get_listener_for_socket(b, fd);
172 
173  int delay = 1;
174  const int max_retries = 10;
175  for (int try = 0; try < max_retries; try++) {
176  if (listener_hold_response(l, fd, seq_id, timeout_sec)) {
177  return true;
178  } else {
179  /* Don't apply much backpressure here since the client
180  * thread will get it when the message is done sending. */
181  poll(NULL, 0, delay);
182  if (delay < 5) { delay++; }
183  }
184  }
185  return false;
186 }

188 static void handle_failure(struct bus *b, boxed_msg *box, bus_send_status_t status) {
189  BUS_LOG_SNPRINTF(b, 5, LOG_SENDER, b->udata, 64,
190  "handle_failure: box %p, <fd:%d, seq_id:%lld>, status %d",
191  (void*)box, box->fd, (long long)box->out_seq_id, status);
192 
193  box->result = (bus_msg_result_t){
194  .status = status,
195  };
196 
197  size_t backpressure = 0;
198 
199  /* Retry until it succeeds. */
200  size_t retries = 0;
201  for (;;) {
202  if (bus_process_boxed_message(b, box, &backpressure)) {
203  BUS_LOG_SNPRINTF(b, 5, LOG_SENDER, b->udata, 64,
204  "deleted box %p", (void*)box);
205  bus_backpressure_delay(b, backpressure,
206  LISTENER_EXPECT_BACKPRESSURE_SHIFT);
207  return;
208  } else {
209  retries++;
210  const int delay = 5;
211  poll(NULL, 0, delay);
212  if (retries > 0 && (retries & 255) == 0) {
213  BUS_LOG_SNPRINTF(b, 0, LOG_SENDER, b->udata, 64,
214  "looping on handle_failure retry: %zd", retries);
215  }
216  }
217  }
218 }

224 static handle_write_res handle_write(bus *b, boxed_msg *box) {
225  SSL *ssl = box->ssl;
226  ssize_t wrsz = 0;
227 
228  /* Attempt a single write to the socket. */
229  if (ssl == BUS_NO_SSL) {
230  wrsz = write_plain(b, box);
231  } else {
232  assert(ssl);
233  wrsz = write_ssl(b, box, ssl);
234  }
235  BUS_LOG_SNPRINTF(b, 5, LOG_SENDER, b->udata, 64,
236  "wrote %zd", wrsz);
237 
238  if (wrsz == -1) {
239  handle_failure(b, box, BUS_SEND_TX_FAILURE);
240  return HW_ERROR;
241  } else if (wrsz == 0) {
242  /* If the OS set POLLOUT but we can't actually write, then
243  * just go back to the poll() loop with no progress.
244  * If we busywait on this, something is deeply wrong. */
245  BUS_LOG_SNPRINTF(b, 0, LOG_SENDER, b->udata, 128,
246  "suspicious: wrote %zd bytes to <fd:%d, seq_id%lld>",
247  wrsz, box->fd, (long long)box->out_seq_id);
248  } else {
249  /* Update amount written so far */
250  box->out_sent_size += wrsz;
251  }
252 
253  size_t msg_size = box->out_msg_size;
254  size_t sent_size = box->out_sent_size;
255  size_t rem = msg_size - sent_size;
256 
257  BUS_LOG_SNPRINTF(b, 5, LOG_SENDER, b->udata, 64,
258  "wrote %zd, rem is %zd", wrsz, rem);
259 
260  if (rem == 0) { /* check if whole message is written */
261  struct timeval tv;
262  if (util_timestamp(&tv, true)) {
263  box->tv_send_done = tv;
264  }
265 
266  if (enqueue_request_sent_message_to_listener(b, box)) {
267  return HW_DONE;
268  } else {
269  handle_failure(b, box, BUS_SEND_TX_TIMEOUT_NOTIFYING_LISTENER);
270  return HW_ERROR;
271  }
272  } else {
273  return HW_OK;
274  }
275 }

277 static ssize_t write_plain(struct bus *b, boxed_msg *box) {
278  int fd = box->fd;
279  uint8_t *msg = box->out_msg;
280  size_t msg_size = box->out_msg_size;
281  size_t sent_size = box->out_sent_size;
282  size_t rem = msg_size - sent_size;
283 
284  BUS_LOG_SNPRINTF(b, 10, LOG_SENDER, b->udata, 64,
285  "write %p to %d, %zd bytes",
286  (void*)&msg[sent_size], fd, rem);
287 
288  /* Attempt a single write. ('for' is due to continue-based retry.) */
289  for (;;) {
290  ssize_t wrsz = write(fd, &msg[sent_size], rem);
291  if (wrsz == -1) {
292  if (util_is_resumable_io_error(errno)) {
293  errno = 0;
294  continue;
295  } else {
296  /* will notify about closed socket upstream */
297  BUS_LOG_SNPRINTF(b, 1, LOG_SENDER, b->udata, 64,
298  "write: socket error writing, %s", strerror(errno));
299  errno = 0;
300  return -1;
301  }
302  } else if (wrsz > 0) {
303  BUS_LOG_SNPRINTF(b, 5, LOG_SENDER, b->udata, 64,
304  "sent: %zd\n", wrsz);
305  return wrsz;
306  } else {
307  return 0;
308  }
309  }
310 }

312 static ssize_t write_ssl(struct bus *b, boxed_msg *box, SSL *ssl) {
313  uint8_t *msg = box->out_msg;
314  size_t msg_size = box->out_msg_size;
315  ssize_t rem = msg_size - box->out_sent_size;
316  int fd = box->fd;
317  ssize_t written = 0;
318  assert(rem >= 0);
319 
320  while (rem > 0) {
321  ssize_t wrsz = SSL_write(ssl, &msg[box->out_sent_size], rem);
322  BUS_LOG_SNPRINTF(b, 5, LOG_SENDER, b->udata, 64,
323  "SSL_write: socket %d, write %zd => wrsz %zd",
324  fd, rem, wrsz);
325  if (wrsz > 0) {
326  written += wrsz;
327  return written;
328  } else if (wrsz < 0) {
329  int reason = SSL_get_error(ssl, wrsz);
330  switch (reason) {
331  case SSL_ERROR_WANT_WRITE:
332  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 64,
333  "SSL_write: socket %d: WANT_WRITE", fd);
334  break;
335 
336  case SSL_ERROR_WANT_READ:
337  BUS_LOG_SNPRINTF(b, 0, LOG_SENDER, b->udata, 64,
338  "SSL_write: socket %d: WANT_READ", fd);
339  assert(false); // shouldn't get this; we're writing.
340  break;
341 
342  case SSL_ERROR_SYSCALL:
343  {
344  if (util_is_resumable_io_error(errno)) {
345  errno = 0;
346  /* don't break; we want to retry on EINTR etc. until
347  * we get WANT_WRITE, otherwise poll(2) may not retry
348  * the socket for too long */
349  continue;
350  } else {
351  /* will notify about socket error upstream */
352  BUS_LOG_SNPRINTF(b, 1, LOG_SENDER, b->udata, 64,
353  "SSL_write on fd %d: SSL_ERROR_SYSCALL -- %s",
354  fd, strerror(errno));
355  errno = 0;
356  return -1;
357  }
358  }
359  case SSL_ERROR_SSL:
360  BUS_LOG_SNPRINTF(b, 1, LOG_SENDER, b->udata, 64,
361  "SSL_write: socket %d: error SSL (wrsz %zd)",
362  box->fd, wrsz);
363 
364  /* Log error messages from OpenSSL */
365  for (;;) {
366  unsigned long e = ERR_get_error();
367  if (e == 0) { break; } // no more errors
368  BUS_LOG_SNPRINTF(b, 1, LOG_SENDER, b->udata, 128,
369  "SSL_write error: %s",
370  ERR_error_string(e, NULL));
371  }
372  return -1;
373  default:
374  {
375  BUS_LOG_SNPRINTF(b, 1, LOG_SENDER, b->udata, 64,
376  "SSL_write on socket %d: match fail: error %d", box->fd, reason);
377  return -1;
378  }
379  }
380  } else {
381  /* SSL_write should give SSL_ERROR_WANT_WRITE when unable
382  * to write further. */
383  }
384  }
385  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 64,
386  "SSL_write: leaving loop, %zd bytes written", written);
387  return written;
388 }

390 /* Notify the listener that the client has finished sending a message, and
391  * transfer all details for handling the response to it. Blocking. */
392 static bool enqueue_request_sent_message_to_listener(bus *b, boxed_msg *box) {
393  /* Notify listener that it should expect a response to a
394  * successfully sent message. */
395  BUS_LOG_SNPRINTF(b, 3, LOG_SENDER, b->udata, 128,
396  "telling listener to expect sent response, with box %p, seq_id %lld",
397  (void *)box, (long long)box->out_seq_id);
398 
399  struct listener *l = bus_get_listener_for_socket(b, box->fd);
400 
401  for (int retries = 0; retries < 10; retries++) {
402  uint16_t backpressure = 0;
403  /* If this succeeds, then this thread cannot touch the box anymore. */
404  if (listener_expect_response(l, box, &backpressure)) {
405  bus_backpressure_delay(b, backpressure,
406  LISTENER_EXPECT_BACKPRESSURE_SHIFT);
407  return true;
408  } else {
409  BUS_LOG_SNPRINTF(b, 5, LOG_SENDER, b->udata, 64,
410  "enqueue_request_sent: failed delivery %d", retries);
411  poll(NULL, 0, 10); // delay
412  }
413  }
414 
415  /* Timeout, will be treated as a TX error */
416  return false;
417 }
