54 static const char *str_of_permission(KineticProto_Command_Security_ACL_Permission perm) {
55  for (size_t i = 0; i < PERM_TABLE_ROWS; i++) {
56  permission_pair *pp = &permission_table[i];

62 static KineticProto_Command_Security_ACL_Permission permission_of_str(const char *str) {
63  for (size_t i = 0; i < PERM_TABLE_ROWS; i++) {
64  permission_pair *pp = &permission_table[i];

71 acl_of_file(const char *path, struct ACL **instance) {
72  if (path == NULL || instance == NULL) {
73  return ACL_ERROR_NULL;
74  }
75 
76  int fd = open(path, O_RDONLY);
77  if (fd == -1) {
78 #ifndef TEST
79  LOGF0("Failed ot open file '%s': %s", path, strerror(errno));
80 #endif
81  errno = 0;
82  return ACL_ERROR_BAD_JSON;
83  }
84  acl_of_file_res res = ACL_ERROR_NULL;
85 
86  const int BUF_START_SIZE = 256;
87  char *buf = calloc(1, BUF_START_SIZE);
88  if (buf == NULL) {
89  res = ACL_ERROR_MEMORY;
90  goto cleanup;
91  }
92  size_t buf_sz = BUF_START_SIZE;
93  size_t buf_used = 0;
94 
95  ssize_t read_sz = 0;
96  for (;;) {
97  read_sz = read(fd, &buf[buf_used], buf_sz - buf_used);
98  if (read_sz == -1) {
99  res = ACL_ERROR_JSON_FILE;
100  goto cleanup;
101  } else if (read_sz == 0) {
102  break;
103  } else {
104  buf_used += read_sz;
105  if (buf_sz == buf_used) {
106  size_t nsz = 2 * buf_sz;
107  char *nbuf = realloc(buf, nsz);
108  if (nbuf) {
109  buf_sz = nsz;
110  buf = nbuf;
111  }
112  }
113  }
114  }
115 
116 #ifndef TEST
117  LOGF2(" -- read %zd bytes, parsing...\n", buf_used);
118 #endif
119  res = acl_of_string(buf, buf_used, instance);
120 cleanup:
121  if (buf) { free(buf); }
122  close(fd);
123  return res;
124 }

126 acl_of_file_res
127 acl_of_string(const char *buf, size_t buf_size, struct ACL **instance) {
128  acl_of_file_res res = ACL_ERROR_MEMORY;
129  struct ACL *acl_group = NULL;
130  KineticProto_Command_Security_ACL **acl_array = NULL;
131  struct json_tokener* tokener = NULL;
132 
133  acl_group = calloc(1, sizeof(*acl_group));
134  if (acl_group == NULL) { goto cleanup; }
135 
136  acl_group->ACLs = calloc(1, sizeof(KineticProto_Command_Security_ACL *));
137  if (acl_group->ACLs == NULL) { goto cleanup; }
138  acl_group->ACL_ceil = 1;
139  acl_group->ACL_count = 0;
140 
141  tokener = json_tokener_new();
142  if (tokener == NULL) { goto cleanup; }
143 
144  size_t offset = 0;
145 
146  while (buf_size - offset > 0) {
147  size_t offset_out = 0;
148  KineticProto_Command_Security_ACL *new_acl = NULL;
149 #ifndef TEST
150  LOGF2(" -- reading next ACL at offset %zd, rem %zd\n", offset, buf_size - offset);
151 #endif
152  res = read_next_ACL(buf, buf_size, offset,
153  &offset_out, tokener, &new_acl);
154  offset += offset_out;
155 #ifndef TEST
156  LOGF2(" -- result %d, offset_out %zd\n", res, offset);
157 #endif
158  if (res == ACL_OK) {
159  if (acl_group->ACL_count == acl_group->ACL_ceil) { /* grow */
160  size_t nsz = 2 * acl_group->ACL_ceil * sizeof(acl_group->ACLs[0]);
161  KineticProto_Command_Security_ACL **nACLs = realloc(acl_group->ACLs, nsz);
162  if (nACLs == NULL) {
163  goto cleanup;
164  } else {
165  acl_group->ACL_ceil *= 2;
166  acl_group->ACLs = nACLs;
167  }
168  }
169  acl_group->ACLs[acl_group->ACL_count] = new_acl;
170  acl_group->ACL_count++;
171  } else {
172  break;
173  }
174  }
175 
176 cleanup:
177  if (tokener) { json_tokener_free(tokener); }
178 
179  if (res == ACL_END_OF_STREAM || res == ACL_OK) {
180  if (acl_group && acl_group->ACL_count == 0) {
181  LOG2("Failed to read any JSON objects\n");
182  return ACL_ERROR_BAD_JSON;
183  } else { /* read at least one ACL */
184  *instance = acl_group;
185  return ACL_OK;
186  }
187  } else {
188  if (acl_group) { free(acl_group); }
189  if (acl_array) { free(acl_array); }
190  return res;
191  }
192 }

194 static acl_of_file_res read_next_ACL(const char *buf, size_t buf_size,
195  size_t offset, size_t *new_offset,
196  struct json_tokener *tokener, KineticProto_Command_Security_ACL **instance) {
197  struct json_object *obj = json_tokener_parse_ex(tokener,
198  &buf[offset], buf_size - offset);
199  if (obj == NULL) {
200  if (json_tokener_get_error(tokener) == json_tokener_error_parse_eof) {
201  return ACL_END_OF_STREAM;
202  } else {
203  LOGF2("JSON error %d\n", json_tokener_get_error(tokener));
204  return ACL_ERROR_BAD_JSON;
205  }
206  }
207 
208  *new_offset = tokener->char_offset;
209 
210  acl_of_file_res res = ACL_ERROR_MEMORY;
211  KineticProto_Command_Security_ACL *acl = NULL;
212  uint8_t *data = NULL;
213 
214  int scope_count = 0;
215  struct json_object *val = NULL;
216  if (json_object_object_get_ex(obj, "scope", &val)) {
217  scope_count = json_object_array_length(val);
218  } else {
219  res = ACL_ERROR_MISSING_FIELD;
220  goto cleanup;
221  }
222 
223  size_t alloc_sz = sizeof(*acl);
224  acl = calloc(1, alloc_sz);
225  if (acl == NULL) { goto cleanup; }
226 
227  KineticProto_command_security_acl__init(acl);
228 
229  /* Copy fields */
230  if (json_object_object_get_ex(obj, "identity", &val)) {
231  acl->has_identity = true;
232  acl->identity = json_object_get_int64(val);
233  } else {
234  acl->has_identity = false;
235  }
236 
237  if (json_object_object_get_ex(obj, "key", &val)) {
238  const char *key = json_object_get_string(val);
239  size_t len = strlen(key);
240 
241  acl->has_hmacAlgorithm = true;
242  acl->hmacAlgorithm = KINETIC_PROTO_COMMAND_SECURITY_ACL_HMACALGORITHM_HmacSHA1;
243 
244  acl->key.len = len;
245  data = calloc(1, len + 1);
246  if (data == NULL) { goto cleanup; }
247  memcpy(data, key, len);
248  data[len] = '\0';
249  acl->key.data = data;
250  data = NULL;
251  acl->has_key = true;
252  }
253 
254  if (json_object_object_get_ex(obj, "HMACAlgorithm", &val)) {
255  const char *hmac = json_object_get_string(val);
256  if (0 != strcmp(hmac, "HmacSHA1")) {
257  res = ACL_ERROR_INVALID_FIELD;
258  goto cleanup;
259  } else {
260  // acl->key->type is already set to HMAC_SHA1.
261  }
262  }
263 
264  struct json_object *scopes = NULL;
265  if (json_object_object_get_ex(obj, "scope", &scopes)) {
266  res = unpack_scopes(acl, scope_count, scopes);
267  if (res != ACL_OK) { goto cleanup; }
268  }
269 
270  /* Decrement JSON object refcount, freeing it. */
271  json_object_put(obj);
272  *instance = acl;
273  return ACL_OK;
274 cleanup:
275  if (obj) { json_object_put(obj); }
276  if (acl) { free(acl); }
277  if (data) { free(data); }
278  return res;
279 }

281 static acl_of_file_res unpack_scopes(KineticProto_Command_Security_ACL *acl,
282  int scope_count, json_object *scopes) {
283  acl_of_file_res res = ACL_ERROR_MEMORY;
284  KineticProto_Command_Security_ACL_Scope **scope_array = NULL;
285  KineticProto_Command_Security_ACL_Permission *perm_array = NULL;
286  KineticProto_Command_Security_ACL_Scope *scope = NULL;
287  uint8_t *data = NULL;
288 
289  scope_array = calloc(scope_count, sizeof(*scope_array));
290  if (scope_array == NULL) { goto cleanup; }
291 
292  acl->scope = scope_array;
293 
294  for (int si = 0; si < scope_count; si++) {
295  struct json_object *cur_scope = json_object_array_get_idx(scopes, si);
296  if (cur_scope) {
297  scope = calloc(1, sizeof(*scope));
298  if (scope == NULL) { goto cleanup; }
299  KineticProto_command_security_acl_scope__init(scope);
300 
301  struct json_object *val = NULL;
302  if (json_object_object_get_ex(cur_scope, "offset", &val)) {
303  scope->offset = json_object_get_int64(val);
304  scope->has_offset = true;
305  } else {
306  scope->has_offset = false;
307  }
308 
309  if (json_object_object_get_ex(cur_scope, "value", &val)) {
310  const char *str = json_object_get_string(val);
311  if (str) {
312  size_t len = strlen(str);
313  data = malloc(len + 1);
314  if (data == NULL) {
315  return ACL_ERROR_MEMORY;
316  } else {
317  memcpy(data, str, len);
318  data[len] = '\0';
319  scope->value.data = data;
320  scope->value.len = len;
321  data = NULL;
322  scope->has_value = true;
323  }
324  }
325  } else {
326  scope->has_value = false;
327  }
328 
329  scope->n_permission = 0;
330  if (json_object_object_get_ex(cur_scope, "permission", &val)) {
331  perm_array = calloc(ACL_MAX_PERMISSIONS, sizeof(*perm_array));
332  if (perm_array == NULL) { goto cleanup; }
333  scope->permission = perm_array;
334  perm_array = NULL;
335 
336  enum json_type perm_type = json_object_get_type(val);
337  if (perm_type == json_type_string) {
338  scope->permission[0] = permission_of_str(json_object_get_string(val));
339  if (scope->permission[0] == KINETIC_PROTO_COMMAND_SECURITY_ACL_PERMISSION_INVALID_PERMISSION) {
340  return ACL_ERROR_INVALID_FIELD;
341  } else {
342  scope->n_permission++;
343  }
344  } else if (perm_type == json_type_array) {
345  int count = json_object_array_length(val);
346  for (int i = 0; i < count; i++) {
347  struct json_object *jperm = json_object_array_get_idx(val, i);
348  KineticProto_Command_Security_ACL_Permission p;
349  p = permission_of_str(json_object_get_string(jperm));
350  if (p == KINETIC_PROTO_COMMAND_SECURITY_ACL_PERMISSION_INVALID_PERMISSION) {
351  return ACL_ERROR_INVALID_FIELD;
352  } else {
353  scope->permission[scope->n_permission] = p;
354  scope->n_permission++;
355  }
356  }
357  } else {
358  return ACL_ERROR_INVALID_FIELD;
359  }
360  }
361  if (json_object_object_get_ex(cur_scope, "TlsRequired", &val)) {
362  scope->TlsRequired = json_object_get_boolean(val);
363  scope->has_TlsRequired = true;
364  }
365 
366  acl->scope[acl->n_scope] = scope;
367  acl->n_scope++;
368  scope = NULL;
369  }
370  }
371  acl->n_scope = scope_count;
372  return ACL_OK;
373 
374 cleanup:
375  if (scope_array) { free(scope_array); }
376  if (scope) { free(scope); }
377  if (perm_array) { free(perm_array); }
378  if (data) { free(data); }
379  return res;
380 }

382 void acl_fprintf(FILE *f, struct ACL *ACLs) {
383  if (ACLs == NULL) {
384  fprintf(f, "NULL\n");
385  return;
386  }
387 
388  fprintf(f, "ACLs [%zd]:\n", ACLs->ACL_count);
389 
390  for (size_t ai = 0; ai < ACLs->ACL_count; ai++) {
391  KineticProto_Command_Security_ACL *acl = ACLs->ACLs[ai];
392  if (acl == NULL) { continue; }
393  if (ai > 0) { fprintf(f, "\n"); }
394 
395  if (acl->has_identity) {
396  fprintf(f, " identity: %lld\n", (long long)acl->identity);
397  }
398 
399  if (acl->has_key) {
400  fprintf(f, " key[%s,%zd]: \"%s\"\n",
401  "HmacSHA1", acl->key.len, acl->key.data);
402  }
403 
404  fprintf(f, " scopes: (%zd)\n", acl->n_scope);
405 
406  for (size_t si = 0; si < acl->n_scope; si++) {
407  KineticProto_Command_Security_ACL_Scope *scope = acl->scope[si];
408  if (si > 0) { fprintf(f, "\n"); }
409  fprintf(f, " scope %zd:\n", si);
410  if (scope->has_offset) {
411  fprintf(f, " offset: %lld\n", (long long)scope->offset);
412  }
413  if (scope->has_value) {
414  fprintf(f, " value[%zd]: \"%s\"\n",
415  scope->value.len, scope->value.data);
416  }
417  for (size_t pi = 0; pi < scope->n_permission; pi++) {
418  fprintf(f, " permission: %s\n",
419  str_of_permission(scope->permission[pi]));
420  }
421 
422  if (scope->has_TlsRequired) {
423  fprintf(f, " TlsRequired: %d\n", scope->TlsRequired);
424  }
425  }
426  }

431  for (size_t ai = 0; ai < ACLs->ACL_count; ai++) {
432  KineticProto_Command_Security_ACL *acl = ACLs->ACLs[ai];

437  free(scope->value.data);

439 
440  if (scope->n_permission > 0) {
441  free(scope->permission);

443  free(scope);
444  }
445  free(acl->scope);
446 
447  if (acl->has_key && acl->key.data) {
448  free(acl->key.data);
449  }
450  free(acl);
451  }
452  }
