227 bool Copy_KineticProto_Command_KeyValue_to_KineticEntry(KineticProto_Command_KeyValue* keyValue, KineticEntry* entry)
228 {
229  bool bufferOverflow = false;
230 
231  if (keyValue != NULL && entry != NULL) {
232  ByteBuffer_Reset(&entry->dbVersion);
233  if (keyValue->has_dbVersion && keyValue->dbVersion.len > 0) {
234  if (entry->dbVersion.array.data == NULL || entry->dbVersion.array.len < keyValue->dbVersion.len) {
235  entry->dbVersion.bytesUsed = keyValue->dbVersion.len;
236  LOG1(" BUFFER_OVERRUN: dbVersion");
237  bufferOverflow = true;
238  }
239  else {
240  ByteBuffer_Append(&entry->dbVersion, keyValue->dbVersion.data, keyValue->dbVersion.len);
241  }
242  }
243 
244  ByteBuffer_Reset(&entry->key);
245  if (keyValue->has_key && keyValue->key.len > 0) {
246  if (entry->key.array.data == NULL || entry->key.array.len < keyValue->key.len) {
247  entry->key.bytesUsed = keyValue->key.len;
248  LOG1(" BUFFER_OVERRUN: key");
249  bufferOverflow = true;
250  }
251  else {
252  ByteBuffer_Append(&entry->key, keyValue->key.data, keyValue->key.len);
253  }
254  }
255 
256  ByteBuffer_Reset(&entry->tag);
257  if (keyValue->has_tag && keyValue->tag.len > 0) {
258  if (entry->tag.array.data == NULL || entry->tag.array.len < keyValue->tag.len) {
259  entry->tag.bytesUsed = keyValue->tag.len;
260  LOG1(" BUFFER_OVERRUN: tag");
261  bufferOverflow = true;
262  }
263  else {
264  ByteBuffer_Append(&entry->tag, keyValue->tag.data, keyValue->tag.len);
265  }
266  }
267 
268  if (keyValue->has_algorithm) {
269  entry->algorithm =
270  KineticAlgorithm_from_KineticProto_Command_Algorithm(
271  keyValue->algorithm);
272  }
273  }
274 
275  return !bufferOverflow;
276 }

278 bool Copy_KineticProto_Command_Range_to_ByteBufferArray(KineticProto_Command_Range* keyRange, ByteBufferArray* keys)
279 {
280  bool bufferOverflow = false;
281  LOGF2("Copying: keyRange=0x%0llX, keys=0x%0llX, max_keys=%lld", keyRange, keys->buffers, keys->count);
282  if (keyRange != NULL && keys->count > 0 && keys != NULL) {
283  for (size_t i = 0; i < MIN((size_t)keys->count, (size_t)keyRange->n_keys); i++) {
284  ByteBuffer_Reset(&keys->buffers[i]);
285  if (ByteBuffer_Append(&keys->buffers[i], keyRange->keys[i].data, keyRange->keys[i].len) == NULL) {
286  LOGF2("WANRNING: Buffer overrun for keys[%zd]", i);
287  bufferOverflow = true;
288  }
289  }
290  keys->used = keyRange->n_keys;
291  }
292  return !bufferOverflow;
293 }
