193 bool Copy_ProtobufCBinaryData_to_ByteBuffer(ByteBuffer dest, ProtobufCBinaryData src)
194 {
195  if (src.data == NULL || src.len == 0) {
196  return false;
197  }
198  if (dest.array.data == NULL || dest.array.len < src.len) {
199  return false;
200  }
201 
202  bool success = (memcpy(dest.array.data, src.data, src.len) == dest.array.data);
203  if (success) {
204  dest.bytesUsed = src.len;
205  }
206 
207  return success;
208 }
