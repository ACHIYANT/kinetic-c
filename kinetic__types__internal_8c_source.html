203 {
204  if (src.data == NULL || src.len == 0) {
205  return false;
206  }
207  if (dest.array.data == NULL || dest.array.len < src.len) {
208  return false;

211  bool success = (memcpy(dest.array.data, src.data, src.len) == dest.array.data);
212  if (success) {
213  dest.bytesUsed = src.len;
214  }

216  return success;

220 bool Copy_KineticProto_Command_KeyValue_to_KineticEntry(KineticProto_Command_KeyValue* keyValue, KineticEntry* entry)
221 {
222  bool bufferOverflow = false;
223 
224  if (keyValue != NULL && entry != NULL) {
225 
226  ByteBuffer_Reset(&entry->newVersion);
227  if (keyValue->has_newVersion) {
228  if (entry->newVersion.array.data == NULL ||
229  entry->newVersion.array.len < keyValue->newVersion.len) {
230  entry->newVersion.bytesUsed = keyValue->newVersion.len;
231  LOG1(" BUFFER_OVERRUN: newVersion");

233  }

235  ByteBuffer_Append(&entry->newVersion, keyValue->newVersion.data, keyValue->newVersion.len);

247  ByteBuffer_Append(&entry->dbVersion, keyValue->dbVersion.data, keyValue->dbVersion.len);
248  }
249  }
250 
251  ByteBuffer_Reset(&entry->key);
252  if (keyValue->has_key) {
253  if (entry->key.array.data == NULL || entry->key.array.len < keyValue->key.len) {
254  entry->key.bytesUsed = keyValue->key.len;
255  LOG1(" BUFFER_OVERRUN: key");
256  bufferOverflow = true;

259  ByteBuffer_Append(&entry->key, keyValue->key.data, keyValue->key.len);

266  entry->tag.bytesUsed = keyValue->tag.len;

271  ByteBuffer_Append(&entry->tag, keyValue->tag.data, keyValue->tag.len);
272  }
273  }
274

276  entry->algorithm =

294 bool Copy_KineticProto_Command_Range_to_ByteBufferArray(KineticProto_Command_Range* keyRange, ByteBufferArray* keys)
295 {
296  bool bufferOverflow = false;
297  LOGF2("Copying: keyRange=0x%0llX, keys=0x%0llX, max_keys=%lld", keyRange, keys->buffers, keys->count);
298  if (keyRange != NULL && keys->count > 0 && keys != NULL) {
299  for (size_t i = 0; i < MIN((size_t)keys->count, (size_t)keyRange->n_keys); i++) {
300  ByteBuffer_Reset(&keys->buffers[i]);
301  if (ByteBuffer_Append(&keys->buffers[i], keyRange->keys[i].data, keyRange->keys[i].len) == NULL) {
302  LOGF2("WANRNING: Buffer overrun for keys[%zd]", i);
303  bufferOverflow = true;
304  }
305  }
306  }
307  return !bufferOverflow;
308 }
