34 KineticOperation* KineticController_CreateOperation(KineticSession const * const session)
35 {
36  if (session == NULL) {
37  LOG0("Specified session is NULL");
38  return NULL;
39  }
40 
41  if (session->connection == NULL) {
42  LOG0("Specified session is not associated with a connection");

44  }
45 
46  LOGF3("--------------------------------------------------\n"

48 
49  KineticOperation* operation = KineticAllocator_NewOperation(session->connection);
50  if (operation == NULL || operation->request == NULL) {
51  return NULL;
52  }

64 static void DefaultCallback(KineticCompletionData* kinetic_data, void* client_data)
65 {
66  DefaultCallbackData * data = client_data;
67  pthread_mutex_lock(&data->receiveCompleteMutex);
68  data->status = kinetic_data->status;

82 KineticStatus KineticController_ExecuteOperation(KineticOperation* operation, KineticCompletionClosure* const closure)
83 {
84  KINETIC_ASSERT(operation != NULL);
85  KINETIC_ASSERT(operation->connection != NULL);
86  KINETIC_ASSERT(operation->connection->pSession != NULL);
87  KineticStatus status = KINETIC_STATUS_INVALID;
88 
89  if (closure != NULL)
90  {
91  operation->closure = *closure;
92  return KineticOperation_SendRequest(operation);
93  }
94  else
95  {
96  DefaultCallbackData data;
97  pthread_mutex_init(&data.receiveCompleteMutex, NULL);
98  pthread_cond_init(&data.receiveComplete, NULL);
99  data.status = KINETIC_STATUS_INVALID;
100  data.completed = false;
101 
102  operation->closure = DefaultClosure(&data);
103 
104  // Send the request
105  status = KineticOperation_SendRequest(operation);
106 
107  if (status == KINETIC_STATUS_SUCCESS) {
108  pthread_mutex_lock(&data.receiveCompleteMutex);
109  while(data.completed == false)
110  { pthread_cond_wait(&data.receiveComplete, &data.receiveCompleteMutex); }
111  status = data.status;
112  pthread_mutex_unlock(&data.receiveCompleteMutex);
113  }
114 
115  pthread_cond_destroy(&data.receiveComplete);
116  pthread_mutex_destroy(&data.receiveCompleteMutex);
117 
118  return status;
119  }
120 }

122 KineticStatus bus_to_kinetic_status(bus_send_status_t const status)
123 {
124  KineticStatus res = KINETIC_STATUS_INVALID;
125 
126  switch(status)
127  {
128  // TODO scrutinize all these mappings
129  case BUS_SEND_SUCCESS:
130  res = KINETIC_STATUS_SUCCESS;
131  break;
132  case BUS_SEND_TX_TIMEOUT:
133  res = KINETIC_STATUS_SOCKET_TIMEOUT;
134  break;
135  case BUS_SEND_TX_FAILURE:
136  res = KINETIC_STATUS_SOCKET_ERROR;
137  break;
138  case BUS_SEND_RX_TIMEOUT:
139  res = KINETIC_STATUS_OPERATION_TIMEDOUT;
140  break;
141  case BUS_SEND_RX_FAILURE:
142  res = KINETIC_STATUS_SOCKET_ERROR;
143  break;
144  case BUS_SEND_BAD_RESPONSE:
145  res = KINETIC_STATUS_SOCKET_ERROR;
146  break;
147  case BUS_SEND_UNREGISTERED_SOCKET:
148  res = KINETIC_STATUS_SOCKET_ERROR;
149  break;
150  case BUS_SEND_RX_TIMEOUT_EXPECT:
151  res = KINETIC_STATUS_OPERATION_TIMEDOUT;
152  break;
153  case BUS_SEND_UNDEFINED:
154  default:
155  {
156  LOGF0("bus_to_kinetic_status: UNMATCHED %d\n", status);
157  KINETIC_ASSERT(false);
158  return KINETIC_STATUS_INVALID;
159  }
160  }
161 
162  LOGF3("bus_to_kinetic_status: mapping status %d => %d\n",
163  status, res);
164  return res;
165 }

191 void KineticController_HandleUnexpectedResponse(void *msg,
192  int64_t seq_id,
193  void *bus_udata,
194  void *socket_udata)
195 {
196  KineticResponse * response = msg;
197  KineticConnection* connection = socket_udata;
198  bool connetionInfoReceived = false;
199  char const * statusTag = "[PDU RX STATUS]";
200  char const * unexpectedTag = "[PDU RX UNEXPECTED]";
201  char const * logTag = unexpectedTag;
202  int logAtLevel, protoLogAtLevel;
203 
204  (void)bus_udata;
205 
206  // Handle unsolicited status PDUs
207  if (response->proto->authType == KINETIC_PROTO_MESSAGE_AUTH_TYPE_UNSOLICITEDSTATUS) {
208  if (response->command != NULL &&
209  response->command->header != NULL &&
210  response->command->header->has_connectionID)
211  {
212  // Extract connectionID from unsolicited status message
213  connection->connectionID = response->command->header->connectionID;
214  LOGF2("Extracted connection ID from unsolicited status PDU (id=%lld)",
215  connection->connectionID);
216  connetionInfoReceived = true;
217  logTag = statusTag;
218  logAtLevel = 2;
219  protoLogAtLevel = 2;
220  }
221  else {
222  LOG0("WARNING: Unsolicited status received. Connection being terminated by remote!");
223  logTag = statusTag;
224  logAtLevel = 0;
225  protoLogAtLevel = 0;
226  }
227  }
228  else
229  {
230  KineticLogger_LogTimestamp(0, "WARNING: Received unexpected response!");
231  logTag = unexpectedTag;
232  logAtLevel = 0;
233  protoLogAtLevel = 0;
234  }
235 
236  KineticLogger_LogPrintf(logAtLevel, "%s pdu: %p, session: %p, bus: %p, "
237  "fd: %6d, seq: %8lld, protoLen: %8u, valueLen: %8u",
238  logTag,
239  (void*)response, (void*)connection->pSession,
240  (void*)connection->messageBus,
241  connection->socket, (long long)seq_id,
242  response->header.protobufLength, response->header.valueLength);
243  KineticLogger_LogProtobuf(protoLogAtLevel, response->proto);
244 
245  KineticAllocator_FreeKineticResponse(response);
246 
247  if (connetionInfoReceived) {
248  KineticResourceWaiter_SetAvailable(&connection->connectionReady);
249  }
250 }

252 void KineticController_HandleResult(bus_msg_result_t *res, void *udata)
253 {
254  KineticOperation* op = udata;
255  KINETIC_ASSERT(op);
256  KINETIC_ASSERT(op->connection);
257 
258  KineticStatus status = bus_to_kinetic_status(res->status);
259 
260  if (status == KINETIC_STATUS_SUCCESS) {
261  KineticResponse * response = res->u.response.opaque_msg;
262  KINETIC_ASSERT(response);
263  KINETIC_ASSERT(response->command);
264  KINETIC_ASSERT(response->command->header);
265 
266  if (response->command != NULL &&
267  response->command->status != NULL &&
268  response->command->status->has_code)
269  {
270  status = KineticProtoStatusCode_to_KineticStatus(response->command->status->code);
271  op->response = response;
272  }
273  else {
274  status = KINETIC_STATUS_INVALID;
275  }
276 
277  LOGF2("[PDU RX] pdu: %p, session: %p, bus: %p, "
278  "fd: %6d, seq: %8lld, protoLen: %8u, valueLen: %8u, op: %p, status: %s",
279  (void*)response,
280  (void*)op->connection->pSession, (void*)op->connection->messageBus,
281  op->connection->socket, response->command->header->ackSequence,
282  response->header.protobufLength, response->header.valueLength,
283  (void*)op,
284  Kinetic_GetStatusDescription(status));
285  KineticLogger_LogHeader(3, &response->header);
286  KineticLogger_LogProtobuf(3, response->proto);
287  }
288  else
289  {
290  // pull out bus error?
291  LOGF0("Error receiving response, got message bus error: %s", bus_error_string(res->status));
292  if (res->status == BUS_SEND_RX_TIMEOUT) {
293  KineticLogger_LogTimestamp(0, "RX_TIMEOUT");
294  }
295  }
296 
297  // Call operation-specific callback, if configured
298  if (op->callback != NULL) {
299  status = op->callback(op, status);
300  }
301 
302  KineticOperation_Complete(op, status);
303 }
