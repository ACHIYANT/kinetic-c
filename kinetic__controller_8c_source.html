95 KineticStatus KineticController_ExecuteOperation(KineticOperation* operation, KineticCompletionClosure* const closure)
96 {
97  assert(operation != NULL);
98  KineticStatus status = KINETIC_STATUS_INVALID;
99 
100  LOGF1("Executing operation: 0x%llX", operation);
101  if (operation->entry != NULL &&
102  operation->entry->value.array.data != NULL &&
103  operation->entry->value.bytesUsed > 0)
104  {
105  LOGF1(" Sending PDU (0x%0llX) w/value (%zu bytes)",
106  operation->request, operation->entry->value.bytesUsed);
107  }
108  else {
109  LOGF1(" Sending PDU (0x%0llX) w/o value", operation->request);
110  }
111 
112  if (closure != NULL)
113  {
114  operation->closure = *closure;
115  return KineticOperation_SendRequest(operation);
116  }
117  else
118  {
119  DefaultCallbackData data;
120  pthread_mutex_init(&data.receiveCompleteMutex, NULL);
121  pthread_cond_init(&data.receiveComplete, NULL);
122  data.status = KINETIC_STATUS_INVALID;
123 
124  operation->closure = DefaultClosure(&data);
125 
126  // Send the request
127  status = KineticOperation_SendRequest(operation);
128 
129  if (status == KINETIC_STATUS_SUCCESS) {
130  pthread_mutex_lock(&data.receiveCompleteMutex);
131  pthread_cond_wait(&data.receiveComplete, &data.receiveCompleteMutex);
132  pthread_mutex_unlock(&data.receiveCompleteMutex);
133  status = data.status;
134  }
135 
136  pthread_cond_destroy(&data.receiveComplete);
137  pthread_mutex_destroy(&data.receiveCompleteMutex);
138 
139  return status;
140  }
141 }

150 void KineticController_HandleIncomingPDU(KineticConnection* const connection)
151 {
152  KineticPDU* response = KineticAllocator_NewPDU(connection);
153  KineticStatus status = KineticPDU_ReceiveMain(response);
154  if (status != KINETIC_STATUS_SUCCESS) {
155  LOGF0("ERROR: PDU receive reported an error: %s",
156  Kinetic_GetStatusDescription(status));
157  }
158 
159  if (response->proto != NULL && response->proto->has_authType) {
160 
161  // Handle unsolicited status PDUs
162  if (response->proto->authType == KINETIC_PROTO_MESSAGE_AUTH_TYPE_UNSOLICITEDSTATUS) {
163  response->type = KINETIC_PDU_TYPE_UNSOLICITED;
164  if (response->command != NULL &&
165  response->command->header != NULL &&
166  response->command->header->has_connectionID)
167  {
168  // Extract connectionID from unsolicited status message
169  response->connection->connectionID = response->command->header->connectionID;
170  LOGF2("Extracted connection ID from unsolicited status PDU (id=%lld)",
171  response->connection->connectionID);
172  }
173  else {
174  LOG0("WARNING: Unsolicited PDU in invalid. Does not specify connection ID!");
175  }
176  KineticAllocator_FreePDU(connection, response);
177  }
178 
179  // Associate solicited response PDUs with their requests
180  else {
181  response->type = KINETIC_PDU_TYPE_RESPONSE;
182  KineticOperation* op = KineticOperation_AssociateResponseWithOperation(response);
183  if (op == NULL) {
184  LOG0("Failed to find request matching received response PDU!");
185  KineticAllocator_FreePDU(connection, response);
186  }
187  else {
188  LOG2("Found associated operation/request for response PDU.");
189  size_t valueLength = KineticPDU_GetValueLength(response);
190  if (valueLength > 0) {
191  // we need to try to read the value off the socket, even in the case of an error
192  // otherwise the stream will get out of sync
193  KineticStatus valueStatus = KineticPDU_ReceiveValue(op->connection->socket,
194  &op->entry->value, valueLength);
195 
196  // so we only care about the status of the value read if the operation
197  // succeeded
198  if (status == KINETIC_STATUS_SUCCESS) {
199  status = valueStatus;
200  }
201  }
202 
203  // Call operation-specific callback, if configured
204  if (status == KINETIC_STATUS_SUCCESS && op->callback != NULL) {
205  status = op->callback(op, KINETIC_STATUS_SUCCESS);
206  }
207 
208  if (status == KINETIC_STATUS_SUCCESS) {
209  status = KineticPDU_GetStatus(response);
210  }
211 
212  LOGF2("Response PDU received w/status %s, %i",
213  Kinetic_GetStatusDescription(status), status);
214 
215  KineticOperation_Complete(op, status);
216  }
217  }
218  }
219  else {
220  // Free invalid PDU
221  KineticAllocator_FreePDU(connection, response);
222  }
223 }

225 static KineticStatus service_controller(KineticConnection* const connection)
226 {
227  // Wait for and receive a PDU
228  KineticWaitStatus wait_status = KineticSocket_WaitUntilDataAvailable(connection->socket, 100);
229  switch(wait_status)
230  {
231  case KINETIC_WAIT_STATUS_DATA_AVAILABLE:
232  {
233  KineticController_HandleIncomingPDU(connection);
234  } break;
235  case KINETIC_WAIT_STATUS_TIMED_OUT:
236  case KINETIC_WAIT_STATUS_RETRYABLE_ERROR:
237  break; // Break upon rcoverable/retryable conditions in order to retry
238  default:
239  case KINETIC_WAIT_STATUS_FATAL_ERROR:
240  {
241  LOG0("ERROR: Socket error while waiting for PDU to arrive");
242  return KINETIC_STATUS_SOCKET_ERROR;
243  } break;
244  }
245 
246  KineticOperation_TimeoutOperations(connection);
247 
248  return KINETIC_STATUS_SUCCESS;
249 }

251 void* KineticController_ReceiveThread(void* thread_arg)
252 {
253  KineticThread* thread = thread_arg;
254 
255  while(!thread->abortRequested && !thread->fatalError) {
256 
257  // Do not service PDUs if thread paused
258  if (thread->paused) {
259  struct timespec sleepDuration = {.tv_nsec = 500000};
260  nanosleep(&sleepDuration, NULL);
261  continue;
262  }
263 
264  if (service_controller(thread->connection) != KINETIC_STATUS_SUCCESS) {
265  thread->fatalError = true;
266  }
267  }
268 
269  LOG3("Worker thread terminated!");
270  return (void*)NULL;
271 }
