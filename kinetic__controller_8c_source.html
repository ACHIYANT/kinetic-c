33 {

36  return NULL;

40  LOG0("Specified session is not associated with a connection");

43 
44  LOGF3("--------------------------------------------------\n"

46

50  }

62 static void DefaultCallback(KineticCompletionData* kinetic_data, void* client_data)
63 {
64  DefaultCallbackData * data = client_data;

80 KineticStatus KineticController_ExecuteOperation(KineticOperation* operation, KineticCompletionClosure* const closure)
81 {
82  assert(operation != NULL);
83  assert(operation->connection != NULL);
84  assert(operation->connection->pSession != NULL);
85  KineticStatus status = KINETIC_STATUS_INVALID;
86 
87  if (closure != NULL)
88  {
89  operation->closure = *closure;
90  return KineticOperation_SendRequest(operation);
91  }
92  else
93  {
94  DefaultCallbackData data;
95  pthread_mutex_init(&data.receiveCompleteMutex, NULL);
96  pthread_cond_init(&data.receiveComplete, NULL);
97  data.status = KINETIC_STATUS_INVALID;
98  data.completed = false;
99 
100  operation->closure = DefaultClosure(&data);
101 
102  // Send the request
103  status = KineticOperation_SendRequest(operation);
104 
105  if (status == KINETIC_STATUS_SUCCESS) {
106  pthread_mutex_lock(&data.receiveCompleteMutex);
107  while(data.completed == false)
108  { pthread_cond_wait(&data.receiveComplete, &data.receiveCompleteMutex); }
109  status = data.status;
110  pthread_mutex_unlock(&data.receiveCompleteMutex);
111  }
112 
113  pthread_cond_destroy(&data.receiveComplete);
114  pthread_mutex_destroy(&data.receiveCompleteMutex);
115 
116  return status;
117  }
118 }

120 KineticStatus bus_to_kinetic_status(bus_send_status_t const status)
121 {
122  KineticStatus res = KINETIC_STATUS_INVALID;
123 
124  switch(status)
125  {
126  // TODO scrutinize all these mappings
127  case BUS_SEND_SUCCESS:
128  res = KINETIC_STATUS_SUCCESS;
129  break;
130  case BUS_SEND_TX_TIMEOUT:
131  res = KINETIC_STATUS_SOCKET_TIMEOUT;
132  break;
133  case BUS_SEND_TX_FAILURE:
134  res = KINETIC_STATUS_SOCKET_ERROR;
135  break;
136  case BUS_SEND_RX_TIMEOUT:
137  res = KINETIC_STATUS_OPERATION_TIMEDOUT;
138  break;
139  case BUS_SEND_RX_FAILURE:
140  res = KINETIC_STATUS_SOCKET_ERROR;
141  break;
142  case BUS_SEND_BAD_RESPONSE:
143  res = KINETIC_STATUS_SOCKET_ERROR;
144  break;
145  case BUS_SEND_UNREGISTERED_SOCKET:
146  res = KINETIC_STATUS_SOCKET_ERROR;
147  break;
148  case BUS_SEND_UNDEFINED:
149  default:
150  {
151  LOGF0("bus_to_kinetic_status: UNMATCHED %d\n", status);
152  assert(false);
153  return KINETIC_STATUS_INVALID;
154  }
155  }
156 
157  LOGF3("bus_to_kinetic_status: mapping status %d => %d\n",
158  status, res);
159  return res;
160 }

182 void KineticController_HandleUnexecpectedResponse(void *msg,
183  int64_t seq_id,
184  void *bus_udata,
185  void *socket_udata)
186 {
187  KineticResponse * response = msg;
188  KineticConnection* connection = socket_udata;
189  bool connetionInfoReceived = false;
190 
191  (void)seq_id;
192  (void)bus_udata;
193 
194  LOGF2("[PDU RX UNSOLICITED] pdu: 0x%0llX, session: 0x%llX, bus: 0x%llX, "
195  "fd: %6d, protoLen: %u, valueLen: %u",
196  response, connection->pSession, connection->messageBus, connection->socket,
197  response->header.protobufLength, response->header.valueLength);
198 
199  // Handle unsolicited status PDUs
200  if (response->proto->authType == KINETIC_PROTO_MESSAGE_AUTH_TYPE_UNSOLICITEDSTATUS) {
201  if (response->command != NULL &&
202  response->command->header != NULL &&
203  response->command->header->has_connectionID)
204  {
205  // Extract connectionID from unsolicited status message
206  connection->connectionID = response->command->header->connectionID;
207  LOGF2("Extracted connection ID from unsolicited status PDU (id=%lld)",
208  connection->connectionID);
209  connetionInfoReceived = true;
210  }
211  else {
212  LOG0("WARNING: Unsolicited PDU received after session initialized!");
213  }
214  }
215  else
216  {
217  LOG0("WARNING: Received unexpected response that was not an unsolicited status.");
218  }
219 
220  KineticLogger_LogProtobuf(2, response->proto);
221 
222  KineticAllocator_FreeKineticResponse(response);
223 
224  if (connetionInfoReceived) {
225  KineticResourceWaiter_SetAvailable(&connection->connectionReady);
226  }
227 }

229 void KineticController_HandleExpectedResponse(bus_msg_result_t *res, void *udata)
230 {
231  KineticOperation* op = udata;
232 
233  KineticStatus status = bus_to_kinetic_status(res->status);
234 
235  if (status == KINETIC_STATUS_SUCCESS) {
236 
237  KineticResponse * response = res->u.response.opaque_msg;
238  if (response->command != NULL &&
239  response->command->status != NULL &&
240  response->command->status->has_code)
241  {
242  status = KineticProtoStatusCode_to_KineticStatus(response->command->status->code);
243  KineticLogger_LogProtobuf(3, response->proto);
244  op->response = response;
245  }
246  else
247  {
248  status = KINETIC_STATUS_INVALID;
249  }
250 
251  LOGF2("[PDU RX] pdu: 0x%0llX, op: 0x%llX, session: 0x%llX, bus: 0x%llX, fd: %6d, "
252  "seq: %5lld, protoLen: %4u, valueLen: %u, status: %s",
253  response, op, op->connection->pSession, op->connection->messageBus, op->connection->socket,
254  response->command->header->ackSequence, response->header.protobufLength, response->header.valueLength,
255  Kinetic_GetStatusDescription(status));
256  }
257  else
258  {
259  // pull out bus error?
260  LOGF1("Error receiving response, got message bus error: %s", bus_error_string(res->status));
261  }
262 
263  // Call operation-specific callback, if configured
264  if (op->callback != NULL) {
265  status = op->callback(op, status);
266  }
267 
268  KineticOperation_Complete(op, status);
269 }
