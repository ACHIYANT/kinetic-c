59 bool bus_init(bus_config *config, struct bus_result *res) {
60  if (res == NULL) { return false; }
61  if (config == NULL) {
62  res->status = BUS_INIT_ERROR_NULL;
63  return false;
64  }
65  set_defaults(config);
66  if (config->sink_cb == NULL) {
67  res->status = BUS_INIT_ERROR_MISSING_SINK_CB;
68  return false;
69  }
70  if (config->unpack_cb == NULL) {
71  res->status = BUS_INIT_ERROR_MISSING_UNPACK_CB;
72  return false;
73  }
74  if (config->log_cb == NULL) {
75  config->log_cb = noop_log_cb;
76  config->log_level = INT_MIN;
77  }
78  if (config->error_cb == NULL) {
79  config->error_cb = noop_error_cb;
80  }
81 
82  res->status = BUS_INIT_ERROR_ALLOC_FAIL;
83 
84  bool log_lock_init = false;
85  struct sender **ss = NULL; /* senders */
86  struct listener **ls = NULL; /* listeners */
87  struct threadpool *tp = NULL;
88  bool *joined = NULL;
89  pthread_t *threads = NULL;
90  struct yacht *fd_set = NULL;
91 
92  bus *b = calloc(1, sizeof(*b));
93  if (b == NULL) { goto cleanup; }
94 
95  if (!bus_ssl_init(b)) { goto cleanup; }
96 
97  b->sink_cb = config->sink_cb;
98  b->unpack_cb = config->unpack_cb;
99  b->unexpected_msg_cb = config->unexpected_msg_cb;
100  b->error_cb = config->error_cb;
101  b->log_cb = config->log_cb;
102  b->log_level = config->log_level;
103  b->udata = config->bus_udata;
104  if (0 != pthread_mutex_init(&b->log_lock, NULL)) {
105  res->status = BUS_INIT_ERROR_MUTEX_INIT_FAIL;
106  goto cleanup;
107  }
108  if (0 != pthread_mutex_init(&b->fd_set_lock, NULL)) {
109  res->status = BUS_INIT_ERROR_MUTEX_INIT_FAIL;
110  goto cleanup;
111  }
112 
113  log_lock_init = true;
114 
115  attempt_to_increase_resource_limits(b);
116 
117  BUS_LOG_SNPRINTF(b, 3, LOG_INITIALIZATION, b->udata, 64,
118  "Initialized bus at %p", (void*)b);
119 
120  ss = calloc(config->sender_count, sizeof(*ss));
121  if (ss == NULL) {
122  goto cleanup;
123  }
124 
125  for (int i = 0; i < config->sender_count; i++) {
126  ss[i] = sender_init(b, config);
127  if (ss[i] == NULL) {
128  res->status = BUS_INIT_ERROR_SENDER_INIT_FAIL;
129  goto cleanup;
130  } else {
131  BUS_LOG_SNPRINTF(b, 3, LOG_INITIALIZATION, b->udata, 64,
132  "Initialized sender %d at %p", i, (void*)ss[i]);
133  }
134  }
135 
136  ls = calloc(config->listener_count, sizeof(*ls));
137  if (ls == NULL) {
138  goto cleanup;
139  }
140 
141  for (int i = 0; i < config->listener_count; i++) {
142  ls[i] = listener_init(b, config);
143  if (ls[i] == NULL) {
144  res->status = BUS_INIT_ERROR_LISTENER_INIT_FAIL;
145  goto cleanup;
146  } else {
147  BUS_LOG_SNPRINTF(b, 3, LOG_INITIALIZATION, b->udata, 64,
148  "Initialized listener %d at %p", i, (void*)ls[i]);
149  }
150  }
151 
152  tp = threadpool_init(&config->threadpool_cfg);
153  if (tp == NULL) {
154  res->status = BUS_INIT_ERROR_THREADPOOL_INIT_FAIL;
155  goto cleanup;
156  }
157 
158  int thread_count = config->sender_count + config->listener_count;
159  joined = calloc(thread_count, sizeof(bool));
160  threads = calloc(thread_count, sizeof(pthread_t));
161  if (joined == NULL || threads == NULL) {
162  goto cleanup;
163  }
164 
165  fd_set = yacht_init(DEF_FD_SET_SIZE2);
166  if (fd_set == NULL) {
167  goto cleanup;
168  }
169 
170  b->sender_count = config->sender_count;
171  b->senders = ss;
172  b->listener_count = config->listener_count;
173  b->listeners = ls;
174  b->threadpool = tp;
175  b->joined = joined;
176  b->threads = threads;
177 
178  for (int i = 0; i < b->sender_count; i++) {
179  int pcres = pthread_create(&b->threads[i], NULL,
180  sender_mainloop, (void *)b->senders[i]);
181  if (pcres != 0) {
182  res->status = BUS_INIT_ERROR_PTHREAD_INIT_FAIL;
183  goto cleanup;
184  }
185  }
186 
187  for (int i = 0; i < b->listener_count; i++) {
188  int pcres = pthread_create(&b->threads[i + b->sender_count], NULL,
189  listener_mainloop, (void *)b->listeners[i]);
190  if (pcres != 0) {
191  res->status = BUS_INIT_ERROR_PTHREAD_INIT_FAIL;
192  goto cleanup;
193  }
194  }
195 
196  b->fd_set = fd_set;
197  res->bus = b;
198  BUS_LOG(b, 2, LOG_INITIALIZATION, "initialized", config->bus_udata);
199  return true;
200 
201 cleanup:
202  if (ss) {
203  for (int i = 0; i < config->sender_count; i++) {
204  if (ss[i]) { sender_free(ss[i]); }
205  }
206  free(ss);
207  }
208  if (ls) {
209  for (int i = 0; i < config->listener_count; i++) {
210  if (ls[i]) { listener_free(ls[i]); }
211  }
212  free(ls);
213  }
214  if (tp) { threadpool_free(tp); }
215  if (joined) { free(joined); }
216  if (b) {
217  if (log_lock_init) {
218  pthread_mutex_destroy(&b->fd_set_lock);
219  pthread_mutex_destroy(&b->log_lock);
220  }
221  free(b);
222  }
223 
224  if (threads) { free(threads); }
225  if (fd_set) { yacht_free(fd_set, NULL, NULL); }
226 
227  return false;
228 }

230 static bool attempt_to_increase_resource_limits(struct bus *b) {
231  struct rlimit info;
232  if (-1 == getrlimit(RLIMIT_NOFILE, &info)) {
233  fprintf(stderr, "getrlimit: %s", strerror(errno));
234  errno = 0;
235  return false;
236  }
237 
238  const unsigned int nval = 1024;
239 
240  if (info.rlim_cur < nval && info.rlim_max > nval) {
241  info.rlim_cur = nval;
242  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 256,
243  "Current FD resource limits, [%lu, %lu], changing to %u",
244  (unsigned long)info.rlim_cur, (unsigned long)info.rlim_max, nval);
245  if (-1 == setrlimit(RLIMIT_NOFILE, &info)) {
246  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 256,
247  "Failed to increase FD resource limit to %u, %s",
248  nval, strerror(errno));
249  fprintf(stderr, "getrlimit: %s", strerror(errno));
250  errno = 0;
251  return false;
252  } else {
253  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 256,
254  "Successfully increased FD resource limit to %u", nval);
255  }
256  } else {
257  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 256,
258  "Current FD resource limits [%lu, %lu] are acceptable",
259  (unsigned long)info.rlim_cur, (unsigned long)info.rlim_max);
260  }
261  return true;
262 }

264 /* Pack message to deliver on behalf of the user into an envelope
265  * that can track status / routing along the way.
266  *
267  * The box should only ever be accessible on a single thread at a time. */
268 static boxed_msg *box_msg(struct bus *b, bus_user_msg *msg) {
269  boxed_msg *box = calloc(1, sizeof(*box));
270  if (box == NULL) { return NULL; }
271 
272  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 64,
273  "Allocated boxed message -- %p", (void*)box);
274 
275  box->fd = msg->fd;
276  assert(msg->fd != 0);
277 
278  /* Lock hash table and check whether this FD uses SSL. */
279  if (0 != pthread_mutex_lock(&b->fd_set_lock)) { assert(false); }
280  void *value = NULL;
281  SSL *ssl = NULL;
282  if (yacht_get(b->fd_set, box->fd, &value)) {
283  ssl = (SSL *)value;
284  assert(ssl != NULL);
285  box->ssl = ssl;
286  }
287  if (0 != pthread_mutex_unlock(&b->fd_set_lock)) { assert(false); }
288 
289  if (ssl == NULL) {
290  /* socket isn't registered, fail out */
291  free(box);
292  return NULL;
293  }
294 
295  box->timeout_sec = (time_t)msg->timeout_sec;
296  if (box->timeout_sec == 0) {
297  box->timeout_sec = BUS_DEFAULT_TIMEOUT_SEC;
298  }
299 
300  box->out_seq_id = msg->seq_id;
301  box->out_msg_size = msg->msg_size;
302 
303  /* Store message by pointer, since the client thread using it is blocked
304  * until we are done sending. */
305  box->out_msg = msg->msg;
306 
307  box->cb = msg->cb;
308  box->udata = msg->udata;
309  return box;
310 }

334 static bool poll_on_completion(struct bus *b, int fd) {
335  /* POLL in a pipe */
336  struct pollfd fds[1];
337  fds[0].fd = fd;
338  fds[0].events = POLLIN;
339 
340  /* TODO REFACTOR this should be reused between bus, sender, and listener,
341  * or be moved into the listener.
342  * The sender has its own blocking polling for commands. */
343 
344  /* FIXME: compare this to TCP timeouts -- try to prevent sender
345  * succeeding but failing to notify client. */
346  const int TIMEOUT_SECONDS = 10;
347  const int ONE_SECOND = 1000; // msec
348 
349  for (int i = 0; i < TIMEOUT_SECONDS; i++) {
350  BUS_LOG(b, 5, LOG_SENDING_REQUEST, "Polling on completion...tick...", b->udata);
351  int res = poll(fds, 1, ONE_SECOND);
352  if (res == -1) {
353  if (util_is_resumable_io_error(errno)) {
354  BUS_LOG(b, 3, LOG_SENDING_REQUEST, "Polling on completion...EAGAIN", b->udata);
355  if (errno == EINTR && i > 0) { i--; }
356  errno = 0;
357  } else {
358  assert(false);
359  break;
360  }
361  } else if (res > 0) {
362  uint16_t msec = 0;
363  uint8_t read_buf[sizeof(msec)];
364 
365  if (fds[0].revents & (POLLERR | POLLHUP | POLLNVAL)) {
366  BUS_LOG(b, 1, LOG_SENDING_REQUEST, "failed (broken alert pipe)", b->udata);
367  return false;
368  }
369 
370  BUS_LOG(b, 3, LOG_SENDING_REQUEST, "Reading alert pipe...", b->udata);
371  ssize_t sz = read(fd, read_buf, sizeof(read_buf));
372 
373  if (sz == sizeof(read_buf)) {
374  /* Payload: little-endian uint16_t, msec of backpressure. */
375  msec = (read_buf[0] << 0) + (read_buf[1] << 8);
376  if (msec > 0) {
377  BUS_LOG_SNPRINTF(b, 5, LOG_SENDING_REQUEST, b->udata, 64,
378  " -- awakening client thread with backpressure of %d msec", msec);
379  (void)poll(NULL, 0, msec);
380  }
381 
382  BUS_LOG(b, 3, LOG_SENDING_REQUEST, "sent!", b->udata);
383  return true;
384  } else if (sz == -1) {
385  if (util_is_resumable_io_error(errno)) {
386  errno = 0;
387  } else {
388  assert(false);
389  break;
390  }
391  }
392  }
393  }
394  BUS_LOG(b, 2, LOG_SENDING_REQUEST, "failed (timeout)", b->udata);
395 
396  #if 0
397  assert(false);
398  #endif
399 
400  return false;
401 }

438 bool bus_register_socket(struct bus *b, bus_socket_t type, int fd, void *udata) {
439  /* Register a socket internally with a sender and listener. */
440  int s_id = sender_id_of_socket(b, fd);
441  int l_id = listener_id_of_socket(b, fd);
442 
443  BUS_LOG_SNPRINTF(b, 2, LOG_SOCKET_REGISTERED, b->udata, 64,
444  "registering socket %d", fd);
445 
446  /* Spread sockets throughout the different sender & listener processes. */
447  struct sender *s = b->senders[s_id];
448  struct listener *l = b->listeners[l_id];
449 
450  int pipes[2];
451  if (0 != pipe(pipes)) {
452  BUS_LOG(b, 1, LOG_SOCKET_REGISTERED, "pipe failure", b->udata);
453  return false;
454  }
455 
456  int pipe_out = pipes[0];
457  int pipe_in = pipes[1];
458 
459  connection_info *ci = malloc(sizeof(*ci));
460  if (ci == NULL) { goto cleanup; }
461 
462  ci->type = type;
463  ci->fd = fd;
464  ci->to_read_size = 0;
465  ci->udata = udata;
466  ci->largest_seq_id_seen = 0;
467 
468  if (type == BUS_SOCKET_SSL) {
469  if (!bus_ssl_connect(b, ci)) { goto cleanup; }
470  } else {
471  ci->ssl = BUS_NO_SSL;
472  }
473 
474  void *old_value = NULL;
475 
476  /* Lock hash table and save whether this FD uses SSL. */
477  if (0 != pthread_mutex_lock(&b->fd_set_lock)) { assert(false); }
478  bool set_ok = yacht_set(b->fd_set, fd, (void *)ci->ssl, &old_value);
479  if (0 != pthread_mutex_unlock(&b->fd_set_lock)) { assert(false); }
480 
481  if (set_ok) {
482  assert(old_value == NULL);
483  } else {
484  goto cleanup;
485  }
486 
487  bool res = false;
488  res = sender_register_socket(s, fd, ci->ssl);
489  if (!res) { goto cleanup; }
490 
491  res = listener_add_socket(l, ci, pipe_in);
492  if (!res) { goto cleanup; }
493 
494  /* FIXME: Move this into listener_add_socket? */
495  BUS_LOG(b, 2, LOG_SOCKET_REGISTERED, "polling on socket add...", b->udata);
496  bool completed = poll_on_completion(b, pipe_out);
497  if (!completed) { goto cleanup; }
498 
499  close(pipe_out);
500  close(pipe_in);
501 
502  BUS_LOG(b, 2, LOG_SOCKET_REGISTERED, "successfully added socket", b->udata);
503  return true;
504 cleanup:
505  if (ci) {
506  free(ci);
507  }
508  close(pipe_out);
509  close(pipe_in);
510  BUS_LOG(b, 2, LOG_SOCKET_REGISTERED, "failed to add socket", b->udata);
511  return false;
512 }

514 /* Free metadata about a socket that has been disconnected. */
515 bool bus_release_socket(struct bus *b, int fd) {
516  int s_id = sender_id_of_socket(b, fd);
517  int l_id = listener_id_of_socket(b, fd);
518 
519  BUS_LOG_SNPRINTF(b, 2, LOG_SOCKET_REGISTERED, b->udata, 64,
520  "forgetting socket %d", fd);
521 
522  struct sender *s = b->senders[s_id];
523  struct listener *l = b->listeners[l_id];
524 
525  if (!sender_remove_socket(s, fd)) {
526  return false;
527  }
528 
529  if (!listener_remove_socket(l, fd)) {
530  return false; /* couldn't send msg to listener */
531  }
532 
533  /* Lock hash table and forget whether this FD uses SSL. */
534  void *old_value = NULL;
535  if (0 != pthread_mutex_lock(&b->fd_set_lock)) { assert(false); }
536  bool rm_ok = yacht_remove(b->fd_set, fd, &old_value);
537  if (0 != pthread_mutex_unlock(&b->fd_set_lock)) { assert(false); }
538  assert(rm_ok);
539 
540  SSL *ssl = (SSL *)old_value;
541  assert(ssl != NULL);
542 
543  if (ssl == BUS_NO_SSL) {
544  return true; /* nothing else to do */
545  } else {
546  return bus_ssl_disconnect(b, ssl);
547  }
548 }

555 bool bus_shutdown(bus *b) {
556  BUS_LOG(b, 2, LOG_SHUTDOWN, "shutting down sender threads", b->udata);
557  for (int i = 0; i < b->sender_count; i++) {
558  int off = 0;
559  if (!b->joined[i + off]) {
560  BUS_LOG_SNPRINTF(b, 3, LOG_SHUTDOWN, b->udata, 128,
561  "sender_shutdown -- %d", i);
562  while (!sender_shutdown(b->senders[i])) {
563  BUS_LOG_SNPRINTF(b, 3, LOG_SHUTDOWN, b->udata, 128,
564  "sender_shutdown -- retry %d", i);
565  sleep(1);
566  }
567  void *unused = NULL;
568  int res = pthread_join(b->threads[i + off], &unused);
569  BUS_LOG_SNPRINTF(b, 3, LOG_SHUTDOWN, b->udata, 128,
570  "sender_shutdown -- joined %d", i);
571  assert(res == 0);
572  b->joined[i + off] = true;
573  }
574  }
575 
576  BUS_LOG(b, 2, LOG_SHUTDOWN, "shutting down listener threads", b->udata);
577  for (int i = 0; i < b->listener_count; i++) {
578  int off = b->sender_count;
579  if (!b->joined[i + off]) {
580  BUS_LOG_SNPRINTF(b, 3, LOG_SHUTDOWN, b->udata, 128,
581  "listener_shutdown -- %d", i);
582  while (!listener_shutdown(b->listeners[i])) {
583  sleep(1);
584  }
585  BUS_LOG_SNPRINTF(b, 3, LOG_SHUTDOWN, b->udata, 128,
586  "listener_shutdown -- joining %d", i);
587  void *unused = NULL;
588  int res = pthread_join(b->threads[i + off], &unused);
589  BUS_LOG_SNPRINTF(b, 3, LOG_SHUTDOWN, b->udata, 128,
590  "listener_shutdown -- joined %d", i);
591  assert(res == 0);
592  b->joined[i + off] = true;
593  }
594  }
595 
596  BUS_LOG(b, 2, LOG_SHUTDOWN, "done with shutdown", b->udata);
597  return true;
598 }

647  bus_shutdown(b);

650  BUS_LOG_SNPRINTF(b, 3, LOG_SHUTDOWN, b->udata, 128,
651  "sender_free -- %d", i);
652  sender_free(b->senders[i]);
653  }
654  free(b->senders);
655 
656  for (int i = 0; i < b->listener_count; i++) {
657  BUS_LOG_SNPRINTF(b, 3, LOG_SHUTDOWN, b->udata, 128,
658  "listener_free -- %d", i);
659  listener_free(b->listeners[i]);
660  }
661  free(b->listeners);
662

664  for (int i = 0; i < limit; i++) {
665  BUS_LOG_SNPRINTF(b, 3, LOG_SHUTDOWN, b->udata, 128,
666  "threadpool_shutdown -- %d", i);
667  if (threadpool_shutdown(b->threadpool, false)) { break; }
668  (void)poll(NULL, 0, 10);
669 
670  if (i == limit - 1) {
671  BUS_LOG_SNPRINTF(b, 3, LOG_SHUTDOWN, b->udata, 128,
672  "threadpool_shutdown -- %d (forced)", i);
673  threadpool_shutdown(b->threadpool, true);
674  }
675  }
676  BUS_LOG(b, 3, LOG_SHUTDOWN, "threadpool_free", b->udata);
677  threadpool_free(b->threadpool);
678 
679  free(b->joined);
680  free(b->threads);
681  yacht_free(b->fd_set, NULL, NULL);
682
