39 struct listener *listener_init(struct bus *b, struct bus_config *cfg) {
40  struct listener *l = calloc(1, sizeof(*l));
41  if (l == NULL) { return NULL; }
42 
43  assert(b);
44  l->bus = b;
45  BUS_LOG(b, 2, LOG_LISTENER, "init", b->udata);
46 
47  int pipes[2];
48  if (0 != pipe(pipes)) {
49  free(l);
50  return NULL;
51  }
52 
53  l->commit_pipe = pipes[1];
54  l->incoming_msg_pipe = pipes[0];
55  l->fds[INCOMING_MSG_PIPE_ID].fd = l->incoming_msg_pipe;
56  l->fds[INCOMING_MSG_PIPE_ID].events = POLLIN;
57  l->shutdown_notify_fd = LISTENER_NO_FD;
58 
59  for (int i = 0; i < MAX_PENDING_MESSAGES; i++) {
60  rx_info_t *info = &l->rx_info[i];
61  info->state = RIS_INACTIVE;
62 
63  uint16_t *p_id = (uint16_t *)&info->id;
64  if (i < MAX_PENDING_MESSAGES - 1) {
65  info->next = &l->rx_info[i + 1];
66  }
67  *p_id = i;
68  }
69  l->rx_info_freelist = &l->rx_info[0];
70 
71  for (int pipe_count = 0; pipe_count < MAX_QUEUE_MESSAGES; pipe_count++) {
72  listener_msg *msg = &l->msgs[pipe_count];
73  uint8_t *p_id = (uint8_t *)&msg->id;
74  *p_id = pipe_count; /* Set (const) ID. */
75 
76  if (0 != pipe(msg->pipes)) {
77  for (int i = 0; i < pipe_count; i++) {
78  listener_msg *msg = &l->msgs[i];
79  close(msg->pipes[0]);
80  close(msg->pipes[1]);
81  }
82  close(l->commit_pipe);
83  close(l->incoming_msg_pipe);
84  free(l);
85  return NULL;
86  }
87 
88  if (pipe_count < MAX_QUEUE_MESSAGES - 1) { /* forward link */
89  msg->next = &l->msgs[pipe_count + 1];
90  }
91  }
92  l->msg_freelist = &l->msgs[0];
93  l->rx_info_max_used = 0;
94 
95  (void)cfg;
96  return l;
97 }

182 void listener_free(struct listener *l) {
183  struct bus *b = l->bus;
184  /* Thread has joined but data has not been freed yet. */
185  assert(l->shutdown_notify_fd == LISTENER_SHUTDOWN_COMPLETE_FD);
186 
187  if (l) {
188  for (int i = 0; i < MAX_PENDING_MESSAGES; i++) {
189  rx_info_t *info = &l->rx_info[i];
190 
191  switch (info->state) {
192  case RIS_INACTIVE:
193  break;
194  case RIS_HOLD:
195  break;
196  case RIS_EXPECT:
197  if (info->u.expect.box) {
198  /* TODO: This can leak memory, since the caller's
199  * callback is not being called. It should be called
200  * with BUS_SEND_RX_FAILURE, if it's safe to do so. */
201  free(info->u.expect.box);
202  info->u.expect.box = NULL;
203  }
204  break;
205  default:
206  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
207  "match fail %d on line %d", info->state, __LINE__);
208  BUS_ASSERT(b, b->udata, false);
209  }
210  }
211 
212  for (int i = 0; i < MAX_QUEUE_MESSAGES; i++) {
213  listener_msg *msg = &l->msgs[i];
214  switch (msg->type) {
215  case MSG_ADD_SOCKET:
216  ListenerCmd_NotifyCaller(msg->u.add_socket.notify_fd);
217  break;
218  case MSG_REMOVE_SOCKET:
219  ListenerCmd_NotifyCaller(msg->u.remove_socket.notify_fd);
220  break;
221  case MSG_EXPECT_RESPONSE:
222  if (msg->u.expect.box) { free(msg->u.expect.box); }
223  break;
224  default:
225  break;
226  }
227 
228  close(msg->pipes[0]);
229  close(msg->pipes[1]);
230  }
231 
232  if (l->read_buf) {
233  free(l->read_buf);
234  }
235 
236  close(l->commit_pipe);
237  close(l->incoming_msg_pipe);
238 
239  free(l);
240  }
241 }

243 static bool push_message(struct listener *l, listener_msg *msg, int *reply_fd) {
244  struct bus *b = l->bus;
245  BUS_ASSERT(b, b->udata, msg);
246 
247  uint8_t msg_buf[sizeof(msg->id)];
248  msg_buf[0] = msg->id;
249 
250  if (reply_fd) { *reply_fd = msg->pipes[0]; }
251 
252  for (;;) {
253  ssize_t wr = write(l->commit_pipe, msg_buf, sizeof(msg_buf));
254  if (wr == sizeof(msg_buf)) {
255  return true; // committed
256  } else {
257  if (errno == EINTR) { /* signal interrupted; retry */
258  errno = 0;
259  } else {
260  BUS_LOG_SNPRINTF(b, 10, LOG_LISTENER, b->udata, 64,
261  "write_commit error, errno %d", errno);
262  errno = 0;
263  ListenerTask_ReleaseMsg(l, msg);
264  return false;
265  }
266  }
267  }
268 }

270 static listener_msg *get_free_msg(listener *l) {
271  struct bus *b = l->bus;
272 
273  BUS_LOG_SNPRINTF(b, 4, LOG_LISTENER, b->udata, 128,
274  "get_free_msg -- in use: %d", l->msgs_in_use);
275 
276  for (;;) {
277  listener_msg *head = l->msg_freelist;
278  if (head == NULL) {
279  BUS_LOG(b, 3, LOG_LISTENER, "No free messages!", b->udata);
280  return NULL;
281  } else if (ATOMIC_BOOL_COMPARE_AND_SWAP(&l->msg_freelist, head, head->next)) {
282  for (;;) {
283  int16_t miu = l->msgs_in_use;
284 
285  if (ATOMIC_BOOL_COMPARE_AND_SWAP(&l->msgs_in_use, miu, miu + 1)) {
286  BUS_LOG(l->bus, 5, LOG_LISTENER, "got free msg", l->bus->udata);
287 
288  /* Add counterpressure between the client and the listener.
289  * 10 * ((n >> 1) ** 2) microseconds */
290  int16_t delay = 10 * (miu >> 1) * (miu >> 1);
291  if (delay > 0) {
292  struct timespec ts = {
293  .tv_sec = 0,
294  .tv_nsec = 1000L * delay,
295  };
296  nanosleep(&ts, NULL);
297  }
298  BUS_ASSERT(b, b->udata, head->type == MSG_NONE);
299  memset(&head->u, 0, sizeof(head->u));
300  return head;
301  }
302  }
303  }
304  }
