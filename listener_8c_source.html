44  struct listener *l = calloc(1, sizeof(*l));
45  if (l == NULL) { return NULL; }
46 
47  l->bus = b;
48  BUS_LOG(b, 2, LOG_LISTENER, "init", b->udata);
49 
50  int pipes[2];
51  if (0 != pipe(pipes)) {
52  free(l);
53  return NULL;
54  }
55 
56  l->commit_pipe = pipes[1];
57  l->incoming_msg_pipe = pipes[0];
58  l->fds[INCOMING_MSG_PIPE_ID].fd = l->incoming_msg_pipe;
59  l->fds[INCOMING_MSG_PIPE_ID].events = POLLIN;
60 
61  for (int i = 0; i < MAX_PENDING_MESSAGES; i++) {
62  rx_info_t *info = &l->rx_info[i];
63  info->state = RIS_INACTIVE;
64 
65  uint16_t *p_id = (uint16_t *)&info->id;
66  if (i < MAX_PENDING_MESSAGES - 1) {
67  info->next = &l->rx_info[i + 1];
68  }
69  *p_id = i;
70  }
71  l->rx_info_freelist = &l->rx_info[0];
72 
73  for (int i = 0; i < MAX_QUEUE_MESSAGES; i++) {
74  listener_msg *msg = &l->msgs[i];
75  uint8_t *p_id = (uint8_t *)&msg->id;
76  *p_id = i;
77  if (i < MAX_QUEUE_MESSAGES - 1) { /* forward link */
78  msg->next = &l->msgs[i + 1];
79  }
80  }
81  l->msg_freelist = &l->msgs[0];
82  l->rx_info_max_used = 0;
83 
84  (void)cfg;
85  return l;
86 }

119 static uint16_t get_backpressure(struct listener *l) {
120  uint16_t msg_fill_pressure = 0;
121 
122  if (l->msgs_in_use < 0.25 * MAX_QUEUE_MESSAGES) {
123  msg_fill_pressure = 0;
124  } else if (l->msgs_in_use < 0.5 * MAX_QUEUE_MESSAGES) {
125  msg_fill_pressure = MSG_BP_1QTR * 2 * l->msgs_in_use;
126  } else if (l->msgs_in_use < 0.75 * MAX_QUEUE_MESSAGES) {
127  msg_fill_pressure = MSG_BP_HALF * 10 * l->msgs_in_use;
128  } else {
129  msg_fill_pressure = MSG_BP_3QTR * 100 * l->msgs_in_use;
130  }
131 
132  uint16_t rx_info_fill_pressure = 0;
133  if (l->rx_info_in_use < 0.25 * MAX_PENDING_MESSAGES) {
134  rx_info_fill_pressure = 0;
135  } else if (l->rx_info_in_use < 0.5 * MAX_PENDING_MESSAGES) {
136  rx_info_fill_pressure = RX_INFO_BP_1QTR * l->rx_info_in_use;
137  } else if (l->rx_info_in_use < 0.75 * MAX_PENDING_MESSAGES) {
138  rx_info_fill_pressure = RX_INFO_BP_HALF * l->rx_info_in_use;
139  } else {
140  rx_info_fill_pressure = RX_INFO_BP_3QTR * l->rx_info_in_use;
141  }
142 
143  uint16_t threadpool_fill_pressure = THREADPOOL_BP * l->upstream_backpressure;
144 
145  struct bus *b = l->bus;
146  BUS_LOG_SNPRINTF(b, 6, LOG_SENDER, b->udata, 64,
147  "lbp: %u, %u (iu %u), %u",
148  msg_fill_pressure, rx_info_fill_pressure, l->rx_info_in_use, threadpool_fill_pressure);
149 
150  return msg_fill_pressure + rx_info_fill_pressure
151  + threadpool_fill_pressure;
152 }

220 void listener_free(struct listener *l) {
221  struct bus *b = l->bus;
222  /* assert: pthread must be join'd. */
223  if (l) {
224  for (int i = 0; i < MAX_PENDING_MESSAGES; i++) {
225  rx_info_t *info = &l->rx_info[i];
226 
227  switch (info->state) {
228  case RIS_INACTIVE:
229  break;
230  case RIS_HOLD:
231  break;
232  case RIS_EXPECT:
233  if (info->u.expect.box) {
234  free(info->u.expect.box);
235  info->u.expect.box = NULL;
236  }
237  break;
238  default:
239  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
240  "match fail %d on line %d", info->state, __LINE__);
241  BUS_ASSERT(b, b->udata, false);
242  }
243  }
244 
245  for (int i = 0; i < MAX_QUEUE_MESSAGES; i++) {
246  listener_msg *msg = &l->msgs[i];
247  switch (msg->type) {
248  case MSG_ADD_SOCKET:
249  if (msg->u.add_socket.info) { free_ci(msg->u.add_socket.info); }
250  break;
251  case MSG_EXPECT_RESPONSE:
252  if (msg->u.expect.box) { free(msg->u.expect.box); }
253  break;
254  default:
255  break;
256  }
257  }
258 
259  while (l->tracked_fds > 0) {
260  /* Remove off the front to stress remove_socket. */
261  remove_socket(l, l->fds[0 + INCOMING_MSG_PIPE].fd);
262  }
263 
264  if (l->read_buf) {
265  free(l->read_buf);
266  }
267 
268  close(l->commit_pipe);
269  close(l->incoming_msg_pipe);
270 
271  free(l);
272  }
273 }

280 void *listener_mainloop(void *arg) {
281  listener *self = (listener *)arg;
282  assert(self);
283  struct bus *b = self->bus;
284  struct timeval tv;
285 
286  gettimeofday(&tv, NULL);
287  time_t last_sec = tv.tv_sec;
288 
289  /* The listener thread has full control over its execution -- the
290  * only thing other threads can do is reserve messages from l->msgs,
291  * write commands into them, and then commit them by writing their
292  * msg->id into the incoming command ID pipe. All cross-thread
293  * communication is managed at the command interface, so it doesn't
294  * need any internal locking. */
295 
296  while (!self->shutdown) {
297  gettimeofday(&tv, NULL); // TODO: clock_gettime
298  time_t cur_sec = tv.tv_sec;
299  if (cur_sec != last_sec) {
300  tick_handler(self);
301  last_sec = cur_sec;
302  }
303 
304  int delay = (self->is_idle ? -1 : TIMEOUT_DELAY);
305  int res = poll(self->fds, self->tracked_fds + INCOMING_MSG_PIPE, delay);
306  BUS_LOG_SNPRINTF(b, (res == 0 ? 6 : 4), LOG_LISTENER, b->udata, 64,
307  "poll res %d", res);
308 
309  if (res < 0) {
310  if (util_is_resumable_io_error(errno)) {
311  errno = 0;
312  } else {
313  /* unrecoverable poll error -- FD count is bad
314  * or FDS is a bad pointer. */
315  BUS_ASSERT(b, b->udata, false);
316  }
317  } else if (res > 0) {
318  check_and_flush_incoming_msg_pipe(self, &res);
319  attempt_recv(self, res);
320  } else {
321  /* nothing to do */
322  }
323  }
324 
325  BUS_LOG(b, 3, LOG_LISTENER, "shutting down", b->udata);
326  return NULL;
327 }

329 static void check_and_flush_incoming_msg_pipe(listener *l, int *res) {
330  struct bus *b = l->bus;
331  short events = l->fds[INCOMING_MSG_PIPE_ID].revents;
332  if (events & (POLLERR | POLLHUP | POLLNVAL)) { /* hangup/error */
333  return;
334  }
335 
336  if (events & POLLIN) {
337  char buf[64];
338  for (;;) {
339  ssize_t rd = read(l->fds[INCOMING_MSG_PIPE_ID].fd, buf, sizeof(buf));
340  if (rd == -1) {
341  if (errno == EINTR) {
342  errno = 0;
343  continue;
344  } else {
345  BUS_LOG_SNPRINTF(b, 6, LOG_LISTENER, b->udata, 128,
346  "check_and_flush_incoming_msg_pipe: %s", strerror(errno));
347  errno = 0;
348  break;
349  }
350  } else {
351  for (ssize_t i = 0; i < rd; i++) {
352  uint8_t msg_id = buf[i];
353  listener_msg *msg = &l->msgs[msg_id];
354  msg_handler(l, msg);
355  }
356  (*res)--;
357  break;
358  }
359  }
360  }

363 static void set_error_for_socket(listener *l, int id, int fd, rx_error_t err) {
364  /* Mark all pending messages on this socket as being failed due to error. */
365  struct bus *b = l->bus;
366  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
367  "set_error_for_socket %d, err %d", fd, err);
368 
369  for (int i = 0; i <= l->rx_info_max_used; i++) {
370  rx_info_t *info = &l->rx_info[i];
371  switch (info->state) {
372  case RIS_INACTIVE:
373  break;
374  case RIS_HOLD:
375  release_rx_info(l, info);
376  break;
377  case RIS_EXPECT:
378  {
379  struct boxed_msg *box = info->u.expect.box;
380  if (box && box->fd == fd) {
381  info->u.expect.error = err;
382  }
383  break;
384  }
385  default:
386  {
387  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
388  "match fail %d on line %d", info->state, __LINE__);
389  BUS_ASSERT(b, b->udata, false);
390  }
391  }
392  }
393  l->fds[id + INCOMING_MSG_PIPE].events &= ~POLLIN;
394 }

411 static void attempt_recv(listener *l, int available) {
412  /* --> failure --> set 'closed' error on socket, don't die */
413  struct bus *b = l->bus;
414  int read_from = 0;
415  BUS_LOG(b, 3, LOG_LISTENER, "attempting receive", b->udata);
416 
417  for (int i = 0; i < l->tracked_fds; i++) {
418  if (read_from == available) { break; }
419  struct pollfd *fd = &l->fds[i + INCOMING_MSG_PIPE];
420  connection_info *ci = l->fd_info[i];
421  BUS_ASSERT(b, b->udata, ci->fd == fd->fd);
422 
423  if (fd->revents & (POLLERR | POLLNVAL)) {
424  read_from++;
425  BUS_LOG(b, 2, LOG_LISTENER,
426  "pollfd: socket error (POLLERR | POLLNVAL)", b->udata);
427  set_error_for_socket(l, i, ci->fd, RX_ERROR_POLLERR);
428  } else if (fd->revents & POLLHUP) {
429  read_from++;
430  BUS_LOG(b, 3, LOG_LISTENER, "pollfd: socket error POLLHUP",
431  b->udata);
432  set_error_for_socket(l, i, ci->fd, RX_ERROR_POLLHUP);
433  } else if (fd->revents & POLLIN) {
434  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
435  "reading %zd bytes from socket (buf is %zd)",
436  ci->to_read_size, l->read_buf_size);
437  BUS_ASSERT(b, b->udata, l->read_buf_size >= ci->to_read_size);
438  read_from++;
439 
440  switch (ci->type) {
441  case BUS_SOCKET_PLAIN:
442  socket_read_plain(b, l, i, ci);
443  break;
444  case BUS_SOCKET_SSL:
445  socket_read_ssl(b, l, i, ci);
446  break;
447  default:
448  BUS_ASSERT(b, b->udata, false);
449  }
450  }
451  }
452 }

477 static bool socket_read_ssl(struct bus *b, listener *l, int pfd_i, connection_info *ci) {
478  BUS_ASSERT(b, b->udata, ci->ssl);
479  for (;;) {
480  // ssize_t pending = SSL_pending(ci->ssl);
481  ssize_t size = (ssize_t)SSL_read(ci->ssl, l->read_buf, ci->to_read_size);
482  // fprintf(stderr, "=== PENDING: %zd, got %zd ===\n", pending, size);
483 
484  if (size == -1) {
485  int reason = SSL_get_error(ci->ssl, size);
486  switch (reason) {
487  case SSL_ERROR_WANT_READ:
488  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
489  "SSL_read fd %d: WANT_READ\n", ci->fd);
490  return true;
491 
492  case SSL_ERROR_WANT_WRITE:
493  BUS_ASSERT(b, b->udata, false);
494 
495  case SSL_ERROR_SYSCALL:
496  {
497  if (errno == 0) {
498  print_SSL_error(b, ci, 1, "SSL_ERROR_SYSCALL errno 0");
499  BUS_ASSERT(b, b->udata, false);
500  } else if (util_is_resumable_io_error(errno)) {
501  errno = 0;
502  } else {
503  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
504  "SSL_read fd %d: errno %d\n", ci->fd, errno);
505  print_SSL_error(b, ci, 1, "SSL_ERROR_SYSCALL");
506  set_error_for_socket(l, pfd_i, ci->fd, RX_ERROR_READ_FAILURE);
507  }
508  break;
509  }
510  case SSL_ERROR_ZERO_RETURN:
511  {
512  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
513  "SSL_read fd %d: ZERO_RETURN (HUP)\n", ci->fd);
514  set_error_for_socket(l, pfd_i, ci->fd, RX_ERROR_POLLHUP);
515  break;
516  }
517 
518  default:
519  print_SSL_error(b, ci, 1, "SSL_ERROR UNKNOWN");
520  set_error_for_socket(l, pfd_i, ci->fd, RX_ERROR_READ_FAILURE);
521  BUS_ASSERT(b, b->udata, false);
522  }
523  } else if (size > 0) {
524  sink_socket_read(b, l, ci, size);
525  }
526  }
527  return true;
528 }

584  case RIS_HOLD:
585  BUS_LOG_SNPRINTF(b, 4, LOG_MEMORY, b->udata, 128,
586  "find_info_by_sequence_id: info (%p) at +%d: <fd:%d, seq_id:%lld>",
587  (void*)info, info->id, fd, (long long)seq_id);
588  if (info->u.hold.fd == fd && info->u.hold.seq_id == seq_id) {
589  return info;
590  }
591  break;
592  case RIS_EXPECT:
593  {
594  struct boxed_msg *box = info->u.expect.box;
595  BUS_LOG_SNPRINTF(b, 4, LOG_MEMORY, b->udata, 128,
596  "find_info_by_sequence_id: info (%p) at +%d [s %d]: box is %p",
597  (void*)info, info->id, info->u.expect.error, (void*)box);
598  if (box != NULL && box->out_seq_id == seq_id && box->fd == fd) {
599  return info;
600  }
601  break;
602  }
603  default:
604  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
605  "match fail %d on line %d", info->state, __LINE__);
606  BUS_ASSERT(b, b->udata, false);
607  }
608  }
609 
610  if (b->log_level > 5 || 0) {
611  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
612  "==== Could not find <fd:%d, seq_id:%lld>, dumping table ====\n",
613  fd, (long long)seq_id);
614  dump_rx_info_table(l);
615  }
616  /* Not found. Probably an unsolicited status message. */
617  return NULL;
618 }

620 static void process_unpacked_message(listener *l,
621  connection_info *ci, bus_unpack_cb_res_t result) {
622  struct bus *b = l->bus;
623 
624  /* NOTE: message may be an unsolicited status message */
625 
626  if (result.ok) {
627  int64_t seq_id = result.u.success.seq_id;
628  void *opaque_msg = result.u.success.msg;
629 
630  if (seq_id < ci->largest_seq_id_seen && ci->largest_seq_id_seen != 0
631  && seq_id != BUS_NO_SEQ_ID) {
632  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 128,
633  "suspicious sequence ID on %d: largest seen is %lld, got %lld\n",
634  ci->fd, (long long)ci->largest_seq_id_seen, (long long)seq_id);
635  }
636  ci->largest_seq_id_seen = seq_id;
637 
638  rx_info_t *info = find_info_by_sequence_id(l, ci->fd, seq_id);
639  if (info) {
640  switch (info->state) {
641  case RIS_HOLD:
642  /* Just save result, to match up later. */
643  BUS_ASSERT(b, b->udata, !info->u.hold.has_result);
644  info->u.hold.has_result = true;
645  info->u.hold.result = result;
646  break;
647  case RIS_EXPECT:
648  {
649  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 128,
650  "marking info %d, seq_id:%lld ready for delivery",
651  info->id, (long long)result.u.success.seq_id);
652  info->u.expect.error = RX_ERROR_READY_FOR_DELIVERY;
653  BUS_ASSERT(b, b->udata, !info->u.hold.has_result);
654  info->u.expect.has_result = true;
655  info->u.expect.result = result;
656  attempt_delivery(l, info);
657  break;
658  }
659  case RIS_INACTIVE:
660  default:
661  BUS_ASSERT(b, b->udata, false);
662  }
663  } else {
664  /* We received a response that we weren't expecting. */
665  if (seq_id != BUS_NO_SEQ_ID) {
666  BUS_LOG_SNPRINTF(b, 2 - 2, LOG_LISTENER, b->udata, 128,
667  "Couldn't find info for fd:%d, seq_id:%lld, msg %p",
668  ci->fd, (long long)seq_id, opaque_msg);
669  }
670  if (b->unexpected_msg_cb) {
671  b->unexpected_msg_cb(opaque_msg, seq_id, b->udata, ci->udata);
672  }
673  }
674  } else {
675  uintptr_t e_id = result.u.error.opaque_error_id;
676  BUS_LOG_SNPRINTF(b, 1, LOG_LISTENER, b->udata, 128,
677  "Got opaque_error_id of %lu (0x%08lx)",
678  e_id, e_id);
679 
680  /* Timeouts will clean up after it; give user code a chance to
681  * clean up after it here, though technically speaking they
682  * could in the unpack_cb above. */
683  b->error_cb(result, ci->udata);
684  }
685 }

687 static void tick_handler(listener *l) {
688  struct bus *b = l->bus;
689  bool any_work = false;
690 
691  BUS_LOG_SNPRINTF(b, 2, LOG_LISTENER, b->udata, 128,
692  "tick... %p: %d of %d msgs in use, %d of %d rx_info in use, %d tracked_fds",
693  (void*)l, l->msgs_in_use, MAX_QUEUE_MESSAGES,
694  l->rx_info_in_use, MAX_PENDING_MESSAGES, l->tracked_fds);
695 
696  if (b->log_level > 5) { /* dump msg table types */
697  for (int i = 0; i < l->msgs_in_use; i++) {
698  printf(" -- msg %d: type %d\n", i, l->msgs[i].type);
699  }
700  }
701 
702  if (b->log_level > 5 || 0) { dump_rx_info_table(l); }
703 
704  for (int i = 0; i <= l->rx_info_max_used; i++) {
705  rx_info_t *info = &l->rx_info[i];
706 
707  switch (info->state) {
708  case RIS_INACTIVE:
709  break;
710  case RIS_HOLD:
711  any_work = true;
712  /* Check timeout */
713  if (info->timeout_sec == 1) {
714  struct timeval tv;
715  if (-1 == gettimeofday(&tv, NULL)) {
716  BUS_LOG(b, 0, LOG_LISTENER,
717  "gettimeofday failure in tick_handler!", b->udata);
718  continue;
719  }
720 
721  /* never got a response, but we don't have the callback
722  * either -- the sender will notify about the timeout. */
723  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
724  "timing out hold info %p -- <fd:%d, seq_id:%lld> at (%ld.%ld)",
725  (void*)info, info->u.hold.fd, (long long)info->u.hold.seq_id,
726  (long)tv.tv_sec, (long)tv.tv_usec);
727 
728  release_rx_info(l, info);
729  } else {
730  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
731  "decrementing countdown on info %p [%u]: %ld",
732  (void*)info, info->id, info->timeout_sec - 1);
733  info->timeout_sec--;
734  }
735  break;
736  case RIS_EXPECT:
737  any_work = true;
738  if (info->u.expect.error == RX_ERROR_READY_FOR_DELIVERY) {
739  BUS_LOG(b, 4, LOG_LISTENER,
740  "retrying RX event delivery", b->udata);
741  retry_delivery(l, info);
742  } else if (info->u.expect.error == RX_ERROR_DONE) {
743  BUS_LOG_SNPRINTF(b, 4, LOG_LISTENER, b->udata, 64,
744  "cleaning up completed RX event at info %p", (void*)info);
745  clean_up_completed_info(l, info);
746  } else if (info->u.expect.error != RX_ERROR_NONE) {
747  BUS_LOG_SNPRINTF(b, 1, LOG_LISTENER, b->udata, 64,
748  "notifying of rx failure -- error %d (info %p)",
749  info->u.expect.error, (void*)info);
750  notify_message_failure(l, info, BUS_SEND_RX_FAILURE);
751  } else if (info->timeout_sec == 1) {
752  struct timeval tv;
753  if (-1 == gettimeofday(&tv, NULL)) {
754  BUS_LOG(b, 0, LOG_LISTENER,
755  "gettimeofday failure in tick_handler!", b->udata);
756  continue;
757  }
758  struct boxed_msg *box = info->u.expect.box;
759  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 256 + 64,
760  "notifying of rx failure -- timeout (info %p) -- "
761  "<fd:%d, seq_id:%lld>, from time (queued:%ld.%ld) to (sent:%ld.%ld) to (now:%ld.%ld)",
762  (void*)info, box->fd, (long long)box->out_seq_id,
763  (long)box->tv_send_start.tv_sec, (long)box->tv_send_start.tv_usec,
764  (long)box->tv_send_done.tv_sec, (long)box->tv_send_done.tv_usec,
765  (long)tv.tv_sec, (long)tv.tv_usec);
766 
767  notify_message_failure(l, info, BUS_SEND_RX_TIMEOUT);
768  } else {
769  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
770  "decrementing countdown on info %p [%u]: %ld",
771  (void*)info, info->id, info->timeout_sec - 1);
772  info->timeout_sec--;
773  }
774  break;
775  default:
776  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
777  "match fail %d on line %d", info->state, __LINE__);
778  BUS_ASSERT(b, b->udata, false);
779  }
780  }
781  if (!any_work) { l->is_idle = true; }
782 }

810 static void retry_delivery(listener *l, rx_info_t *info) {
811  struct bus *b = l->bus;
812  BUS_ASSERT(b, b->udata, info->state == RIS_EXPECT);
813  BUS_ASSERT(b, b->udata, info->u.expect.error == RX_ERROR_READY_FOR_DELIVERY);
814  BUS_ASSERT(b, b->udata, info->u.expect.box);
815 
816  struct boxed_msg *box = info->u.expect.box;
817  info->u.expect.box = NULL; /* release */
818  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 128,
819  "releasing box %p at line %d", (void*)box, __LINE__);
820  BUS_ASSERT(b, b->udata, box->result.status == BUS_SEND_SUCCESS);
821 
822  size_t backpressure = 0;
823  if (bus_process_boxed_message(l->bus, box, &backpressure)) {
824  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 128,
825  "successfully delivered box %p (seq_id %lld) from info %d at line %d (retry)",
826  (void*)box, (long long)box->out_seq_id, info->id, __LINE__);
827  info->u.expect.error = RX_ERROR_DONE;
828  release_rx_info(l, info);
829  } else {
830  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 128,
831  "returning box %p at line %d", (void*)box, __LINE__);
832  info->u.expect.box = box; /* retry in tick_handler */
833  }
834 
835  observe_backpressure(l, backpressure);
836 }

941 static void release_rx_info(struct listener *l, rx_info_t *info) {
942  struct bus *b = l->bus;
943  BUS_ASSERT(b, b->udata, info);
944  BUS_LOG_SNPRINTF(b, 5, LOG_LISTENER, b->udata, 128,
945  "releasing RX info %d (%p)", info->id, (void *)info);
946  BUS_ASSERT(b, b->udata, info->id < MAX_PENDING_MESSAGES);
947  BUS_ASSERT(b, b->udata, info == &l->rx_info[info->id]);
948 
949  switch (info->state) {
950  case RIS_HOLD:
951  if (info->u.hold.has_result) {
952  /* FIXME: If we have a message that timed out, we need to
953  * free it, but don't know how. We should never get here,
954  * because it means the sender finished sending the message,
955  * but the listener never got the handler callback. */
956 
957  if (info->u.hold.result.ok) {
958  void *msg = info->u.hold.result.u.success.msg;
959  int64_t seq_id = info->u.hold.result.u.success.seq_id;
960 
961  connection_info *ci = get_connection_info(l, info->u.hold.fd);
962  if (ci && b->unexpected_msg_cb) {
963  BUS_LOG_SNPRINTF(b, 1, LOG_LISTENER, b->udata, 128,
964  "CALLING UNEXPECTED_MSG_CB ON RESULT %p", (void *)&info->u.hold.result);
965  b->unexpected_msg_cb(msg, seq_id, b->udata, ci->udata);
966  } else {
967  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 128,
968  "LEAKING RESULT %p", (void *)&info->u.hold.result);
969  }
970 
971  }
972  }
973  break;
974  case RIS_EXPECT:
975  BUS_ASSERT(b, b->udata, info->u.expect.error == RX_ERROR_DONE);
976  BUS_ASSERT(b, b->udata, info->u.expect.box == NULL);
977  break;
978  default:
979  case RIS_INACTIVE:
980  BUS_ASSERT(b, b->udata, false);
981  }
982 
983  /* Set to no longer active and push on the freelist. */
984  BUS_LOG_SNPRINTF(b, 5, LOG_LISTENER, b->udata, 128,
985  "releasing rx_info_t %d (%p), was %d",
986  info->id, (void *)info, info->state);
987 
988  BUS_ASSERT(b, b->udata, info->state != RIS_INACTIVE);
989  info->state = RIS_INACTIVE;
990  memset(&info->u, 0, sizeof(info->u));
991  info->next = l->rx_info_freelist;
992  l->rx_info_freelist = info;
993 
994  if (l->rx_info_max_used == info->id && info->id > 0) {
995  BUS_LOG_SNPRINTF(b, 5, LOG_LISTENER, b->udata, 128,
996  "rx_info_max_used--, from %d to %d",
997  l->rx_info_max_used, l->rx_info_max_used - 1);
998  while (l->rx_info[l->rx_info_max_used].state == RIS_INACTIVE) {
999  l->rx_info_max_used--;
1000  if (l->rx_info_max_used == 0) { break; }
1001  }
1002  BUS_ASSERT(b, b->udata, l->rx_info_max_used < MAX_PENDING_MESSAGES);
1003  }
1004 
1005  l->rx_info_in_use--;
1006 }

1008 static listener_msg *get_free_msg(listener *l) {
1009  struct bus *b = l->bus;
1010 
1011  BUS_LOG_SNPRINTF(b, 4, LOG_LISTENER, b->udata, 128,
1012  "get_free_msg -- in use: %d", l->msgs_in_use);
1013 
1014  for (;;) {
1015  listener_msg *head = l->msg_freelist;
1016  if (head == NULL) {
1017  BUS_LOG(b, 3, LOG_LISTENER, "No free messages!", b->udata);

1022

1027  * 10 * ((n >> 1) ** 2) microseconds */

1029  if (delay > 0) {
1030  struct timespec ts = {
1031  .tv_sec = 0,
1032  .tv_nsec = 1000L * delay,
1033  };
1034  nanosleep(&ts, NULL);
1035  }
1036  BUS_ASSERT(b, b->udata, head->type == MSG_NONE);
1037  return head;
1038  }
1039  }
1040  }
1041  }

1065 static bool push_message(struct listener *l, listener_msg *msg) {
1066  struct bus *b = l->bus;
1067  BUS_ASSERT(b, b->udata, msg);
1068

1070  msg_buf[0] = msg->id;
1071

1075  return true; // committed
1076  } else {

1078  errno = 0;
1079  } else {

1082  errno = 0;
1083  release_msg(l, msg);
1084  return false;
1085  }
1086  }
1087  }
