61 bool KineticPDU_Send(KineticPDU* const request)
62 {
63  int fd = request->exchange->connection->socketDescriptor;
64 
65  // Send the PDU header
66  if (!KineticSocket_Write(fd, &request->header, sizeof(KineticPDUHeader)))
67  {
68  LOG("Failed to send PDU header!");
69  return false;
70  }
71 
72  // Populate the HMAC for the protobuf
73  KineticHMAC_Init(&request->hmac, KINETIC_PROTO_SECURITY_ACL_HMACALGORITHM_HmacSHA1);
74  KineticHMAC_Populate(&request->hmac, request->protobuf, request->exchange->key, request->exchange->keyLength);
75 
76  // Send the protobuf message
77  if (!KineticSocket_WriteProtobuf(fd, request->protobuf))
78  {
79  LOG("Failed to send PDU protobuf message!");
80  return false;
81  }
82 
83  // Send the value/payload, if specified
84  if ((request->valueLength > 0) && (request->value != NULL))
85  {
86  if (!KineticSocket_Write(fd, request->value, request->valueLength))
87  {
88  LOG("Failed to send PDU value payload!");
89  return false;
90  }
91  }
92 
93  return true;
94 }

96 bool KineticPDU_Receive(KineticPDU* const response)
97 {
98  const int fd = response->exchange->connection->socketDescriptor;
99 
100  // Send the PDU header
101  if (!KineticSocket_Read(fd, &response->header, sizeof(KineticPDUHeader)))
102  {
103  LOG("Failed to receive PDU header!");
104  return false;
105  }
106 
107  // Send the protobuf message
108  if (!KineticSocket_ReadProtobuf(fd, &response->protobuf->proto,
109  (KineticProto*)response->protobufScratch,
110  response->header.protobufLength))
111  {
112  LOG("Failed to receive PDU protobuf message!");
113  return false;
114  }
115 
116  // Validate the HMAC for the recevied protobuf message
117  if (!KineticHMAC_Validate(&response->protobuf->proto,
118  response->exchange->key, response->exchange->keyLength))
119  {
120  LOG("Received PDU protobuf message has invalid HMAC!");
121  return false;
122  }
123 
124  // Send the value payload, if specified
125  if (response->header.valueLength)
126  {
127  if (!KineticSocket_Read(fd, response->value, response->valueLength))
128  {
129  LOG("Failed to receive PDU value payload!");
130  return false;
131  }
132  }
133 
134  return true;
135 }

137 void KineticPDU_PackInt32(uint8_t* const buffer, int32_t value)
138 {
139  int i;
140  for (i = sizeof(int32_t) - 1; i >= 0; i--)
141  {
142  buffer[i] = value & 0xFF;
143  value = value >> 8;
144  }
145 }
