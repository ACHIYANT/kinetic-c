58 bool KineticPDU_Send(KineticPDU* request)
59 {
60  assert(request != NULL);
61  assert(request->connection != NULL);
62  LOGF("Attempting to receive PDU via fd=%d",
63  request->connection->socketDescriptor);
64 
65  // Populate the HMAC for the protobuf
66  KineticHMAC_Init(&request->hmac,
67  KINETIC_PROTO_SECURITY_ACL_HMACALGORITHM_HmacSHA1);
68  KineticHMAC_Populate(&request->hmac,
69  &request->protoData.message.proto, request->connection->key);
70 
71  // Configure PDU header length fields
72  request->header.versionPrefix = 'F';
73  request->header.protobufLength =
74  KineticProto__get_packed_size(&request->protoData.message.proto);
75  request->header.valueLength = request->value.len;
76  KineticLogger_LogHeader(&request->header);
77 
78  // Create NBO copy of header for sending
79  request->headerNBO.versionPrefix = request->header.versionPrefix;
80  request->headerNBO.protobufLength =
81  KineticNBO_FromHostU32(request->header.protobufLength);
82  request->headerNBO.valueLength =
83  KineticNBO_FromHostU32(request->header.valueLength);
84 
85  // Pack and send the PDU header
86  ByteArray headerNBO = {
87  .data = (uint8_t*)&request->headerNBO,
88  .len = sizeof(KineticPDUHeader)
89  };
90  if (!KineticSocket_Write(request->connection->socketDescriptor,
91  headerNBO))
92  {
93  LOG("Failed to send PDU header!");
94  return false;
95  }
96 
97  // Send the protobuf message
98  LOG("Sending PDU Protobuf:");
99  KineticLogger_LogProtobuf(&request->protoData.message.proto);
100  if (!KineticSocket_WriteProtobuf(request->connection->socketDescriptor,
101  request))
102  {
103  LOG("Failed to send PDU protobuf message!");
104  return false;
105  }
106 
107  // Send the value/payload, if specified
108  if ((request->value.len > 0) && (request->value.data != NULL))
109  {
110  if (!KineticSocket_Write(request->connection->socketDescriptor,
111  request->value))
112  {
113  LOG("Failed to send PDU value payload!");
114  return false;
115  }
116  }
117 
118  return true;
119 }

121 bool KineticPDU_Receive(KineticPDU* const response)
122 {
123  const int fd = response->connection->socketDescriptor;
124  LOGF("Attempting to receive PDU via fd=%d", fd);
125  assert(fd >= 0);
126  assert(response != NULL);
127  assert(response->proto != NULL);
128  ByteArray rawHeader = {
129  .data = (uint8_t*)&response->headerNBO,
130  .len = sizeof(KineticPDUHeader) };
131 
132  // Receive the PDU header
133  if (!KineticSocket_Read(fd, rawHeader))
134  {
135  LOG("Failed to receive PDU header!");
136  return false;
137  }
138  else
139  {
140  LOG("PDU header received successfully");
141  response->header = (KineticPDUHeader)
142  {
143  .versionPrefix = response->headerNBO.versionPrefix,
144  .protobufLength =
145  KineticNBO_ToHostU32(response->headerNBO.protobufLength),
146  .valueLength =
147  KineticNBO_ToHostU32(response->headerNBO.valueLength)
148  };
149  response->value.len = response->header.valueLength;
150  KineticLogger_LogHeader(&response->header);
151  }
152 
153  // Receive the protobuf message
154  if (!KineticSocket_ReadProtobuf(fd, response))
155  {
156  LOG("Failed to receive PDU protobuf message!");
157  return false;
158  }
159  else
160  {
161  LOG("Received PDU protobuf");
162  KineticLogger_LogProtobuf(&response->protoData.message.proto);
163  }
164 
165  // Validate the HMAC for the recevied protobuf message
166  if (!KineticHMAC_Validate(response->proto, response->connection->key))
167  {
168  LOG("Received PDU protobuf message has invalid HMAC!");
169  response->protoData.message.proto.command =
170  &response->protoData.message.command;
171  response->protoData.message.command.status =
172  &response->protoData.message.status;
173  response->protoData.message.status.code =
174  KINETIC_PROTO_STATUS_STATUS_CODE_DATA_ERROR;
175 
176  return false;
177  }
178  else
179  {
180  LOG("Received protobuf HMAC validation succeeded");
181  }
182 
183  // Receive the value payload, if specified
184  response->value.len = response->header.valueLength;
185  if (response->header.valueLength > 0)
186  {
187  assert(response->value.data != NULL);
188 
189  LOGF("Attempting to receive value payload (%lld bytes)...",
190  (long long)response->header.valueLength);
191 
192  LOG_LOCATION; LOGF("value.len=%u, value.data=0x%zu", response->value.len, response->value.data);
193 
194  if (!KineticSocket_Read(fd, response->value))
195  {
196  LOG("Failed to receive PDU value payload!");
197  return false;
198  }
199  else
200  {
201  LOG("Received value payload successfully");
202  response->value.len = response->header.valueLength;
203  KineticLogger_LogByteArray("Value Payload", response->value);
204  }
205  }
206 
207  return true;
208 }
