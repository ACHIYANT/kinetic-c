42 KineticStatus KineticPDU_Send(KineticPDU* request)
43 {
44  assert(request != NULL);
45  assert(request->connection != NULL);
46  LOGF("Receiving PDU via fd=%d", request->connection->socket);
47 
48  KineticStatus status = KINETIC_STATUS_INVALID;
49 
50  // Populate the HMAC for the protobuf
51  KineticHMAC_Init(
52  &request->hmac,
53  KINETIC_PROTO_SECURITY_ACL_HMACALGORITHM_HmacSHA1);
54  KineticHMAC_Populate(
55  &request->hmac,
56  &request->protoData.message.proto,
57  request->connection->session.hmacKey);
58 
59  // Configure PDU header length fields
60  request->header.versionPrefix = 'F';
61  request->header.protobufLength =
62  KineticProto__get_packed_size(&request->protoData.message.proto);
63  request->header.valueLength =
64  (request->entry.value.array.data == NULL) ? 0 : request->entry.value.bytesUsed;
65  KineticLogger_LogHeader(&request->header);
66 
67  // Create NBO copy of header for sending
68  request->headerNBO.versionPrefix = 'F';
69  request->headerNBO.protobufLength =
70  KineticNBO_FromHostU32(request->header.protobufLength);
71  request->headerNBO.valueLength =
72  KineticNBO_FromHostU32(request->header.valueLength);
73 
74  // Pack and send the PDU header
75  ByteBuffer hdr = ByteBuffer_Create(&request->headerNBO, sizeof(KineticPDUHeader));
76  hdr.bytesUsed = hdr.array.len;
77  status = KineticSocket_Write(request->connection->socket, &hdr);
78  if (status != KINETIC_STATUS_SUCCESS) {
79  LOG("Failed to send PDU header!");
80  return status;
81  }
82 
83  // Send the protobuf message
84  #ifdef KINETIC_LOG_PDU_OPERATIONS
85  LOG("Sending PDU Protobuf:");
86  #endif
87  KineticLogger_LogProtobuf(&request->protoData.message.proto);
88  status = KineticSocket_WriteProtobuf(
89  request->connection->socket, request);
90  if (status != KINETIC_STATUS_SUCCESS) {
91  LOG("Failed to send PDU protobuf message!");
92  return status;
93  }
94 
95  // Send the value/payload, if specified
96  ByteBuffer* value = &request->entry.value;
97  if ((value->array.data != NULL) && (value->array.len > 0) && (value->bytesUsed > 0)) {
98  status = KineticSocket_Write(request->connection->socket, value);
99  if (status != KINETIC_STATUS_SUCCESS) {
100  LOG("Failed to send PDU value payload!");
101  return status;
102  }
103  }
104 
105  return KINETIC_STATUS_SUCCESS;
106 }

108 KineticStatus KineticPDU_Receive(KineticPDU* const response)
109 {
110  assert(response != NULL);
111  const int fd = response->connection->socket;
112  assert(fd >= 0);
113  LOGF("Receiving PDU via fd=%d", fd);
114 
115  KineticStatus status;
116 
117  // Receive the PDU header
118  ByteBuffer rawHeader =
119  ByteBuffer_Create(&response->headerNBO, sizeof(KineticPDUHeader));
120  status = KineticSocket_Read(fd, &rawHeader, rawHeader.array.len);
121  if (status != KINETIC_STATUS_SUCCESS) {
122  LOG("Failed to receive PDU header!");
123  return status;
124  }
125  else {
126  #ifdef KINETIC_LOG_PDU_OPERATIONS
127  LOG("PDU header received successfully");
128  #endif
129  KineticPDUHeader* headerNBO = &response->headerNBO;
130  response->header = (KineticPDUHeader) {
131  .versionPrefix = headerNBO->versionPrefix,
132  .protobufLength = KineticNBO_ToHostU32(headerNBO->protobufLength),
133  .valueLength = KineticNBO_ToHostU32(headerNBO->valueLength),
134  };
135  KineticLogger_LogHeader(&response->header);
136  }
137 
138  // Receive the protobuf message
139  status = KineticSocket_ReadProtobuf(fd, response);
140  if (status != KINETIC_STATUS_SUCCESS) {
141  LOG("Failed to receive PDU protobuf message!");
142  return status;
143  }
144  else {
145  #ifdef KINETIC_LOG_PDU_OPERATIONS
146  LOG("Received PDU protobuf");
147  #endif
148  KineticLogger_LogProtobuf(response->proto);
149  }
150 
151  // Validate the HMAC for the recevied protobuf message
152  if (!KineticHMAC_Validate(response->proto,
153  response->connection->session.hmacKey)) {
154  LOG("Received PDU protobuf message has invalid HMAC!");
155  KineticMessage* msg = &response->protoData.message;
156  msg->proto.command = &msg->command;
157  msg->command.status = &msg->status;
158  msg->status.code = KINETIC_PROTO_STATUS_STATUS_CODE_DATA_ERROR;
159  return KINETIC_STATUS_DATA_ERROR;
160  }
161  else {
162  #ifdef KINETIC_LOG_PDU_OPERATIONS
163  LOG("Received protobuf HMAC validation succeeded");
164  #endif
165  }
166 
167  // Receive the value payload, if specified
168  if (response->header.valueLength > 0) {
169  assert(response->entry.value.array.data != NULL);
170  #ifdef KINETIC_LOG_PDU_OPERATIONS
171  LOGF("Receiving value payload (%lld bytes)...",
172  (long long)response->header.valueLength);
173  #endif
174 
175  response->entry.value.bytesUsed = 0;
176  status = KineticSocket_Read(fd,
177  &response->entry.value, response->header.valueLength);
178  if (status != KINETIC_STATUS_SUCCESS) {
179  LOG("Failed to receive PDU value payload!");
180  return status;
181  }
182  #ifdef KINETIC_LOG_PDU_OPERATIONS
183  else {
184  LOG("Received value payload successfully");
185  }
186  #endif
187 
188  // KineticLogger_LogByteBuffer("Value Buffer", response->entry.value);
189  }
190 
191  // Update connectionID to match value returned from device, if provided
192  KineticProto_Command* cmd = response->proto->command;
193  if ((cmd != NULL) && (cmd->header != NULL) && (cmd->header->has_connectionID)) {
194  response->connection->connectionID = cmd->header->connectionID;
195  }
196 
197  return KineticPDU_GetStatus(response);
198 }
