63 bool KineticPDU_Send(KineticPDU* const request)
64 {
65  int fd = request->exchange->connection->socketDescriptor;
66 
67  assert(request != NULL);
68  assert(request->message != NULL);
69 
70  // Populate the HMAC for the protobuf
71  KineticHMAC_Init(&request->hmac, KINETIC_PROTO_SECURITY_ACL_HMACALGORITHM_HmacSHA1);
72  KineticHMAC_Populate(&request->hmac, &request->message->proto, request->exchange->key, request->exchange->keyLength);
73 
74  // Repack PDU header length fields
75  request->protobufLength = KineticProto_get_packed_size(&request->message->proto);
76  LOGF("Packing PDU with protobuf of %d bytes", request->protobufLength);
77  KineticPDU_PackInt32((uint8_t*)&request->header.protobufLength, request->protobufLength);
78  LOGF("Packing PDU with value payload of %d bytes", request->valueLength);
79  KineticPDU_PackInt32((uint8_t*)&request->header.valueLength, request->valueLength);
80 
81  // Send the PDU header
82  if (!KineticSocket_Write(fd, &request->header, sizeof(KineticPDUHeader)))
83  {
84  LOG("Failed to send PDU header!");
85  return false;
86  }
87 
88  // Send the protobuf message
89  LOG("Sending PDU Protobuf:");
90  KineticPDU_LogProtobuf(&request->message->proto);
91  if (!KineticSocket_WriteProtobuf(fd, &request->message->proto))
92  {
93  LOG("Failed to send PDU protobuf message!");
94  return false;
95  }
96 
97  // Send the value/payload, if specified
98  if ((request->valueLength > 0) && (request->value != NULL))
99  {
100  if (!KineticSocket_Write(fd, request->value, request->valueLength))
101  {
102  LOG("Failed to send PDU value payload!");
103  return false;
104  }
105  }
106 
107  return true;
108 }

110 bool KineticPDU_Receive(KineticPDU* const response)
111 {
112  const int fd = response->exchange->connection->socketDescriptor;
113  LOGF("Attempting to receive PDU via fd=%d", fd);
114  assert(fd >= 0);
115 
116  // Receive the PDU header
117  if (!KineticSocket_Read(fd, response->rawHeader, sizeof(KineticPDUHeader)))
118  {
119  LOG("Failed to receive PDU header!");
120  return false;
121  }
122  else
123  {
124  LOG("PDU header received successfully");
125  response->header.versionPrefix = response->rawHeader[0];
126  response->header.protobufLength = KineticPDU_UnpackInt32(&response->rawHeader[1]);
127  response->header.valueLength = KineticPDU_UnpackInt32(&response->rawHeader[1 + sizeof(uint32_t)]);
128  KineticPDU_LogHeader(&response->header);
129  }
130 
131  // Receive the protobuf message
132  if (!KineticSocket_ReadProtobuf(fd,
133  &response->proto,
134  (KineticProto*)response->protobufScratch,
135  response->header.protobufLength))
136  {
137  LOG("Failed to receive PDU protobuf message!");
138  return false;
139  }
140  else
141  {
142  LOG("Received PDU protobuf");
143  KineticPDU_LogProtobuf(response->proto);
144  }
145 
146  // Validate the HMAC for the recevied protobuf message
147  if (!KineticHMAC_Validate(response->proto,
148  response->exchange->key, response->exchange->keyLength))
149  {
150  LOG("Received PDU protobuf message has invalid HMAC!");
151  return false;
152  }
153  else
154  {
155  LOG("Received protobuf HMAC validation succeeded");
156  }
157 
158  // Receive the value payload, if specified
159  if (response->header.valueLength > 0)
160  {
161  LOG("Attempting to receive value payload...");
162  if (!KineticSocket_Read(fd, response->value, response->valueLength))
163  {
164  LOG("Failed to receive PDU value payload!");
165  return false;
166  }
167  else
168  {
169  LOG("Received value payload successfully");
170  }
171  }
172 
173  return true;
174 }

176 void KineticPDU_PackInt32(uint8_t* const buffer, int32_t value)
177 {
178  int i;
179  for (i = sizeof(int32_t) - 1; i >= 0; i--)
180  {
181  buffer[i] = value & 0xFF;
182  value = value >> 8;
183  }
184 }

186 int32_t KineticPDU_UnpackInt32(const uint8_t* const buffer)
187 {
188  int i;
189  int32_t value = 0;
190  for (i = 0; i < sizeof(int32_t); i++)
191  {
192  value <<= 8;
193  value += buffer[i];
194  }
195  return value;
196 }
