36 KineticStatus KineticPDU_ReceiveMain(KineticPDU* const response)
37 {
38  assert(response != NULL);
39  assert(response->connection != NULL);
40  const int fd = response->connection->socket;
41  assert(fd >= 0);
42  LOGF1("\nReceiving PDU via fd=%d", fd);
43 
44  KineticStatus status;
45  KineticMessage* msg = &response->protoData.message;
46 
47  // Receive the PDU header
48  ByteBuffer rawHeader =
49  ByteBuffer_Create(&response->headerNBO, sizeof(KineticPDUHeader), 0);
50  status = KineticSocket_Read(fd, &rawHeader, rawHeader.array.len);
51  if (status != KINETIC_STATUS_SUCCESS) {
52  LOG0("Failed to receive PDU header!");
53  return status;
54  }
55  else {
56  LOG3("PDU header received successfully");
57  KineticPDUHeader* headerNBO = &response->headerNBO;
58  response->header = (KineticPDUHeader) {
59  .versionPrefix = headerNBO->versionPrefix,
60  .protobufLength = KineticNBO_ToHostU32(headerNBO->protobufLength),
61  .valueLength = KineticNBO_ToHostU32(headerNBO->valueLength),
62  };
63  KineticLogger_LogHeader(1, &response->header);
64  }
65 
66  // Receive the protobuf message
67  status = KineticSocket_ReadProtobuf(fd, response);
68  if (status != KINETIC_STATUS_SUCCESS) {
69  LOG0("Failed to receive PDU protobuf message!");
70  return status;
71  }
72  else {
73  LOG3("Received PDU protobuf");
74  }
75 
76  // Validate the HMAC for the recevied protobuf message
77  if (response->proto->authType == KINETIC_PROTO_MESSAGE_AUTH_TYPE_HMACAUTH) {
78  KineticLogger_LogProtobuf(2, response->proto);
79  if(!KineticHMAC_Validate(
80  response->proto, response->connection->session.config.hmacKey)) {
81  LOG0("Received PDU protobuf message has invalid HMAC!");
82  msg->has_command = true;
83  msg->command.status = &msg->status;
84  msg->status.code = KINETIC_PROTO_COMMAND_STATUS_STATUS_CODE_DATA_ERROR;
85  return KINETIC_STATUS_DATA_ERROR;
86  }
87  else {
88  LOG3("Received protobuf HMAC validation succeeded");
89  }
90  }
91  else if (response->proto->authType == KINETIC_PROTO_MESSAGE_AUTH_TYPE_PINAUTH) {
92  KineticLogger_LogProtobuf(2, response->proto);
93  LOG0("PIN-based authentication not yet supported!");
94  return KINETIC_STATUS_DATA_ERROR;
95  }
96  else if (response->proto->authType == KINETIC_PROTO_MESSAGE_AUTH_TYPE_UNSOLICITEDSTATUS) {
97  KineticLogger_LogProtobuf(3, response->proto);
98  }
99 
100  // Extract embedded command, if supplied
101  KineticProto_Message* pMsg = response->proto;
102  if (pMsg->has_commandBytes &&
103  pMsg->commandBytes.data != NULL &&
104  pMsg->commandBytes.len > 0) {
105  response->command = KineticProto_command__unpack(
106  NULL,
107  pMsg->commandBytes.len,
108  pMsg->commandBytes.data);
109  }
110 
111  status = KineticPDU_GetStatus(response);
112  if (status == KINETIC_STATUS_SUCCESS) {
113  LOG2("PDU received successfully!");
114  }
115  return status;
116 }

118 KineticStatus KineticPDU_ReceiveValue(int socket_desc, ByteBuffer* value, size_t value_length)
119 {
120  assert(socket_desc >= 0);
121  assert(value != NULL);
122  assert(value->array.data != NULL);
123 
124  // Receive value payload
125  LOGF1("Receiving value payload (%lld bytes)...", value_length);
126  ByteBuffer_Reset(value);
127  KineticStatus status = KineticSocket_Read(socket_desc, value, value_length);
128  if (status != KINETIC_STATUS_SUCCESS) {
129  LOG0("Failed to receive PDU value payload!");
130  return status;
131  }
132  LOG1("Received value payload successfully");
133  KineticLogger_LogByteBuffer(3, "Value Buffer", *value);
134  return status;
135 }
