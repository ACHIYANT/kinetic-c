62 STATIC bool unpack_header(uint8_t const * const read_buf, size_t const read_size, KineticPDUHeader * const header)
63 {
64  if (read_size != sizeof(KineticPDUHeader)) {
65  return false;
66  // TODO this will fail if we don't get all of the header bytes in one read
67  // we should fix this
68  }
69  KineticPDUHeader const * const buf_header = (KineticPDUHeader const * const)read_buf;
70  uint32_t protobufLength = KineticNBO_ToHostU32(buf_header->protobufLength);
71  uint32_t valueLength = KineticNBO_ToHostU32(buf_header->valueLength);
72  uint8_t versionPrefix = buf_header->versionPrefix;
73 
74  if (protobufLength <= PDU_PROTO_MAX_LEN &&
75  valueLength <= PDU_PROTO_MAX_LEN) {
76 
77  *header = (KineticPDUHeader){
78  .versionPrefix = versionPrefix,
79  .protobufLength = protobufLength,
80  .valueLength = valueLength,
81  };
82  return true;
83  } else {
84  return false;
85  }
86 }

88 STATIC bus_sink_cb_res_t sink_cb(uint8_t *read_buf,
89  size_t read_size, void *socket_udata) {
90 
91  KineticConnection * connection = (KineticConnection *)socket_udata;
92  assert(connection);
93  socket_info *si = connection->si;
94  assert(si);
95 
96  switch (si->state) {
97  case STATE_UNINIT:
98  {
99  assert(read_size == 0);
100  return reset_transfer(si);
101  }
102  case STATE_AWAITING_HEADER:
103  {
104  memcpy(&si->buf[si->accumulated], read_buf, read_size);
105  si->accumulated += read_size;
106 
107  uint32_t remaining = PDU_HEADER_LEN - si->accumulated;
108 
109  if (remaining == 0) {
110  if (unpack_header(&si->buf[0], PDU_HEADER_LEN, &si->header))
111  {
112  si->accumulated = 0;
113  si->unpack_status = UNPACK_ERROR_SUCCESS;
114  si->state = STATE_AWAITING_BODY;
115  bus_sink_cb_res_t res = {
116  .next_read = si->header.protobufLength + si->header.valueLength,
117  };
118  return res;
119  } else {
120  si->accumulated = 0;
121  si->unpack_status = UNPACK_ERROR_INVALID_HEADER;
122  si->state = STATE_AWAITING_HEADER;
123  bus_sink_cb_res_t res = {
124  .next_read = sizeof(KineticPDUHeader),
125  .full_msg_buffer = si,
126  };
127  return res;
128  }
129  }
130  else
131  {
132  bus_sink_cb_res_t res = {
133  .next_read = remaining,
134  };
135  return res;
136  }
137  break;
138  }
139  case STATE_AWAITING_BODY:
140  {
141  memcpy(&si->buf[si->accumulated], read_buf, read_size);
142  si->accumulated += read_size;
143 
144  uint32_t remaining = si->header.protobufLength + si->header.valueLength - si->accumulated;
145 
146  if (remaining == 0) {
147  si->state = STATE_AWAITING_HEADER;
148  si->accumulated = 0;
149  bus_sink_cb_res_t res = {
150  .next_read = sizeof(KineticPDUHeader),
151  // returning the whole si, because we need access to the pdu header as well
152  // as the protobuf and value bytes
153  .full_msg_buffer = si,
154  };
155  return res;
156  } else {
157  bus_sink_cb_res_t res = {
158  .next_read = remaining,
159  };
160  return res;
161  }
162  break;
163  }
164  default:
165  assert(false);
166  }
167 }

169 STATIC bus_unpack_cb_res_t unpack_cb(void *msg, void *socket_udata) {
170  (void)socket_udata;
171  /* just got .full_msg_buffer from sink_cb -- pass it along as-is */
172  socket_info *si = (socket_info *)msg;
173 
174  if (si->unpack_status != UNPACK_ERROR_SUCCESS)
175  {
176  return (bus_unpack_cb_res_t) {
177  .ok = false,
178  .u.error.opaque_error_id = si->unpack_status,
179  };
180  }
181 
182  KineticResponse * response = KineticAllocator_NewKineticResponse(si->header.valueLength);
183 
184  if (response == NULL) {
185  bus_unpack_cb_res_t res = {
186  .ok = false,
187  .u.error.opaque_error_id = UNPACK_ERROR_PAYLOAD_MALLOC_FAIL,
188  };
189  return res;
190  } else {
191  response->header = si->header;
192 
193  response->proto = KineticPDU_unpack_message(NULL, si->header.protobufLength, si->buf);
194  if (response->proto->has_commandBytes &&
195  response->proto->commandBytes.data != NULL &&
196  response->proto->commandBytes.len > 0)
197  {
198  response->command = KineticPDU_unpack_command(NULL,
199  response->proto->commandBytes.len, response->proto->commandBytes.data);
200  } else {
201  response->command = NULL;
202  }
203 
204  if (response->header.valueLength > 0)
205  {
206  memcpy(response->value, &si->buf[si->header.protobufLength], si->header.valueLength);
207  }
208 
209  int64_t seq_id = 0;
210  if (response->command != NULL &&
211  response->command->header != NULL) {
212 
213  seq_id = response->command->header->ackSequence;
214  }
215 
216  bus_unpack_cb_res_t res = {
217  .ok = true,
218  .u.success = {
219  .seq_id = seq_id,
220  .msg = response,
221  },
222  };
223  return res;
224  }
225 }

228 {
229  int log_level = config->logLevel;
230 
231  bus_config cfg = {
232  .log_cb = log_cb,

234  .sink_cb = sink_cb,

236  .unexpected_msg_cb = KineticController_HandleUnexpectedResponse,
237  .bus_udata = NULL,
238  .sender_count = config->writerThreads,

243  };
244  bus_result res;
245  memset(&res, 0, sizeof(res));
246  if (!bus_init(&cfg, &res)) {
247  LOGF0("failed to init bus: %d\n", res.status);
248  return false;
249  }
250  client->bus = res.bus;
251  return true;
252 }
