44 int KineticSocket_Connect(char* host, int port, bool blocking)
45 {
46  char port_str[32];
47  struct addrinfo hints;
48  struct addrinfo* ai_result = NULL;
49  struct addrinfo* ai = NULL;
50  socket99_result result;
51 
52  // Setup server address info
53  socket99_config cfg = {
54  .host = host,
55  .port = port,
56  .nonblocking = !blocking,
57  };
58  sprintf(port_str, "%d", port);
59 
60  // Open socket
61  LOGF("\nConnecting to %s:%d", host, port);
62  if (!socket99_open(&cfg, &result))
63  {
64  LOGF("Failed to open socket connection with host: status %d, errno %d\n",
65  result.status, result.saved_errno);
66  return -1;
67  }
68 
69  // Configure the socket
70  socket99_set_hints(&cfg, &hints);
71  if (getaddrinfo(cfg.host, port_str, &hints, &ai_result) != 0)
72  {
73  LOGF("Failed to get socket address info: errno %d\n", errno);
74  close(result.fd);
75  return -1;
76  }
77 
78  for (ai = ai_result; ai != NULL; ai = ai->ai_next)
79  {
80  // OSX won't let us set close-on-exec when creating the socket, so set it separately
81  int current_fd_flags = fcntl(result.fd, F_GETFD);
82  if (current_fd_flags == -1)
83  {
84  LOG("Failed to get socket fd flags");
85  close(result.fd);
86  continue;
87  }
88  if (fcntl(result.fd, F_SETFD, current_fd_flags | FD_CLOEXEC) == -1)
89  {
90  LOG("Failed to set socket close-on-exit");
91  close(result.fd);
92  continue;
93  }
94 
95  #if defined(SO_NOSIGPIPE) && !defined(__APPLE__)
96  // On BSD-like systems we can set SO_NOSIGPIPE on the socket to prevent it from sending a
97  // PIPE signal and bringing down the whole application if the server closes the socket
98  // forcibly
99  {
100  int set = 1;
101  int setsockopt_result = setsockopt(socket_fd, SOL_SOCKET, SO_NOSIGPIPE, &set, sizeof(set));
102  // Allow ENOTSOCK because it allows tests to use pipes instead of real sockets
103  if (setsockopt_result != 0 && setsockopt_result != ENOTSOCK)
104  {
105  LOG("Failed to set SO_NOSIGPIPE on socket");
106  close(result.fd);
107  continue;
108  }
109  }
110  #endif
111 
112  break;
113  }
114 
115  freeaddrinfo(ai_result);
116 
117  if (ai == NULL)
118  {
119  // we went through all addresses without finding one we could bind to
120  LOGF("Could not connect to %s:%d", host, port);
121  return -1;
122  }
123 
124  return result.fd;
125 }

149 bool KineticSocket_Read(int socketDescriptor, void* buffer, size_t length)
150 {
151  size_t count;
152 
153  for (count = 0; count < length; )
154  {
155  int status;
156  fd_set readSet;
157  struct timeval timeout;
158 
159  // Time out after 5 seconds
160  timeout.tv_sec = 5;
161  timeout.tv_usec = 0;
162 
163  FD_ZERO(&readSet);
164  FD_SET(socketDescriptor, &readSet);
165  status = select(socketDescriptor+1, &readSet, NULL, NULL, &timeout);
166 
167  if (status < 0) // Error occurred
168  {
169  char msg[128];
170  sprintf(msg, "Failed waiting to read from socket! status=%d, errno=%d\n", status, errno);
171  LOG(msg);
172  return false;
173  }
174  else if (status == 0) // Timeout occurred
175  {
176  LOG("Timed out waiting for socket data to arrive!");
177  return false;
178  }
179  else if (status > 0) // Data available to read
180  {
181  // The socket is ready for reading
182  status = read(socketDescriptor, &((uint8_t*)buffer)[count], length - count);
183  if (status == -1 && errno == EINTR)
184  {
185  continue;
186  }
187  else if (status <= 0)
188  {
189  char msg[128];
190  sprintf(msg, "Failed to read from socket! status=%d, errno=%d\n", status, errno);
191  LOG(msg);
192  return false;
193  }
194  else
195  {
196  count += status;
197  }
198  }
199  }
200 
201  return true;
202 }

240 bool KineticSocket_Write(int socketDescriptor, const void* buffer, size_t length)
241 {
242  int status;
243  size_t count;
244  char msg[128];
245 
246  for (count = 0; count < length; )
247  {
248  LOG("Sending data to client...");
249  sprintf(msg, "Attempting to write %zd bytes to socket", (length - count));
250  LOG(msg);
251  status = write(socketDescriptor, &((uint8_t*)buffer)[count], length - count);
252  if (status == -1 && errno == EINTR)
253  {
254  LOG("Write interrupted... retrying...");
255  continue;
256  }
257  else if (status <= 0)
258  {
259  LOG("Failed writing data to socket!");
260  sprintf(msg, "Failed to write to socket! status=%d, errno=%d\n", status, errno);
261  LOG(msg);
262  return false;
263  }
264  else
265  {
266  count += status;
267  sprintf(msg, "Wrote %d bytes to socket! %zd more to send...", status, (length - count));
268  LOG(msg);
269  }
270  }
271 
272  LOG("Write completed successfully!");
273  return true;
274 }
