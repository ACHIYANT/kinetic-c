49 int KineticSocket_Connect(const char* host, int port, bool nonBlocking)
50 {
51  char port_str[32];
52  struct addrinfo hints;
53  struct addrinfo* ai_result = NULL;
54  struct addrinfo* ai = NULL;
55  socket99_result result;
56 
57  // Setup server address info
58  socket99_config cfg = {
59  .host = (char*)host,
60  .port = port,
61  .nonblocking = nonBlocking,
62  };
63  sprintf(port_str, "%d", port);
64 
65  // Open socket
66  LOGF("Connecting to %s:%d", host, port);
67  if (!socket99_open(&cfg, &result)) {
68  LOGF("Failed to open socket connection"
69  "with host: status %d, errno %d",
70  result.status, result.saved_errno);
71  return KINETIC_SOCKET_DESCRIPTOR_INVALID;
72  }
73 
74  // Configure the socket
75  socket99_set_hints(&cfg, &hints);
76  if (getaddrinfo(cfg.host, port_str, &hints, &ai_result) != 0) {
77  LOGF("Failed to get socket address info: errno %d", errno);
78  close(result.fd);
79  return KINETIC_SOCKET_DESCRIPTOR_INVALID;
80  }
81 
82  for (ai = ai_result; ai != NULL; ai = ai->ai_next) {
83  int setsockopt_result;
84  int buffer_size = PDU_VALUE_MAX_LEN;
85 
86 #if defined(SO_NOSIGPIPE) && !defined(__APPLE__)
87  // On BSD-like systems we can set SO_NOSIGPIPE on the socket to
88  // prevent it from sending a PIPE signal and bringing down the whole
89  // application if the server closes the socket forcibly
90  int enable = 1;
91  setsockopt_result = setsockopt(result.fd,
92  SOL_SOCKET, SO_NOSIGPIPE,
93  &enable, sizeof(enable));
94  // Allow ENOTSOCK because it allows tests to use pipes instead of
95  // real sockets
96  if (setsockopt_result != 0 && setsockopt_result != ENOTSOCK) {
97  LOG("Failed to set SO_NOSIGPIPE on socket");
98  continue;
99  }
100 #endif
101 
102  // Increase send buffer to PDU_VALUE_MAX_LEN
103  // Note: OS allocates 2x this value for its overhead
104  setsockopt_result = setsockopt(result.fd,
105  SOL_SOCKET, SO_SNDBUF,
106  &buffer_size, sizeof(buffer_size));
107  if (setsockopt_result == -1) {
108  LOG("Error setting socket send buffer size");
109  continue;
110  }
111 
112  // Increase receive buffer to PDU_VALUE_MAX_LEN
113  // Note: OS allocates 2x this value for its overheadbuffer_size
114  setsockopt_result = setsockopt(result.fd,
115  SOL_SOCKET, SO_RCVBUF,
116  &buffer_size, sizeof(buffer_size));
117  if (setsockopt_result == -1) {
118  LOG("Error setting socket receive buffer size");
119  continue;
120  }
121 
122  break;
123  }
124 
125  freeaddrinfo(ai_result);
126 
127  if (ai == NULL || result.fd == KINETIC_SOCKET_DESCRIPTOR_INVALID) {
128  // we went through all addresses without finding one we could bind to
129  LOGF("Could not connect to %s:%d", host, port);
130  return KINETIC_SOCKET_DESCRIPTOR_INVALID;
131  }
132  else {
133  LOGF("Successfully connected to %s:%d (fd=%d)", host, port, result.fd);
134  return result.fd;
135  }
136 }

156 KineticStatus KineticSocket_Read(int socket, ByteBuffer* dest, size_t len)
157 {
158  #ifdef KINETIC_LOG_SOCKET_OPERATIONS
159  LOGF("Reading %zd bytes into buffer @ 0x%zX from fd=%d",
160  len, (size_t)dest->array.data, socket);
161  #endif
162 
163  KineticStatus status = KINETIC_STATUS_INVALID;
164 
165  // Read "up to" the allocated number of bytes into dest buffer
166  size_t bytesToReadIntoBuffer = len;
167  if (dest->array.len < len) {
168  bytesToReadIntoBuffer = dest->array.len;
169  }
170  while (dest->bytesUsed < bytesToReadIntoBuffer) {
171  int opStatus;
172  fd_set readSet;
173  struct timeval timeout;
174 
175  // Time out after 5 seconds
176  timeout.tv_sec = 5;
177  timeout.tv_usec = 0;
178 
179  FD_ZERO(&readSet);
180  FD_SET(socket, &readSet);
181  opStatus = select(socket + 1, &readSet, NULL, NULL, &timeout);
182 
183  if (opStatus < 0) { // Error occurred
184  LOGF("Failed waiting to read from socket!"
185  " status=%d, errno=%d, desc='%s'",
186  opStatus, errno, strerror(errno));
187  return KINETIC_STATUS_SOCKET_ERROR;
188  }
189  else if (opStatus == 0) { // Timeout occurred
190  LOG("Timed out waiting for socket data to arrive!");
191  return KINETIC_STATUS_SOCKET_TIMEOUT;
192  }
193  else if (opStatus > 0) { // Data available to read
194  // The socket is ready for reading
195  opStatus = read(socket,
196  &dest->array.data[dest->bytesUsed],
197  dest->array.len - dest->bytesUsed);
198  // Retry if no data yet...
199  if (opStatus == -1 &&
200  ((errno == EINTR) ||
201  (errno == EAGAIN) ||
202  (errno == EWOULDBLOCK)
203  )) {
204  continue;
205  }
206  else if (opStatus <= 0) {
207  LOGF("Failed to read from socket!"
208  " status=%d, errno=%d, desc='%s'",
209  opStatus, errno, strerror(errno));
210  return KINETIC_STATUS_SOCKET_ERROR;
211  }
212  else {
213  dest->bytesUsed += opStatus;
214  #ifdef KINETIC_LOG_SOCKET_OPERATIONS
215  LOGF("Received %d bytes (%zd of %zd)",
216  opStatus, dest->bytesUsed, len);
217  #endif
218  }
219  }
220  }
221 
222  // Flush any remaining data, in case of a truncated read w/short dest buffer
223  if (dest->bytesUsed < len) {
224  bool abortFlush = false;
225 
226  uint8_t* discardedBytes = malloc(len - dest->bytesUsed);
227  if (discardedBytes == NULL) {
228  LOG("Failed allocating a socket read discard buffer!");
229  abortFlush = true;
230  status = KINETIC_STATUS_MEMORY_ERROR;
231  }
232 
233  while (!abortFlush && dest->bytesUsed < len) {
234  int opStatus;
235  fd_set readSet;
236  struct timeval timeout;
237  size_t remainingLen = len - dest->bytesUsed;
238 
239  // Time out after 5 seconds
240  timeout.tv_sec = 5;
241  timeout.tv_usec = 0;
242 
243  FD_ZERO(&readSet);
244  FD_SET(socket, &readSet);
245  opStatus = select(socket + 1, &readSet, NULL, NULL, &timeout);
246 
247  if (opStatus < 0) { // Error occurred
248  LOGF("Failure trying to flush read socket data!"
249  " status=%d, errno=%d, desc='%s'",
250  status, errno, strerror(errno));
251  abortFlush = true;
252  status = KINETIC_STATUS_SOCKET_ERROR;
253  continue;
254  }
255  else if (opStatus == 0) { // Timeout occurred
256  LOG("Timed out waiting to flush socket data!");
257  abortFlush = true;
258  status = KINETIC_STATUS_SOCKET_TIMEOUT;
259  continue;
260  }
261  else if (opStatus > 0) { // Data available to read
262  // The socket is ready for reading
263  opStatus = read(socket, discardedBytes, remainingLen);
264  // Retry if no data yet...
265  if (opStatus == -1 &&
266  ((errno == EINTR) ||
267  (errno == EAGAIN) ||
268  (errno == EWOULDBLOCK)
269  )) {
270  continue;
271  }
272  else if (opStatus <= 0) {
273  LOGF("Failed to read from socket while flushing!"
274  " status=%d, errno=%d, desc='%s'",
275  opStatus, errno, strerror(errno));
276  abortFlush = true;
277  status = KINETIC_STATUS_SOCKET_ERROR;
278  }
279  else {
280  dest->bytesUsed += opStatus;
281  LOGF("Flushed %d bytes from socket read pipe (%zd of %zd)",
282  opStatus, dest->bytesUsed, len);
283  }
284  }
285  }
286 
287  // Free up dynamically allocated memory before returning
288  if (discardedBytes != NULL) {
289  free(discardedBytes);
290  }
291 
292  // Report any error that occurred during socket flush
293  if (abortFlush) {
294  LOG("Socket read pipe flush aborted!");
295  assert(status == KINETIC_STATUS_SUCCESS);
296  return status;
297  }
298 
299  // Report truncation of data for any variable length byte arrays
300  LOGF("Socket read buffer was truncated due to buffer overrun!"
301  " received=%zu, copied=%zu",
302  len, dest->array.len);
303  return KINETIC_STATUS_BUFFER_OVERRUN;
304  }
305 
306  #ifdef KINETIC_LOG_SOCKET_OPERATIONS
307  LOGF("Received %zd of %zd bytes requested", dest->bytesUsed, len);
308  #endif
309 
310  return KINETIC_STATUS_SUCCESS;
311 }

313 KineticStatus KineticSocket_ReadProtobuf(int socket, KineticPDU* pdu)
314 {
315  size_t bytesToRead = pdu->header.protobufLength;
316  #ifdef KINETIC_LOG_SOCKET_OPERATIONS
317  LOGF("Reading %zd bytes of protobuf", bytesToRead);
318  #endif
319 
320  uint8_t* packed = (uint8_t*)malloc(bytesToRead);
321  if (packed == NULL) {
322  LOG("Failed allocating memory for protocol buffer");
323  return KINETIC_STATUS_MEMORY_ERROR;
324  }
325 
326  ByteBuffer recvBuffer = ByteBuffer_Create(packed, bytesToRead);
327  KineticStatus status = KineticSocket_Read(socket, &recvBuffer, bytesToRead);
328 
329  if (status != KINETIC_STATUS_SUCCESS) {
330  LOG("Protobuf read failed!");
331  free(packed);
332  return status;
333  }
334  else {
335  pdu->proto = KineticProto__unpack(
336  NULL, recvBuffer.bytesUsed, recvBuffer.array.data);
337  }
338 
339  free(packed);
340 
341  if (pdu->proto == NULL) {
342  pdu->protobufDynamicallyExtracted = false;
343  LOG("Error unpacking incoming Kinetic protobuf message!");
344  return KINETIC_STATUS_DATA_ERROR;
345  }
346  else {
347  pdu->protobufDynamicallyExtracted = true;
348  #ifdef KINETIC_LOG_SOCKET_OPERATIONS
349  LOG("Protobuf unpacked successfully!");
350  #endif
351  return KINETIC_STATUS_SUCCESS;
352  }
353 }

355 KineticStatus KineticSocket_Write(int socket, ByteBuffer* src)
356 {
357  #ifdef KINETIC_LOG_SOCKET_OPERATIONS
358  LOGF("Writing %zu bytes to socket...", src->bytesUsed);
359  #endif
360  for (unsigned int bytesSent = 0; bytesSent < src->bytesUsed;) {
361  int bytesRemaining = src->bytesUsed - bytesSent;
362  int status = write(socket, &src->array.data[bytesSent], bytesRemaining);
363  if (status == -1 &&
364  ((errno == EINTR) || (errno == EAGAIN) || (errno == EWOULDBLOCK))) {
365  #ifdef KINETIC_LOG_SOCKET_OPERATIONS
366  LOG("Write interrupted. retrying...");
367  #endif
368  continue;
369  }
370  else if (status <= 0) {
371  LOGF("Failed to write to socket! status=%d, errno=%d\n", status, errno);
372  return KINETIC_STATUS_SOCKET_ERROR;
373  }
374  else {
375  bytesSent += status;
376  #ifdef KINETIC_LOG_SOCKET_OPERATIONS
377  LOGF("Wrote %d bytes (%d of %zu sent)", status, bytesSent, src->bytesUsed);
378  #endif
379  }
380  }
381 
382  #ifdef KINETIC_LOG_SOCKET_OPERATIONS
383  LOG("Socket write completed successfully");
384  #endif
385 
386  return KINETIC_STATUS_SUCCESS;
387 }
