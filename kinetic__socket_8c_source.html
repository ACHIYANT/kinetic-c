57 bool KineticSocket_HaveInput(int fd, double timeout)
58 {
59  int status;
60  fd_set fds;
61  struct timeval tv;
62  FD_ZERO(&fds);
63  FD_SET(fd, &fds);
64  tv.tv_sec = (long)timeout; // cast needed for C++
65  tv.tv_usec = (long)((timeout - tv.tv_sec) * 1000000); // 'suseconds_t'
66 
67  while (1)
68  {
69  if (!(status = select(fd + 1, &fds, 0, 0, &tv)))
70  {
71  return false;
72  }
73  else if (status > 0 && FD_ISSET(fd, &fds))
74  {
75  return true;
76  }
77  else if (status > 0)
78  {
79  LOG("I am confused!!!");
80  }
81  else if (errno != EINTR)
82  {
83  LOG("Failed select upon flush!"); // tbd EBADF: man page "an error has occurred"
84  }
85  }

113 bool KineticSocket_CloseSocket(int fd)
114 {
115  if (fd >= 0)
116  {
117  char msg[64];
118  int err;
119 
120  LOG("KineticSocket_CloseSocket");
121 
122  if (!KineticSocket_FlushSocketBeforeClose(fd))
123  {
124  LOG("Failed to gracefully close socket!");
125  }
126  else
127  {
128  LOG("Socket flushed successfully");
129  }
130 
131  err = KineticSocket_GetSocketError(fd); // first clear any errors, which can cause close to fail
132  if (err)
133  {
134  sprintf(msg, "Socket error had occurred: err=%d", err);
135  LOG(msg);
136  }
137  else
138  {
139  LOG("Socket errors cleared successfully!");
140  }
141 
142  if (shutdown(fd, SHUT_RDWR) < 0) // secondly, terminate the 'reliable' delivery
143  {
144  if (errno != ENOTCONN && errno != EINVAL) // SGI causes EINVAL
145  {
146  sprintf(msg, "Socket error occurred! errno=%d", errno);
147  return false;
148  }
149 
150  if (close(fd) < 0) // finally call close()
151  {
152  LOG("Failed closing socket!");
153  return false;
154  }
155  else
156  {
157  LOG("Succeeded closing socket");
158  }
159  }
160  else
161  {
162  LOG("Call to shutdown socket failed!");
163  return false;
164  }
165  }
166 
167  return true;
168 }

171 {
172  char port_str[32];

174  struct addrinfo* ai_result = NULL;
175  struct addrinfo* ai = NULL;
176  socket99_result result;
177 
178  // Setup server address info
179  socket99_config cfg = {
180  .host = host,
181  .port = port,
182  .nonblocking = !blocking,
183  };
184  sprintf(port_str, "%d", port);
185 
186  // Open socket
187  LOGF("\nConnecting to %s:%d", host, port);
188  if (!socket99_open(&cfg, &result))
189  {
190  LOGF("Failed to open socket connection with host: status %d, errno %d\n",
191  result.status, result.saved_errno);
192  return -1;
193  }
194 
195  // Configure the socket
196  socket99_set_hints(&cfg, &hints);
197  if (getaddrinfo(cfg.host, port_str, &hints, &ai_result) != 0)
198  {
199  LOGF("Failed to get socket address info: errno %d\n", errno);
200  close(result.fd);
201  return -1;
202  }
203 
204  for (ai = ai_result; ai != NULL; ai = ai->ai_next)
205  {
206  // OSX won't let us set close-on-exec when creating the socket, so set it separately
207  int current_fd_flags = fcntl(result.fd, F_GETFD);
208  if (current_fd_flags == -1)
209  {
210  LOG("Failed to get socket fd flags");
211  close(result.fd);
212  continue;
213  }
214  if (fcntl(result.fd, F_SETFD, current_fd_flags | FD_CLOEXEC) == -1)
215  {
216  LOG("Failed to set socket close-on-exit");
217  close(result.fd);
218  continue;
219  }
220 
221  // On BSD-like systems we can set SO_NOSIGPIPE on the socket to prevent it from sending a
222  // PIPE signal and bringing down the whole application if the server closes the socket
223  // forcibly
224 // #if defined(SO_NOSIGPIPE) && !defined(__APPLE__)
225 // {
226 // int set = 1;
227 // int setsockopt_result = setsockopt(socket_fd, SOL_SOCKET, SO_NOSIGPIPE, &set, sizeof(set));

231 // LOG("Failed to set SO_NOSIGPIPE on socket");
232 // close(result.fd);
233 // continue;
234 // }
235 // }
236 // #endif
237 
238  break;
239  }
240

242 
243  if (ai == NULL)
244  {
245  // we went through all addresses without finding one we could bind to
246  LOGF("Could not connect to %s:%d", host, port);
247  return -1;

276 bool KineticSocket_Read(int socketDescriptor, void* buffer, size_t length)
277 {
278  size_t count;
279  uint8_t* data = (uint8_t*)buffer;
280 
281  for (count = 0; count < length; )
282  {
283  int status;
284  fd_set readSet;
285  struct timeval timeout;
286 
287  // Time out after 5 seconds
288  timeout.tv_sec = 5;
289  timeout.tv_usec = 0;
290 
291  FD_ZERO(&readSet);
292  FD_SET(socketDescriptor, &readSet);
293  status = select(socketDescriptor+1, &readSet, NULL, NULL, &timeout);
294 
295  if (status < 0) // Error occurred
296  {
297  char msg[128];
298  sprintf(msg, "Failed waiting to read from socket! status=%d, errno=%d\n", status, errno);
299  LOG(msg);
300  return false;
301  }
302  else if (status == 0) // Timeout occurred
303  {
304  LOG("Timed out waiting for socket data to arrive!");
305  return false;
306  }
307  else if (status > 0) // Data available to read
308  {
309  // The socket is ready for reading
310  status = read(socketDescriptor, &((uint8_t*)buffer)[count], length - count);
311  if (status == -1 && errno == EINTR)
312  {
313  continue;
314  }
315  else if (status <= 0)
316  {
317  char msg[128];
318  sprintf(msg, "Failed to read from socket! status=%d, errno=%d\n", status, errno);
319  LOG(msg);
320  return false;
321  }
322  else
323  {
324  count += status;
325  }
326  }
327  }
328 
329  return true;
330 }
