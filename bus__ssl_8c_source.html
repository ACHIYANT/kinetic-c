88 static bool init_client_SSL_CTX(SSL_CTX **ctx_out) {
89  SSL_CTX *ctx = NULL;
90  assert(ctx_out);
91 
92  /* Create TLS context */
93  const SSL_METHOD *method = NULL;
94 
95  #if KINETIC_USE_TLS_1_2
96  method = TLSv1_2_client_method();
97  #else
98  method = TLSv1_1_client_method();
99  #endif
100 
101  assert(method);
102  ctx = SSL_CTX_new(method);
103  if (ctx == NULL) {
104  ERR_print_errors_fp(stderr);
105  return false;
106  }
107 
108  disable_SSL_compression();
109  disable_known_bad_ciphers(ctx);
110  *ctx_out = ctx;
111  return true;
112 }

131 static bool do_blocking_connection(struct bus *b, connection_info *ci) {
132  BUS_LOG_SNPRINTF(b, 2, LOG_SOCKET_REGISTERED, b->udata, 128,
133  "SSL_Connect handshake for socket %d", ci->fd);
134 
135  struct pollfd fds[1];
136  fds[0].fd = ci->fd;
137  fds[0].events = POLLOUT;
138 
139  bool connected = false;
140  size_t elapsed = 0;
141 
142  while (!connected) {
143  int pres = poll(fds, 1, TIMEOUT_MSEC);
144  BUS_LOG_SNPRINTF(b, 5, LOG_SOCKET_REGISTERED, b->udata, 128,
145  "SSL_Connect handshake for socket %d, poll res %d", ci->fd, pres);
146 
147  if (pres < 0) {
148  if (util_is_resumable_io_error(errno)) {
149  errno = 0;
150  } else {
151  /* */
152  assert(false);
153  }
154  } else if (pres > 0) {
155  if (fds[0].revents & (POLLOUT | POLLIN)) {
156  int connect_res = SSL_connect(ci->ssl);
157  BUS_LOG_SNPRINTF(b, 5, LOG_SOCKET_REGISTERED, b->udata, 128,
158  "socket %d: connect_res %d", ci->fd, connect_res);
159 
160  if (connect_res == 1) {
161  BUS_LOG_SNPRINTF(b, 5, LOG_SOCKET_REGISTERED, b->udata, 128,
162  "socket %d: successfully connected", ci->fd);
163  connected = true;
164  } else if (connect_res < 0) {
165  int reason = SSL_get_error(ci->ssl, connect_res);
166 
167  switch (reason) {
168  case SSL_ERROR_WANT_WRITE:
169  BUS_LOG(b, 4, LOG_SOCKET_REGISTERED, "WANT_WRITE", b->udata);
170  fds[0].events = POLLOUT;
171  break;
172 
173  case SSL_ERROR_WANT_READ:
174  BUS_LOG(b, 4, LOG_SOCKET_REGISTERED, "WANT_READ", b->udata);
175  fds[0].events = POLLIN;
176  break;
177 
178  case SSL_ERROR_SYSCALL:
179  {
180  if (util_is_resumable_io_error(errno)) {
181  errno = 0;
182  break;
183  } else {
184  unsigned long errval = ERR_get_error();
185  char ebuf[256];
186  BUS_LOG_SNPRINTF(b, 5, LOG_SOCKET_REGISTERED, b->udata, 128,
187  "socket %d: ERROR -- %s", ci->fd, ERR_error_string(errval, ebuf));
188  }
189  }
190  break;
191  default:
192  {
193  unsigned long errval = ERR_get_error();
194  char ebuf[256];
195  BUS_LOG_SNPRINTF(b, 5, LOG_SOCKET_REGISTERED, b->udata, 128,
196  "socket %d: ERROR %d -- %s", ci->fd, reason, ERR_error_string(errval, ebuf));
197  assert(false);
198  }
199  }
200 
201  } else {
202  BUS_LOG_SNPRINTF(b, 5, LOG_SOCKET_REGISTERED, b->udata, 128,
203  "socket %d: unknown state, setting event bask to (POLLIN | POLLOUT)",
204  ci->fd);
205  fds[0].events = (POLLIN | POLLOUT);
206  }
207  } else if (fds[0].revents & POLLHUP) {
208  BUS_LOG_SNPRINTF(b, 2, LOG_SOCKET_REGISTERED, b->udata, 128,
209  "SSL_Connect: HUP on %d", ci->fd);
210  return false;
211  } else if (fds[0].revents & POLLERR) {
212  BUS_LOG_SNPRINTF(b, 2, LOG_SOCKET_REGISTERED, b->udata, 128,
213  "SSL_Connect: ERR on %d", ci->fd);
214  return false;
215  }
216  } else {
217  BUS_LOG(b, 4, LOG_SOCKET_REGISTERED, "poll timeout", b->udata);
218  elapsed += TIMEOUT_MSEC;
219  if (elapsed > MAX_TIMEOUT) {
220  BUS_LOG(b, 2, LOG_SOCKET_REGISTERED, "timed out", b->udata);
221  return false;
222  }
223  }
224  }
225 
226  return connected;
227 }
