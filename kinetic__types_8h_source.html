107 // Kinetic Device Client Connection
108 typedef struct _KineticConnection
109 {
110  bool connected;
111  bool nonBlocking;
112  int port;
113  int socketDescriptor;
114  int64_t connectionID;
115  char host[HOST_NAME_MAX];
116 
117  // Optional field - default value is 0
118  // The version number of this cluster definition. If this is not equal to
119  // the value on the device, the request is rejected and will return a
120  // `VERSION_FAILURE` `statusCode` in the `Status` message.
121  int64_t clusterVersion;
122 
123  // Required field
124  // The identity associated with this request. See the ACL discussion above.
125  // The Kinetic Device will use this identity value to lookup the
126  // HMAC key (shared secret) to verify the HMAC.
127  int64_t identity;
128 
129  // Required field
130  // This is the identity's HMAC Key. This is a shared secret between the
131  // client and the device, used to sign requests.
132  uint8_t keyData[KINETIC_MAX_KEY_LEN];
133  ByteArray key;
134 
135  // Required field
136  // A monotonically increasing number for each request in a TCP connection.
137  int64_t sequence;
138 } KineticConnection;
139 #define KINETIC_CONNECTION_INIT(_con, _id, _key) { \
140  (*_con) = (KineticConnection) { \
141  .socketDescriptor = -1, \
142  .connectionID = time(NULL), \
143  .identity = (_id), \
144  .sequence = 0, \
145  }; \
146  (*_con).key = (ByteArray){.data = (*_con).keyData, .len = (_key).len}; \
147  if ((_key).data != NULL && (_key).len > 0) { \
148  memcpy((_con)->keyData, (_key).data, (_key).len); } \
149 }
