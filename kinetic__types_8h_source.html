153 #define KINETIC_SESSION_INIT(_session, _host, _clusterVersion, _identity, _hmacKey) { \
154  (*_session).config = (KineticSessionConfig) { \
155  .port = KINETIC_PORT, \
156  .clusterVersion = (_clusterVersion), \
157  .identity = (_identity), \
158  .hmacKey = {.data = (_session)->config.keyData, .len = (_hmacKey).len}, \
159  }; \
160  strcpy((_session)->config.host, (_host)); \
161  memcpy((_session)->config.hmacKey.data, (_hmacKey).data, (_hmacKey).len); \
162 }
