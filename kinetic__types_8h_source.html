138 typedef struct _KineticExchange
139 {
140  // Defaults to false, since clusterVersion is optional
141  // Set to true to enable clusterVersion for the exchange
142  bool has_clusterVersion;
143 
144  // Optional field - default value is 0
145  // The version number of this cluster definition. If this is not equal to
146  // the value on the device, the request is rejected and will return a
147  // `VERSION_FAILURE` `statusCode` in the `Status` message.
148  int64_t clusterVersion;
149 
150  // Required field
151  // The identity associated with this request. See the ACL discussion above.
152  // The Kinetic Device will use this identity value to lookup the
153  // HMAC key (shared secret) to verify the HMAC.
154  int64_t identity;
155 
156  // Required field
157  // This is the identity's HMAC Key. This is a shared secret between the
158  // client and the device, used to sign requests.
159  bool has_key;
160  size_t keyLength;
161  char key[KINETIC_MAX_KEY_LEN+1];
162 
163  // Required field
164  // A unique number for this connection between the source and target.
165  // On the first request to the drive, this should be the time of day in
166  // seconds since 1970. The drive can change this number and the client must
167  // continue to use the new number and the number must remain constant
168  // during the session
169  int64_t connectionID;
170 
171  // Required field
172  // A monotonically increasing number for each request in a TCP connection.
173  int64_t sequence;
174 
175  // Associated Kinetic connection
176  KineticConnection* connection;
177 } KineticExchange;

179 #define KINETIC_EXCHANGE_INIT(exchg, id, k, klen, con) { \
180  memset((exchg), 0, sizeof(KineticExchange)); \
181  (exchg)->identity = id; \
182  if (k != NULL && klen > 0) \
183  { \
184  memcpy((exchg)->key, k, klen); \
185  (exchg)->key[klen] = '\0'; \

187  (exchg)->has_key = true; \
188  } \
189  (exchg)->connection = con; \
190  (exchg)->sequence = -1; \
191 }
