99 typedef struct _KineticSession {
100  // Host name/IP address of Kinetic Device
101  char host[HOST_NAME_MAX];
102 
103  // Port for Kinetic Device session
104  int port;
105 
106  // Set to true to enable non-blocking/asynchronous I/O
107  bool nonBlocking;
108 
109  // The version number of this cluster definition. If this is not equal to
110  // the value on the Kinetic Device, the request is rejected and will return
111  // `KINETIC_STATUS_VERSION_FAILURE`
112  int64_t clusterVersion;
113 
114  // The identity associated with this request. See the ACL discussion above.
115  // The Kinetic Device will use this identity value to lookup the
116  // HMAC key (shared secret) to verify the HMAC.
117  int64_t identity;
118 
119  // This is the identity's HMAC Key. This is a shared secret between the
120  // client and the device, used to sign requests.
121  uint8_t keyData[KINETIC_MAX_KEY_LEN];
122  ByteArray hmacKey;
123 } KineticSession;

125 #define KINETIC_SESSION_INIT(_session, _host, _clusterVersion, _identity, _hmacKey) { \
126  *(_session) = (KineticSession) { \
127  .port = KINETIC_PORT, \
128  .clusterVersion = (_clusterVersion), \
129  .identity = (_identity), \
130  .hmacKey = {.data = (_session)->keyData, .len = (_hmacKey).len}, \
131  }; \
132  strcpy((_session)->host, (_host)); \
133  memcpy((_session)->hmacKey.data, (_hmacKey).data, (_hmacKey).len); \
134 }

137 typedef enum {
138  KINETIC_STATUS_INVALID = -1, // Status not available (no reponse/status available)
139  KINETIC_STATUS_NOT_ATTEMPTED = 0, // No operation has been attempted

143  KINETIC_STATUS_HOST_EMPTY, // Host was empty in request
144  KINETIC_STATUS_HMAC_EMPTY, // HMAC key is empty or NULL
145  KINETIC_STATUS_NO_PDUS_AVAVILABLE, // All PDUs for the session have been allocated
146  KINETIC_STATUS_DEVICE_BUSY, // Device busy (retry later)
147  KINETIC_STATUS_CONNECTION_ERROR, // No connection/disconnected
148  KINETIC_STATUS_INVALID_REQUEST, // Something about the request is invalid
149  KINETIC_STATUS_OPERATION_INVALID, // Operation was invalid
150  KINETIC_STATUS_OPERATION_FAILED, // Device reported an operation error
151  KINETIC_STATUS_OPERATION_TIMEDOUT, // Device did not respond to the operation in time
152  KINETIC_STATUS_CLUSTER_MISMATCH, // Specified cluster version does not match device
153  KINETIC_STATUS_VERSION_MISMATCH, // The specified object version info for a PUT/GET do not match stored object
154  KINETIC_STATUS_DATA_ERROR, // Device reported data error, no space or HMAC failure
155  KINETIC_STATUS_NOT_FOUND, // The requested object does not exist
156  KINETIC_STATUS_BUFFER_OVERRUN, // One or more of byte buffers did not fit all data
157  KINETIC_STATUS_MEMORY_ERROR, // Failed allocating/deallocating memory
158  KINETIC_STATUS_SOCKET_TIMEOUT, // A timeout occurred while waiting for a socket operation
159  KINETIC_STATUS_SOCKET_ERROR, // An I/O error occurred during a socket operation
160  KINETIC_STATUS_COUNT // Number of status codes in KineticStatusDescriptor
161 } KineticStatus;
