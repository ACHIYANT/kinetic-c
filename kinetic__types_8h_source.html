99 typedef struct _KineticSession {
100  // Host name/IP address of Kinetic Device
101  char host[HOST_NAME_MAX];
102 
103  // Port for Kinetic Device session
104  int port;
105 
106  // Set to true to enable non-blocking/asynchronous I/O
107  bool nonBlocking;
108 
109  // The version number of this cluster definition. If this is not equal to
110  // the value on the Kinetic Device, the request is rejected and will return
111  // `KINETIC_STATUS_VERSION_FAILURE`
112  int64_t clusterVersion;
113 
114  // The identity associated with this request. See the ACL discussion above.
115  // The Kinetic Device will use this identity value to lookup the
116  // HMAC key (shared secret) to verify the HMAC.
117  int64_t identity;
118 
119  // This is the identity's HMAC Key. This is a shared secret between the
120  // client and the device, used to sign requests.
121  uint8_t keyData[KINETIC_MAX_KEY_LEN];
122  ByteArray hmacKey;
123 } KineticSession;

125 #define KINETIC_SESSION_INIT(_session, _host, _clusterVersion, _identity, _hmacKey) { \
126  *(_session) = (KineticSession) { \

128  .clusterVersion = (_clusterVersion), \
129  .identity = (_identity), \
130  .hmacKey = {.data = (_session)->keyData, .len = (_hmacKey).len}, \
