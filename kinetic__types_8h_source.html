98 typedef struct _KineticSessionConfig {
99  // Host name/IP address of Kinetic Device
100  char host[HOST_NAME_MAX];
101 
102  // Port for Kinetic Device session
103  int port;
104 
105  // The version number of this cluster definition. If this is not equal to
106  // the value on the Kinetic Device, the request is rejected and will return
107  // `KINETIC_STATUS_VERSION_FAILURE`
108  int64_t clusterVersion;
109 
110  // The identity associated with this request. See the ACL discussion above.
111  // The Kinetic Device will use this identity value to lookup the
112  // HMAC key (shared secret) to verify the HMAC.
113  int64_t identity;
114 
115  // This is the identity's HMAC Key. This is a shared secret between the
116  // client and the device, used to sign requests.
117  uint8_t keyData[KINETIC_MAX_KEY_LEN];

119 } KineticSessionConfig;

134 #define KINETIC_SESSION_INIT(_session, _host, _clusterVersion, _identity, _hmacKey) { \
135  (*_session).config = (KineticSessionConfig) { \
136  .port = KINETIC_PORT, \
137  .clusterVersion = (_clusterVersion), \
138  .identity = (_identity), \
139  .hmacKey = {.data = (_session)->config.keyData, .len = (_hmacKey).len}, \
140  }; \
141  strcpy((_session)->config.host, (_host)); \
142  memcpy((_session)->config.hmacKey.data, (_hmacKey).data, (_hmacKey).len); \
143 }

146 typedef enum {
147  KINETIC_STATUS_INVALID = -1, // Status not available (no reponse/status available)
148  KINETIC_STATUS_NOT_ATTEMPTED = 0, // No operation has been attempted

154  KINETIC_STATUS_NO_PDUS_AVAVILABLE, // All PDUs for the session have been allocated
155  KINETIC_STATUS_DEVICE_BUSY, // Device busy (retry later)
156  KINETIC_STATUS_CONNECTION_ERROR, // No connection/disconnected
157  KINETIC_STATUS_INVALID_REQUEST, // Something about the request is invalid
158  KINETIC_STATUS_OPERATION_INVALID, // Operation was invalid
159  KINETIC_STATUS_OPERATION_FAILED, // Device reported an operation error
160  KINETIC_STATUS_OPERATION_TIMEDOUT, // Device did not respond to the operation in time
161  KINETIC_STATUS_CLUSTER_MISMATCH, // Specified cluster version does not match device
162  KINETIC_STATUS_VERSION_MISMATCH, // The specified object version info for a PUT/GET do not match stored object
163  KINETIC_STATUS_DATA_ERROR, // Device reported data error, no space or HMAC failure
164  KINETIC_STATUS_NOT_FOUND, // The requested object does not exist
165  KINETIC_STATUS_BUFFER_OVERRUN, // One or more of byte buffers did not fit all data
166  KINETIC_STATUS_MEMORY_ERROR, // Failed allocating/deallocating memory
167  KINETIC_STATUS_SOCKET_TIMEOUT, // A timeout occurred while waiting for a socket operation
168  KINETIC_STATUS_SOCKET_ERROR, // An I/O error occurred during a socket operation
169  KINETIC_STATUS_MISSING_KEY, // An operation is missing a required key
170  KINETIC_STATUS_MISSING_VALUE_BUFFER,// An operation is missing a required value buffer
171  KINETIC_STATUS_COUNT // Number of status codes in KineticStatusDescriptor
172 } KineticStatus;
