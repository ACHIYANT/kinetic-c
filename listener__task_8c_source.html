31 void *ListenerTask_MainLoop(void *arg) {
32  listener *self = (listener *)arg;
33  assert(self);
34  struct bus *b = self->bus;
35  struct timeval tv;
36 
37  if (!util_timestamp(&tv, true)) {
38  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
39  "timestamp failure: %d", errno);
40  }
41 
42  time_t last_sec = tv.tv_sec;
43 
44  /* The listener thread has full control over its execution -- the
45  * only thing other threads can do is reserve messages from l->msgs,
46  * write commands into them, and then commit them by writing their
47  * msg->id into the incoming command ID pipe. All cross-thread
48  * communication is managed at the command interface, so it doesn't
49  * need any internal locking. */
50 
51  while (self->shutdown_notify_fd == LISTENER_NO_FD) {
52  if (!util_timestamp(&tv, true)) {
53  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
54  "timestamp failure: %d", errno);
55  }
56  time_t cur_sec = tv.tv_sec;
57  if (cur_sec != last_sec) {
58  tick_handler(self);
59  last_sec = cur_sec;
60  }
61 
62  int delay = (self->is_idle ? -1 : TIMEOUT_DELAY);
63  int res = poll(self->fds, self->tracked_fds + INCOMING_MSG_PIPE, delay);
64  BUS_LOG_SNPRINTF(b, (res == 0 ? 6 : 4), LOG_LISTENER, b->udata, 64,
65  "poll res %d", res);
66 
67  if (res < 0) {
68  if (util_is_resumable_io_error(errno)) {
69  errno = 0;
70  } else {
71  /* unrecoverable poll error -- FD count is bad
72  * or FDS is a bad pointer. */
73  BUS_ASSERT(b, b->udata, false);
74  }
75  } else if (res > 0) {
76  ListenerCmd_CheckIncomingMessages(self, &res);
77  ListenerIO_AttemptRecv(self, res);
78  } else {
79  /* nothing to do */
80  }
81  }
82 
83  BUS_LOG(b, 3, LOG_LISTENER, "shutting down", b->udata);
84 
85  if (self->tracked_fds > 0) {
86  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
87  "%d connections still open!", self->tracked_fds);
88  }
89 
90  ListenerCmd_NotifyCaller(self->shutdown_notify_fd);
91  self->shutdown_notify_fd = LISTENER_SHUTDOWN_COMPLETE_FD;
92  return NULL;
93 }

95 static void tick_handler(listener *l) {
96  struct bus *b = l->bus;
97  bool any_work = false;
98 
99  BUS_LOG_SNPRINTF(b, 2, LOG_LISTENER, b->udata, 128,
100  "tick... %p: %d of %d msgs in use, %d of %d rx_info in use, %d tracked_fds",
101  (void*)l, l->msgs_in_use, MAX_QUEUE_MESSAGES,
102  l->rx_info_in_use, MAX_PENDING_MESSAGES, l->tracked_fds);
103 
104  if (b->log_level > 5) { /* dump msg table types */
105  for (int i = 0; i < l->msgs_in_use; i++) {
106  printf(" -- msg %d: type %d\n", i, l->msgs[i].type);
107  }
108  }
109 
110  if (b->log_level > 5 || 0) { ListenerTask_DumpRXInfoTable(l); }
111 
112  for (int i = 0; i <= l->rx_info_max_used; i++) {
113  rx_info_t *info = &l->rx_info[i];
114 
115  switch (info->state) {
116  case RIS_INACTIVE:
117  break;
118  case RIS_HOLD:
119  any_work = true;
120  /* Check timeout */
121  if (info->timeout_sec == 1) {
122  struct timeval tv;
123  if (!util_timestamp(&tv, false)) {
124  BUS_LOG(b, 0, LOG_LISTENER,
125  "gettimeofday failure in tick_handler!", b->udata);
126  continue;
127  }
128 
129  /* never got a response, but we don't have the callback
130  * either -- the client will notify about the timeout. */
131  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 128,
132  "timing out hold info %p -- <fd:%d, seq_id:%lld> at (%ld.%ld)",
133  (void*)info, info->u.hold.fd, (long long)info->u.hold.seq_id,
134  (long)tv.tv_sec, (long)tv.tv_usec);
135 
136  ListenerTask_ReleaseRXInfo(l, info);
137  } else {
138  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
139  "decrementing countdown on info %p [%u]: %ld",
140  (void*)info, info->id, info->timeout_sec - 1);
141  info->timeout_sec--;
142  }
143  break;
144  case RIS_EXPECT:
145  any_work = true;
146  if (info->u.expect.error == RX_ERROR_READY_FOR_DELIVERY) {
147  BUS_LOG(b, 4, LOG_LISTENER,
148  "retrying RX event delivery", b->udata);
149  retry_delivery(l, info);
150  } else if (info->u.expect.error == RX_ERROR_DONE) {
151  BUS_LOG_SNPRINTF(b, 4, LOG_LISTENER, b->udata, 64,
152  "cleaning up completed RX event at info %p", (void*)info);
153  clean_up_completed_info(l, info);
154  } else if (info->u.expect.error != RX_ERROR_NONE) {
155  BUS_LOG_SNPRINTF(b, 1, LOG_LISTENER, b->udata, 64,
156  "notifying of rx failure -- error %d (info %p)",
157  info->u.expect.error, (void*)info);
158  ListenerTask_NotifyMessageFailure(l, info, BUS_SEND_RX_FAILURE);
159  } else if (info->timeout_sec == 1) {
160  struct timeval tv;
161  if (!util_timestamp(&tv, false)) {
162  BUS_LOG(b, 0, LOG_LISTENER,
163  "gettimeofday failure in tick_handler!", b->udata);
164  continue;
165  }
166  struct boxed_msg *box = info->u.expect.box;
167  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 256 + 64,
168  "notifying of rx failure -- timeout (info %p) -- "
169  "<fd:%d, seq_id:%lld>, from time (queued:%ld.%ld) to (sent:%ld.%ld) to (now:%ld.%ld)",
170  (void*)info, box->fd, (long long)box->out_seq_id,
171  (long)box->tv_send_start.tv_sec, (long)box->tv_send_start.tv_usec,
172  (long)box->tv_send_done.tv_sec, (long)box->tv_send_done.tv_usec,
173  (long)tv.tv_sec, (long)tv.tv_usec);
174 
175  ListenerTask_NotifyMessageFailure(l, info, BUS_SEND_RX_TIMEOUT);
176  } else {
177  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
178  "decrementing countdown on info %p [%u]: %ld",
179  (void*)info, info->id, info->timeout_sec - 1);
180  info->timeout_sec--;
181  }
182  break;
183  default:
184  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
185  "match fail %d on line %d", info->state, __LINE__);
186  BUS_ASSERT(b, b->udata, false);
187  }
188  }
189  if (!any_work) { l->is_idle = true; }
190 }

218 static void retry_delivery(listener *l, rx_info_t *info) {
219  struct bus *b = l->bus;
220  BUS_ASSERT(b, b->udata, info->state == RIS_EXPECT);
221  BUS_ASSERT(b, b->udata, info->u.expect.error == RX_ERROR_READY_FOR_DELIVERY);
222  BUS_ASSERT(b, b->udata, info->u.expect.box);
223 
224  struct boxed_msg *box = info->u.expect.box;
225  info->u.expect.box = NULL; /* release */
226  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 128,
227  "releasing box %p at line %d", (void*)box, __LINE__);
228  BUS_ASSERT(b, b->udata, box->result.status == BUS_SEND_SUCCESS);
229 
230  size_t backpressure = 0;
231  if (bus_process_boxed_message(l->bus, box, &backpressure)) {
232  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 128,
233  "successfully delivered box %p (seq_id %lld) from info %d at line %d (retry)",
234  (void*)box, (long long)box->out_seq_id, info->id, __LINE__);
235  info->u.expect.error = RX_ERROR_DONE;
236  ListenerTask_ReleaseRXInfo(l, info);
237  } else {
238  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 128,
239  "returning box %p at line %d", (void*)box, __LINE__);
240  info->u.expect.box = box; /* retry in tick_handler */
241  }
242 
243  observe_backpressure(l, backpressure);
244 }

246 static void clean_up_completed_info(listener *l, rx_info_t *info) {
247  struct bus *b = l->bus;
248  BUS_ASSERT(b, b->udata, info->state == RIS_EXPECT);
249  BUS_ASSERT(b, b->udata, info->u.expect.error == RX_ERROR_DONE);
250  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 128,
251  "info %p, box is %p at line %d", (void*)info,
252  (void*)info->u.expect.box, __LINE__);
253  size_t backpressure = 0;
254  if (info->u.expect.box) {
255  struct boxed_msg *box = info->u.expect.box;
256  if (box->result.status != BUS_SEND_SUCCESS) {
257  printf("*** info %d: info->timeout %ld\n",
258  info->id, info->timeout_sec);
259  printf(" info->error %d\n", info->u.expect.error);
260  printf(" info->box == %p\n", (void*)box);
261  printf(" info->box->result.status == %d\n", box->result.status);
262  printf(" info->box->fd %d\n", box->fd);
263  printf(" info->box->out_seq_id %lld\n", (long long)box->out_seq_id);
264  printf(" info->box->out_msg %p\n", (void*)box->out_msg);
265 
266  }
267  BUS_ASSERT(b, b->udata, box->result.status == BUS_SEND_SUCCESS);
268  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 128,
269  "releasing box %p at line %d", (void*)box, __LINE__);
270  info->u.expect.box = NULL; /* release */
271  if (bus_process_boxed_message(l->bus, box, &backpressure)) {
272  ListenerTask_ReleaseRXInfo(l, info);
273  } else {
274  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 128,
275  "returning box %p at line %d", (void*)box, __LINE__);
276  info->u.expect.box = box; /* retry in tick_handler */
277  }
278  } else { /* already processed, just release it */
279  ListenerTask_ReleaseRXInfo(l, info);
280  }
281 
282  observe_backpressure(l, backpressure);
283 }

285 void ListenerTask_NotifyMessageFailure(listener *l,
286  rx_info_t *info, bus_send_status_t status) {
287  size_t backpressure = 0;
288  struct bus *b = l->bus;
289  BUS_ASSERT(b, b->udata, info->state == RIS_EXPECT);
290  BUS_ASSERT(b, b->udata, info->u.expect.box);
291 
292  info->u.expect.box->result.status = status;
293 
294  boxed_msg *box = info->u.expect.box;
295  info->u.expect.box = NULL;
296  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 128,
297  "releasing box %p at line %d", (void*)box, __LINE__);
298  if (bus_process_boxed_message(l->bus, box, &backpressure)) {
299  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 128,
300  "delivered box %p with failure message %d at line %d (info %p)",
301  (void*)box, status, __LINE__, (void*)info);
302  info->u.expect.error = RX_ERROR_DONE;
303  ListenerTask_ReleaseRXInfo(l, info);
304  } else {
305  /* Return to info, will be released on retry. */
306  info->u.expect.box = box;
307  }
308 
309  observe_backpressure(l, backpressure);
310 }

322 void ListenerTask_ReleaseRXInfo(struct listener *l, rx_info_t *info) {
323  struct bus *b = l->bus;
324  BUS_ASSERT(b, b->udata, info);
325  BUS_LOG_SNPRINTF(b, 5, LOG_LISTENER, b->udata, 128,
326  "releasing RX info %d (%p)", info->id, (void *)info);
327  BUS_ASSERT(b, b->udata, info->id < MAX_PENDING_MESSAGES);
328  BUS_ASSERT(b, b->udata, info == &l->rx_info[info->id]);
329 
330  switch (info->state) {
331  case RIS_HOLD:
332  if (info->u.hold.has_result) {
333  /* If we have a message that timed out, we need to free it,
334  * but don't know how. We should never get here, because it
335  * means the client finished sending the message, but the
336  * listener never got the handler callback. */
337 
338  if (info->u.hold.result.ok) {
339  void *msg = info->u.hold.result.u.success.msg;
340  int64_t seq_id = info->u.hold.result.u.success.seq_id;
341 
342  connection_info *ci = get_connection_info(l, info->u.hold.fd);
343  if (ci && b->unexpected_msg_cb) {
344  BUS_LOG_SNPRINTF(b, 1, LOG_LISTENER, b->udata, 128,
345  "CALLING UNEXPECTED_MSG_CB ON RESULT %p", (void *)&info->u.hold.result);
346  b->unexpected_msg_cb(msg, seq_id, b->udata, ci->udata);
347  } else {
348  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 128,
349  "LEAKING RESULT %p", (void *)&info->u.hold.result);
350  }
351  }
352  }
353  break;
354  case RIS_EXPECT:
355  BUS_ASSERT(b, b->udata, info->u.expect.error == RX_ERROR_DONE);
356  BUS_ASSERT(b, b->udata, info->u.expect.box == NULL);
357  break;
358  default:
359  case RIS_INACTIVE:
360  BUS_ASSERT(b, b->udata, false);
361  }
362 
363  /* Set to no longer active and push on the freelist. */
364  BUS_LOG_SNPRINTF(b, 5, LOG_LISTENER, b->udata, 128,
365  "releasing rx_info_t %d (%p), was %d",
366  info->id, (void *)info, info->state);
367 
368  BUS_ASSERT(b, b->udata, info->state != RIS_INACTIVE);
369  info->state = RIS_INACTIVE;
370  memset(&info->u, 0, sizeof(info->u));
371  info->next = l->rx_info_freelist;
372  l->rx_info_freelist = info;
373 
374  if (l->rx_info_max_used == info->id && info->id > 0) {
375  BUS_LOG_SNPRINTF(b, 5, LOG_LISTENER, b->udata, 128,
376  "rx_info_max_used--, from %d to %d",
377  l->rx_info_max_used, l->rx_info_max_used - 1);
378  while (l->rx_info[l->rx_info_max_used].state == RIS_INACTIVE) {
379  l->rx_info_max_used--;
380  if (l->rx_info_max_used == 0) { break; }
381  }
382  BUS_ASSERT(b, b->udata, l->rx_info_max_used < MAX_PENDING_MESSAGES);
383  }
384 
385  l->rx_info_in_use--;
386 }

388 void ListenerTask_ReleaseMsg(struct listener *l, listener_msg *msg) {
389  struct bus *b = l->bus;
390  BUS_ASSERT(b, b->udata, msg->id < MAX_QUEUE_MESSAGES);
391  msg->type = MSG_NONE;
392 
393  for (;;) {
394  listener_msg *fl = l->msg_freelist;
395  msg->next = fl;
396  if (ATOMIC_BOOL_COMPARE_AND_SWAP(&l->msg_freelist, fl, msg)) {
397  for (;;) {
398  int16_t miu = l->msgs_in_use;
399  if (ATOMIC_BOOL_COMPARE_AND_SWAP(&l->msgs_in_use, miu, miu - 1)) {
400  BUS_ASSERT(b, b->udata, miu >= 0);
401  BUS_LOG(b, 3, LOG_LISTENER, "Releasing msg", b->udata);
402  return;
403  }
404  }
405  }
406  }
407 }

427 void ListenerTask_AttemptDelivery(listener *l, struct rx_info_t *info) {
428  struct bus *b = l->bus;
429 
430  struct boxed_msg *box = info->u.expect.box;
431  info->u.expect.box = NULL; /* release */
432  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
433  "attempting delivery of %p", (void*)box);
434  BUS_LOG_SNPRINTF(b, 5, LOG_MEMORY, b->udata, 128,
435  "releasing box %p at line %d", (void*)box, __LINE__);
436 
437  bus_msg_result_t *result = &box->result;
438  result->status = BUS_SEND_SUCCESS;
439 
440  bus_unpack_cb_res_t unpacked_result;
441  switch (info->state) {
442  case RIS_EXPECT:
443  BUS_ASSERT(b, b->udata, info->u.expect.has_result);
444  unpacked_result = info->u.expect.result;
445  break;
446  default:
447  case RIS_HOLD:
448  case RIS_INACTIVE:
449  BUS_ASSERT(b, b->udata, false);
450  }
451 
452  BUS_ASSERT(b, b->udata, unpacked_result.ok);
453  int64_t seq_id = unpacked_result.u.success.seq_id;
454  void *opaque_msg = unpacked_result.u.success.msg;
455  result->u.response.seq_id = seq_id;
456  result->u.response.opaque_msg = opaque_msg;
457 
458  size_t backpressure = 0;
459  if (bus_process_boxed_message(b, box, &backpressure)) {
460  /* success */
461  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 256,
462  "successfully delivered box %p (seq_id:%lld), marking info %d as DONE",
463  (void*)box, (long long)seq_id, info->id);
464  info->u.expect.error = RX_ERROR_DONE;
465  BUS_LOG_SNPRINTF(b, 4, LOG_LISTENER, b->udata, 128,
466  "initial clean-up attempt for completed RX event at info +%d", info->id);
467  clean_up_completed_info(l, info);
468  info = NULL; /* drop out of scope, likely to be stale */
469  } else {
470  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 128,
471  "returning box %p at line %d", (void*)box, __LINE__);
472  info->u.expect.box = box; /* retry in tick_handler */
473  }
474  observe_backpressure(l, backpressure);
475 }

478  size_t cur = l->upstream_backpressure;

491 uint16_t ListenerTask_GetBackpressure(struct listener *l) {
492  uint16_t msg_fill_pressure = 0;
493 
494  if (l->msgs_in_use < 0.25 * MAX_QUEUE_MESSAGES) {
495  msg_fill_pressure = 0;
496  } else if (l->msgs_in_use < 0.5 * MAX_QUEUE_MESSAGES) {
497  msg_fill_pressure = MSG_BP_1QTR * 2 * l->msgs_in_use;
498  } else if (l->msgs_in_use < 0.75 * MAX_QUEUE_MESSAGES) {
499  msg_fill_pressure = MSG_BP_HALF * 10 * l->msgs_in_use;
500  } else {
501  msg_fill_pressure = MSG_BP_3QTR * 100 * l->msgs_in_use;
502  }
503 
504  uint16_t rx_info_fill_pressure = 0;
505  if (l->rx_info_in_use < 0.25 * MAX_PENDING_MESSAGES) {
506  rx_info_fill_pressure = 0;
507  } else if (l->rx_info_in_use < 0.5 * MAX_PENDING_MESSAGES) {
508  rx_info_fill_pressure = RX_INFO_BP_1QTR * l->rx_info_in_use;
509  } else if (l->rx_info_in_use < 0.75 * MAX_PENDING_MESSAGES) {
510  rx_info_fill_pressure = RX_INFO_BP_HALF * l->rx_info_in_use;
511  } else {
