111 static bus_sink_cb_res_t sink_cb(uint8_t *read_buf,
112  size_t read_size, void *socket_udata) {
113 
114  socket_info *si = (socket_info *)socket_udata;
115  assert(si);
116 
117  switch (si->state) {
118  case STATE_UNINIT:
119  {
120  assert(read_size == 0);
121  return reset_transfer(si);
122  }
123  case STATE_AWAITING_HEADER:
124  {
125  bool valid_header = true;
126  bool split_header = false;
127 
128  size_t header_rem = sizeof(prot_header_t) - si->used;
129  if (read_size > header_rem) {
130  printf("surplus read_size %zd\n", read_size);
131  printf("header_rem %zd (sizeof(prot_header_t) %zd)\n", header_rem, sizeof(prot_header_t));
132  assert(false);
133  } else if (read_size < sizeof(prot_header_t)) {
134  //printf("split header, %zd\n", read_size);
135  split_header = true;
136  }
137 
138  size_t copied = read_size;
139  if (copied > header_rem) { copied = header_rem; }
140 
141  memcpy(&si->buf[si->used], read_buf, copied);
142  si->used += copied;
143 
144  if (si->used < sizeof(prot_header_t)) {
145  bus_sink_cb_res_t res = {
146  .next_read = sizeof(prot_header_t) - si->used,
147  };
148  si->state = STATE_AWAITING_HEADER;
149  return res;
150  }
151 
152  assert(si->used == sizeof(prot_header_t));
153 
154  prot_header_t *header = (prot_header_t *)&si->buf[0];
155 
156  if (si->used < sizeof(prot_header_t)) {
157  printf("INVALID HEADER A: read_size %zd\n", si->used);
158  valid_header = false;
159  } else if (header->magic_number != MAGIC_NUMBER) {
160  printf("INVALID HEADER B: magic number 0x%08x\n", header->magic_number);
161  valid_header = false;
162  }
163 
164  if (valid_header) {
165  prot_header_t *header = (prot_header_t *)&si->buf[0];
166  si->cur_payload_size = header->size;
167  memcpy(&si->buf[si->used], read_buf + copied, read_size - copied);
168  si->used += read_size - copied;
169  bus_sink_cb_res_t res = {
170  .next_read = header->size,
171  };
172  si->state = STATE_AWAITING_BODY;
173  return res;
174  } else {
175  assert(false);
176  return reset_transfer(si);
177  }
178 
179  break;
180  }
181  case STATE_AWAITING_BODY:
182  {
183  assert(DEFAULT_BUF_SIZE - si->used >= read_size);
184  memcpy(&si->buf[si->used], read_buf, read_size);
185  si->used += read_size;
186  assert(si->used <= si->cur_payload_size + sizeof(prot_header_t));
187  size_t rem = si->cur_payload_size + sizeof(prot_header_t) - si->used;
188 
189  if (rem == 0) {
190  bus_sink_cb_res_t res = {
191  .next_read = sizeof(prot_header_t),
192  .full_msg_buffer = read_buf,
193  };
194  si->state = STATE_AWAITING_HEADER;
195  si->used = 0;
196  return res;
197  } else {
198  bus_sink_cb_res_t res = {
199  .next_read = rem,
200  };
201  return res;
202  }
203  }
204  default:
205  assert(false);
206  }
207 }

306 static void parse_args(int argc, char **argv, example_state *s) {
307  int a = 0;
308 
309  s->buf_size = DEFAULT_BUF_SIZE;
310 
311  while ((a = getopt(argc, argv, "b:l:h:s:v")) != -1) {
312  switch (a) {
313  case 'b': /* buffer size */
314  s->buf_size = atol(optarg);
315  break;
316  case 'l': /* low port */
317  s->port_low = atoi(optarg);
318  break;
319  case 'h': /* high port */
320  s->port_high = atoi(optarg);
321  break;
322  case 's': /* stop loop at sequence ID */
323  s->max_seq_id = atoi(optarg);
324  break;
325  case 'v': /* verbosity */
326  s->verbosity++;
327  break;
328  default:
329  fprintf(stderr, "illegal option: -- %c\n", a);
330  usage();
331  }
332  }
333 
334  if (s->port_low == 0) { s->port_low = s->port_high; }
335  if (s->port_high == 0) { s->port_high = s->port_low; }
336  if (s->port_high < s->port_low || s->port_low == 0) { usage(); }
337 
338  if (s->verbosity > 0) {
339  printf("bus_size: %zd\n", s->buf_size);
340  printf("port_low: %d\n", s->port_low);
341  printf("port_high: %d\n", s->port_high);
342  printf("verbosity: %d\n", s->verbosity);
343  }

346 static void open_sockets(example_state *s) {
347  int socket_count = s->port_high - s->port_low + 1;
348 
349  size_t info_size = sizeof(socket_info) + s->buf_size;
350 
351  /* open socket(s) */
352  for (int i = 0; i < socket_count; i++) {
353  int port = i + s->port_low;
354 
355  socket99_config cfg = {
356  .host = "127.0.0.1",
357  .port = port,
358  .nonblocking = true,
359  };
360  socket99_result res;
361 
362  if (!socket99_open(&cfg, &res)) {
363  socket99_fprintf(stderr, &res);
364  exit(1);
365  }
366 
367  s->sockets[i] = res.fd;
368  s->sockets_used++;
369  socket_info *si = malloc(info_size);
370  assert(si);
371  memset(si, 0, info_size);
372  s->info[i] = si;

376 static size_t construct_msg(uint8_t *buf, size_t buf_size, size_t payload_size, int64_t seq_id) {
377  size_t header_size = sizeof(prot_header_t);
378  assert(buf_size > header_size);
379  prot_header_t *header = (prot_header_t *)buf;
380 
381  uint8_t *payload = &buf[header_size];
382  for (int i = 0; i < payload_size; i++) {
383  payload[i] = (uint8_t)(0xFF & i);
384  }
385 
386  header->magic_number = MAGIC_NUMBER;
387  header->seq_id = seq_id;
388  header->size = payload_size;
389 
390  return header_size + payload_size;
391 }

397 static void completion_cb(bus_msg_result_t *res, void *udata) {
398  example_state *s = &state;
399  socket_info *si = (socket_info *)udata;
400 
401  switch (res->status) {
402  case BUS_SEND_SUCCESS:
403  {
404 #if INCREMENT_COMPLETION_COUNTER
405  for (;;) {
406  size_t cur = s->completed_deliveries;
407  if (ATOMIC_BOOL_COMPARE_AND_SWAP(&s->completed_deliveries, cur, cur + 1)) {
408  LOG(3, " -- ! got %zd bytes, seq_id 0x%08llx, %p\n",
409  si->cur_payload_size, res->u.response.seq_id,
410  res->u.response.opaque_msg);
411  break;
412  }
413  }
414 #else
415  (void)s;
416  (void)si;
417 #endif
418  }
419  break;
420  case BUS_SEND_TX_TIMEOUT:
421  LOG(1, "BUS_SEND_TIMEOUT\n");
422  break;
423  case BUS_SEND_TX_FAILURE:
424  LOG(1, "BUS_SEND_TX_FAILURE\n");
425  break;
426  case BUS_SEND_RX_FAILURE:
427  LOG(1, "BUS_SEND_RX_FAILURE\n");
428  break;
429  case BUS_SEND_RX_TIMEOUT:
430  LOG(1, "BUS_SEND_RX_TIMEOUT\n");
431  break;
432  default:
433  fprintf(stderr, "match fail: %d\n", res->status);
434  assert(false);
435  }
436 }

451 static void run_bus(example_state *s, struct bus *b) {
452  register_signal_handlers();
453  open_sockets(s);
454 
455  for (int i = 0; i < s->sockets_used; i++) {
456  bus_register_socket(b, BUS_SOCKET_PLAIN, s->sockets[i], s->info[i]);
457  }
458 
459  int cur_socket_i = 0;
460  int64_t seq_id = 1;
461 
462  uint8_t msg_buf[DEFAULT_BUF_SIZE];
463  size_t buf_size = sizeof(msg_buf);
464  size_t payload_size = seq_id;
465 
466  s->last_second = get_cur_second();
467 
468  int sleep_counter = 0;
469  int dropped = 0;
470 
471  while (loop_flag) {
472  time_t cur_second = get_cur_second();
473  if (cur_second != s->last_second) {
474  tick_handler(s);
475  s->last_second = cur_second;
476  payload_size = 8;
477  if (sleep_counter > 0) {
478  sleep_counter--;
479  if (sleep_counter == 0) {
480  printf(" -- resuming\n");
481  }
482  } else if ((cur_second & 0x3f) == 0x00) {
483  printf(" -- sleeping for 10 seconds\n");
484  sleep_counter = 10;
485  }
486  }
487 
488  if (sleep_counter == 0) {
489  size_t msg_size = construct_msg(msg_buf, buf_size,
490  100 * /*payload_size * */ 1024L, seq_id);
491  LOG(3, " @@ sending message with %zd bytes\n", msg_size);
492  bus_user_msg msg = {
493  .fd = s->sockets[cur_socket_i],
494  .seq_id = seq_id,
495  .msg = msg_buf,
496  .msg_size = msg_size,
497  .cb = completion_cb,
498  .udata = s->info[cur_socket_i],
499  };
500 
501  s->sent_msgs++;
502  payload_size++;
503  if (!bus_send_request(b, &msg)) {
504  LOG(1, " @@@ bus_send_request failed!\n");
505  dropped++;
506  if (dropped >= 100) {
507  LOG(1, " @@@ more than 100 send failures, halting\n");
508  loop_flag = false;
509  }
510  }
511 
512  cur_socket_i++;
513  if (cur_socket_i == s->sockets_used) {
514  cur_socket_i = 0;
515  }
516  seq_id++;
517  if (seq_id == s->max_seq_id) {
518  loop_flag = false;
519  }
520  } else {
521  poll(NULL, 0, 1000);
522  }
523  }
524 }
