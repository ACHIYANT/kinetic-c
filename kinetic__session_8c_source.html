37 KineticStatus KineticSession_Create(KineticSession * const session, KineticClient * const client)
38 {
39  if (session == NULL) {
40  return KINETIC_STATUS_SESSION_EMPTY;
41  }
42 
43  if (client == NULL) {

46

64  KineticAllocator_FreeConnection(session->connection);

85 KineticStatus KineticSession_Connect(KineticSession const * const session)
86 {
87  if (session == NULL) {
88  return KINETIC_STATUS_SESSION_EMPTY;
89  }
90  KineticConnection* connection = session->connection;
91  if (connection == NULL) {
92  return KINETIC_STATUS_CONNECTION_ERROR;
93  }
94 
95  // Establish the connection
96  assert(session != NULL);
97  assert(session->connection != NULL);
98  assert(strlen(session->config.host) > 0);
99  connection->socket = KineticSocket_Connect(
100  session->config.host, session->config.port);
101  if (connection->socket == KINETIC_SOCKET_DESCRIPTOR_INVALID) {
102  LOG0("Session connection failed!");
103  connection->socket = KINETIC_SOCKET_DESCRIPTOR_INVALID;
104  connection->connected = false;
105  return KINETIC_STATUS_CONNECTION_ERROR;
106  }
107  connection->connected = true;
108 
109  // Kick off the worker thread
110  connection->si = calloc(1, sizeof(socket_info) + 2 * PDU_PROTO_MAX_LEN);
111  if (connection->si == NULL) { return KINETIC_STATUS_MEMORY_ERROR; }
112  bool success = bus_register_socket(connection->messageBus, BUS_SOCKET_PLAIN, connection->socket, connection);
113  if (!success) {
114  LOG0("Failed registering connection with client!");
115  goto connection_error_cleanup;
116  }
117 
118  // Wait for initial unsolicited status to be received in order to obtain connection ID
119  success = KineticResourceWaiter_WaitTilAvailable(&connection->connectionReady, KINETIC_CONNECTION_TIMEOUT_SECS);
120  if (!success) {
121  LOG0("Timed out waiting for connection ID from device!");
122  goto connection_error_cleanup;
123  }
124  // nanosleep(&sleepDuration, NULL);
125  LOGF1("Received connection ID %lld for session %p",
126  (long long)connection->connectionID, (void*)session);
127  // nanosleep(&sleepDuration, NULL);
128 
129  return KINETIC_STATUS_SUCCESS;
130 
131 connection_error_cleanup:
132 
133  LOG_LOCATION;
134 
135  if (connection->si != NULL) {
136  free(connection->si);
137  connection->si = NULL;
138  }
139  if (connection->socket != KINETIC_SOCKET_DESCRIPTOR_INVALID) {
140  KineticSocket_Close(connection->socket);
141  connection->socket = KINETIC_SOCKET_DESCRIPTOR_INVALID;
142  }
143  connection->connected = false;
144  return KINETIC_STATUS_CONNECTION_ERROR;
145 }

147 KineticStatus KineticSession_Disconnect(KineticSession const * const session)
148 {
149  if (session == NULL) {
150  return KINETIC_STATUS_SESSION_EMPTY;
151  }
152  KineticConnection* connection = session->connection;
153  if (connection == NULL || !session->connection->connected || connection->socket < 0) {
154  return KINETIC_STATUS_CONNECTION_ERROR;
155  }
156 
157  // Close the connection
158  bus_release_socket(connection->messageBus, connection->socket);
159  free(connection->si);
160  connection->si = NULL;
161  connection->socket = KINETIC_HANDLE_INVALID;
162  connection->connected = false;
163  pthread_mutex_destroy(&connection->sendMutex);
164 
165  return KINETIC_STATUS_SUCCESS;
166 }
