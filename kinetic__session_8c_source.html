37 KineticStatus KineticSession_Create(KineticSession * const session, KineticClient * const client)
38 {
39  if (session == NULL) {
40  LOG0("Session is NULL");

42  }
43 
44  if (client == NULL) {

61  }
62 
63  session->connection->outstandingOperations =
64  KineticCountingSemaphore_Create(KINETIC_MAX_OUTSTANDING_OPERATIONS_PER_SESSION);
65  if (session->connection->outstandingOperations == NULL) {
66  LOG0("Failed initializing session counting semaphore!");
67  KineticAllocator_FreeConnection(session->connection);
68  return KINETIC_STATUS_MEMORY_ERROR;
69  }
70 
71  return KINETIC_STATUS_SUCCESS;

91 KineticStatus KineticSession_Connect(KineticSession * const session)
92 {
93  if (session == NULL) {
94  return KINETIC_STATUS_SESSION_EMPTY;
95  }
96  KineticConnection* connection = session->connection;
97  if (connection == NULL) {
98  return KINETIC_STATUS_CONNECTION_ERROR;
99  }
100 
101  // Establish the connection
102  KINETIC_ASSERT(session != NULL);
103  KINETIC_ASSERT(session->connection != NULL);
104  KINETIC_ASSERT(strlen(session->config.host) > 0);
105  connection->socket = KineticSocket_Connect(
106  session->config.host, session->config.port);
107  if (connection->socket == KINETIC_SOCKET_DESCRIPTOR_INVALID) {
108  LOG0("Session connection failed!");
109  connection->socket = KINETIC_SOCKET_DESCRIPTOR_INVALID;
110  connection->connected = false;
111  return KINETIC_STATUS_CONNECTION_ERROR;
112  }
113  connection->connected = true;
114 
115  bus_socket_t socket_type = session->config.useSsl ? BUS_SOCKET_SSL : BUS_SOCKET_PLAIN;
116  connection->si = calloc(1, sizeof(socket_info) + 2 * PDU_PROTO_MAX_LEN);
117  if (connection->si == NULL) { return KINETIC_STATUS_MEMORY_ERROR; }
118  bool success = bus_register_socket(connection->messageBus, socket_type, connection->socket, connection);
119  if (!success) {
120  LOG0("Failed registering connection with client!");
121  goto connection_error_cleanup;
122  }
123 
124  // Wait for initial unsolicited status to be received in order to obtain connection ID
125  success = KineticResourceWaiter_WaitTilAvailable(&connection->connectionReady, KINETIC_CONNECTION_TIMEOUT_SECS);
126  if (!success) {
127  LOG0("Timed out waiting for connection ID from device!");
128  goto connection_error_cleanup;
129  }
130  // nanosleep(&sleepDuration, NULL);
131  LOGF1("Received connection ID %lld for session %p",
132  (long long)connection->connectionID, (void*)session);
133  // nanosleep(&sleepDuration, NULL);
134 
135  return KINETIC_STATUS_SUCCESS;
136 
137 connection_error_cleanup:
138 
139  if (connection->si != NULL) {
140  free(connection->si);
141  connection->si = NULL;
142  }
143  if (connection->socket != KINETIC_SOCKET_DESCRIPTOR_INVALID) {
144  KineticSocket_Close(connection->socket);
145  connection->socket = KINETIC_SOCKET_DESCRIPTOR_INVALID;
146  }
147  connection->connected = false;
148  return KINETIC_STATUS_CONNECTION_ERROR;
149 }

151 KineticStatus KineticSession_Disconnect(KineticSession const * const session)
152 {
153  if (session == NULL) {
154  return KINETIC_STATUS_SESSION_EMPTY;
155  }
156  KineticConnection* connection = session->connection;
157  if (connection == NULL || !session->connection->connected || connection->socket < 0) {
158  return KINETIC_STATUS_CONNECTION_ERROR;
159  }
160 
161  // Close the connection
162  bus_release_socket(connection->messageBus, connection->socket, NULL);
163  free(connection->si);
164  connection->si = NULL;
165  connection->socket = KINETIC_HANDLE_INVALID;
166  connection->connected = false;
167  pthread_mutex_destroy(&connection->sendMutex);
168 
169  return KINETIC_STATUS_SUCCESS;
170 }
