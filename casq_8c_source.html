102 void *casq_pop(struct casq *q) {
103  for (;;) {
104  if (q->head != NULL) {
105  casq_link *l = NULL;
106  for (;;) { /* spin, pop */
107  l = q->head;
108  if (ATOMIC_BOOL_COMPARE_AND_SWAP(&q->head, l, l->next)) {
109  break;
110  }
111  }
112  void *res = l->data;
113  l->data = NULL;
114  for (;;) { /* spin, push */
115  l->next = q->free_links;
116  if (ATOMIC_BOOL_COMPARE_AND_SWAP(&q->free_links, q->free_links, l)) {
117  break;
118  }
119  }
120  return res;
121  } else if (q->accum != NULL) {
122  /* If no more links are available, reverse the accumulated ones on demand. */
123  if (ATOMIC_BOOL_COMPARE_AND_SWAP(&q->is_reversing, 0, 1)) { /* spin, lock */
124  reverse(q);
125  q->is_reversing = 0;
126  } else {
127  continue;
128  }
129  } else { /* empty */
130  return NULL;
131  }
132  }
133 }
