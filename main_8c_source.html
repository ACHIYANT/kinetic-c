60 int main(int argc, char** argv)
61 {
62  // Parse command line options
63  ParseOptions(argc, argv, &SessionConfig, &Entry);
64 
65  // Establish a session/connection with the Kinetic Device
66  KineticSessionHandle sessionHandle;
67  KineticStatus status = KineticClient_Connect(&SessionConfig, &sessionHandle);
68  if (status != KINETIC_STATUS_SUCCESS) {
69  printf("Failed connecting to host %s:%d (status: %s)",
70  SessionConfig.host, SessionConfig.port,
71  Kinetic_GetStatusDescription(status));
72  return -1;
73  }
74 
75  // Execute all specified operations in order
76  for (int optionIndex = 1; optionIndex < argc; optionIndex++) {
77  const char* operation = argv[optionIndex];
78  ReportOperationConfiguration(operation, &SessionConfig, &Entry);
79  ExecuteOperation(operation, sessionHandle, &Entry);
80  }
81 
82  // Shutdown the Kinetic Device session
83  KineticClient_Disconnect(&sessionHandle);
84  printf("\nKinetic client session terminated!\n\n");
85 
86  return 0;
87 }

89 KineticStatus ExecuteOperation(
90  const char* operation,
91  KineticSessionHandle sessionHandle,
92  KineticEntry* entry)
93 {
94  KineticStatus status = KINETIC_STATUS_INVALID;
95 
96  if (strcmp("noop", operation) == 0) {
97  status = KineticClient_NoOp(sessionHandle);
98  if (status == KINETIC_STATUS_SUCCESS) {
99  printf("\nNoOp operation completed successfully."
100  " Kinetic Device is alive and well!\n");
101  }
102  }
103 
104  else if (strcmp("put", operation) == 0) {
105  status = KineticClient_Put(sessionHandle, entry);
106  if (status == KINETIC_STATUS_SUCCESS) {
107  printf("\nPut operation completed successfully."
108  " Your data has been stored!\n\n");
109  }
110  }
111 
112  else if (strcmp("get", operation) == 0) {
113  status = KineticClient_Get(sessionHandle, entry);
114  if (status == 0) {
115  printf("\nGet executed successfully."
116  "The entry has been retrieved!\n\n");
117  }
118  }
119 
120  else if (strcmp("delete", operation) == 0) {
121  status = KineticClient_Delete(sessionHandle, entry);
122  if (status == 0) {
123  printf("\nDelete executed successfully."
124  " The entry has been destroyed!\n\n");
125  }
126  }
127 
128  else {
129  printf("\nSpecified operation '%s' is invalid!\n", operation);
130  return -1;
131  }
132 
133  // Print out status code description if operation was not successful
134  if (status != KINETIC_STATUS_SUCCESS) {
135  printf("\nERROR: Operation '%s' failed! with status = %s)\n\n",
136  operation, Kinetic_GetStatusDescription(status));
137  }
138 
139  return status;
140 }

196 void ParseOptions(
197  const int argc,
198  char** const argv,
199  KineticSession* sessionConfig,
200  KineticEntry* entry)
201 {
202  // Create an ArgP processor to parse arguments
203  struct {
204  char host[HOST_NAME_MAX];
205  int port;
206  int nonBlocking;
207  int useTls;
208  int64_t clusterVersion;
209  int64_t identity;
210  char hmacKey[KINETIC_MAX_KEY_LEN];
211  char key[64];
212  char version[64];
213  char tag[64];
214  KineticAlgorithm algorithm;
215  } cfg = {
216  .host = "localhost",
217  .port = KINETIC_PORT,
218  .nonBlocking = false,
219  .useTls = false,
220  .clusterVersion = 0,
221  .identity = 1,
222  .hmacKey = "asdfasdf",
223  .key = "SomeObjectKeyValue",
224  .version = "v1.0",
225  .tag = "SomeTagValue",
226  .algorithm = KINETIC_ALGORITHM_SHA1,
227  };
228 
229  // Create configuration for long format options
230  struct option long_options[] = {
231  {"non-blocking", no_argument, &cfg.nonBlocking, true},
232  {"blocking", no_argument, &cfg.nonBlocking, false},
233  {"tls", no_argument, &cfg.port, KINETIC_TLS_PORT},
234  {"host", required_argument, 0, 'h'},
235  {0, 0, 0, 0},
236  };
237 
238  // Parse the options from the command line
239  int option, optionIndex = 0;
240  while ((option = getopt_long(argc, argv, "h", long_options, &optionIndex)) != -1) {
241  // Parse options until we reach the end of the argument list
242  switch (option) {
243  // If this option set a flag, do nothing else now
244  case 0: if (long_options[optionIndex].flag != 0) {
245  break;
246  }
247  // Configure host
248  case 'h': strcpy(cfg.host, optarg); break;
249  // Discard '?', since getopt_long already printed info
250  case '?': break;
251  // Abort upon error
252  default: assert(false);
253  }
254  }
255 
256  // Configure session for connection
257  *sessionConfig = (KineticSession) {
258  .port = cfg.port,
259  .clusterVersion = cfg.clusterVersion,
260  .identity = cfg.identity,
261  .nonBlocking = cfg.nonBlocking,
262  .hmacKey = ByteArray_Create(HmacData, strlen(cfg.hmacKey)),
263  };
264  memcpy(HmacData, cfg.hmacKey, strlen(cfg.hmacKey));
265  strncpy(sessionConfig->host, cfg.host, HOST_NAME_MAX);
266 
267  // Populate and configure the entry to be used for operations
268  ConfigureEntry(entry,
269  cfg.key, cfg.tag, cfg.version, cfg.algorithm, TestDataString);
270 }
