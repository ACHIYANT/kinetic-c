165 //------------------------------------------------------------------------------
166 // Main Entry Point Definition
167 int main(int argc, char** argv)
168 {
169  struct UtilConfig cfg;
170  memset(&cfg, 0, sizeof(cfg));
171 
172  // Parse command line options
173  ParseOptions(argc, argv, &cfg);
174  if (cfg.opID == OPT_NONE) {
175  fprintf(stderr, "No operation specified!\n");
176  PrintUsage(argv[0]);
177  exit(1);
178  }
179 
180  // Create a client instance
181  KineticClientConfig client_config = {
182  .logFile = "stdout",
183  .logLevel = cfg.logLevel,
184  };
185  cfg.client = KineticClient_Init(&client_config);
186  if (cfg.client == NULL) {
187  fprintf(stderr, "Failed creating a client instance!\n");
188  return 1;
189  }
190 
191  // Establish a session with the Kinetic Device
192  KineticStatus status = KineticClient_CreateSession(&cfg.config, cfg.client, &cfg.session);
193  if (status != KINETIC_STATUS_SUCCESS) {
194  fprintf(stderr, "Failed connecting to host %s:%d (status: %s)\n",
195  cfg.config.host, cfg.config.port,
196  Kinetic_GetStatusDescription(status));
197  return 1;
198  }
199 
200  // Establish an admin session with the Kinetic Device
201  cfg.adminConfig.useSsl = true;
202  status = KineticClient_CreateSession(&cfg.adminConfig, cfg.client, &cfg.adminSession);
203  if (status != KINETIC_STATUS_SUCCESS) {
204  fprintf(stderr, "Failed connecting to host %s:%d (status: %s)\n",
205  cfg.config.host, cfg.config.port,
206  Kinetic_GetStatusDescription(status));
207  return 1;
208  }
209 
210  // Execute the specified operation
211  // ReportConfiguration(cmd, &cfg);
212  status = ExecuteOperation(&cfg);
213  if (status != KINETIC_STATUS_SUCCESS) {
214  fprintf(stderr, "Failed executing operation!\n");
215  return 1;
216  }
217 
218  return 0;
219 }

319 KineticStatus ExecuteOperation(
320  struct UtilConfig * cfg)
321 {
322  KineticStatus status = KINETIC_STATUS_INVALID;
323  KineticLogInfo * logInfo;
324 
325  switch (cfg->opID) {
326 
327  case OPT_NOOP:
328  status = KineticClient_NoOp(cfg->session);
329  if (status == KINETIC_STATUS_SUCCESS) {
330  printf("NoOp operation completed successfully.\n"
331  " Kinetic Device is alive and well!\n"); }
332  break;
333 
334  case OPT_PUT:
335  status = KineticClient_Put(cfg->session, &cfg->entry, NULL);
336  if (status == KINETIC_STATUS_SUCCESS) {
337  printf("Put operation completed successfully.\n"
338  " Your data has been stored!\n"); }
339  break;
340 
341  case OPT_GET:
342  status = KineticClient_Get(cfg->session, &cfg->entry, NULL);
343  if (status == KINETIC_STATUS_SUCCESS) {
344  printf("Get executed successfully.\n");
345  PrintEntry(&cfg->entry);
346  }
347  break;
348 
349  case OPT_GETNEXT:
350  status = KineticClient_GetNext(cfg->session, &cfg->entry, NULL);
351  if (status == KINETIC_STATUS_SUCCESS) {
352  printf("GetNext executed successfully.\n");
353  PrintEntry(&cfg->entry);
354  }
355  break;
356 
357  case OPT_GETPREVIOUS:
358  status = KineticClient_GetPrevious(cfg->session, &cfg->entry, NULL);
359  if (status == KINETIC_STATUS_SUCCESS) {
360  printf("GetPrevious executed successfully.\n");
361  PrintEntry(&cfg->entry);
362  }
363  break;
364 
365  case OPT_DELETE:
366  status = KineticClient_Delete(cfg->session, &cfg->entry, NULL);
367  if (status == KINETIC_STATUS_SUCCESS) {
368  printf("Delete executed successfully. The entry has been destroyed!\n");
369  PrintEntry(&cfg->entry);
370  }
371  break;
372 
373  case OPT_GETLOG:
374  status = KineticAdminClient_GetLog(cfg->adminSession, cfg->logType, &logInfo, NULL);
375  if (status == KINETIC_STATUS_SUCCESS) {
376  printf("GetLog executed successfully.\n"
377  "The device log info has been retrieved!\n");
378  // TODO: report returned log info per specified type
379  PrintLogInfo(cfg->logType, logInfo);
380  }
381  break;
382 
383 #if 0
384  // TODO: Is this necesary? We could add vendor-specific name as opt-arg to "getlog"
385  // case OPT_GETVENDORLOG:
386  {
387  // status = KineticClient_GetLog(cfg->session, info_type, &log_info, NULL);
388  // if (status == KINETIC_STATUS_SUCCESS) {
389  // printf("GetLog executed successfully."
390  // "The vendor-specific device log info has been retrieved!\n");
391  // // TODO: report returned log info per specified type
392  // } break; // }
393 #endif
394 
395  case OPT_SETERASEPIN:
396  status = KineticAdminClient_SetErasePin(cfg->adminSession,
397  ByteArray_Create(cfg->pin, strlen(cfg->pin)),
398  ByteArray_Create(cfg->newPin, strlen(cfg->newPin)));
399  if (status == KINETIC_STATUS_SUCCESS) {
400  printf("SetErasePin executed successfully.\n"
401  "The kinetic device erase pin has been changed!\n"); }
402  break;
403 
404  case OPT_INSTANTERASE:
405  status = KineticAdminClient_InstantErase(cfg->adminSession,
406  ByteArray_Create(cfg->pin, strlen(cfg->pin)));

410  break;
411

417  "The kinetic device has been erased!\n"); }
418  break;
419

421  status = KineticAdminClient_SetLockPin(cfg->adminSession,
422  ByteArray_Create(cfg->pin, strlen(cfg->pin)),
423  ByteArray_Create(cfg->newPin, strlen(cfg->newPin)));

427  break;
428 
429  case OPT_LOCKDEVICE:
430  status = KineticAdminClient_LockDevice(cfg->adminSession,
431  ByteArray_Create(cfg->pin, strlen(cfg->pin)));
432  if (status == KINETIC_STATUS_SUCCESS) {
433  printf("LockDevice executed successfully.\n"
434  "The kinetic device is now locked!\n"); }
435  break;
436 
437  case OPT_UNLOCKDEVICE:
438  status = KineticAdminClient_UnlockDevice(cfg->adminSession,
439  ByteArray_Create(cfg->pin, strlen(cfg->pin)));
440  if (status == KINETIC_STATUS_SUCCESS) {
441  printf("UnlockDevice executed successfully.\n"
442  "The kinetic device is now unlocked!\n"); }
443  break;
444 
445  case OPT_SETCLUSTERVERSION:
446  status = KineticAdminClient_SetClusterVersion(cfg->adminSession, cfg->newClusterVersion);
447  if (status == KINETIC_STATUS_SUCCESS) {
448  printf("SetClusterVersion executed successfully.\n"
449  "The kinetic device's cluster version has been updated!\n"); }
450  break;
451 
452  case OPT_SETACL:
453  status = KineticAdminClient_SetACL(cfg->adminSession, cfg->file);
454  if (status == KINETIC_STATUS_SUCCESS) {
455  printf("SetACL executed successfully.\n"
456  "The kinetic device ACLs have been replaced!\n"); }
457  break;
458 
459  case OPT_UPDATEFIRMWARE:
460  status = KineticAdminClient_UpdateFirmware(cfg->session, cfg->file);
461  if (status == KINETIC_STATUS_SUCCESS) {
462  printf("SecureErase executed successfully.\n"
463  "The kinetic device has been restored to empty status!\n"); }
464  break;
465 
466  default:
467  fprintf(stderr, "Specified operation '%d' is invalid!\n",
468  (int)cfg->opID);
469  exit(-1);
470  };
471 
472  // Print out status code description if operation was not successful
473  if (status != KINETIC_STATUS_SUCCESS) {
474  fprintf(stderr, "\nERROR: Operation '%s' failed with status '%s'\n",
475  GetOptString(cfg->opID), Kinetic_GetStatusDescription(status));
476  }
477 
478  return status;
479 }

522  const int argc,

525 {

583  // Options

601  // Parse the options from the command line
602  extern char *optarg;
603  extern int optind;
604  int option, optionIndex = 0;
605  while ((option = getopt_long(argc, argv, "?lhptics:", long_options, &optionIndex)) != -1) {
606  switch (option) {
607  case 0:
608  // If this option, just set the flag
609  if (long_options[optionIndex].flag != 0) {break;}
610  // assert(false);
611  // break;
612  case OPT_LOGLEVEL:
613  opts.logLevel = atoi(optarg);
614  break;
615  case OPT_HOST:
616  strncpy(opts.host, optarg, sizeof(opts.host));
617  break;
618  case OPT_PORT:
619  opts.port = atoi(optarg);
620  break;
621  case OPT_TLSPORT:
622  opts.tlsPort = atoi(optarg);
623  break;
624  case OPT_IDENTITY:
625  opts.identity = atoi(optarg);
626  break;
627  case OPT_HMACKEY:
628  strncpy(opts.hmacKey, optarg, sizeof(opts.hmacKey)-1);
629  break;
630  case OPT_KEY:
631  strncpy(opts.key, optarg, sizeof(opts.key)-1);
632  break;
633  case OPT_VALUE:
634  strncpy(opts.value, optarg, sizeof(opts.value)-1);
635  break;
