61 int main(int argc, char** argv)
62 {
63  KineticClientConfig client_config = {
64  .logFile = "stdout",
65  .logLevel = 0,
66  };
67  KineticClient * client = KineticClient_Init(&client_config);
68  if (client == NULL) {
69  return 1;
70  }
71 
72  // Parse command line options
73  int operationsArgsIndex = ParseOptions(argc, argv, &Session, &Entry);
74 
75  // Establish a session/connection with the Kinetic Device
76  KineticStatus status = KineticClient_CreateConnection(&Session, client);
77  if (status != KINETIC_STATUS_SUCCESS) {
78  printf("Failed connecting to host %s:%d (status: %s)\n\n",
79  Session.config.host, Session.config.port,
80  Kinetic_GetStatusDescription(status));
81  return 1;
82  }
83 
84  // Execute all specified operations in order
85  for (int optionIndex = operationsArgsIndex; optionIndex < argc; optionIndex++) {
86  const char* operation = argv[optionIndex];
87  ReportOperationConfiguration(operation, &Session, &Entry);
88  ExecuteOperation(operation, &Session, &Entry);
89  }
90 
91  // Shutdown the Kinetic Device session
92  KineticClient_DestroyConnection(&Session);
93  KineticClient_Shutdown(client);
94  printf("\nKinetic client session terminated!\n\n");
95 
96  return 0;
97 }

103 KineticStatus ExecuteOperation(
104  const char* operation,
105  KineticSession const * const session,
106  KineticEntry* entry)
107 {
108  KineticStatus status = KINETIC_STATUS_INVALID;
109 
110  if (strcmp("instanterase", operation) == 0) {
111  status = KineticClient_InstantSecureErase(session);
112  if (status == 0) {
113  printf("\nInstantSecureErase executed successfully."
114  " The device has been erased!\n\n");
115  }
116  }
117 
118  else if (strcmp("noop", operation) == 0) {
119  status = KineticClient_NoOp(session);
120  if (status == KINETIC_STATUS_SUCCESS) {
121  printf("\nNoOp operation completed successfully."
122  " Kinetic Device is alive and well!\n\n");
123  }
124  }
125 
126  else if (strcmp("put", operation) == 0) {
127  status = KineticClient_Put(session, entry, NULL);
128  if (status == KINETIC_STATUS_SUCCESS) {
129  printf("\nPut operation completed successfully."
130  " Your data has been stored!\n\n");
131  }
132  }
133 
134  else if (strcmp("get", operation) == 0) {
135  status = KineticClient_Get(session, entry, NULL);
136  if (status == 0) {
137  printf("\nGet executed successfully."
138  "The entry has been retrieved!\n\n");
139  }
140  }
141 
142  else if (strcmp("delete", operation) == 0) {
143  status = KineticClient_Delete(session, entry, NULL);
144  if (status == 0) {
145  printf("\nDelete executed successfully."
146  " The entry has been destroyed!\n\n");
147  }
148  }
149 
150  else if (strcmp("getnext", operation) == 0) {
151  status = KineticClient_GetNext(session, entry, NULL);
152  if (status == 0) {
153  printf("\nGetNext executed successfully."
154  "The entry has been retrieved!\n\n");
155  }
156  }
157 
158  else if (strcmp("getprevious", operation) == 0) {
159  status = KineticClient_GetPrevious(session, entry, NULL);
160  if (status == 0) {
161  printf("\nGetPrevious executed successfully."
162  "The entry has been retrieved!\n\n");
163  }
164  }
165 
166  else {
167  printf("\nSpecified operation '%s' is invalid!\n\n", operation);
168  return -1;
169  }
170 
171  // Print out status code description if operation was not successful
172  if (status != KINETIC_STATUS_SUCCESS) {
173  printf("\nERROR: Operation '%s' failed! with status = %s)\n\n",
174  operation, Kinetic_GetStatusDescription(status));
175  }
176 
177  return status;
178 }

232 int ParseOptions(
233  const int argc,
234  char** const argv,
235  KineticSession * const session,
236  KineticEntry* entry)
237 {
238  // Create an ArgP processor to parse arguments
239  struct {
240  char host[HOST_NAME_MAX];
241  int port;
242  int nonBlocking;
243  int useTls;
244  int64_t clusterVersion;
245  int64_t identity;
246  char hmacKey[KINETIC_MAX_KEY_LEN];
247  char key[64];
248  char version[64];
249  char tag[64];
250  KineticAlgorithm algorithm;
251  } cfg = {
252  .host = "localhost",
253  .port = KINETIC_PORT,
254  .useTls = false,
255  .clusterVersion = 0,
256  .identity = 1,
257  .hmacKey = "asdfasdf",
258  .key = "SomeObjectKeyValue",
259  .version = "v1.0",
260  .tag = "SomeTagValue",
261  .algorithm = KINETIC_ALGORITHM_SHA1,
262  };
263 
264  // Create configuration for long format options
265  struct option long_options[] = {
266  {"non-blocking", no_argument, &cfg.nonBlocking, true},
267  {"blocking", no_argument, &cfg.nonBlocking, false},
268  {"tls", no_argument, &cfg.port, KINETIC_TLS_PORT},
269  {"host", required_argument, 0, 'h'},
270  {0, 0, 0, 0},
271  };
272 
273  // Parse the options from the command line
274  extern char *optarg;
275  extern int optind;
276  int option, optionIndex = 0;
277  while ((option = getopt_long(argc, argv, "h:", long_options, &optionIndex)) != -1) {
278  // Parse options until we reach the end of the argument list
279  switch (option) {
280  // If this option set a flag, do nothing else now
281  case 0: if (long_options[optionIndex].flag != 0) {break;}
282  // Configure host
283  case 'h': strcpy(cfg.host, optarg); break;
284  // Discard '?', since getopt_long already printed info
285  case '?': break;
286  // Abort upon error
287  default: assert(false);
288  }
289  }
290 
291  // Configure session for connection
292  *session = (KineticSession) {
293  .config = (KineticSessionConfig) {
294  .port = cfg.port,
295  .clusterVersion = cfg.clusterVersion,
296  .identity = cfg.identity,
297  .hmacKey = ByteArray_Create(HmacData, strlen(cfg.hmacKey)),
298  }
299  };
300  memcpy(HmacData, cfg.hmacKey, strlen(cfg.hmacKey));
301  strncpy(session->config.host, cfg.host, HOST_NAME_MAX);
302 
303  // Populate and configure the entry to be used for operations
304  ConfigureEntry(entry, cfg.key, cfg.tag, cfg.version, cfg.algorithm, TestDataString);
305 
306  return optind;
307 }
