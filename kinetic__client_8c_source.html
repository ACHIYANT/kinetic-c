33 static KineticStatus KineticClient_CreateOperation(
34  KineticOperation** operation,
35  KineticSessionHandle handle)
36 {
37  if (handle == KINETIC_HANDLE_INVALID) {
38  LOG0("Specified session has invalid handle value");
39  return KINETIC_STATUS_SESSION_EMPTY;
40  }
41 
42  KineticConnection* connection = KineticConnection_FromHandle(handle);
43  if (connection == NULL) {
44  LOG0("Specified session is not associated with a connection");
45  return KINETIC_STATUS_SESSION_INVALID;
46  }
47 
48  LOGF1("\n"
49  "--------------------------------------------------\n"
50  "Building new operation on connection @ 0x%llX", connection);
51 
52  *operation = KineticAllocator_NewOperation(connection);
53  if (*operation == NULL) {
54  return KINETIC_STATUS_MEMORY_ERROR;
55  }
56  if ((*operation)->request == NULL) {
57  return KINETIC_STATUS_NO_PDUS_AVAVILABLE;
58  }
59 
60  return KINETIC_STATUS_SUCCESS;
61 }

63 static KineticStatus KineticClient_ExecuteOperation(KineticOperation* operation)
64 {
65  assert(operation != NULL);
66  KineticStatus status = KINETIC_STATUS_INVALID;
67 
68  LOGF1("Executing operation: 0x%llX", operation);
69  if (operation->entry != NULL &&
70  operation->entry->value.array.data != NULL &&
71  operation->entry->value.bytesUsed > 0)
72  {
73  LOGF1(" Sending PDU (0x%0llX) w/value (%zu bytes)",
74  operation->request, operation->entry->value.bytesUsed);
75  }
76  else {
77  LOGF1(" Sending PDU (0x%0llX) w/o value", operation->request);
78  }
79 
80  // Send the request
81  status = KineticOperation_SendRequest(operation);
82  if (status != KINETIC_STATUS_SUCCESS) {
83  return status;
84  }
85 
86  return KineticOperation_ReceiveAsync(operation);
87 }

99 KineticStatus KineticClient_Connect(const KineticSession* config,
100  KineticSessionHandle* handle)
101 {
102  if (handle == NULL) {
103  LOG0("Session handle is NULL!");
104  return KINETIC_STATUS_SESSION_EMPTY;
105  }
106  *handle = KINETIC_HANDLE_INVALID;
107 
108  if (config == NULL) {
109  LOG0("KineticSession is NULL!");
110  return KINETIC_STATUS_SESSION_EMPTY;
111  }
112 
113  if (strlen(config->host) == 0) {
114  LOG0("Host is empty!");
115  return KINETIC_STATUS_HOST_EMPTY;
116  }
117 
118  if (config->hmacKey.len < 1 || config->hmacKey.data == NULL) {
119  LOG0("HMAC key is NULL or empty!");
120  return KINETIC_STATUS_HMAC_EMPTY;
121  }
122 
123  // Obtain a new connection/handle
124  *handle = KineticConnection_NewConnection(config);
125  if (*handle == KINETIC_HANDLE_INVALID) {
126  LOG0("Failed connecting to device!");
127  return KINETIC_STATUS_SESSION_INVALID;
128  }
129  KineticConnection* connection = KineticConnection_FromHandle(*handle);
130  if (connection == NULL) {
131  LOG0("Failed getting valid connection from handle!");
132  return KINETIC_STATUS_CONNECTION_ERROR;
133  }
134 
135  // Create the connection
136  KineticStatus status = KineticConnection_Connect(connection);
137  if (status != KINETIC_STATUS_SUCCESS) {
138  LOGF0("Failed creating connection to %s:%d", config->host, config->port);
139  KineticConnection_FreeConnection(handle);
140  *handle = KINETIC_HANDLE_INVALID;
141  return status;
142  }
143 
144  // Wait for initial unsolicited status to be received in order to obtain connectionID
145  while(connection->connectionID == 0) {sleep(1);}
146 
147  return status;
148 }
