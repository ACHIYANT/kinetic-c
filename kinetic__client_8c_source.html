187 KineticProto_Status_StatusCode KineticClient_ExecuteOperation(KineticOperation* operation)
188 {
189  KineticProto_Status_StatusCode status =
190  KINETIC_PROTO_STATUS_STATUS_CODE_INVALID_STATUS_CODE;
191 
192  // Send the request
193  if (KineticPDU_Send(operation->request))
194  {
195  // Associate response with same exchange as request
196  operation->response->connection = operation->request->connection;
197 
198  // Receive the response
199  if (KineticPDU_Receive(operation->response))
200  {
201  status = KineticPDU_Status(operation->response);
202  }
203  }
204 
205  return status;
206 }
