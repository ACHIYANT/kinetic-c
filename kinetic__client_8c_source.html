40 KineticStatus KineticClient_Connect(const KineticSession* config,
41  KineticSessionHandle* handle)
42 {
43  if (handle == NULL) {
44  LOG0("Session handle is NULL!");
45  return KINETIC_STATUS_SESSION_EMPTY;
46  }
47  *handle = KINETIC_HANDLE_INVALID;
48 
49  if (config == NULL) {
50  LOG0("KineticSession is NULL!");
51  return KINETIC_STATUS_SESSION_EMPTY;
52  }
53 
54  if (strlen(config->host) == 0) {
55  LOG0("Host is empty!");
56  return KINETIC_STATUS_HOST_EMPTY;
57  }
58 
59  if (config->hmacKey.len < 1 || config->hmacKey.data == NULL) {
60  LOG0("HMAC key is NULL or empty!");
61  return KINETIC_STATUS_HMAC_EMPTY;
62  }
63 
64  // Obtain a new connection/handle
65  *handle = KineticConnection_NewConnection(config);
66  if (*handle == KINETIC_HANDLE_INVALID) {
67  LOG0("Failed connecting to device!");
68  return KINETIC_STATUS_SESSION_INVALID;
69  }
70  KineticConnection* connection = KineticConnection_FromHandle(*handle);
71  if (connection == NULL) {
72  LOG0("Failed getting valid connection from handle!");
73  return KINETIC_STATUS_CONNECTION_ERROR;
74  }
75 
76  // Create the connection
77  KineticStatus status = KineticConnection_Connect(connection);
78  if (status != KINETIC_STATUS_SUCCESS) {
79  LOGF0("Failed creating connection to %s:%d", config->host, config->port);
80  KineticConnection_FreeConnection(handle);
81  *handle = KINETIC_HANDLE_INVALID;
82  return status;
83  }
84 
85  // Wait for initial unsolicited status to be received in order to obtain connectionID
86  while(connection->connectionID == 0) {sleep(1);}
87 
88  return status;
89 }

92 {
93  if (*handle == KINETIC_HANDLE_INVALID) {
94  LOG0("Invalid KineticSessionHandle specified!");
95  return KINETIC_STATUS_SESSION_INVALID;

97  KineticConnection* connection = KineticConnection_FromHandle(*handle);
98  if (connection == NULL) {
99  LOG0("Failed getting valid connection from handle!");
100  return KINETIC_STATUS_CONNECTION_ERROR;
101  }
102 
103  // Disconnect
104  KineticStatus status = KineticConnection_Disconnect(connection);
105  if (status != KINETIC_STATUS_SUCCESS) {LOG0("Disconnection failed!");}
106  KineticConnection_FreeConnection(handle);
107  *handle = KINETIC_HANDLE_INVALID;
108 
109  return status;
