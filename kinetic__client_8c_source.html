38  if (client == NULL) { return NULL; }
39 
40  /* Use defaults if set to 0. */

45  config->maxThreadpoolThreads = KINETIC_CLIENT_DEFAULT_MAX_THREADPOOL_THREADS;
46  }
47 
48  bool success = KineticBus_Init(client, config);
49  if (!success) {

63 KineticStatus KineticClient_CreateSession(KineticSessionConfig* const config,
64  KineticClient * const client, KineticSession** session)
65 {
66  if (config == NULL) {
67  LOG0("KineticSessionConfig is NULL!");
68  return KINETIC_STATUS_SESSION_INVALID;
69  }
70 
71  if (session == NULL) {
72  LOG0("Pointer to KineticSession pointer is NULL!");
73  return KINETIC_STATUS_SESSION_EMPTY;
74  }
75 
76  if (strlen(config->host) == 0) {
77  LOG0("Host is empty!");
78  return KINETIC_STATUS_HOST_EMPTY;
79  }
80 
81  if (config->hmacKey.len < 1 || config->hmacKey.data == NULL)
82  {
83  LOG0("HMAC key is NULL or empty!");
84  return KINETIC_STATUS_HMAC_REQUIRED;
85  }
86 
87  // Create a new session
88  KineticSession* s = KineticAllocator_NewSession(client->bus, config);
89  if (s == NULL) {
90  LOG0("Failed to create session instance!");
91  return KINETIC_STATUS_MEMORY_ERROR;
92  }
93  KineticSession_Create(s, client);
94  if (s->connection == NULL) {
95  LOG0("Failed to create connection instance!");
96  KineticAllocator_FreeSession(s);
97  return KINETIC_STATUS_CONNECTION_ERROR;
98  }
99 
100  // Establish the connection
101  KineticStatus status = KineticSession_Connect(s);
102  if (status != KINETIC_STATUS_SUCCESS) {
103  LOGF0("Failed creating connection to %s:%d", config->host, config->port);
104  s->connection = NULL;
105  KineticSession_Destroy(s);
106  KineticAllocator_FreeSession(s);
107  return status;
108  }
109 
110  *session = s;
111 
112  return status;
113 }

146 KineticStatus KineticClient_Put(KineticSession const * const session,
147  KineticEntry* const entry,
148  KineticCompletionClosure* closure)
149 {
150  KINETIC_ASSERT(session != NULL);
151  KINETIC_ASSERT(session->connection != NULL);
152  KINETIC_ASSERT(entry != NULL);
153 
154  // Assert non-NULL value upon non-zero length
155  if (entry->value.array.len > 0) {
156  KINETIC_ASSERT(entry->value.array.data != NULL);
157  }
158 
159  KINETIC_ASSERT(session->connection->pSession == session);
160  KINETIC_ASSERT(session->connection == session->connection->pSession->connection);
161 
162  KineticOperation* operation = KineticAllocator_NewOperation(session->connection);
163  if (operation == NULL) {return KINETIC_STATUS_MEMORY_ERROR;}
164 
165  // Initialize request
166  KineticStatus status = KineticOperation_BuildPut(operation, entry);
167  if (status != KINETIC_STATUS_SUCCESS) {
168  KineticAllocator_FreeOperation(operation);
169  return status;
170  }
171 
172  // Execute the operation
173  KINETIC_ASSERT(operation->connection == session->connection);
174  KineticStatus res = KineticController_ExecuteOperation(operation, closure);
175  return res;
176 }

210 static KineticStatus handle_get_command(GET_COMMAND cmd,
211  KineticSession const * const session,
212  KineticEntry* const entry,
213  KineticCompletionClosure* closure)
214 {
215  KINETIC_ASSERT(session != NULL);
216  KINETIC_ASSERT(session->connection != NULL);
217  KINETIC_ASSERT(entry != NULL);
218 
219  if (!has_key(entry)) {return KINETIC_STATUS_MISSING_KEY;}
220  if (!has_value_buffer(entry) && !entry->metadataOnly) {
221  return KINETIC_STATUS_MISSING_VALUE_BUFFER;
222  }
223 
224  KineticOperation* operation = KineticAllocator_NewOperation(session->connection);
225  if (operation == NULL) {
226  return KINETIC_STATUS_MEMORY_ERROR;
227  }
228 
229  // Initialize request
230  switch (cmd)
231  {
232  case CMD_GET:
233  KineticOperation_BuildGet(operation, entry);
234  break;
235  case CMD_GET_NEXT:
236  KineticOperation_BuildGetNext(operation, entry);
237  break;
238  case CMD_GET_PREVIOUS:
239  KineticOperation_BuildGetPrevious(operation, entry);
240  break;
241  default:
242  KINETIC_ASSERT(false);
243  }
244 
245  // Execute the operation
246  return KineticController_ExecuteOperation(operation, closure);
247 }
