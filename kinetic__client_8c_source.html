31 static KineticStatus KineticClient_CreateOperation(
32  KineticOperation* const operation,
33  KineticSessionHandle handle)
34 {
35  if (handle == KINETIC_HANDLE_INVALID) {
36  LOG("Specified session has invalid handle value");
37  return KINETIC_STATUS_SESSION_EMPTY;
38  }
39 
40  KineticConnection* connection = KineticConnection_FromHandle(handle);
41  if (connection == NULL) {
42  LOG("Specified session is not associated with a connection");
43  return KINETIC_STATUS_SESSION_INVALID;
44  }
45 
46  *operation = KineticOperation_Create(connection);
47  if (operation->request == NULL || operation->response == NULL) {
48  return KINETIC_STATUS_NO_PDUS_AVAVILABLE;
49  }
50 
51  return KINETIC_STATUS_SUCCESS;
52 }

54 static KineticStatus KineticClient_ExecuteOperation(KineticOperation* operation)
55 {
56  KineticStatus status = KINETIC_STATUS_INVALID;
57 
58  LOGF("Executing operation: 0x%llX", operation);
59  if (operation->request->entry.value.array.data != NULL
60  && operation->request->entry.value.bytesUsed > 0) {
61  LOGF(" Sending PDU w/value (%zu bytes)",
62  operation->request->entry.value.bytesUsed);
63  }
64  else {
65  LOG(" Sending PDU w/o value");
66  }
67 
68  // Send the request
69  status = KineticPDU_Send(operation->request);
70  if (status == KINETIC_STATUS_SUCCESS) {
71  // Associate response with same exchange as request
72  operation->response->connection = operation->request->connection;
73 
74  // Receive the response
75  status = KineticPDU_Receive(operation->response);
76  if (status == KINETIC_STATUS_SUCCESS) {
77  status = KineticOperation_GetStatus(operation);
78  }
79  }
80 
81  return status;
82 }

89 KineticStatus KineticClient_Connect(const KineticSession* config,
90  KineticSessionHandle* handle)
91 {
92  if (handle == NULL) {
93  LOG("Session handle is NULL!");
94  return KINETIC_STATUS_SESSION_EMPTY;
95  }
96  *handle = KINETIC_HANDLE_INVALID;
97 
98  if (config == NULL) {
99  LOG("KineticSession is NULL!");
100  return KINETIC_STATUS_SESSION_EMPTY;
101  }
102 
103  if (strlen(config->host) == 0) {
104  LOG("Host is empty!");
105  return KINETIC_STATUS_HOST_EMPTY;
106  }
107 
108  if (config->hmacKey.len < 1 || config->hmacKey.data == NULL) {
109  LOG("HMAC key is NULL or empty!");
110  return KINETIC_STATUS_HMAC_EMPTY;
111  }
112 
113  *handle = KineticConnection_NewConnection(config);
114  if (handle == KINETIC_HANDLE_INVALID) {
115  LOG("Failed connecting to device!");
116  return KINETIC_STATUS_SESSION_INVALID;
117  }
118 
119  KineticConnection* connection = KineticConnection_FromHandle(*handle);
120  if (connection == NULL) {
121  LOG("Failed getting valid connection from handle!");
122  return KINETIC_STATUS_CONNECTION_ERROR;
123  }
124 
125  KineticStatus status = KineticConnection_Connect(connection);
126  if (status != KINETIC_STATUS_SUCCESS) {
127  LOGF("Failed creating connection to %s:%d", config->host, config->port);
128  KineticConnection_FreeConnection(handle);
129  *handle = KINETIC_HANDLE_INVALID;
130  return status;
131  }
132 
133  return KINETIC_STATUS_SUCCESS;
134 }

181 KineticStatus KineticClient_Put(KineticSessionHandle handle,
182  KineticEntry* const entry)

186 
187  status = KineticClient_CreateOperation(&operation, handle);
188  if (status != KINETIC_STATUS_SUCCESS) {
189  return status;
190  }
191 
192  // Initialize request
193  KineticOperation_BuildPut(&operation, entry);
194 
195  // Execute the operation
196  status = KineticClient_ExecuteOperation(&operation);
197 
198  if (status == KINETIC_STATUS_SUCCESS) {
199  // Propagate newVersion to dbVersion in metadata, if newVersion specified
200  if (entry->newVersion.array.data != NULL && entry->newVersion.array.len > 0) {
201  entry->dbVersion = entry->newVersion;
202  entry->newVersion = BYTE_BUFFER_NONE;
203  }
204  }
205 
206  KineticOperation_Free(&operation);

211 KineticStatus KineticClient_Get(KineticSessionHandle handle,
212  KineticEntry* const entry)
213 {
214  assert(entry != NULL);
215  if (!entry->metadataOnly) {
216  assert(entry->value.array.data != NULL);
217  }
218 
219  KineticStatus status;
220  KineticOperation operation;
221 
222  status = KineticClient_CreateOperation(&operation, handle);
223  if (status != KINETIC_STATUS_SUCCESS) {
224  return status;
225  }
226 
227  // Initialize request
228  KineticOperation_BuildGet(&operation, entry);
229 
230  // Execute the operation
231  status = KineticClient_ExecuteOperation(&operation);
232 
233 
234  // Update the entry upon success
235  // entry->value.array.len = 0;
236  if (status == KINETIC_STATUS_SUCCESS) {
237  KineticProto_KeyValue* keyValue = KineticPDU_GetKeyValue(operation.response);
238  if (keyValue != NULL) {
239  if (!Copy_KineticProto_KeyValue_to_KineticEntry(keyValue, entry)) {
240  status = KINETIC_STATUS_BUFFER_OVERRUN;
241  }
242  }
243  }
244 
245  KineticOperation_Free(&operation);
246 
247  return status;
248 }
