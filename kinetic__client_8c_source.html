33 KineticClient * KineticClient_Init(KineticClientConfig *config)
34 {
35  KineticLogger_Init(config->logFile, config->logLevel);
36  KineticClient * client = KineticCalloc(1, sizeof(*client));
37  if (client == NULL) { return NULL; }
38 
39  /* Use defaults if set to 0. */
40  if (config->writerThreads == 0) {
41  config->writerThreads = KINETIC_CLIENT_DEFAULT_WRITER_THREADS;
42  }
43  if (config->readerThreads == 0) {
44  config->readerThreads = KINETIC_CLIENT_DEFAULT_READER_THREADS;
45  }
46  if (config->maxThreadpoolThreads == 0) {
47  config->maxThreadpoolThreads = KINETIC_CLIENT_DEFAULT_MAX_THREADPOOL_THREADS;
48  }
49 
50  bool success = KineticPDU_InitBus(client, config);
51  if (!success)
52  {
53  KineticFree(client);
54  return NULL;
55  }
56  return client;
57 }

66 KineticStatus KineticClient_CreateConnection(KineticSession* const session, KineticClient * const client)
67 {
68  if (session == NULL) {
69  LOG0("KineticSession is NULL!");
70  return KINETIC_STATUS_SESSION_EMPTY;
71  }
72 
73  if (strlen(session->config.host) == 0) {
74  LOG0("Host is empty!");
75  return KINETIC_STATUS_HOST_EMPTY;
76  }
77 
78  if (session->config.hmacKey.len < 1 || session->config.hmacKey.data == NULL) {
79  LOG0("HMAC key is NULL or empty!");
80  return KINETIC_STATUS_HMAC_EMPTY;
81  }
82 
83  KineticStatus res = KineticSession_Create(session, client);
84  if (res != KINETIC_STATUS_SUCCESS) {
85  LOGF0("Failed to create connection instance: %s",
86  Kinetic_GetStatusDescription(res));
87  return res;
88  }
89 
90  if (session->connection == NULL) {
91  LOG0("Failed to create connection instance!");
92  return KINETIC_STATUS_CONNECTION_ERROR;
93  }
94 
95  // Create the connection
96  KineticStatus status = KineticSession_Connect(session);
97  if (status != KINETIC_STATUS_SUCCESS) {
98  LOGF0("Failed creating connection to %s:%d", session->config.host, session->config.port);
99  KineticSession_Destroy(session);
100  session->connection = NULL;
101  return status;
102  }
103 
104  return status;
105 }

195 static KineticStatus handle_get_command(GET_COMMAND cmd,
196  KineticSession const * const session,
197  KineticEntry* const entry,
198  KineticCompletionClosure* closure)
199 {
200  assert(session != NULL);
201  assert(session->connection != NULL);
202  assert(entry != NULL);
203 
204  if (!has_key(entry)) {return KINETIC_STATUS_MISSING_KEY;}
205  if (!has_value_buffer(entry) && !entry->metadataOnly) {
206  return KINETIC_STATUS_MISSING_VALUE_BUFFER;
207  }
208 
209  KineticOperation* operation = KineticController_CreateOperation(session);
210  if (operation == NULL) {
211  return KINETIC_STATUS_MEMORY_ERROR;
212  }
213 
214  // Initialize request
215  switch (cmd)
216  {
217  case CMD_GET:
218  KineticOperation_BuildGet(operation, entry);
219  break;
220  case CMD_GET_NEXT:
221  KineticOperation_BuildGetNext(operation, entry);
222  break;
223  case CMD_GET_PREVIOUS:
224  KineticOperation_BuildGetPrevious(operation, entry);
225  break;
226  default:
227  assert(false);
228  }
229 
230  // Execute the operation
231  return KineticController_ExecuteOperation(operation, closure);
232 }
