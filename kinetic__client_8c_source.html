40 KineticStatus KineticClient_CreateConnection(KineticSession* const session)
41 {
42  if (session == NULL) {
43  LOG0("KineticSession is NULL!");
44  return KINETIC_STATUS_SESSION_EMPTY;
45  }
46 
47  if (strlen(session->config.host) == 0) {
48  LOG0("Host is empty!");
49  return KINETIC_STATUS_HOST_EMPTY;
50  }
51 
52  if (session->config.hmacKey.len < 1 || session->config.hmacKey.data == NULL) {
53  LOG0("HMAC key is NULL or empty!");
54  return KINETIC_STATUS_HMAC_EMPTY;
55  }
56 
57  KineticSession_Create(session);
58  if (session->connection == NULL) {
59  LOG0("Failed to create connection instance!");
60  return KINETIC_STATUS_CONNECTION_ERROR;
61  }
62 
63  // Create the connection
64  KineticStatus status = KineticSession_Connect(session);
65  if (status != KINETIC_STATUS_SUCCESS) {
66  LOGF0("Failed creating connection to %s:%d", session->config.host, session->config.port);
67  KineticSession_Destroy(session);
68  session->connection = NULL;
69  return status;
70  }
71 
72  return status;
73 }

158 static KineticStatus handle_get_command(GET_COMMAND cmd,
159  KineticSession const * const session,
160  KineticEntry* const entry,
161  KineticCompletionClosure* closure)
162 {
163  assert(session != NULL);
164  assert(session->connection != NULL);
165  assert(entry != NULL);
166 
167  if (!has_key(entry)) {return KINETIC_STATUS_MISSING_KEY;}
168  if (!has_value_buffer(entry) && !entry->metadataOnly) {
169  return KINETIC_STATUS_MISSING_VALUE_BUFFER;
170  }
171 
172  KineticOperation* operation = KineticController_CreateOperation(session);
173  if (operation == NULL) {
174  return KINETIC_STATUS_MEMORY_ERROR;
175  }
176 
177  // Initialize request
178  switch (cmd)
179  {
180  case CMD_GET:
181  KineticOperation_BuildGet(operation, entry);
182  break;
183  case CMD_GET_NEXT:
184  KineticOperation_BuildGetNext(operation, entry);
185  break;
186  case CMD_GET_PREVIOUS:
187  KineticOperation_BuildGetPrevious(operation, entry);
188  break;
189  default:
190  assert(false);
191  }
192 
193  // Execute the operation
194  return KineticController_ExecuteOperation(operation, closure);
195 }
