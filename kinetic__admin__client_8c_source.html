56 KineticStatus KineticAdminClient_SetErasePin(KineticSession const * const session,
57  ByteArray old_pin, ByteArray new_pin)
58 {
59  KineticStatus status;
60  status = KineticAuth_EnsureSslEnabled(&session->config);
61  if (status != KINETIC_STATUS_SUCCESS) {return status;}
62 
63  // Ensure PIN arrays have data if non-empty
64  if ((old_pin.len > 0 && old_pin.data == NULL) ||
65  (new_pin.len > 0 && new_pin.data == NULL)) {
66  return KINETIC_STATUS_MISSING_PIN;
67  }
68 
69  KineticOperation* operation = KineticAllocator_NewOperation(session->connection);
70  if (operation == NULL) {return KINETIC_STATUS_MEMORY_ERROR;}
71 
72  KineticOperation_BuildSetPin(operation, old_pin, new_pin, false);
73  return KineticController_ExecuteOperation(operation, NULL);
74 }
