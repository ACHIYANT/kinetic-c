62 bool KineticHMAC_Validate(const KineticProto* proto,
63  const ByteArray key)
64 {
65  bool success = false;
66  size_t i;
67  int result = 0;
68  KineticHMAC tempHMAC;
69 
70  if (proto->has_hmac)
71  {
72  KineticHMAC_Init(&tempHMAC, KINETIC_PROTO_SECURITY_ACL_HMACALGORITHM_HmacSHA1);
73  KineticHMAC_Compute(&tempHMAC, proto, key);
74  if (proto->hmac.len == tempHMAC.len)
75  {
76  for (i = 0; i < tempHMAC.len; i++)
77  {
78  result |= proto->hmac.data[i] ^ tempHMAC.data[i];
79  }
80  success = (result == 0);
81  }
82 
83  if (!success)
84  {
85  LOG("HMAC did not compare!");
86  ByteArray expected = {.data = proto->hmac.data, .len = proto->hmac.len};
87  KineticLogger_LogByteArray("expected HMAC", expected);
88  ByteArray actual = {.data = tempHMAC.data, .len = tempHMAC.len};
89  KineticLogger_LogByteArray("actual HMAC", actual);
90  }
91  }
92  return success;
93 }

97 static void KineticHMAC_Compute(KineticHMAC* hmac,
98  const KineticProto* proto,
99  const ByteArray key)
100 {
101  assert(proto->command);
102  uint32_t len = protobuf_c_message_get_packed_size((ProtobufCMessage*)proto->command);
103  uint32_t lenNBO = KineticNBO_FromHostU32(len);
104  uint8_t* packed = malloc(len);
105  assert(packed);
106  uint32_t lenPacked = protobuf_c_message_pack((ProtobufCMessage*)proto->command, packed);
107  assert(lenPacked == len);
108 
109  HMAC_CTX ctx;
110  HMAC_CTX_init(&ctx);
111  HMAC_Init_ex(&ctx, key.data, key.len, EVP_sha1(), NULL);
112  HMAC_Update(&ctx, (uint8_t*)&lenNBO, sizeof(uint32_t));
113  HMAC_Update(&ctx, packed, len);
114  HMAC_Final(&ctx, hmac->data, &hmac->len);
115  HMAC_CTX_cleanup(&ctx);
116 
117  free(packed);
118 }
