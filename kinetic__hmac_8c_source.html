53 bool KineticHMAC_Validate(const KineticProto* proto, const char* const key, size_t keyLen)
54 {
55  size_t i;
56  int result = 0;
57  KineticHMAC tempHMAC;
58 
59  if (!proto->has_hmac)
60  {
61  return false;
62  }
63 
64  KineticHMAC_Init(&tempHMAC, KINETIC_PROTO_SECURITY_ACL_HMACALGORITHM_HmacSHA1);
65  KineticHMAC_Compute(&tempHMAC, proto, key, keyLen);
66 
67  if (proto->hmac.len != tempHMAC.valueLength)
68  {
69  return false;
70  }
71 
72  for (i = 0; i < tempHMAC.valueLength; i++)
73  {
74  result |= proto->hmac.data[i] ^ tempHMAC.value[i];
75  }
76 
77  return (result == 0);
