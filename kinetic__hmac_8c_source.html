65 bool KineticHMAC_Validate(const KineticProto_Message* msg,
66  const ByteArray key)
67 {
68  assert(msg != NULL);
69  assert(key.data != NULL);
70  assert(key.len > 0);
71 
72  bool success = false;
73  size_t i;
74  int result = 0;
75  KineticHMAC tempHMAC;
76 
77  if (!msg->has_authType
78  || msg->authType != KINETIC_PROTO_MESSAGE_AUTH_TYPE_HMACAUTH
79  || msg->hmacAuth == NULL
80  || !msg->hmacAuth->has_hmac
81  || msg->hmacAuth->hmac.data == NULL
82  || msg->hmacAuth->hmac.len == 0) {
83  return false;
84  }
85 
86  KineticHMAC_Init(&tempHMAC, KINETIC_PROTO_COMMAND_SECURITY_ACL_HMACALGORITHM_HmacSHA1);
87  KineticHMAC_Compute(&tempHMAC, msg, key);
88  if (msg->hmacAuth->hmac.len == tempHMAC.len) {
89  for (i = 0; i < tempHMAC.len; i++) {
90  result |= msg->hmacAuth->hmac.data[i] ^ tempHMAC.data[i];
91  }
92  success = (result == 0);
93  }
94 
95  if (!success) {
96  LOG0("HMAC did not compare!");
97  ByteArray expected = {.data = msg->hmacAuth->hmac.data, .len = msg->hmacAuth->hmac.len};
98  KineticLogger_LogByteArray(1, "expected HMAC", expected);
99  ByteArray actual = {.data = tempHMAC.data, .len = tempHMAC.len};
100  KineticLogger_LogByteArray(1, "actual HMAC", actual);
101  }
102 
103  return success;
104 }

108 static void KineticHMAC_Compute(KineticHMAC* hmac,
109  const KineticProto_Message* msg,
110  const ByteArray key)
111 {
112  assert(hmac != NULL);
113  assert(hmac->data != NULL);
114  assert(hmac->len > 0);
115  assert(msg != NULL);
116  assert(msg->has_commandBytes);
117  assert(msg->commandBytes.data != NULL);
118  assert(msg->commandBytes.len > 0);
119 
120  uint32_t lenNBO = KineticNBO_FromHostU32(msg->commandBytes.len);
121 
122  HMAC_CTX ctx;
123  HMAC_CTX_init(&ctx);
124  HMAC_Init_ex(&ctx, key.data, key.len, EVP_sha1(), NULL);
125  HMAC_Update(&ctx, (uint8_t*)&lenNBO, sizeof(uint32_t));
126  HMAC_Update(&ctx, msg->commandBytes.data, msg->commandBytes.len);
127  HMAC_Final(&ctx, hmac->data, &hmac->len);
128  HMAC_CTX_cleanup(&ctx);
129 }
