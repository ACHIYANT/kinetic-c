29 void ListenerCmd_NotifyCaller(int fd) {
30  if (fd == -1) { return; }
31  uint8_t reply_buf[sizeof(uint8_t) + sizeof(uint16_t)] = {LISTENER_MSG_TAG};
32 
33  /* TODO: reply_buf[1:2] can be little-endian backpressure. */
34 
35  for (;;) {
36  ssize_t wres = write(fd, reply_buf, sizeof(reply_buf));
37  if (wres == sizeof(reply_buf)) { break; }
38  if (wres == -1) {
39  if (errno == EINTR) {
40  errno = 0;
41  } else {
42  err(1, "write");
43  }
44  }
45  }
46 }

48 void ListenerCmd_CheckIncomingMessages(listener *l, int *res) {
49  struct bus *b = l->bus;
50  short events = l->fds[INCOMING_MSG_PIPE_ID].revents;
51  if (events & (POLLERR | POLLHUP | POLLNVAL)) { /* hangup/error */
52  return;
53  }
54 
55  if (events & POLLIN) {
56  char buf[64];
57  for (;;) {
58  ssize_t rd = read(l->fds[INCOMING_MSG_PIPE_ID].fd, buf, sizeof(buf));
59  if (rd == -1) {
60  if (errno == EINTR) {
61  errno = 0;
62  continue;
63  } else {
64  BUS_LOG_SNPRINTF(b, 6, LOG_LISTENER, b->udata, 128,
65  "check_and_flush_incoming_msg_pipe: %s", strerror(errno));
66  errno = 0;
67  break;
68  }
69  } else {
70  for (ssize_t i = 0; i < rd; i++) {
71  uint8_t msg_id = buf[i];
72  listener_msg *msg = &l->msgs[msg_id];
73  msg_handler(l, msg);
74  }
75  (*res)--;
76  break;
77  }
78  }
79  }
80 }

82 static void msg_handler(listener *l, listener_msg *pmsg) {
83  struct bus *b = l->bus;
84  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 128,
85  "Handling message -- %p", (void*)pmsg);
86 
87  l->is_idle = false;
88 
89  listener_msg msg = *pmsg;
90  switch (msg.type) {
91 
92  case MSG_ADD_SOCKET:
93  add_socket(l, msg.u.add_socket.info, msg.u.add_socket.notify_fd);
94  break;
95  case MSG_REMOVE_SOCKET:
96  remove_socket(l, msg.u.remove_socket.fd, msg.u.remove_socket.notify_fd);
97  break;
98  case MSG_HOLD_RESPONSE:
99  hold_response(l, msg.u.hold.fd, msg.u.hold.seq_id,
100  msg.u.hold.timeout_sec);
101  break;
102  case MSG_EXPECT_RESPONSE:
103  expect_response(l, msg.u.expect.box);
104  break;
105  case MSG_SHUTDOWN:
106  shutdown(l, msg.u.shutdown.notify_fd);
107  break;
108 
109  case MSG_NONE:
110  default:
111  BUS_ASSERT(b, b->udata, false);
112  break;
113  }
114  ListenerTask_ReleaseMsg(l, pmsg);
115 }

117 static void add_socket(listener *l, connection_info *ci, int notify_fd) {
118  /* TODO: if epoll, just register with the OS. */
119  struct bus *b = l->bus;
120  BUS_LOG(b, 3, LOG_LISTENER, "adding socket", b->udata);
121 
122  if (l->tracked_fds == MAX_FDS) {
123  /* error: full */
124  BUS_LOG(b, 3, LOG_LISTENER, "FULL", b->udata);
125  }
126  for (int i = 0; i < l->tracked_fds; i++) {
127  if (l->fds[i + INCOMING_MSG_PIPE].fd == ci->fd) {
128  free(ci);
129  ListenerCmd_NotifyCaller(notify_fd);
130  return; /* already present */
131  }
132  }
133 
134  int id = l->tracked_fds;
135  l->fd_info[id] = ci;
136  l->fds[id + INCOMING_MSG_PIPE].fd = ci->fd;
137  l->fds[id + INCOMING_MSG_PIPE].events = POLLIN;
138  l->tracked_fds++;
139 
140  /* Prime the pump by sinking 0 bytes and getting a size to expect. */
141  bus_sink_cb_res_t sink_res = b->sink_cb(l->read_buf, 0, ci->udata);
142  BUS_ASSERT(b, b->udata, sink_res.full_msg_buffer == NULL); // should have nothing to handle yet
143  ci->to_read_size = sink_res.next_read;
144 
145  //if (!grow_read_buf(l, ci->to_read_size)) {
146  if (!ListenerTask_GrowReadBuf(l, ci->to_read_size)) {
147  free(ci);
148  ListenerCmd_NotifyCaller(notify_fd);
149  return; /* alloc failure */
150  }
151 
152  BUS_LOG(b, 3, LOG_LISTENER, "added socket", b->udata);
153  ListenerCmd_NotifyCaller(notify_fd);
154 }

156 static void remove_socket(listener *l, int fd, int notify_fd) {
157  struct bus *b = l->bus;
158  BUS_LOG_SNPRINTF(b, 2, LOG_LISTENER, b->udata, 128,
159  "removing socket %d", fd);
160 
161  /* Don't really close it, just drop info about it in the listener.
162  * The client thread will actually free the structure, close SSL, etc. */
163  for (int i = 0; i < l->tracked_fds; i++) {
164  if (l->fds[i + INCOMING_MSG_PIPE].fd == fd) {
165  if (l->tracked_fds > 1) {
166  /* Swap pollfd CI and last ones. */
167  struct pollfd pfd = l->fds[i + INCOMING_MSG_PIPE];
168  l->fds[i + INCOMING_MSG_PIPE] = l->fds[l->tracked_fds - 1 + INCOMING_MSG_PIPE];
169  l->fds[l->tracked_fds - 1 + INCOMING_MSG_PIPE] = pfd;
170  connection_info *ci = l->fd_info[i];
171  l->fd_info[i] = l->fd_info[l->tracked_fds - 1];
172  l->fd_info[l->tracked_fds - 1] = ci;
173  }
174  l->tracked_fds--;
175  }
176  }
177  /* CI will be freed by the client thread. */
178  ListenerCmd_NotifyCaller(notify_fd);
179 }

181 static void hold_response(listener *l, int fd, int64_t seq_id, int16_t timeout_sec) {
182  struct bus *b = l->bus;
183 
184  rx_info_t *info = get_free_rx_info(l);
185  if (info == NULL) {
186  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 128,
187  "failed to get free rx_info for <fd:%d, seq_id:%lld>, dropping it",
188  fd, (long long)seq_id);
189  return;
190  }
191  BUS_ASSERT(b, b->udata, info);
192  BUS_ASSERT(b, b->udata, info->state == RIS_INACTIVE);
193  BUS_LOG_SNPRINTF(b, 5, LOG_LISTENER, b->udata, 128,
194  "setting info %p(+%d) to hold response <fd:%d, seq_id:%lld>",
195  (void *)info, info->id, fd, (long long)seq_id);
196 
197  info->state = RIS_HOLD;
198  info->timeout_sec = timeout_sec;
199  info->u.hold.fd = fd;
200  info->u.hold.seq_id = seq_id;
201  info->u.hold.has_result = false;
202  memset(&info->u.hold.result, 0, sizeof(info->u.hold.result));
203 }

205 static rx_info_t *get_hold_rx_info(listener *l, int fd, int64_t seq_id) {
206  for (int i = 0; i <= l->rx_info_max_used; i++) {
207  rx_info_t *info = &l->rx_info[i];
208  if (info->state == RIS_HOLD &&
209  info->u.hold.fd == fd &&
210  info->u.hold.seq_id == seq_id) {
211  return info;
212  }
213  }
214  return NULL;

217 static void expect_response(listener *l, struct boxed_msg *box) {
218  struct bus *b = l->bus;
219  BUS_ASSERT(b, b->udata, box);
220  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 128,
221  "notifying to expect response <box:%p, fd:%d, seq_id:%lld>",
222  (void *)box, box->fd, (long long)box->out_seq_id);
223 
224  /* If there's a pending HOLD message, convert it. */
225  rx_info_t *info = get_hold_rx_info(l, box->fd, box->out_seq_id);
226  if (info) {
227  BUS_ASSERT(b, b->udata, info->state == RIS_HOLD);
228  if (info->u.hold.has_result) {
229  bus_unpack_cb_res_t result = info->u.hold.result;
230 
231  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 256,
232  "converting HOLD to EXPECT for info %d (%p) with result, attempting delivery <box:%p, fd:%d, seq_id:%lld>",
233  info->id, (void *)info,
234  (void *)box, info->u.hold.fd, (long long)info->u.hold.seq_id);
235 
236  info->state = RIS_EXPECT;
237  info->u.expect.error = RX_ERROR_READY_FOR_DELIVERY;
238  info->u.expect.box = box;
239  info->u.expect.has_result = true;
240  info->u.expect.result = result;
241  ListenerTask_AttemptDelivery(l, info);
242  } else {
243  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 256,
244  "converting HOLD to EXPECT info %d (%p), attempting delivery <box:%p, fd:%d, seq_id:%lld>",
245  info->id, (void *)info,
246  (void *)box, info->u.hold.fd, (long long)info->u.hold.seq_id);
247  info->state = RIS_EXPECT;
248  info->u.expect.box = box;
249  info->u.expect.error = RX_ERROR_NONE;
250  info->u.expect.has_result = false;
251  /* Switch over to client's transferred timeout */
252  info->timeout_sec = box->timeout_sec;
253  }
254  } else { /* use free info */
255  /* If we get here, the listener thinks the HOLD message timed out,
256  * but the client doesn't think things timed out badly enough to
257  * itself expose an error. We also don't know if we're going to
258  * get a response or not. */
259 
260  BUS_LOG_SNPRINTF(b, 0, LOG_MEMORY, b->udata, 128,
261  "get_hold_rx_info FAILED: fd %d, seq_id %lld",
262  box->fd, (long long)box->out_seq_id);
263 
264  /* This should be treated like a send timeout. */
265  info = get_free_rx_info(l);
266  BUS_ASSERT(b, b->udata, info);
267  BUS_ASSERT(b, b->udata, info->state == RIS_INACTIVE);
268 
269  BUS_LOG_SNPRINTF(b, 0, LOG_MEMORY, b->udata, 256,
270  "Setting info %p (+%d)'s box to %p, which will be expired immediately (timeout %lld)",
271  (void*)info, info->id, (void*)box, (long long)box->timeout_sec);
272 
273  info->state = RIS_EXPECT;
274  BUS_ASSERT(b, b->udata, info->u.expect.box == NULL);
275  info->u.expect.box = box;
276  info->u.expect.error = RX_ERROR_NONE;
277  info->u.expect.has_result = false;
278  /* Switch over to client's transferred timeout */
279  ListenerTask_NotifyMessageFailure(l, info, BUS_SEND_RX_TIMEOUT_EXPECT);
280  }
281 }

287 static rx_info_t *get_free_rx_info(struct listener *l) {
288  struct bus *b = l->bus;
289 
290  struct rx_info_t *head = l->rx_info_freelist;
291  if (head == NULL) {
292  BUS_LOG(b, 6, LOG_SENDER, "No rx_info cells left!", b->udata);
293  return NULL;
294  } else {
295  l->rx_info_freelist = head->next;
296  head->next = NULL;
297  l->rx_info_in_use++;
298  BUS_LOG(l->bus, 4, LOG_LISTENER, "reserving RX info", l->bus->udata);
299  BUS_ASSERT(b, b->udata, head->state == RIS_INACTIVE);
300  if (l->rx_info_max_used < head->id) {
301  BUS_LOG_SNPRINTF(b, 5, LOG_LISTENER, b->udata, 128,
302  "rx_info_max_used <- %d", head->id);
303  l->rx_info_max_used = head->id;
304  BUS_ASSERT(b, b->udata, l->rx_info_max_used < MAX_PENDING_MESSAGES);
305  }
306 
307  BUS_LOG_SNPRINTF(b, 5, LOG_LISTENER, b->udata, 128,
308  "got free rx_info_t %d (%p)", head->id, (void *)head);
309  BUS_ASSERT(b, b->udata, head == &l->rx_info[head->id]);
310  return head;
311  }
