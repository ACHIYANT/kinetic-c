43 {
44  assert(operation != NULL);
45  assert(operation->connection != NULL);
46  assert(operation->request != NULL);

48  KineticStatus status = KineticOperation_SendRequestInner(operation);
49  if (status != KINETIC_STATUS_SUCCESS)

51  //cleanup

53  if (request->message.message.commandBytes.data != NULL) {

55  request->message.message.commandBytes.data = NULL;
56  }
57  }
58  return status;
59 }

62 static KineticStatus KineticOperation_SendRequestInner(KineticOperation* const operation)
63 {
64  LOGF3("\nSending PDU via fd=%d", operation->connection->socket);
65 
66  KineticStatus status = KINETIC_STATUS_INVALID;
67  uint8_t * msg = NULL;
68  KineticPDU* request = operation->request;
69  KineticProto_Message* proto = &operation->request->message.message;
70  pthread_mutex_t* sendMutex = &operation->connection->sendMutex;
71 
72  // Acquire lock
73  pthread_mutex_lock(sendMutex);
74  KineticSession *session = operation->connection->pSession;
75 
76  // Populate sequence count and increment it for next operation
77  assert(request->message.header.sequence == KINETIC_SEQUENCE_NOT_YET_BOUND);
78 
79  // int seq_id = ATOMIC_FETCH_AND_INCREMENT(&operation->connection->sequence);
80  // request->message.header.sequence = seq_id;
81  int64_t seq_id = operation->connection->sequence++;
82  request->message.header.sequence = seq_id;
83 
84  // Pack the command, if available
85  size_t expectedLen = KineticProto_command__get_packed_size(&request->message.command);
86  request->message.message.commandBytes.data = (uint8_t*)malloc(expectedLen);
87  if(request->message.message.commandBytes.data == NULL)
88  {
89  LOG0("Failed to allocate command bytes!");
90  status = KINETIC_STATUS_MEMORY_ERROR;
91  goto cleanup;
92  }
93  size_t packedLen = KineticProto_command__pack(
94  &request->message.command,
95  request->message.message.commandBytes.data);
96  assert(packedLen == expectedLen);
97  request->message.message.commandBytes.len = packedLen;
98  request->message.message.has_commandBytes = true;
99  KineticLogger_LogByteArray(3, "commandBytes", (ByteArray){
100  .data = request->message.message.commandBytes.data,
101  .len = request->message.message.commandBytes.len,
102  });
103 
104  switch (proto->authType) {
105  case KINETIC_PROTO_MESSAGE_AUTH_TYPE_PINAUTH:
106  /* TODO: If operation uses PIN AUTH, then init that */
107  break;
108  case KINETIC_PROTO_MESSAGE_AUTH_TYPE_HMACAUTH:
109  {
110  KineticHMAC hmac;
111  memset(&hmac, 0, sizeof(hmac));
112 
113  // Populate the HMAC for the protobuf
114  KineticHMAC_Init(&hmac, KINETIC_PROTO_COMMAND_SECURITY_ACL_HMACALGORITHM_HmacSHA1);
115  KineticHMAC_Populate(&hmac, proto, session->config.hmacKey);
116  } break;
117  default:
118  break;
119  }
120 
121  KineticPDUHeader header;
122  memset(&header, 0, sizeof(header));
123 
124  // Configure PDU header length fields
125  header.versionPrefix = 'F';
126  header.protobufLength = KineticProto_Message__get_packed_size(proto);
127  if (operation->entry != NULL && operation->sendValue) {
128  header.valueLength = operation->entry->value.bytesUsed;
129  }
130  else {
131  header.valueLength = 0;
132  }
133  if (header.valueLength > PDU_PROTO_MAX_LEN) {
134  // Packed value exceeds max size.
135  LOGF2("\nPacked value exceeds maximum size. Packed size is: %d, Max size is: %d", header.valueLength, PDU_PROTO_MAX_LEN);
136  status = KINETIC_STATUS_BUFFER_OVERRUN;
137  goto cleanup;
138  }
139  uint32_t nboProtoLength = KineticNBO_FromHostU32(header.protobufLength);
140  uint32_t nboValueLength = KineticNBO_FromHostU32(header.valueLength);
141 
142  // Allocate and pack message
143  size_t offset = 0;
144  msg = malloc(PDU_HEADER_LEN + header.protobufLength + header.valueLength);
145  if (msg == NULL) {
146  LOG0("Failed to allocate outgoing message!");
147  status = KINETIC_STATUS_MEMORY_ERROR;
148  goto cleanup;
149  }
150 
151  msg[offset] = header.versionPrefix;
152  offset += sizeof(header.versionPrefix);
153  memcpy(&msg[offset], &nboProtoLength, sizeof(nboProtoLength));
154  offset += sizeof(nboProtoLength);
155  memcpy(&msg[offset], &nboValueLength, sizeof(nboValueLength));
156  offset += sizeof(nboValueLength);
157  size_t len = KineticProto_Message__pack(&request->message.message, &msg[offset]);
158  assert(len == header.protobufLength);
159  offset += header.protobufLength;
160 
161  assert(operation);
162  assert(operation->request);
163  assert(operation->connection);
164  assert(request);
165 
166  LOGF2("[PDU TX] pdu: %p, session: %p, bus: %p, "
167  "fd: %6d, seq: %8lld, protoLen: %8u, valueLen: %8u, op: %p",
168  (void*)operation->request,
169  (void*)operation->connection->pSession, (void*)operation->connection->messageBus,
170  operation->connection->socket, request->message.header.sequence,
171  header.protobufLength, header.valueLength,
172  (void*)operation);
173  KineticLogger_LogHeader(3, &header);
174  KineticLogger_LogProtobuf(3, proto);
175 
176  free(request->message.message.commandBytes.data);
177  request->message.message.commandBytes.data = NULL;
178 
179  // Send the value/payload, if specified
180  if (header.valueLength > 0) {
181  memcpy(&msg[offset], operation->entry->value.array.data, operation->entry->value.bytesUsed);
182  offset += operation->entry->value.bytesUsed;
183  }
184  assert((PDU_HEADER_LEN + header.protobufLength + header.valueLength) == offset);
185 
186  // Claim a send PDU for proper throttling of concurrent requests to avoid flooding the drive
187  KineticCountingSemaphore * const sem = operation->connection->outstandingOperations;
188  KineticCountingSemaphore_Take(sem);
189  int fd = operation->connection->socket;
190  if (!bus_send_request(operation->connection->messageBus, &(bus_user_msg){
191  .fd = fd,
192  .type = BUS_SOCKET_PLAIN,
193  .seq_id = seq_id,
194  .msg = msg,
195  .msg_size = offset,
196  .cb = KineticController_HandleResult,
197  .udata = operation,
198  }))
199  {
200  LOGF0("Failed queuing request %p for transmit on fd=%d w/seq=%lld",
201  (void*)request, fd, (long long)seq_id);
202  // Since the request PDU failed getting queued into the bus,
203  // release this request from the throttle to clean up properly
204  KineticCountingSemaphore_Give(sem);
205  status = KINETIC_STATUS_SOCKET_ERROR;
206  }
207  else {
208  status = KINETIC_STATUS_SUCCESS;
209  }
210 
211 cleanup:
212 
213  pthread_mutex_unlock(sendMutex);
214 
215  if (msg != NULL) { free(msg); }
216 
217  return status;
218 }

249 KineticStatus KineticOperation_PutCallback(KineticOperation* const operation, KineticStatus const status)
250 {
251  assert(operation != NULL);
252  assert(operation->connection != NULL);
253  LOGF3("PUT callback w/ operation (0x%0llX) on connection (0x%0llX)",
254  operation, operation->connection);
255  assert(operation->entry != NULL);
256 
257  if (status == KINETIC_STATUS_SUCCESS)
258  {
259  assert(operation->response != NULL);
260  // Propagate newVersion to dbVersion in metadata, if newVersion specified
261  KineticEntry* entry = operation->entry;
262  if (entry->newVersion.array.data != NULL && entry->newVersion.array.len > 0) {
263  // If both buffers supplied, copy newVersion into dbVersion, and clear newVersion
264  if (entry->dbVersion.array.data != NULL && entry->dbVersion.array.len > 0) {
265  ByteBuffer_Reset(&entry->dbVersion);
266  ByteBuffer_Append(&entry->dbVersion, entry->newVersion.array.data, entry->newVersion.bytesUsed);
267  ByteBuffer_Reset(&entry->newVersion);
268  }
269 
270  // If only newVersion buffer supplied, move newVersion buffer into dbVersion,
271  // and set newVersion to NULL buffer
272  else {
273  entry->dbVersion = entry->newVersion;
274  entry->newVersion = BYTE_BUFFER_NONE;
275  }
276  }
277  }
278  return status;
279 }

297 static KineticStatus get_cb(const char *cmd_name, KineticOperation* const operation, KineticStatus const status)
298 {
299  assert(operation != NULL);
300  assert(operation->connection != NULL);
301  LOGF3("%s callback w/ operation (0x%0llX) on connection (0x%0llX)",
302  cmd_name, operation, operation->connection);
303  assert(operation->entry != NULL);
304 
305  if (status == KINETIC_STATUS_SUCCESS)
306  {
307  assert(operation->response != NULL);
308  // Update the entry upon success
309  KineticProto_Command_KeyValue* keyValue = KineticPDU_GetKeyValue(operation->response);
310  if (keyValue != NULL) {
311  if (!Copy_KineticProto_Command_KeyValue_to_KineticEntry(keyValue, operation->entry)) {
312  return KINETIC_STATUS_BUFFER_OVERRUN;
313  }
314  }
315 
316  if (!operation->entry->metadataOnly &&
317  !ByteBuffer_IsNull(operation->entry->value))
318  {
319  ByteBuffer_AppendArray(&operation->entry->value, (ByteArray){
320  .data = operation->response->value,
321  .len = operation->response->header.valueLength,
322  });
323  }
324  }
325 
326  return status;
327 }

552 KineticProto_Command_P2POperation* build_p2pOp(uint32_t nestingLevel, KineticP2P_Operation const * const p2pOp)

556  LOG0("P2P operation nesting level is too deep. Max is 1000.");
557  return NULL;
558  }

561  if (proto_p2pOp == NULL) { goto error_cleanup; }
562 
563  KineticProto_command_p2_poperation__init(proto_p2pOp);
564 
565  proto_p2pOp->peer = calloc(1, sizeof(KineticProto_Command_P2POperation_Peer));
566  if (proto_p2pOp->peer == NULL) { goto error_cleanup; }
567 
568  KineticProto_command_p2_poperation_peer__init(proto_p2pOp->peer);
569 
570  proto_p2pOp->peer->hostname = p2pOp->peer.hostname;
571  proto_p2pOp->peer->has_port = true;
572  proto_p2pOp->peer->port = p2pOp->peer.port;
573  proto_p2pOp->peer->has_tls = true;
574  proto_p2pOp->peer->tls = p2pOp->peer.tls;
575 
576  proto_p2pOp->n_operation = p2pOp->numOperations;
577  proto_p2pOp->operation = calloc(p2pOp->numOperations, sizeof(KineticProto_Command_P2POperation_Operation*));
578  if (proto_p2pOp->operation == NULL) { goto error_cleanup; }
579 
580  for(size_t i = 0; i < proto_p2pOp->n_operation; i++) {
581  assert(!ByteBuffer_IsNull(p2pOp->operations[i].key)); // TODO return invalid operand?
582 
583  KineticProto_Command_P2POperation_Operation * p2p_op_op = calloc(1, sizeof(KineticProto_Command_P2POperation_Operation));
584  if (p2p_op_op == NULL) { goto error_cleanup; }
585 
586  KineticProto_command_p2_poperation_operation__init(p2p_op_op);
587 
588  p2p_op_op->has_key = true;
589  p2p_op_op->key.data = p2pOp->operations[i].key.array.data;
590  p2p_op_op->key.len = p2pOp->operations[i].key.bytesUsed;
591 
592  p2p_op_op->has_newKey = !ByteBuffer_IsNull(p2pOp->operations[i].newKey);
593  p2p_op_op->newKey.data = p2pOp->operations[i].newKey.array.data;
594  p2p_op_op->newKey.len = p2pOp->operations[i].newKey.bytesUsed;
595 
596  p2p_op_op->has_version = !ByteBuffer_IsNull(p2pOp->operations[i].version);
597  p2p_op_op->version.data = p2pOp->operations[i].version.array.data;
598  p2p_op_op->version.len = p2pOp->operations[i].version.bytesUsed;
599
