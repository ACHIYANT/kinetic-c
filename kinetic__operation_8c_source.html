26 KineticStatus KineticOperation_GetStatus(const KineticOperation* const operation)
27 {
28  KineticStatus status = KINETIC_STATUS_INVALID;
29 
30  if (operation != NULL &&
31  operation->response != NULL &&
32  operation->response->proto != NULL &&
33  operation->response->proto->command != NULL &&
34  operation->response->proto->command->status != NULL &&
35  operation->response->proto->command->status->has_code != false &&
36  operation->response->proto->command->status->code !=
37  KINETIC_PROTO_STATUS_STATUS_CODE_INVALID_STATUS_CODE)
38  {
39  switch(operation->response->proto->command->status->code)
40  {
41  case KINETIC_PROTO_STATUS_STATUS_CODE_SUCCESS:
42  status = KINETIC_STATUS_SUCCESS; break;
43 
44  case KINETIC_PROTO_STATUS_STATUS_CODE_REMOTE_CONNECTION_ERROR:
45  status = KINETIC_STATUS_CONNECTION_ERROR; break;
46 
47  case KINETIC_PROTO_STATUS_STATUS_CODE_SERVICE_BUSY:
48  status = KINETIC_STATUS_DEVICE_BUSY; break;
49 
50  case KINETIC_PROTO_STATUS_STATUS_CODE_INVALID_REQUEST:
51  case KINETIC_PROTO_STATUS_STATUS_CODE_NOT_ATTEMPTED:
52  case KINETIC_PROTO_STATUS_STATUS_CODE_HEADER_REQUIRED:
53  case KINETIC_PROTO_STATUS_STATUS_CODE_NO_SUCH_HMAC_ALGORITHM:
54  status = KINETIC_STATUS_INVALID_REQUEST; break;
55 
56  case KINETIC_PROTO_STATUS_STATUS_CODE_VERSION_MISMATCH:
57  case KINETIC_PROTO_STATUS_STATUS_CODE_VERSION_FAILURE:
58  status = KINETIC_STATUS_VERSION_FAILURE; break;
59 
60  case KINETIC_PROTO_STATUS_STATUS_CODE_DATA_ERROR:
61  case KINETIC_PROTO_STATUS_STATUS_CODE_HMAC_FAILURE:
62  case KINETIC_PROTO_STATUS_STATUS_CODE_PERM_DATA_ERROR:
63  case KINETIC_PROTO_STATUS_STATUS_CODE_NOT_FOUND:
64  status = KINETIC_STATUS_DATA_ERROR; break;
65 
66  case KINETIC_PROTO_STATUS_STATUS_CODE_INTERNAL_ERROR:
67  case KINETIC_PROTO_STATUS_STATUS_CODE_NOT_AUTHORIZED:
68  case KINETIC_PROTO_STATUS_STATUS_CODE_EXPIRED:
69  case KINETIC_PROTO_STATUS_STATUS_CODE_NO_SPACE:
70  case KINETIC_PROTO_STATUS_STATUS_CODE_NESTED_OPERATION_ERRORS:
71  status = KINETIC_STATUS_OPERATION_FAILED; break;
72 
73  default:
74  case KINETIC_PROTO_STATUS_STATUS_CODE_INVALID_STATUS_CODE:
75  case _KINETIC_PROTO_STATUS_STATUS_CODE_IS_INT_SIZE:
76  status = KINETIC_STATUS_INVALID; break;
77  }
78  }
79  return status;
80 }
