36 static KineticStatus WritePDU(KineticOperation* const operation)
37 {
38  KineticPDU* request = operation->request;
39 
40  // Cork the PDU to begin aggregation
41  KineticSocket_BeginPacket(request->connection->socket);
42 
43  // Pack and send the PDU header
44  ByteBuffer hdr = ByteBuffer_Create(&request->headerNBO, sizeof(KineticPDUHeader), sizeof(KineticPDUHeader));
45  KineticStatus status = KineticSocket_Write(request->connection->socket, &hdr);
46  if (status != KINETIC_STATUS_SUCCESS) {
47  LOG0("Failed to send PDU header!");
48  KineticSocket_FinishPacket(request->connection->socket);
49  return status;
50  }
51 
52  // Send the protobuf message
53  LOG1("Sending PDU Protobuf:");
54  KineticLogger_LogProtobuf(2, request->proto);
55  status = KineticSocket_WriteProtobuf(request->connection->socket, request);
56  if (status != KINETIC_STATUS_SUCCESS) {
57  LOG0("Failed to send PDU protobuf message!");
58  KineticSocket_FinishPacket(request->connection->socket);
59  return status;
60  }
61 
62  // Send the value/payload, if specified
63  if (operation->valueEnabled && operation->sendValue) {
64  LOGF1("Sending PDU Value Payload (%zu bytes)", operation->entry->value.bytesUsed);
65  status = KineticSocket_Write(request->connection->socket, &operation->entry->value);
66  if (status != KINETIC_STATUS_SUCCESS) {
67  LOG0("Failed to send PDU value payload!");
68  KineticSocket_FinishPacket(request->connection->socket);
69  return status;
70  }
71  }
72 
73  // Uncork the PDU to begin transmission
74  KineticSocket_FinishPacket(request->connection->socket);
75 
76  LOG2("PDU sent successfully!");
77  return KINETIC_STATUS_SUCCESS;
78 }

80 KineticStatus KineticOperation_SendRequest(KineticOperation* const operation)
81 {
82  assert(operation != NULL);
83  assert(operation->connection != NULL);
84  assert(operation->request != NULL);
85  assert(operation->request->connection == operation->connection);
86  LOGF1("\nSending PDU via fd=%d", operation->connection->socket);
87  KineticPDU* request = operation->request;
88  request->proto = &operation->request->protoData.message.message;
89 
90  // Pack the command, if available
91  if (request->protoData.message.has_command) {
92  size_t expectedLen = KineticProto_command__get_packed_size(&request->protoData.message.command);
93  request->protoData.message.message.commandBytes.data = (uint8_t*)malloc(expectedLen);
94  assert(request->protoData.message.message.commandBytes.data != NULL);
95  size_t packedLen = KineticProto_command__pack(
96  &request->protoData.message.command,
97  request->protoData.message.message.commandBytes.data);
98  assert(packedLen == expectedLen);
99  request->protoData.message.message.commandBytes.len = packedLen;
100  request->protoData.message.message.has_commandBytes = true;
101  KineticLogger_LogByteArray(2, "commandBytes", (ByteArray){
102  .data = request->protoData.message.message.commandBytes.data,
103  .len = request->protoData.message.message.commandBytes.len,
104  });
105  }
106 
107  // Populate the HMAC for the protobuf
108  KineticHMAC_Init(&request->hmac, KINETIC_PROTO_COMMAND_SECURITY_ACL_HMACALGORITHM_HmacSHA1);
109  KineticHMAC_Populate(&request->hmac, request->proto, request->connection->session.config.hmacKey);
110 
111  // Configure PDU header length fields
112  request->header.versionPrefix = 'F';
113  request->header.protobufLength = KineticProto_Message__get_packed_size(request->proto);
114  if (operation->entry != NULL && operation->sendValue) {
115  request->header.valueLength = operation->entry->value.bytesUsed;
116  }
117  else {
118  request->header.valueLength = 0;
119  }
120  KineticLogger_LogHeader(1, &request->header);
121 
122  // Create NBO copy of header for sending
123  request->headerNBO.versionPrefix = 'F';
124  request->headerNBO.protobufLength = KineticNBO_FromHostU32(request->header.protobufLength);
125  request->headerNBO.valueLength = KineticNBO_FromHostU32(request->header.valueLength);
126 
127  pthread_mutex_lock(&operation->connection->writeMutex);
128  KineticOperation_SetTimeoutTime(operation, KINETIC_OPERATION_TIMEOUT_SECS);
129  KineticStatus status = WritePDU(operation);
130  pthread_mutex_unlock(&operation->connection->writeMutex);
131  return status;
132 }

167 KineticOperation* KineticOperation_AssociateResponseWithOperation(KineticPDU* response)
168 {
169  if (response == NULL ||
170  response->command == NULL ||
171  response->command->header == NULL ||
172  !response->command->header->has_ackSequence ||
173  response->type != KINETIC_PDU_TYPE_RESPONSE)
174  {
175  LOG0("Response to associate with request is invalid!");
176  return NULL;
177  }
178 
179  const int64_t targetSequence = response->command->header->ackSequence;
180  KineticOperation* operation = KineticAllocator_GetFirstOperation(response->connection);
181  if (operation == NULL) {
182  LOG2("ERROR: No pending operations found!");
183  return NULL;
184  }
185 
186  while (operation != NULL) {
187  if (operation->request != NULL &&
188  operation->request->type == KINETIC_PDU_TYPE_REQUEST &&
189  operation->request->command != NULL &&
190  operation->request->command->header != NULL &&
191  operation->request->command->header->has_sequence)
192  {
193  LOGF3("Comparing received PDU w/ ackSequence=%lld with request with sequence=%lld",
194  targetSequence, operation->request->command->header->sequence);
195  if (operation->request->command->header->sequence == targetSequence) {
196  operation->response = response;
197  return operation;
198  }
199  }
200  operation = KineticAllocator_GetNextOperation(response->connection, operation);
201  }
202  return NULL;
203 }

225 KineticStatus KineticOperation_PutCallback(KineticOperation* const operation, KineticStatus const status)
226 {
227  assert(operation != NULL);
228  assert(operation->connection != NULL);
229  LOGF3("PUT callback w/ operation (0x%0llX) on connection (0x%0llX)",
230  operation, operation->connection);
231  assert(operation->response != NULL);
232  assert(operation->entry != NULL);
233 
234  if (status == KINETIC_STATUS_SUCCESS)
235  {
236  // Propagate newVersion to dbVersion in metadata, if newVersion specified
237  KineticEntry* entry = operation->entry;
238  if (entry->newVersion.array.data != NULL && entry->newVersion.array.len > 0) {
239  // If both buffers supplied, copy newVersion into dbVersion, and clear newVersion
240  if (entry->dbVersion.array.data != NULL && entry->dbVersion.array.len > 0) {
241  ByteBuffer_Reset(&entry->dbVersion);
242  ByteBuffer_Append(&entry->dbVersion, entry->newVersion.array.data, entry->newVersion.bytesUsed);
243  ByteBuffer_Reset(&entry->newVersion);
244  }
245 
246  // If only newVersion buffer supplied, move newVersion buffer into dbVersion,
247  // and set newVersion to NULL buffer
248  else {
249  entry->dbVersion = entry->newVersion;
250  entry->newVersion = BYTE_BUFFER_NONE;
251  }
252  }
253  }
254  return status;
255 }
