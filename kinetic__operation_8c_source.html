40 KineticOperation KineticOperation_Create(KineticConnection* const connection)
41 {
42  LOGF("\n"
43  "--------------------------------------------------\n"
44  "Building new operation on connection @ 0x%llX", connection);
45 
46  KineticOperation operation = {
47  .connection = connection,
48  .request = KineticAllocator_NewPDU(&connection->pdus),
49  .response = KineticAllocator_NewPDU(&connection->pdus),
50  };
51 
52  if (operation.request == NULL) {
53  LOG("Request PDU could not be allocated!"
54  " Try reusing or freeing a PDU.");
55  return (KineticOperation) {
56  .request = NULL, .response = NULL
57  };
58  }
59  KineticPDU_Init(operation.request, connection);
60  KINETIC_PDU_INIT_WITH_MESSAGE(operation.request, connection);
61  operation.request->proto = &operation.request->protoData.message.proto;
62 
63  if (operation.response == NULL) {
64  LOG("Response PDU could not be allocated!"
65  " Try reusing or freeing a PDU.");
66  return (KineticOperation) {
67  .request = NULL, .response = NULL
68  };
69  }
70 
71  KineticPDU_Init(operation.response, connection);
72 
73  return operation;
74 }

76 KineticStatus KineticOperation_Free(KineticOperation* const operation)
77 {
78  if (operation == NULL) {
79  LOG("Specified operation is NULL so nothing to free");
80  return KINETIC_STATUS_SESSION_INVALID;
81  }
82 
83  if (operation->request != NULL) {
84  KineticAllocator_FreePDU(&operation->connection->pdus, operation->request);
85  operation->request = NULL;
86  }
87 
88  if (operation->response != NULL) {
89  KineticAllocator_FreePDU(&operation->connection->pdus, operation->response);
90  operation->response = NULL;
91  }
92 
93  return KINETIC_STATUS_SUCCESS;
94 }
