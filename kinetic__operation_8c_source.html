36 static KineticStatus WritePDU(KineticOperation* const operation)
37 {
38  KineticPDU* request = operation->request;
39  // Pack and send the PDU header
40  ByteBuffer hdr = ByteBuffer_Create(&request->headerNBO, sizeof(KineticPDUHeader), sizeof(KineticPDUHeader));
41  KineticStatus status = KineticSocket_Write(request->connection->socket, &hdr);
42  if (status != KINETIC_STATUS_SUCCESS) {
43  LOG0("Failed to send PDU header!");
44  return status;
45  }
46 
47  // Send the protobuf message
48  LOG1("Sending PDU Protobuf:");
49  KineticLogger_LogProtobuf(2, request->proto);
50  status = KineticSocket_WriteProtobuf(request->connection->socket, request);
51  if (status != KINETIC_STATUS_SUCCESS) {
52  LOG0("Failed to send PDU protobuf message!");
53  return status;
54  }
55 
56  // Send the value/payload, if specified
57  if (operation->valueEnabled && operation->sendValue) {
58  LOGF1("Sending PDU Value Payload (%zu bytes)", operation->entry->value.bytesUsed);
59  status = KineticSocket_Write(request->connection->socket, &operation->entry->value);
60  if (status != KINETIC_STATUS_SUCCESS) {
61  LOG0("Failed to send PDU value payload!");
62  return status;
63  }
64  }
65 
66  LOG2("PDU sent successfully!");
67  return KINETIC_STATUS_SUCCESS;
68 }

70 KineticStatus KineticOperation_SendRequest(KineticOperation* const operation)
71 {
72  assert(operation != NULL);
73  assert(operation->connection != NULL);
74  assert(operation->request != NULL);
75  assert(operation->request->connection == operation->connection);
76  LOGF1("\nSending PDU via fd=%d", operation->connection->socket);
77  KineticPDU* request = operation->request;
78  request->proto = &operation->request->protoData.message.message;
79 
80  // Pack the command, if available
81  if (request->protoData.message.has_command) {
82  size_t expectedLen = KineticProto_command__get_packed_size(&request->protoData.message.command);
83  request->protoData.message.message.commandBytes.data = (uint8_t*)malloc(expectedLen);
84  assert(request->protoData.message.message.commandBytes.data != NULL);
85  size_t packedLen = KineticProto_command__pack(
86  &request->protoData.message.command,
87  request->protoData.message.message.commandBytes.data);
88  assert(packedLen == expectedLen);
89  request->protoData.message.message.commandBytes.len = packedLen;
90  request->protoData.message.message.has_commandBytes = true;
91  KineticLogger_LogByteArray(2, "commandBytes", (ByteArray){
92  .data = request->protoData.message.message.commandBytes.data,
93  .len = request->protoData.message.message.commandBytes.len,
94  });
95  }
96 
97  // Populate the HMAC for the protobuf
98  KineticHMAC_Init(&request->hmac, KINETIC_PROTO_COMMAND_SECURITY_ACL_HMACALGORITHM_HmacSHA1);
99  KineticHMAC_Populate(&request->hmac, request->proto, request->connection->session.hmacKey);
100 
101  // Configure PDU header length fields
102  request->header.versionPrefix = 'F';
103  request->header.protobufLength = KineticProto_Message__get_packed_size(request->proto);
104  if (operation->entry != NULL && operation->sendValue) {
105  request->header.valueLength = operation->entry->value.bytesUsed;
106  }
107  else {
108  request->header.valueLength = 0;
109  }
110  KineticLogger_LogHeader(1, &request->header);
111 
112  // Create NBO copy of header for sending
113  request->headerNBO.versionPrefix = 'F';
114  request->headerNBO.protobufLength = KineticNBO_FromHostU32(request->header.protobufLength);
115  request->headerNBO.valueLength = KineticNBO_FromHostU32(request->header.valueLength);
116 
117  pthread_mutex_lock(&operation->connection->writeMutex);
118  KineticStatus status = WritePDU(operation);
119  pthread_mutex_unlock(&operation->connection->writeMutex);
120  return status;
121 }

156 KineticOperation* KineticOperation_AssociateResponseWithOperation(KineticPDU* response)
157 {
158  if (response == NULL ||
159  response->command == NULL ||
160  response->command->header == NULL ||
161  !response->command->header->has_ackSequence ||
162  response->type != KINETIC_PDU_TYPE_RESPONSE)
163  {
164  LOG0("Response to associate with request is invalid!");
165  return NULL;
166  }
167 
168  const int64_t targetSequence = response->command->header->ackSequence;
169  KineticOperation* operation = KineticAllocator_GetFirstOperation(response->connection);
170  if (operation == NULL) {
171  LOG2("ERROR: No pending operations found!");
172  return NULL;
173  }
174 
175  while (operation != NULL) {
176  LOGF2("Comparing received PDU w/ ackSequence=%lld with request with sequence=%lld",
177  targetSequence, operation->request->command->header->sequence);
178  if (operation->request != NULL &&
179  operation->request->type == KINETIC_PDU_TYPE_REQUEST &&
180  operation->request->command != NULL &&
181  operation->request->command->header != NULL &&
182  operation->request->command->header->has_sequence &&
183  operation->request->command->header->sequence == targetSequence)
184  {
185  operation->response = response;
186  return operation;
187  }
188  operation = KineticAllocator_GetNextOperation(response->connection, operation);
189  }
190 
191  return NULL;
192 }

214 KineticStatus KineticOperation_PutCallback(KineticOperation* operation)
215 {
216  assert(operation != NULL);
217  assert(operation->connection != NULL);
218  LOGF3("PUT callback w/ operation (0x%0llX) on connection (0x%0llX)",
219  operation, operation->connection);
220  assert(operation->response != NULL);
221  assert(operation->entry != NULL);
222 
223  // Propagate newVersion to dbVersion in metadata, if newVersion specified
224  KineticEntry* entry = operation->entry;
225  if (entry->newVersion.array.data != NULL && entry->newVersion.array.len > 0) {
226  // If both buffers supplied, copy newVersion into dbVersion, and clear newVersion
227  if (entry->dbVersion.array.data != NULL && entry->dbVersion.array.len > 0) {
228  ByteBuffer_Reset(&entry->dbVersion);
229  ByteBuffer_Append(&entry->dbVersion, entry->newVersion.array.data, entry->newVersion.bytesUsed);
230  ByteBuffer_Reset(&entry->newVersion);
231  }
232 
233  // If only newVersion buffer supplied, move newVersion buffer into dbVersion,
234  // and set newVersion to NULL buffer
235  else {
236  entry->dbVersion = entry->newVersion;
237  entry->newVersion = BYTE_BUFFER_NONE;
238  }
239  }
240  return KINETIC_STATUS_SUCCESS;
241 }

435 void KineticOperation_TimeoutOperations(KineticConnection* const connection)
436 {
437  struct timeval currentTime;
438  gettimeofday(&currentTime, NULL);
439 
440  for (KineticOperation* operation = KineticAllocator_GetFirstOperation(connection);
441  operation != NULL;
442  operation = KineticAllocator_GetNextOperation(connection, operation)) {
443 
444  struct timeval timeoutTime = KineticOperation_GetTimeoutTime(operation);
445 
446  // if this operation has a nonzero timeout
447  // and it's timed out
448  if (!Kinetic_TimevalIsZero(timeoutTime) &&
449  Kinetic_TimevalCmp(currentTime, operation->timeoutTime) >= 0)
450  {
451  KineticOperation_Complete(operation, KINETIC_STATUS_OPERATION_TIMEDOUT);
452  }
453  }
