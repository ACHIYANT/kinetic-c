82 /* Send request.
83  * Note: This whole function operates with operation->connection->sendMutex locked. */
84 static KineticStatus KineticOperation_SendRequestInLock(KineticOperation* const operation)
85 {
86  LOGF3("\nSending PDU via fd=%d", operation->connection->socket);
87  KINETIC_ASSERT(operation);
88  KINETIC_ASSERT(operation->request);
89  KINETIC_ASSERT(operation->connection);
90  KineticRequest* request = operation->request;
91 
92  int64_t seq_id = KineticSession_GetNextSequenceCount(operation->connection->pSession);
93  KINETIC_ASSERT(request->message.header.sequence == KINETIC_SEQUENCE_NOT_YET_BOUND);
94  request->message.header.sequence = seq_id;
95 
96  size_t expectedLen = KineticRequest_PackCommand(request);
97  if (expectedLen == KINETIC_REQUEST_PACK_FAILURE) {
98  return KINETIC_STATUS_MEMORY_ERROR;
99  }
100  uint8_t * commandData = request->message.message.commandBytes.data;
101 
102  log_request_seq_id(operation->connection->socket, seq_id, request->message.header.messageType);
103 
104  KineticSession *session = operation->connection->pSession;
105  KineticStatus status = KineticRequest_PopulateAuthentication(&session->config,
106  operation->request, operation->pin);
107  if (status != KINETIC_STATUS_SUCCESS) {
108  if (commandData) { free(commandData); }
109  return status;
110  }
111 
112  #ifndef TEST
113  uint8_t * msg = NULL;
114  size_t msgSize = 0;
115  #endif
116  status = KineticRequest_PackMessage(operation, &msg, &msgSize);
117  if (status != KINETIC_STATUS_SUCCESS) {
118  if (commandData) { free(commandData); }
119  return status;
120  }
121 
122  if (commandData) { free(commandData); }
123  KineticCountingSemaphore * const sem = operation->connection->outstandingOperations;
124  KineticCountingSemaphore_Take(sem); // limit total concurrent requests
125 
126  if (!KineticRequest_SendRequest(operation, msg, msgSize)) {
127  LOGF0("Failed queuing request %p for transmit on fd=%d w/seq=%lld",
128  (void*)request, operation->connection->socket, (long long)seq_id);
129  /* A false result from bus_send_request means that the request was
130  * rejected outright, so the usual asynchronous, callback-based
131  * error handling for errors during the request or response will
132  * not be used. */
133  KineticCountingSemaphore_Give(sem);
134  status = KINETIC_STATUS_REQUEST_REJECTED;
135  } else {
136  status = KINETIC_STATUS_SUCCESS;
137  }
138 
139  if (msg != NULL) { free(msg); }
140  return status;
141 }

162 void KineticOperation_Complete(KineticOperation* operation, KineticStatus status)
163 {
164  KINETIC_ASSERT(operation != NULL);
165  // ExecuteOperation should ensure a callback exists (either a user supplied one, or the a default)
166  KineticCompletionData completionData = {.status = status};
167 
168  // Release this request so that others can be unblocked if at max (request PDUs throttled)
169  KineticCountingSemaphore_Give(operation->connection->outstandingOperations);
170 
171  if(operation->closure.callback != NULL) {
172  operation->closure.callback(&completionData, operation->closure.clientData);
173  }
174 
175  KineticAllocator_FreeOperation(operation);
176 }

201 KineticStatus KineticOperation_PutCallback(KineticOperation* const operation, KineticStatus const status)
202 {
203  KINETIC_ASSERT(operation != NULL);
204  KINETIC_ASSERT(operation->connection != NULL);
205  LOGF3("PUT callback w/ operation (0x%0llX) on connection (0x%0llX)",
206  operation, operation->connection);
207  KINETIC_ASSERT(operation->entry != NULL);
208 
209  if (status == KINETIC_STATUS_SUCCESS)
210  {
211  KINETIC_ASSERT(operation->response != NULL);
212  // Propagate newVersion to dbVersion in metadata, if newVersion specified
213  KineticEntry* entry = operation->entry;
214  if (entry->newVersion.array.data != NULL && entry->newVersion.array.len > 0) {
215  // If both buffers supplied, copy newVersion into dbVersion, and clear newVersion
216  if (entry->dbVersion.array.data != NULL && entry->dbVersion.array.len > 0) {
217  ByteBuffer_Reset(&entry->dbVersion);
218  ByteBuffer_Append(&entry->dbVersion, entry->newVersion.array.data, entry->newVersion.bytesUsed);
219  ByteBuffer_Reset(&entry->newVersion);
220  }
221 
222  // If only newVersion buffer supplied, move newVersion buffer into dbVersion,
223  // and set newVersion to NULL buffer
224  else {
225  entry->dbVersion = entry->newVersion;
226  entry->newVersion = BYTE_BUFFER_NONE;
227  }
228  }
229  }
230  return status;
231 }

256 static KineticStatus get_cb(const char *cmd_name, KineticOperation* const operation, KineticStatus const status)
257 {
258  KINETIC_ASSERT(operation != NULL);
259  KINETIC_ASSERT(operation->connection != NULL);
260  LOGF3("%s callback w/ operation (0x%0llX) on connection (0x%0llX)",
261  cmd_name, operation, operation->connection);
262  KINETIC_ASSERT(operation->entry != NULL);
263 
264  if (status == KINETIC_STATUS_SUCCESS)
265  {
266  KINETIC_ASSERT(operation->response != NULL);
267  // Update the entry upon success
268  KineticProto_Command_KeyValue* keyValue = KineticResponse_GetKeyValue(operation->response);
269  if (keyValue != NULL) {
270  if (!Copy_KineticProto_Command_KeyValue_to_KineticEntry(keyValue, operation->entry)) {
271  return KINETIC_STATUS_BUFFER_OVERRUN;
272  }
273  }
274 
275  if (!operation->entry->metadataOnly &&
276  !ByteBuffer_IsNull(operation->entry->value))
277  {
278  ByteBuffer_AppendArray(&operation->entry->value, (ByteArray){
279  .data = operation->response->value,
280  .len = operation->response->header.valueLength,
281  });
282  }
283  }
284 
285  return status;
286 }

506 KineticProto_Command_P2POperation* build_p2pOp(uint32_t nestingLevel, KineticP2P_Operation const * const p2pOp)
507 {
508  // limit nesting level to KINETIC_P2P_MAX_NESTING
509  if (nestingLevel >= KINETIC_P2P_MAX_NESTING) {
510  LOGF0("P2P operation nesting level is too deep. Max is %d.", KINETIC_P2P_MAX_NESTING);
511  return NULL;
512  }
513 
514  KineticProto_Command_P2POperation* proto_p2pOp = calloc(1, sizeof(KineticProto_Command_P2POperation));
515  if (proto_p2pOp == NULL) { goto error_cleanup; }
516 
517  KineticProto_command_p2_poperation__init(proto_p2pOp);
518 
519  proto_p2pOp->peer = calloc(1, sizeof(KineticProto_Command_P2POperation_Peer));
520  if (proto_p2pOp->peer == NULL) { goto error_cleanup; }
521 
522  KineticProto_command_p2_poperation_peer__init(proto_p2pOp->peer);
523 
524  proto_p2pOp->peer->hostname = p2pOp->peer.hostname;
525  proto_p2pOp->peer->has_port = true;
526  proto_p2pOp->peer->port = p2pOp->peer.port;
527  proto_p2pOp->peer->has_tls = true;
528  proto_p2pOp->peer->tls = p2pOp->peer.tls;
529 
530  proto_p2pOp->n_operation = p2pOp->numOperations;
531  proto_p2pOp->operation = calloc(p2pOp->numOperations, sizeof(KineticProto_Command_P2POperation_Operation*));
532  if (proto_p2pOp->operation == NULL) { goto error_cleanup; }
533 
534  for(size_t i = 0; i < proto_p2pOp->n_operation; i++) {
535  KINETIC_ASSERT(!ByteBuffer_IsNull(p2pOp->operations[i].key)); // TODO return invalid operand?
536 
537  KineticProto_Command_P2POperation_Operation * p2p_op_op = calloc(1, sizeof(KineticProto_Command_P2POperation_Operation));
538  if (p2p_op_op == NULL) { goto error_cleanup; }
539 
540  KineticProto_command_p2_poperation_operation__init(p2p_op_op);
541 
542  p2p_op_op->has_key = true;
543  p2p_op_op->key.data = p2pOp->operations[i].key.array.data;
544  p2p_op_op->key.len = p2pOp->operations[i].key.bytesUsed;
545 
546  p2p_op_op->has_newKey = !ByteBuffer_IsNull(p2pOp->operations[i].newKey);
547  p2p_op_op->newKey.data = p2pOp->operations[i].newKey.array.data;
548  p2p_op_op->newKey.len = p2pOp->operations[i].newKey.bytesUsed;
549 
550  p2p_op_op->has_version = !ByteBuffer_IsNull(p2pOp->operations[i].version);
551  p2p_op_op->version.data = p2pOp->operations[i].version.array.data;
552  p2p_op_op->version.len = p2pOp->operations[i].version.bytesUsed;
553 
554  // force if no version was specified
555  p2p_op_op->has_force = ByteBuffer_IsNull(p2pOp->operations[i].version);
556  p2p_op_op->force = ByteBuffer_IsNull(p2pOp->operations[i].version);
557 
558  if (p2pOp->operations[i].chainedOperation == NULL) {
559  p2p_op_op->p2pop = NULL;
560  } else {
561  p2p_op_op->p2pop = build_p2pOp(nestingLevel + 1, p2pOp->operations[i].chainedOperation);
562  if (p2p_op_op->p2pop == NULL) { goto error_cleanup; }
563  }
564 
565  p2p_op_op->status = NULL;
566 
567  proto_p2pOp->operation[i] = p2p_op_op;
568  }
569  return proto_p2pOp;
570 
571 error_cleanup:
572  destroy_p2pOp(proto_p2pOp);
573  return NULL;
574 }

576 static void populateP2PStatusCodes(KineticP2P_Operation* const p2pOp, KineticProto_Command_P2POperation const * const p2pOperation)
577 {
578  if (p2pOperation == NULL) { return; }
579  for(size_t i = 0; i < p2pOp->numOperations; i++)
580  {
581  if (i < p2pOperation->n_operation)
582  {
583  if ((p2pOperation->operation[i]->status != NULL) &&

828

831  status = KINETIC_STATUS_INVALID_FILE;

870  status = KINETIC_STATUS_INVALID_FILE;
871  goto cleanup;
872  }
873  fclose(fp);

877  operation->request->message.command.header->messageType = KINETIC_PROTO_COMMAND_MESSAGE_TYPE_SETUP;
878  operation->request->message.command.header->has_messageType = true;
