147 
148  // Initialize request
149  KineticExchange_IncrementSequence(operation->exchange);
150  KineticOperation_BuildNoop(operation);
151 
152  // Send the request
153  KineticPDU_Send(operation->request);
154 
155  // Associate response with same exchange as request
156  operation->response->exchange = operation->request->exchange;
157 
158  // Receive the response
