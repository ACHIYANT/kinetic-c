367 /* Is FILTER a subset of NAME? */ \
368 static int greatest_name_match(const char *name, \
369  const char *filter) { \
370  size_t offset = 0; \
371  size_t filter_len = strlen(filter); \
372  while (name[offset] != '\0') { \
373  if (name[offset] == filter[0]) { \
374  if (0 == strncmp(&name[offset], filter, filter_len)) { \
375  return 1; \
376  } \
377  } \
378  offset++; \
379  } \
380  \
381  return 0; \
382 } \

384 int greatest_pre_test(const char *name) { \
385  if (!GREATEST_LIST_ONLY() \
386  && (!GREATEST_FIRST_FAIL() || greatest_info.suite.failed == 0) \
387  && (greatest_info.test_filter == NULL || \
388  greatest_name_match(name, greatest_info.test_filter))) { \
389  GREATEST_SET_TIME(greatest_info.suite.pre_test); \
390  if (greatest_info.setup) { \
391  greatest_info.setup(greatest_info.setup_udata); \
392  } \
393  return 1; /* test should be run */ \
394  } else { \
395  return 0; /* skipped */ \
396  } \
397 } \

399 void greatest_post_test(const char *name, int res) { \
400  GREATEST_SET_TIME(greatest_info.suite.post_test); \
401  if (greatest_info.teardown) { \
402  void *udata = greatest_info.teardown_udata; \
403  greatest_info.teardown(udata); \
404  } \
405  \
406  if (res < 0) { \
407  greatest_do_fail(name); \
408  } else if (res > 0) { \
409  greatest_do_skip(name); \
410  } else if (res == 0) { \
411  greatest_do_pass(name); \
412  } \
413  greatest_info.suite.tests_run++; \
414  greatest_info.col++; \
415  if (GREATEST_IS_VERBOSE()) { \
416  GREATEST_CLOCK_DIFF(greatest_info.suite.pre_test, \
417  greatest_info.suite.post_test); \
418  fprintf(GREATEST_STDOUT, "\n"); \
419  } else if (greatest_info.col % greatest_info.width == 0) { \
420  fprintf(GREATEST_STDOUT, "\n"); \
421  greatest_info.col = 0; \
422  } \
423  if (GREATEST_STDOUT == stdout) fflush(stdout); \
424 } \

426 static void greatest_run_suite(greatest_suite_cb *suite_cb, \
427  const char *suite_name) { \
428  if (greatest_info.suite_filter && \
429  !greatest_name_match(suite_name, greatest_info.suite_filter)) { \
430  return; \
431  } \
432  if (GREATEST_FIRST_FAIL() && greatest_info.failed > 0) { return; } \
433  memset(&greatest_info.suite, 0, sizeof(greatest_info.suite)); \
434  greatest_info.col = 0; \
435  fprintf(GREATEST_STDOUT, "\n* Suite %s:\n", suite_name); \
436  GREATEST_SET_TIME(greatest_info.suite.pre_suite); \
437  suite_cb(); \
438  GREATEST_SET_TIME(greatest_info.suite.post_suite); \
439  if (greatest_info.suite.tests_run > 0) { \
440  fprintf(GREATEST_STDOUT, \
441  "\n%u tests - %u pass, %u fail, %u skipped", \
442  greatest_info.suite.tests_run, \
443  greatest_info.suite.passed, \
444  greatest_info.suite.failed, \
445  greatest_info.suite.skipped); \
446  GREATEST_CLOCK_DIFF(greatest_info.suite.pre_suite, \
447  greatest_info.suite.post_suite); \
448  fprintf(GREATEST_STDOUT, "\n"); \
449  } \
450  greatest_info.setup = NULL; \
451  greatest_info.setup_udata = NULL; \
452  greatest_info.teardown = NULL; \
453  greatest_info.teardown_udata = NULL; \
454  greatest_info.passed += greatest_info.suite.passed; \
455  greatest_info.failed += greatest_info.suite.failed; \
456  greatest_info.skipped += greatest_info.suite.skipped; \
457  greatest_info.tests_run += greatest_info.suite.tests_run; \
458 } \

504 int greatest_do_assert_equal_t(const void *exp, const void *got, \
505  greatest_type_info *type_info, void *udata) { \
506  if (type_info == NULL || type_info->equal == NULL) { \
507  return 0; \
508  } \
509  int eq = type_info->equal(exp, got, udata); \
510  if (!eq) { \
511  if (type_info->print != NULL) { \
512  fprintf(GREATEST_STDOUT, "Expected: "); \
513  (void)type_info->print(exp, udata); \
514  fprintf(GREATEST_STDOUT, "\nGot: "); \
515  (void)type_info->print(got, udata); \
516  fprintf(GREATEST_STDOUT, "\n"); \
517  } else { \
518  fprintf(GREATEST_STDOUT, \
519  "GREATEST_ASSERT_EQUAL_T failure at %s:%dn", \
520  greatest_info.fail_file, \
521  greatest_info.fail_line); \
522  } \
523  } \
524  return eq; \
525 } \
