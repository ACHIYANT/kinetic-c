40 void* KineticSerialAllocator_AllocateChunk(KineticSerialAllocator* allocator, size_t len)
41 {
42  void* data = NULL;
43 
44  if (allocator->buffer != NULL ||
45  ((unsigned long long)allocator->buffer) % sizeof(uint64_t) == 0 || // Ensure root of buffer is long-aligned
46  allocator->used < allocator->total)
47  {
48  // Align to the next allocated hunk of data to the next 64-bit boundary
49  size_t newUsed = allocator->used + len;
50 
51  // Assign pointer to new hunk of data
52  if (newUsed <= allocator->total) {
53  data = (void*)&allocator->buffer[allocator->used];
54  }
55 
56  // Mark buffer full, if the current allocation has exhausted allocated data
57  const size_t padding = (sizeof(uint64_t) - 1);
58  newUsed += padding;
59  newUsed &= ~padding;
60  allocator->used = MIN(newUsed, allocator->total);
61  }
62 
63  return data;
64 }
