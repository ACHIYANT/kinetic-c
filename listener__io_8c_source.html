29 void ListenerIO_AttemptRecv(listener *l, int available) {
30  /* --> failure --> set 'closed' error on socket, don't die */
31  struct bus *b = l->bus;
32  int read_from = 0;
33  BUS_LOG(b, 3, LOG_LISTENER, "attempting receive", b->udata);
34 
35  for (int i = 0; i < l->tracked_fds; i++) {
36  if (read_from == available) { break; }
37  struct pollfd *fd = &l->fds[i + INCOMING_MSG_PIPE];
38  connection_info *ci = l->fd_info[i];
39  BUS_ASSERT(b, b->udata, ci->fd == fd->fd);
40 
41  if (fd->revents & (POLLERR | POLLNVAL)) {
42  read_from++;
43  BUS_LOG(b, 2, LOG_LISTENER,
44  "pollfd: socket error (POLLERR | POLLNVAL)", b->udata);
45  set_error_for_socket(l, i, ci->fd, RX_ERROR_POLLERR);
46  } else if (fd->revents & POLLHUP) {
47  read_from++;
48  BUS_LOG(b, 3, LOG_LISTENER, "pollfd: socket error POLLHUP",
49  b->udata);
50  set_error_for_socket(l, i, ci->fd, RX_ERROR_POLLHUP);
51  } else if (fd->revents & POLLIN) {
52  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
53  "reading %zd bytes from socket (buf is %zd)",
54  ci->to_read_size, l->read_buf_size);
55  BUS_ASSERT(b, b->udata, l->read_buf_size >= ci->to_read_size);
56  read_from++;
57 
58  switch (ci->type) {
59  case BUS_SOCKET_PLAIN:
60  socket_read_plain(b, l, i, ci);
61  break;
62  case BUS_SOCKET_SSL:
63  socket_read_ssl(b, l, i, ci);
64  break;
65  default:
66  BUS_ASSERT(b, b->udata, false);
67  }
68  }
69  }
70 }

72 static bool socket_read_plain(struct bus *b, listener *l, int pfd_i, connection_info *ci) {
73  ssize_t size = read(ci->fd, l->read_buf, ci->to_read_size);
74  if (size == -1) {
75  if (util_is_resumable_io_error(errno)) {
76  errno = 0;
77  } else {
78  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
79  "read: socket error reading, %d", errno);
80  set_error_for_socket(l, pfd_i, ci->fd, RX_ERROR_READ_FAILURE);
81  errno = 0;
82  }
83  }
84 
85  if (size > 0) {
86  BUS_LOG_SNPRINTF(b, 5, LOG_LISTENER, b->udata, 64,
87  "read: %zd", size);
88 
89  return sink_socket_read(b, l, ci, size);
90  } else {
91  return false;
92  }
93 }

103 static bool socket_read_ssl(struct bus *b, listener *l, int pfd_i, connection_info *ci) {
104  BUS_ASSERT(b, b->udata, ci->ssl);
105  for (;;) {
106  // ssize_t pending = SSL_pending(ci->ssl);
107  ssize_t size = (ssize_t)SSL_read(ci->ssl, l->read_buf, ci->to_read_size);
108  // fprintf(stderr, "=== PENDING: %zd, got %zd ===\n", pending, size);
109 
110  if (size == -1) {
111  int reason = SSL_get_error(ci->ssl, size);
112  switch (reason) {
113  case SSL_ERROR_WANT_READ:
114  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
115  "SSL_read fd %d: WANT_READ\n", ci->fd);
116  return true;
117 
118  case SSL_ERROR_WANT_WRITE:
119  BUS_ASSERT(b, b->udata, false);
120 
121  case SSL_ERROR_SYSCALL:
122  {
123  if (errno == 0) {
124  print_SSL_error(b, ci, 1, "SSL_ERROR_SYSCALL errno 0");
125  BUS_ASSERT(b, b->udata, false);
126  } else if (util_is_resumable_io_error(errno)) {
127  errno = 0;
128  } else {
129  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
130  "SSL_read fd %d: errno %d\n", ci->fd, errno);
131  print_SSL_error(b, ci, 1, "SSL_ERROR_SYSCALL");
132  set_error_for_socket(l, pfd_i, ci->fd, RX_ERROR_READ_FAILURE);
133  }
134  break;
135  }
136  case SSL_ERROR_ZERO_RETURN:
137  {
138  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
139  "SSL_read fd %d: ZERO_RETURN (HUP)\n", ci->fd);
140  set_error_for_socket(l, pfd_i, ci->fd, RX_ERROR_POLLHUP);
141  break;
142  }
143 
144  default:
145  print_SSL_error(b, ci, 1, "SSL_ERROR UNKNOWN");
146  set_error_for_socket(l, pfd_i, ci->fd, RX_ERROR_READ_FAILURE);
147  BUS_ASSERT(b, b->udata, false);
148  }
149  } else if (size > 0) {
150  sink_socket_read(b, l, ci, size);
151  }
152  }
153  return true;
154 }

158 static bool sink_socket_read(struct bus *b,
159  listener *l, connection_info *ci, ssize_t size) {
160  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
161  "read %zd bytes, calling sink CB", size);
162 
163 #if DUMP_READ
164  bus_lock_log(b);
165  printf("\n");
166  for (int i = 0; i < size; i++) {
167  if (i > 0 && (i & 15) == 0) { printf("\n"); }
168  printf("%02x ", l->read_buf[i]);
169  }
170  printf("\n\n");
171  bus_unlock_log(b);
172 #endif
173 
174  bus_sink_cb_res_t sres = b->sink_cb(l->read_buf, size, ci->udata);
175  if (sres.full_msg_buffer) {
176  BUS_LOG(b, 3, LOG_LISTENER, "calling unpack CB", b->udata);
177  bus_unpack_cb_res_t ures = b->unpack_cb(sres.full_msg_buffer, ci->udata);
178  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
179  "process_unpacked_message: ok? %d, seq_id:%lld",
180  ures.ok, (long long)ures.u.success.seq_id);
181  process_unpacked_message(l, ci, ures);
182  }
183 
184  ci->to_read_size = sres.next_read;
185 
186  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
187  "expecting next read to have %zd bytes", ci->to_read_size);
188 
189  /* Grow read buffer if necessary. */
190  if (ci->to_read_size > l->read_buf_size) {
191  if (!ListenerTask_GrowReadBuf(l, ci->to_read_size)) {
192  BUS_LOG_SNPRINTF(b, 3, LOG_MEMORY, b->udata, 128,
193  "Read buffer realloc failure for %p (%zd to %zd)",
194  l->read_buf, l->read_buf_size, ci->to_read_size);
195  BUS_ASSERT(b, b->udata, false);
196  }
197  }
198  return true;
199 }

201 static void set_error_for_socket(listener *l, int id, int fd, rx_error_t err) {
202  /* Mark all pending messages on this socket as being failed due to error. */
203  struct bus *b = l->bus;
204  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 64,
205  "set_error_for_socket %d, err %d", fd, err);
206 
207  for (int i = 0; i <= l->rx_info_max_used; i++) {
208  rx_info_t *info = &l->rx_info[i];
209  switch (info->state) {
210  case RIS_INACTIVE:
211  break;
212  case RIS_HOLD:
213  ListenerTask_ReleaseRXInfo(l, info);
214  break;
215  case RIS_EXPECT:
216  {
217  struct boxed_msg *box = info->u.expect.box;
218  if (box && box->fd == fd) {
219  info->u.expect.error = err;
220  }
221  break;
222  }
223  default:
224  {
225  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
226  "match fail %d on line %d", info->state, __LINE__);
227  BUS_ASSERT(b, b->udata, false);
228  }
229  }
230  }
231  l->fds[id + INCOMING_MSG_PIPE].events &= ~POLLIN;
232 }

235 static rx_info_t *find_info_by_sequence_id(listener *l,
236  int fd, int64_t seq_id) {
237  struct bus *b = l->bus;
238  for (int i = 0; i <= l->rx_info_max_used; i++) {
239  rx_info_t *info = &l->rx_info[i];
240 
241  switch (info->state) {
242  case RIS_INACTIVE:
243  break; /* skip */
244  case RIS_HOLD:
245  BUS_LOG_SNPRINTF(b, 4, LOG_MEMORY, b->udata, 128,
246  "find_info_by_sequence_id: info (%p) at +%d: <fd:%d, seq_id:%lld>",
247  (void*)info, info->id, fd, (long long)seq_id);
248  if (info->u.hold.fd == fd && info->u.hold.seq_id == seq_id) {
249  return info;
250  }
251  break;
252  case RIS_EXPECT:
253  {
254  struct boxed_msg *box = info->u.expect.box;
255  BUS_LOG_SNPRINTF(b, 4, LOG_MEMORY, b->udata, 128,
256  "find_info_by_sequence_id: info (%p) at +%d [s %d]: box is %p",
257  (void*)info, info->id, info->u.expect.error, (void*)box);
258  if (box != NULL && box->out_seq_id == seq_id && box->fd == fd) {
259  return info;
260  }
261  break;
262  }
263  default:
264  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
265  "match fail %d on line %d", info->state, __LINE__);
266  BUS_ASSERT(b, b->udata, false);
267  }
268  }
269 
270  if (b->log_level > 5 || 0) {
271  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 64,
272  "==== Could not find <fd:%d, seq_id:%lld>, dumping table ====\n",
273  fd, (long long)seq_id);
274  ListenerTask_DumpRXInfoTable(l);
275  }
276  /* Not found. Probably an unsolicited status message. */
277  return NULL;
278 }

280 static void process_unpacked_message(listener *l,
281  connection_info *ci, bus_unpack_cb_res_t result) {
282  struct bus *b = l->bus;
283 
284  /* NOTE: message may be an unsolicited status message */
285 
286  if (result.ok) {
287  int64_t seq_id = result.u.success.seq_id;
288  void *opaque_msg = result.u.success.msg;
289 
290  if ((seq_id < ci->largest_rd_seq_id_seen) &&
291  (ci->largest_rd_seq_id_seen != BUS_NO_SEQ_ID) &&
292  (seq_id != BUS_NO_SEQ_ID)) {
293  BUS_LOG_SNPRINTF(b, 0, LOG_LISTENER, b->udata, 128,
294  "suspicious sequence ID on %d: largest seen is %lld, got %lld\n",
295  ci->fd, (long long)ci->largest_rd_seq_id_seen, (long long)seq_id);
296  }
297  ci->largest_rd_seq_id_seen = seq_id;
298 
299  rx_info_t *info = find_info_by_sequence_id(l, ci->fd, seq_id);
300  if (info) {
301  switch (info->state) {
302  case RIS_HOLD:
303  /* Just save result, to match up later. */
304  BUS_ASSERT(b, b->udata, !info->u.hold.has_result);
305  info->u.hold.has_result = true;
306  info->u.hold.result = result;
307  break;
308  case RIS_EXPECT:
309  {
310  BUS_LOG_SNPRINTF(b, 3, LOG_LISTENER, b->udata, 128,
311  "marking info %d, seq_id:%lld ready for delivery",
312  info->id, (long long)result.u.success.seq_id);
313  info->u.expect.error = RX_ERROR_READY_FOR_DELIVERY;
314  BUS_ASSERT(b, b->udata, !info->u.hold.has_result);
315  info->u.expect.has_result = true;
316  info->u.expect.result = result;
317  ListenerTask_AttemptDelivery(l, info);
318  break;
319  }
320  case RIS_INACTIVE:
321  default:
322  BUS_ASSERT(b, b->udata, false);
323  }
324  } else {
325  /* We received a response that we weren't expecting. */
326  if (seq_id != BUS_NO_SEQ_ID) {
327  BUS_LOG_SNPRINTF(b, 2 - 2, LOG_LISTENER, b->udata, 128,
328  "Couldn't find info for fd:%d, seq_id:%lld, msg %p",
329  ci->fd, (long long)seq_id, opaque_msg);
330  }
331  if (b->unexpected_msg_cb) {
332  b->unexpected_msg_cb(opaque_msg, seq_id, b->udata, ci->udata);
333  }
334  }
335  } else {
336  uintptr_t e_id = result.u.error.opaque_error_id;
337  BUS_LOG_SNPRINTF(b, 1, LOG_LISTENER, b->udata, 128,
338  "Got opaque_error_id of %lu (0x%08lx)",
339  e_id, e_id);
340 
341  /* Timeouts will clean up after it; give user code a chance to
342  * clean up after it here, though technically speaking they
343  * could in the unpack_cb above. */
344  b->error_cb(result, ci->udata);
345  }
346 }
