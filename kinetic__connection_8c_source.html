66 KineticStatus KineticSession_Connect(KineticSession const * const session)
67 {
68  if (session == NULL) {
69  return KINETIC_STATUS_SESSION_EMPTY;
70  }
71  KineticConnection* connection = session->connection;
72  if (connection == NULL) {
73  return KINETIC_STATUS_CONNECTION_ERROR;
74  }
75 
76  // Establish the connection
77  assert(session != NULL);
78  assert(session->connection != NULL);
79  assert(strlen(session->config.host) > 0);
80  connection->connected = false;
81  connection->socket = KineticSocket_Connect(
82  session->config.host, session->config.port);
83  connection->connected = (connection->socket >= 0);
84  if (!connection->connected) {
85  LOG0("Session connection failed!");
86  connection->socket = KINETIC_SOCKET_DESCRIPTOR_INVALID;
87  return KINETIC_STATUS_CONNECTION_ERROR;
88  }
89 
90  // Kick off the worker thread
91  session->connection->thread.connection = connection;
92  KineticController_Init(session);
93 
94  connection->session = *session;
95 
96  // Wait for initial unsolicited status to be received in order to obtain connectionID
97  const long maxWaitMicrosecs = 2000000;
98  long microsecsWaiting = 0;
99  struct timespec sleepDuration = {.tv_nsec = 500000};
100  while(connection->connectionID == 0) {
101  if (microsecsWaiting > maxWaitMicrosecs) {
102  LOG0("Timed out waiting for connection ID from device!");
103  return KINETIC_STATUS_SOCKET_TIMEOUT;
104  }
105  nanosleep(&sleepDuration, NULL);
106  microsecsWaiting += (sleepDuration.tv_nsec / 1000);
107  }
108 
109  return KINETIC_STATUS_SUCCESS;
110 }

112 KineticStatus KineticSession_Disconnect(KineticSession const * const session)
113 {
114  if (session == NULL) {
115  return KINETIC_STATUS_SESSION_EMPTY;
116  }
117  KineticConnection* connection = session->connection;
118  if (connection == NULL || !session->connection->connected || connection->socket < 0) {
119  return KINETIC_STATUS_CONNECTION_ERROR;
120  }
121 
122  // Shutdown the worker thread
123  KineticStatus status = KINETIC_STATUS_SUCCESS;
124  connection->thread.abortRequested = true;
125  LOG3("\nSent abort request to worker thread!\n");
126  int pthreadStatus = pthread_join(connection->threadID, NULL);
127  if (pthreadStatus != 0) {
128  char errMsg[256];
129  Kinetic_GetErrnoDescription(pthreadStatus, errMsg, sizeof(errMsg));
130  LOGF0("Failed terminating worker thread w/error: %s", errMsg);
131  status = KINETIC_STATUS_CONNECTION_ERROR;
132  }
133 
134  // Close the connection
135  close(connection->socket);
136  connection->socket = KINETIC_HANDLE_INVALID;
137  connection->connected = false;
138 
139  return status;
140 }
