39 KineticSessionHandle KineticConnection_NewConnection(
40  const KineticSession* const config)
41 {
42  KineticSessionHandle handle = KINETIC_HANDLE_INVALID;
43  if (config == NULL) {
44  return KINETIC_HANDLE_INVALID;
45  }
46  for (int idx = 0; idx < KINETIC_SESSIONS_MAX; idx++) {
47  if (Connections[idx] == NULL) {
48  KineticConnection* connection = &ConnectionInstances[idx];
49  Connections[idx] = connection;
50  KINETIC_CONNECTION_INIT(connection);
51  connection->session = *config;
52  handle = (KineticSessionHandle)(idx + 1);
53  return handle;
54  }
55  }
56  return KINETIC_HANDLE_INVALID;
57 }

79 KineticStatus KineticConnection_Connect(KineticConnection* const connection)
80 {
81  if (connection == NULL) {
82  return KINETIC_STATUS_SESSION_EMPTY;
83  }
84 
85  // Establish the connection
86  connection->connected = false;
87  connection->socket = KineticSocket_Connect(
88  connection->session.host,
89  connection->session.port,
90  connection->session.nonBlocking);
91  connection->connected = (connection->socket >= 0);
92  if (!connection->connected) {
93  LOG0("Session connection failed!");
94  connection->socket = KINETIC_SOCKET_DESCRIPTOR_INVALID;
95  return KINETIC_STATUS_CONNECTION_ERROR;
96  }
97 
98  // Kick off the worker thread
99  connection->thread.connection = connection;
100  KineticController_CreateWorkerThreads(connection);
101 
102  return KINETIC_STATUS_SUCCESS;
103 }

105 KineticStatus KineticConnection_Disconnect(KineticConnection* const connection)
106 {
107  if (connection == NULL || !connection->connected || connection->socket < 0) {
108  return KINETIC_STATUS_SESSION_INVALID;
109  }
110 
111  // Shutdown the worker thread
112  KineticStatus status = KINETIC_STATUS_SUCCESS;
113  connection->thread.abortRequested = true;
114  LOG2("\nSent abort request to worker thread!\n");
115  int pthreadStatus = pthread_join(connection->threadID, NULL);
116  if (pthreadStatus != 0) {
117  char errMsg[256];
118  Kinetic_GetErrnoDescription(pthreadStatus, errMsg, sizeof(errMsg));
119  LOGF0("Failed terminating worker thread w/error: %s", errMsg);
120  status = KINETIC_STATUS_CONNECTION_ERROR;
121  }
122 
123  // Close the connection
124  close(connection->socket);
125  connection->socket = KINETIC_HANDLE_INVALID;
126  connection->connected = false;
127 
128  return status;
129 }
