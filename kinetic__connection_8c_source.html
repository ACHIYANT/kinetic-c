31 KineticSessionHandle KineticConnection_NewConnection(
32  const KineticSession* const config)
33 {
34  KineticSessionHandle handle = KINETIC_HANDLE_INVALID;
35  if (config == NULL) {
36  return KINETIC_HANDLE_INVALID;
37  }
38  for (int idx = 0; idx < KINETIC_SESSIONS_MAX; idx++) {
39  if (Connections[idx] == NULL) {
40  KineticConnection* connection = &ConnectionInstances[idx];
41  Connections[idx] = connection;
42  KINETIC_CONNECTION_INIT(connection);
43  connection->session = *config;
44  handle = (KineticSessionHandle)(idx + 1);
45  return handle;
46  }
47  }
48  return KINETIC_HANDLE_INVALID;
49 }

70 KineticStatus KineticConnection_Connect(KineticConnection* const connection)
71 {
72  if (connection == NULL) {
73  return KINETIC_STATUS_SESSION_EMPTY;
74  }
75 
76  connection->connected = false;
77  connection->socket = KineticSocket_Connect(
78  connection->session.host,
79  connection->session.port,
80  connection->session.nonBlocking);
81  connection->connected = (connection->socket >= 0);
82 
83  if (!connection->connected) {
84  LOG("Session connection failed!");
85  connection->socket = KINETIC_SOCKET_DESCRIPTOR_INVALID;
86  return KINETIC_STATUS_CONNECTION_ERROR;
87  }
88 
89  return KINETIC_STATUS_SUCCESS;
90 }
