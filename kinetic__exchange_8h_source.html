28 typedef struct _KineticExchange
29 {
30  // Defaults to false, since clusterVersion is optional
31  // Set to true to enable clusterVersion for the exchange
32  bool has_clusterVersion;
33 
34  // Optional field - default value is 0
35  // The version number of this cluster definition. If this is not equal to
36  // the value on the device, the request is rejected and will return a
37  // `VERSION_FAILURE` `statusCode` in the `Status` message.
38  int64_t clusterVersion;
39 
40  // Required field
41  // The identity associated with this request. See the ACL discussion above.
42  // The Kinetic Device will use this identity value to lookup the
43  // HMAC key (shared secret) to verify the HMAC.
44  int64_t identity;
45 
46  // Required field
47  // This is the identity's HMAC Key. This is a shared secret between the
48  // client and the device, used to sign requests.
49  bool has_key;
50  size_t keyLength;
51  char key[KINETIC_MAX_KEY_LEN+1];
52 
53  // Required field
54  // A unique number for this connection between the source and target.
55  // On the first request to the drive, this should be the time of day in
56  // seconds since 1970. The drive can change this number and the client must
57  // continue to use the new number and the number must remain constant
58  // during the session
59  int64_t connectionID;
60 
61  // Required field
62  // A monotonically increasing number for each request in a TCP connection.
63  int64_t sequence;
64

69 #define KINETIC_EXCHANGE_INIT(exchg, id, k, klen, con) { \
70  memset((exchg), 0, sizeof(KineticExchange)); \
71  (exchg)->identity = id; \
72  if (k != NULL && klen > 0) \
73  { \
74  memcpy((exchg)->key, k, klen); \
75  (exchg)->key[klen] = '\0'; \
76  (exchg)->keyLength = klen; \
77  (exchg)->has_key = true; \
78  } \
79  (exchg)->connection = con; \
80  (exchg)->sequence = -1; \
81 }
