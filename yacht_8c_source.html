48 /* Init a hash table with approx. 2 ** sz2 buckets. */
49 struct yacht *yacht_init(uint8_t sz2) {
50  if (sz2 == 0) { sz2 = DEF_SZ2; }
51  size_t size = 1 << sz2;
52  struct yacht *y = calloc(1, sizeof(*y));
53  int *buckets = calloc(size, sizeof(*buckets));
54  void **values = calloc(size, sizeof(*values));
55  if (y && buckets && values) {
56  y->size = size;
57  y->mask = size - 1;
58  y->buckets = buckets;
59  y->values = values;
60  LOG(" -- init with size %zd\n", size);
61  for (size_t i = 0; i < size; i++) {
62  y->buckets[i] = YACHT_NO_KEY;
63  }
64  return y;
65  } else {
66  if (y) { free(y); }
67  if (buckets) { free(buckets); }
68  if (values) { free(values); }
69  return NULL;
70  }
71 }

79 bool yacht_get(struct yacht *y, int key, void **value) {
80  size_t b = hash(key) & y->mask;
81  LOG(" -- getting key %d with bucket %zd\n", key, b);
82 
83  for (size_t o = 0; o < y->size; o++) {
84  size_t i = (b + o) & y->mask; // wrap as necessary
85  int bv = y->buckets[i];
86  if (bv == key) {
87  if (value) { *value = y->values[i]; }
88  return true;
89  } else if (bv == YACHT_NO_KEY || bv == YACHT_DELETED) {
90  return false;
91  }
92  }
93  return false;
94 }

117 static bool insert(int *buckets, void **values,
118  size_t mask, size_t max_fill,
119  int key, void *value, void **old_value) {
120  size_t b = hash(key) & mask;
121 
122  for (size_t o = 0; o < max_fill; o++) {
123  if (o > 0) { LOG(" -- o %zd (max_fill %zd)\n", o, max_fill); }
124  size_t i = (b + o) & mask; // wrap as necessary
125  int bv = buckets[i];
126  LOG(" -- b %zd: %d\n", i, bv);
127  if (bv == key) {
128  if (old_value) { *old_value = values[i]; }
129  values[i] = value;
130  return true; /* already present */
131  } else if (bv == YACHT_NO_KEY || bv == YACHT_DELETED) {
132  buckets[i] = key;
133  values[i] = value;
134  return true;
135  }
136  }
137  return false; /* too full */
138 }

140 static bool grow(struct yacht *y) {
141  size_t nsize = 2 * y->size;
142  LOG(" -- growing %zd => %zd\n", y->size, nsize);
143  size_t nmask = nsize - 1;
144  int *nbuckets = NULL;
145  void **nvalues = NULL;
146  nbuckets = malloc(nsize * sizeof(*nbuckets));
147  if (nbuckets == NULL) { return false; }
148  for (size_t i = 0; i < nsize; i++) {
149  nbuckets[i] = YACHT_NO_KEY;
150  }
151  nvalues = calloc(nsize, sizeof(*nvalues));
152  if (nvalues == NULL) {
153  free(nbuckets);
154  return false;
155  }
156 
157  for (size_t i = 0; i < y->size; i++) {
158  int key = y->buckets[i];
159  if (key != YACHT_NO_KEY && key != YACHT_DELETED) {
160  if (!insert(nbuckets, nvalues, nmask, nsize, key, y->values[i], NULL)) {
161  goto cleanup;
162  }
163  }
164  }
165  y->size = nsize;
166  free(y->buckets);
167  free(y->values);
168  y->buckets = nbuckets;
169  y->values = nvalues;
170  y->mask = nmask;
171  return true;
172 cleanup:
173  if (nbuckets) { free(nbuckets); }
174  if (nvalues) { free(nvalues); }
175 
176  return false;
177 }

179 /* Remove KEY from the table. */
180 bool yacht_remove(struct yacht *y, int key, void **old_value) {
181  size_t b = hash(key) & y->mask;
182  LOG(" -- removing %d with bucket %zd\n", key, b);
183 
184  for (size_t o = 0; o < y->size; o++) {
185  size_t i = (b + o) & y->mask; // wrap as necessary
186  int bv = y->buckets[i];
187  LOG(" -- b %zd: %d\n", i, bv);
188  if (bv == key) {
189  if (y->buckets[(i + 1) & y->mask] == YACHT_NO_KEY) {
190  y->buckets[i] = YACHT_NO_KEY;
191  } else {
192  y->buckets[i] = YACHT_DELETED;
193  }
194  if (old_value) {
195  *old_value = y->values[i];
196  y->values[i] = NULL;
197  }
198  return true;
199  } else if (bv == YACHT_NO_KEY) {
200  return false; /* not present */
201  }
202  }
